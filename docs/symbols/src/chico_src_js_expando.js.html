  2 /**
  3  * Expando is a UI-Component.
  4  * @name Expando
  5  * @class Expando
  6  * @augments ch.Navs
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @return {Chico-UI Object}
 10  */
 11  
 12 ch.expando = function(conf){
 13 
 14     /**
 15      * Reference to a internal component instance, saves all the information and configuration properties.
 16      * @private
 17      * @name that
 18      * @type {Object}
 19      * @memberOf ch.Expando
 20      */
 21     var that = this;
 22 	
 23 	that.$element.addClass("ch-expando")
 24 		.children(":first").wrapInner("<span class=\"ch-expando-trigger\"></span>");
 25 		
 26     conf = ch.clon(conf);
 27     conf.open = conf.open || false;
 28 
 29 	that.conf = conf;
 30 	
 31 /**
 32  *	Inheritance
 33  */
 34 
 35     that = ch.navs.call(that);
 36     that.parent = ch.clon(that);
 37 
 38 /**
 39  *  Protected Members
 40  */ 
 41 
 42 	that.$content = that.$element.children().eq(1);
 43 	that.$trigger = that.$element.find(".ch-expando-trigger");
 44 	
 45 	that.show = function(event){
 46 		that.prevent(event);
 47 		
 48 		// Toggle
 49 		if ( that.active ) {
 50 			return that.hide(event);
 51 		};
 52 		
 53 		that.parent.show(event);
 54 		
 55 		return that;
 56 	};
 57 
 58 /**
 59  *  Public Members
 60  */
 61  
 62     /**
 63      * The component's instance unique identifier.
 64      * @public
 65      * @name uid
 66      * @type {Number}
 67      * @memberOf ch.Expando
 68      */
 69    	that["public"].uid = that.uid;
 70     
 71     /**
 72      * The element reference.
 73      * @public
 74      * @name element
 75      * @type {HTMLElement}
 76      * @memberOf ch.Expando
 77      */
 78  	that["public"].element = that.element;
 79  	
 80     /**
 81      * The component's type.
 82      * @public
 83      * @name type
 84      * @type {String}
 85      * @memberOf ch.Expando
 86      */
 87 	that["public"].type = that.type;
 88 	
 89     /**
 90      * Shows component's content.
 91      * @public
 92      * @name show
 93      * @return {Chico-UI Object}
 94      * @memberOf ch.Expando
 95      */
 96 	that["public"].show = function(){
 97 		that.show();
 98 		
 99 		return that["public"];
100 	};
101 
102     /**
103      * Hides component's content.
104      * @public
105      * @name hide
106      * @return {Chico-UI Object}
107      * @memberOf ch.Expando
108      */	
109 	that["public"].hide = function(){
110 		that.hide();
111 		
112 		return that["public"];
113 	};
114 	
115 
116 /**
117  *  Default event delegation
118  */		
119     
120 	// Trigger
121 	that.$trigger
122 		.bind('click', function(event){	that.show(event); })
123 		.addClass('ch-expando-trigger');
124 		
125 	// Content
126 	that.$content.addClass('ch-expando-content ch-hide');
127 	
128 	// Change default behaivor (close)
129 	if( conf.open ) that.show();
130 	
131     
132     // Create the publish object to be returned
133     conf.publish = that.publish;
134 
135 	return that;
136 
137 };
