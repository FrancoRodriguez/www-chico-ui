  2 /**
  3 * AutoComplete is a UI-Component.
  4 * @name AutoComplete
  5 * @class AutoComplete
  6 * @augments ch.Controls
  7 * @standalone
  8 * @memberOf ch
  9 * @param {Object} [conf] Object with configuration properties.
 10 * @param {String} [conf.url] The url pointing to the suggestions's service.
 11 * @param {String} [conf.message] It represent the text when no options are shown.
 12 * @param {Array} [conf.suggestions] The list of suggestions. Use it when you don't have server side suggestions service. Don't use conf.url with this option.
 13 * @returns itself
 14 * @example
 15 * // Create a new autoComplete with configuration.
 16 * var me = $(".example").autoComplete({
 17 *     "url": "http://site.com/mySuggestions?q=",
 18 *     "message": "Write..."
 19 * });
 20 */
 21  
 22 ch.autoComplete = function(conf){
 23 
 24 	/**
 25 	* Reference to a internal component instance, saves all the information and configuration properties.
 26 	* @private
 27 	* @name ch.AutoComplete#that
 28 	* @type object
 29 	*/
 30 	var that = this;
 31 		
 32 	conf = ch.clon(conf);
 33 	
 34 	that.conf = conf;
 35 	that.conf.icon = false;
 36 	that.conf.type = "autoComplete";
 37 	that.conf.message = conf.message || "Please write to be suggested";
 38 	that.conf.suggestions = conf.suggestions;
 39 	that.conf.showAll = conf.showAll;
 40 	
 41 /**
 42 *	Inheritance
 43 */
 44 	
 45 	that = ch.controls.call(that);
 46 	that.parent = ch.clon(that);
 47 	
 48 /**
 49 *  Private Members
 50 */
 51 
 52 	/**
 53 	* Adds keyboard events.
 54 	* @private
 55 	* @type Function
 56 	* @name ch.AutoComplete#shortcuts
 57 	*/
 58 	var shortcuts = function (items) {
 59 		$.each(items, function (i, e) {
 60 			$(e).bind("mouseenter", function () {
 61 				items.eq(that.selected).removeClass("ch-autoComplete-selected");
 62 				that.selected = i;
 63 				items.eq(that.selected).addClass("ch-autoComplete-selected");
 64 				
 65 				
 66 			}).bind("mouseleave",function () {
 67 				
 68 			}).bind("mousedown",function () {
 69 				that.$trigger.val(items.eq(that.selected).text());
 70 				that.$trigger.blur();
 71 			});
 72 		});
 73 	};
 74 
 75 	/**
 76 	* Select an item.
 77 	* @private
 78 	* @type Function
 79 	* @name ch.AutoComplete#selectItem
 80 	*/
 81 	var selectItem = function (arrow, event) {
 82 		that.prevent(event);
 83 		if (that.selected === (arrow === "bottom" ? that.items.length - 1 : 0)) { return; }
 84 		that.items.eq(that.selected).removeClass("ch-autoComplete-selected");
 85 		
 86 		if (arrow === "bottom") { that.selected += 1; } else { that.selected -= 1; }
 87 		that.items.eq(that.selected).addClass("ch-autoComplete-selected");
 88 		
 89 		
 90 	};
 91 
 92 /**
 93 *  Protected Members
 94 */
 95 
 96 	/**
 97 	* The number of the selected item.
 98 	* @protected
 99 	* @type Number
100 	* @name ch.AutoComplete#selected
101 	*/
102 	that.selected = -1;
103 
104 	/**
105 	* List of the shown suggestions.
106 	* @protected
107 	* @type Array
108 	* @name ch.AutoComplete#suggestions
109 	*/
110 	that.suggestions = that.conf.suggestions;
111 
112 	/**
113 	* The input where the AutoComplete works.
114 	* @protected
115 	* @type jQuery
116 	* @name ch.AutoComplete#$trigger
117 	*/
118 	that.$trigger = $(that.element);
119 
120 	/**
121 	* Inner reference to content container. Here is where the content will be added.
122 	* @protected
123 	* @type jQuery
124 	* @name ch.AutoComplete#$content
125 	*/
126 	that.$content = $("<ul>");
127 
128 	/**
129 	* The Float that show the suggestions's list.
130 	* @protected
131 	* @type ch.Layer
132 	* @name ch.AutoComplete#float
133 	*/
134 	that.float = that.$element.attr("autocomplete","off")
135 		// Initialize Layer component
136 		.layer({
137 			"event": "click",
138 			"content": that.$content,
139 			"points": conf.points,
140 			"classes": "ch-autoComplete",
141 			"closeButton": false,
142 			"offset": "0 0"
143 		})
144 		// Show callback
145 		.on("show", function () {
146 			// Old callback system
147 			that.callbacks.call(that, "onShow");
148 			// New callback
149 			that.float.width(that.$trigger.outerWidth());
150 			that.trigger("show");
151 		})
152 		// Hide callback
153 		.on("hide", function () {
154 			// Old callback
155 			that.callbacks.call(that, "onHide");
156 			// New callback
157 			that.trigger("hide");
158 		}).on("ready", function () {
159 			that.float.width((that.$trigger.outerWidth()));
160 		});
161 
162 	/**
163 	* It fills the content inside the element represented by the float.
164 	* @protected
165 	* @type Function
166 	* @name ch.AutoComplete#populateContent
167 	*/
168 	that.populateContent = function(result){
169 		var content = "<li>"+result.join("</li><li>")+"</li>";
170 		that.$content.html(content);
171 		that.float.content(that.$content);
172 		that.items = that.$content.find("li");
173 		that.selected = -1;
174 		shortcuts(that.items);
175 		return that;
176 	}
177 
178 	/**
179 	* It does the query to the server if configured an URL, or it does the query inside the array given.
180 	* @protected
181 	* @type Function
182 	* @name ch.AutoComplete#doQuery
183 	*/
184 	that.doQuery = function(event){
185 		var q = that.$trigger.val().toLowerCase();
186 		// When URL is configured it will execute an ajax request.
187 		if(that.conf.url!==undefined && q!==""){
188 			var result = $.ajax({
189 				url: that.conf.url + q,
190 				dataType:"jsonp",
191 				jsonp:conf.doQuery,
192 				crossDomain:true,
193 				success: function(data){}
194 			});
195 		// When not URL configured and suggestions array were configured it search inside the suggestions array.
196 		} else if(that.conf.url===undefined && q!=="") {
197 			var result = [];
198 			for(var a=(that.suggestions.length-1);(a+1);a--){
199 				var word = that.suggestions[a].toLowerCase();
200 				var exist = word.search(q);
201 				if(!exist){
202 					result.push(that.suggestions[a]);
203 				}
204 			}
205 			that.populateContent(result);
206 		}
207 		return that;
208 	}
209 
210 	/**
211 	* It gives the main behavior(focus and blur) to the $trigger.
212 	* @protected
213 	* @type Function
214 	* @name ch.AutoComplete#configBehavior
215 	*/
216 	that.configBehavior = function () {
217 		that.$trigger
218 			.addClass("ch-" + that.type + "-trigger")
219 			.bind("focus", function (event) { 
220 				  that.show(event);
221 			})
222 			.bind("blur", function (event) { 
223 				 that.hide(event);
224 			});
225 		that.$content.addClass("ch-" + that.type + "-content");
226 		return that;
227 	};
228 
229 	/**
230 	* Internal show method. It adds the behavior.
231 	* @protected
232 	* @type Function
233 	* @name ch.AutoComplete#show
234 	*/
235 	that.show = function(event){
236 		// BACKSPACE key bheavior. When backspace go to the start show the message
237 		ch.utils.document.bind(ch.events.KEY.BACKSPACE, function (x, event) { if( event.target.value.length<=1 ){ that.populateContent([that.conf.message]); }});
238 		// UP ARROW key behavior
239 		ch.utils.document.bind(ch.events.KEY.UP_ARROW, function (x, event) { selectItem("up", event); });
240 		// DOWN ARROW key behavior
241 		ch.utils.document.bind(ch.events.KEY.DOWN_ARROW, function (x, event) { selectItem("bottom", event); });
242 		// ESC key behavior
243 		ch.utils.document.bind(ch.events.KEY.ESC, function (x, event) { that.$trigger.trigger("blur"); });
244 		// ENTER key behavior
245 		ch.utils.document.bind(ch.events.KEY.ENTER, function (x, event) {  that.$trigger.val(that.items.eq(that.selected).text()); that.$trigger.trigger("blur"); });
246 		// Global keyup behavior
247 		ch.utils.document.bind("keyup", function (event) { 
248 			if(event.keyCode!==38 && event.keyCode!==40  && event.keyCode!==13  && event.keyCode!==27){that.doQuery(event);}
249 			 
250 		});
251 		return that;
252 	}
253 
254 	/**
255 	* Internal hide method. It removes the behavior.
256 	* @protected
257 	* @type Function
258 	* @name ch.AutoComplete#hide
259 	*/
260 	that.hide = function(event){
261 		that.doQuery(event);
262 		ch.utils.document.unbind("keyup " + ch.events.KEY.ENTER + " " + ch.events.KEY.ESC + " " + ch.events.KEY.UP_ARROW + " " + ch.events.KEY.DOWN_ARROW + " " + ch.events.KEY.BACKSPACE);
263 		that.float.hide();
264 		return that;
265 	}
266 	//Fills the Float with the message.
267 	that.populateContent([that.conf.message]);
268 /**
269 *  Public Members
270 */
271  
272 	/**
273 	* The component's instance unique identifier.
274 	* @public
275 	* @name ch.AutoComplete#uid
276 	* @type number
277 	*/
278 	
279 	/**
280 	* The element reference.
281 	* @public
282 	* @name ch.AutoComplete#element
283 	* @type HTMLElement
284 	*/
285 	
286 	/**
287 	* The component's type.
288 	* @public
289 	* @name ch.AutoComplete#type
290 	* @type string
291 	*/
292 	
293 	/**
294 	* Shows component's content.
295 	* @public
296 	* @function
297 	* @name ch.AutoComplete#show
298 	* @returns itself
299 	*/
300 	that["public"].show = function(){
301 		that.show();
302 		return that["public"];
303 	};
304 
305 	/**
306 	* Hides component's content.
307 	* @public
308 	* @function
309 	* @name ch.AutoComplete#hide
310 	* @returns itself
311 	*/	
312 	that["public"].hide = function(){
313 		that.hide(ch.events.KEY.ESC);
314 		return that["public"];
315 	};
316 
317 	/**
318 	* Add suggestions to be shown.
319 	* @public
320 	* @function
321 	* @name ch.AutoComplete#suggest
322 	* @returns itself
323 	*/	
324 	that["public"].suggest = function(data){
325 		that.suggestions = data;
326 		that.populateContent(that.suggestions);
327 		return that["public"];
328 	};
329 
330 /**
331 *  Default event delegation
332 */	
333 	
334 	that.configBehavior();
335 	
336 	/**
337 	* Triggers when the component is ready to use (Since 0.8.0).
338 	* @name ch.AutoComplete#ready
339 	* @event
340 	* @public
341 	* @example
342 	* // Following the first example, using 'me' as autoComplete's instance controller:
343 	* me.on("ready",function () {
344 	*	this.show();
345 	* });
346 	*/
347 	setTimeout(function(){ that.trigger("ready")}, 50);
348 
349 	return that;
350 };
