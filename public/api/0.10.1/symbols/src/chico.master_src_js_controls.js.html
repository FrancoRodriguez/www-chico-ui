1 /**
  2 * Abstract class that brings the functionality of all form controls.
  3 * @abstract
  4 * @name Controls
  5 * @class Controls 
  6 * @augments ch.Uiobject
  7 * @memberOf ch
  8 * @returns itself
  9 * @see ch.Countdown
 10 * @see ch.Validator
 11 * @see ch.AutoComplete
 12 * @see ch.Calendar
 13 * @see ch.Mask
 14 */
 15 
 16 ch.controls = function () {
 17 
 18 	/**
 19 	* Reference to a internal component instance, saves all the information and configuration properties.
 20 	* @name ch.Controls#that
 21 	* @type Object
 22 	*/
 23 	var that = this,
 24 
 25 		conf = that.conf;
 26 
 27 /**
 28 *  Inheritance
 29 */
 30 	that = ch.uiobject.call(that);
 31 	that.parent = ch.clon(that);
 32 
 33 /**
 34 *  Protected Members
 35 */
 36 
 37 	/**
 38 	* Reference to the elements to be watched (a.k.a. form inputs).
 39 	* @protected
 40 	* @type Object
 41 	* @name ch.Controls#watcher
 42 	*/
 43 	that.watcher = {
 44 		"elements": that.element,
 45 		"$elements": that.$element,
 46 		"form": that.$element.parents("form"),
 47 		"on": function (event, callback) {
 48 			// TODO: pluralize this
 49 			this.$elements.bind(event, callback);
 50 		},
 51 		"content": function (data) {
 52 			// TODO: pluralize this
 53 			this.$elements.val(data);
 54 		}
 55 	};
 56 	
 57 	/**
 58 	* Reference to the Float component instanced.
 59 	* @protected
 60 	* @type Object
 61 	* @name ch.Controls#float
 62 	*/
 63 	that.float = {};
 64 
 65 /**
 66 *  Public Members
 67 */
 68 
 69 	/**
 70 	* Monitors all form controls associated with this component
 71 	* @public
 72 	* @name ch.Controls#watcher
 73 	* @type Object
 74 	*/
 75 	that["public"].watcher = that.watcher;
 76 
 77 	return that;
 78 };
