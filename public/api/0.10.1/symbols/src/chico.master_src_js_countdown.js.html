1 /**
  2 * Counts the amount of characters that user can enter in a form control and limit the length of value of input.
  3 * @name Countdown
  4 * @class Countdown
  5 * @augments ch.Controls
  6 * @memberOf ch
  7 * @param {Object} conf Object with configuration properties.
  8 * @param {Number} conf.max Number of the maximum amount of characters user can input in form control.
  9 * @param {String} [conf.plural] Message of remaining amount of characters, when it's different to 1. The variable that represents the number to be replaced, should be a hash. By default this parameter is "# characters left.".
 10 * @param {String} [conf.singular] Message of remaining amount of characters, when it's only 1. The variable that represents the number to be replaced, should be a hash. By default this parameter is "# character left.".
 11 * @returns itself
 12 * @example
 13 * // Create a new Countdown with configuration.
 14 * var me = $(".some-form-control").countdown({
 15 *     "max": 500,
 16 *     "plural": "Restan # caracteres.",
 17 *     "singular": "Resta # caracter."
 18 * });
 19 * @example
 20 * // Create a simple Countdown
 21 * var me = $(".some-form-control").countdown(500);
 22 * // Now 'me' is a reference to the Countdown instance controller.
 23 */
 24 
 25 ch.countdown = function (conf) {
 26 
 27 	/**
 28 	* Reference to an internal component instance, saves all the information and configuration properties.
 29 	* @private
 30 	* @name ch.Countdown#that
 31 	* @type Object
 32 	*/
 33 	var that = this;
 34 	
 35 	conf = ch.clon(conf);
 36 	
 37 	// Max length of content
 38 	conf.max = parseInt(conf.max) || conf.value ||Â parseInt(conf.msg) || 500;
 39 	
 40 	// Messages
 41 	conf.plural = conf.plural || "# characters left.";
 42 	conf.singular = conf.singular || "# character left.";
 43 
 44 	that.conf = conf;
 45 
 46 /**
 47 *	Inheritance
 48 */
 49 
 50 	that = ch.controls.call(that);
 51 	that.parent = ch.clon(that);
 52 
 53 /**
 54 *	Private Members
 55 */
 56 	/**
 57 	* Length of value of form control.
 58 	* @private
 59 	* @name ch.Countdown#contentLength
 60 	* @type Number
 61 	*/
 62 	var contentLength = that.$element.val().length,
 63 	
 64 	/**
 65 	* Amount of free characters until full the field.
 66 	* @private
 67 	* @name ch.Countdown#remaining
 68 	* @type Number
 69 	*/
 70 		remaining = conf.max - contentLength;
 71 	
 72 	/**
 73 	* Change the visible message of remaining characters.
 74 	* @private
 75 	* @name ch.Countdown#updateRemaining
 76 	* @function
 77 	* @param num {Number} Remaining characters.
 78 	*/
 79 		updateRemaining = (function () {
 80 			
 81 			// Singular or Plural message depending on amount of remaining characters
 82 			var message = (remaining === 1) ? conf.singular : conf.plural,
 83 			
 84 			// Create the DOM Element when message will be shown
 85 				$display = $("<p class=\"ch-form-hint\">" + message.replace("#", remaining) + "</p>").insertAfter(that.$element);
 86 			
 87 			// Real function
 88 			return function (num) {
 89 				
 90 				// Singular or Plural message depending on amount of remaining characters
 91 				var message = (num === 1) ? conf.singular : conf.plural;
 92 				
 93 				// Update DOM text
 94 				$display.text(message.replace("#", num));
 95 				
 96 				// Update amount of remaining characters
 97 				remaining = num;
 98 				
 99 			};
100 		
101 		}());
102 
103 /**
104 *	Protected Members
105 */
106 
107 	/**
108 	* Process input of data on form control and updates remaining amount of characters or limits the content length
109 	* @protected
110 	* @name ch.Countdown#process
111 	* @function
112 	*/
113 	that.process = function () {
114 		
115 		// TODO: fix watcher
116 		var len = that.watcher.$elements.val().length;
117 		
118 		// Countdown or Countup
119 		if ((len > contentLength && len <= conf.max) || (len < contentLength && len >= 0)) {
120 			
121 			// Change visible message of remaining characters
122 			updateRemaining(remaining - (len - contentLength));
123 			
124 			// Update length of value of form control.
125 			contentLength = len;
126 		
127 		// Limit Count
128 		} else if (len > contentLength && len > conf.max) {
129 			
130 			// Cut the string value of form control
131 			// TODO: fix watcher
132 			//that.watcher.content(that.watcher.content().substr(0, conf.max));
133 			that.watcher.$elements.val(that.watcher.$elements.val().substr(0, conf.max));
134 			
135 		};
136 		
137 	};

171

177 	that.watcher.on("keyup keypress paste", function () { setTimeout(that.process, 0); });

186 	* me.on("ready",function () {
187 	*	this.element;
188 	* });
189 	*/
190 	setTimeout(function () { that.trigger("ready"); }, 50);
191 
192 	return that;
193 };
