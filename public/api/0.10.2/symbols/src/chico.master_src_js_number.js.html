  2 /**
  3 * Validate numbers.
  4 * @name Number
  5 * @class Number
  6 * @interface
  7 * @augments ch.Controls
  8 * @requires ch.Validation
  9 * @memberOf ch
 10 * @param {String} [message] Validation message.
 11 * @returns itself
 12 * @see ch.Validation
 13 * @see ch.Required
 14 * @see ch.Custom
 15 * @see ch.String
 16 * @see ch.Validator
 17 * @see ch.Condition
 18 * @example
 19 * // Create a number validation
 20 * $("input").number("This field must be a number.");
 21 */
 22 ch.extend("validation").as("number", function(conf) {
 23 	
 24 	// Define the conditions of this interface
 25 	conf.condition = {
 26 		name: "number",
 27 		patt: /^(-?[0-9\s]+)$/,
 28 		message: conf.msg || conf.message
 29 	};
 30 
 31 	return conf;
 32 
 33 });

 49 * @see ch.String

 54 */
 55 ch.extend("validation").as("min", function (conf) {
 56 
 57 	conf.condition = {
 58 		name: "min",
 59 		expr: function(a,b) { return a >= b },
 60 		message: conf.msg || conf.message,
 61 		value: conf.value
 62 	};
 63 
 64 	return conf;
