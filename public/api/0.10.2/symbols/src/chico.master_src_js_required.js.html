1 /**
  2 * Creates required validations.
  3 * @name Required
  4 * @class Required
  5 * @interface
  6 * @augments ch.Controls
  7 * @requires ch.Validation
  8 * @memberOf ch
  9 * @param {String} [message] Validation message.
 10 * @returns itself
 11 * @see ch.Validation
 12 * @see ch.Custom
 13 * @see ch.Number
 14 * @see ch.String
 15 * @see ch.Validator
 16 * @see ch.Condition
 17 * @example
 18 * // Simple validation
 19 * $("input").required("This field is required");
 20 */
 21 ch.extend("validation").as("required", function(conf) {
 22 
 23 	conf.condition = {
 24 		name: "required",
 25 		expr: function(e) {
 26 
 27 			var $e = $(e);
 28 
 29 			var tag = ( $e.hasClass("options") || $e.hasClass("ch-form-options")) ? "OPTIONS" : e.tagName;
 30 			switch (tag) {
 31 				case 'OPTIONS':
 32 					return $e.find('input:checked').length !== 0;
 33 				break;
 34 
 35 				case 'SELECT':
 36 					var val = $e.val();
 37 					return (val != "-1" && val != "");
 38 				break;
 39 
 40 				case 'INPUT':
 41 				case 'TEXTAREA':
 42 					return $.trim($e.val()).length !== 0;
 43 				break;
 44 			};
 45 		},
 46 		message: conf.msg || conf.message,
 47 		value: conf.value
 48 	};
 49 
 50 	return conf;
 51 
 52 });
