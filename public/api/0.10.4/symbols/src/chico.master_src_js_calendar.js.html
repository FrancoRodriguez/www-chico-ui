236 	/**
237 	* Indicates if an specific date is selected or not (including date ranges and simple dates).
238 	* @private
239 	* @name ch.Calendar#isSelectable
240 	* @function
241 	* @param year
242 	* @param month
243 	* @param day
244 	* @return Boolean
245 	*/
246 		isSelectable = function (year, month, day) {
247 			
248 			if (!selected) { return; }
249 			
250 			var yepnope = false;
251 			
252 			// Simple selection
253 			if (!ch.utils.isArray(selected)) {
254 				if (year === selected.year && month === selected.month && day === selected.day) {
255 					return yepnope = true;
256 				}
257 			// Multiple selection (ranges)
258 			} else {
259 				$.each(selected, function (i, e) {
260 					// Simple date
261 					if (!ch.utils.isArray(e)) {
262 						if (year === e.year && month === e.month && day === e.day) {
263 							return yepnope = true;
264 						}
265 					// Range
266 					} else {
267 						if (
268 							(year >= e[0].year && month >= e[0].month && day >= e[0].day) &&
269 							(year <= e[1].year && month <= e[1].month && day <= e[1].day)
270 						) {
271 							return yepnope = true;
272 						}
273 					}
274 				});
275 			}
276 			
277 			return yepnope;
278 		},

304 	/**
305 	* Creates a complete month in a table.
306 	* @private
307 	* @function
308 	* @name ch.Calendar#createTable
309 	* @param date {Object} Date from will be created the entire month.
310 	* @return jQuery Object
311 	*/
312 		createTable = function (date) {
313 
314 			// Total amount of days into month
315 			var cells = (function () {
316 
317 				// Amount of days of current month
318 				var currentMonth = new Date(date.year, date.month, 0).getDate(),
319 
320 				// Amount of days of previous month
321 					prevMonth = new Date([date.year, date.month, "01"].join("/")).getDay(),
322 
323 				// Merge amount of previous and current month
324 					subtotal = prevMonth + currentMonth,
325 
326 				// Amount of days into last week of month
327 					latest = subtotal % 7,
328 
329 				// Amount of days of next month
330 					nextMonth = (latest > 0) ? 7 - latest : 0;
331 
332 				return {
333 					"previous": prevMonth,
334 					"subtotal": subtotal,
335 					"total": subtotal + nextMonth
336 				};
337 
338 			}()),
339 
340 			// Final array with month table structure
341 				r = [
342 					"<table class=\"ch-calendar-month\" role=\"grid\" id=\"ch-calendar-grid-" + that.uid + "\">",
343 					"<caption>" + MONTHS_NAMES[date.month - 1] + " - " + date.year + "</caption>",
344 					thead,
345 					"<tbody>",
346 					"<tr class=\"ch-week\" role=\"row\">"
347 				];
348 
349 			// Iteration of weekdays
350 			for (var i = 0; i < cells.total; i += 1) {
351 
352 				// Push an empty cell on previous and next month
353 				if (i < cells.previous || i > cells.subtotal - 1) {
354 					r.push("<td role=\"gridcell\" class=\"ch-calendar-other\">X</td>");
355 					continue;
356 				}
357 
358 				// Positive number of iteration
359 				var positive = i + 1,
360 
361 				// Day number
362 					day = positive - cells.previous,
363 
364 				// Define if it's the day selected
365 					isSelected = isSelectable(date.year, date.month, day);
366 
367 				// Create cell
368 				r.push(
369 					// Open cell structure including WAI-ARIA and classnames space opening
370 					"<td role=\"gridcell\"" + (isSelected ? " aria-selected=\"true\"" : "") + " class=\"ch-calendar-day",
371 
372 					// Add Today classname if it's necesary
373 					(date.year === today.year && date.month === today.month && day === today.day) ? " ch-calendar-today" : null,
374 
375 					// Add Selected classname if it's necesary
376 					(isSelected ? " ch-calendar-selected" : null),
377 
378 					// From/to range. Disabling cells
379 					(
380 						// Disable cell if it's out of FROM range
381 						(from && day < from.day && date.month === from.month && date.year === from.year) ||
382 
383 						// Disable cell if it's out of TO range
384 						(to && day > to.day && date.month === to.month && date.year === to.year)
385 
386 					) ? " ch-calendar-disabled" : null,
387 
388 					// Close classnames attribute and print content closing cell structure
389 					"\">" + day + "</td>"
390 				);
391 
392 				// Cut week if there are seven days
393 				if (positive % 7 === 0) {
394 					r.push("</tr><tr class=\"ch-calendar-week\" role=\"row\">");
395 				}
396 
397 			};
398 
399 			// Return table object
400 			return r.join("");
401 
402 		},

730 	* @name ch.Calendar#select
731 	* @param {string} "YYYY/MM/DD".
732 	* @return itself
733 	*/
734 	that["public"].select = function (date) {
735 
736 		// Getter
737 		if (!date) { return FORMAT_DATE[conf.format](selected); }
738 
739 		// Setter
740 		select((date === "today") ? today : createDateObject(parseDate(date)));
741 
742 		return that["public"];
743 
744 	};

779 	* @public
780 	* @name ch.Calendar#next

784 	* @default Next month
785 	*/
