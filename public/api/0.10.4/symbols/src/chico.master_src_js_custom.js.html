  2 /**
  3 * Create custom validation interfaces for Validator validation engine.
  4 * @name Custom
  5 * @class Custom
  6 * @interface
  7 * @augments ch.Controls
  8 * @requires ch.Validation
  9 * @memberOf ch
 10 * @param {Object} [conf] Object with configuration properties.
 11 * @param {String} [conf.message] Validation message.
 12 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
 13 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 14 * @param {String} [conf.context] It's a reference to position the validation-bubble
 15 * @param {Function} function Custom function to evaluete a value.
 16 * @returns itself
 17 * @see ch.Validation
 18 * @see ch.Required
 19 * @see ch.Number
 20 * @see ch.String
 21 * @see ch.Validator
 22 * @see ch.Condition
 23 * @example
 24 * // Validate a even number
 25 * $("input").custom(function (value) {
 26 * 	return (value%2==0) ? true : false;
 27 * }, "Enter a even number");
 28 */
 29 ch.extend("validation").as("custom", function(conf) {
 30 	
 31 	if (!conf.lambda) {
 32 		alert("Custom Validation fatal error: Need a function to evaluate, try $().custom(function(){},\"Message\");");
 33 	}
 34 
 35 	// Define the conditions of this interface
 36 	conf.condition = {
 37 		// I don't have pre-conditions, comes within conf.lambda argument
 38 		name: "custom",
 39 		func: conf.lambda,
 40 		message: conf.msg || conf.message || "Error"
 41 	};
 42 
 43 	return conf;
 44 });
