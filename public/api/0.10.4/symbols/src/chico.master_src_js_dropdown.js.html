1 /**
  2 * A navegable list of items, UI-Object.
  3 * @name Dropdown
  4 * @class Dropdown
  5 * @augments ch.Navs
  6 * @standalone
  7 * @requires ch.Positioner
  8 * @memberOf ch
  9 * @param {Object} [conf] Object with configuration properties.
 10 * @param {Boolean} [conf.open] Shows the dropdown open when component was loaded. By default, the value is false.
 11 * @param {Boolean} [conf.icon] Shows an arrow as icon. By default, the value is true.
 12 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or centered by default: "cm cm".
 13 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are disable.
 14 * @returns itself
 15 * @example
 16 * // Create a new dropdown with configuration.
 17 * var me = $(".example").dropdown({
 18 *     "open": true,
 19 *     "icon": false,
 20 *     "points": "lt lt",
 21 *     "fx": true
 22 * });
 23 * @example
 24 * // Create a new dropdown without configuration.
 25 * var me = $(".example").dropdown();
 26 */
 27 
 28 ch.dropdown = function (conf) {
 29 
 30 	/**
 31 	* Reference to a internal component instance, saves all the information and configuration properties.
 32 	* @private
 33 	* @name ch.Dropdown#that
 34 	* @type object
 35 	*/
 36 	var that = this;
 37 
 38 	conf = ch.clon(conf);
 39 	
 40 	conf.reposition = ch.utils.hasOwn(conf, "reposition") ? conf.reposition : true;
 41 	
 42 	that.conf = conf;
 43 
 44 /**
 45 *	Inheritance
 46 */
 47 
 48 	that = ch.navs.call(that);
 49 	that.parent = ch.clon(that);
 50 
 51 /**
 52 *  Private Members
 53 */
 54 	/**
 55 	* Adds keyboard events.
 56 	* @private
 57 	* @function
 58 	* @name ch.Dropdown#shortcuts
 59 	*/
 60 	var shortcuts = function (items) {
 61 
 62 		// Keyboard support
 63 		var selected = 0;
 64 
 65 		// Item selected by mouseover
 66 		// TODO: It's over keyboard selection and it is generating double selection.
 67 		$.each(items, function (i, e) {
 68 			$(e).bind("mouseenter", function () {
 69 				selected = i;
 70 				items.eq(selected).focus();
 71 			});
 72 		});
 73 
 74 		var selectItem = function (arrow, event) {
 75 			that.prevent(event);
 76 
 77 			if (selected === (arrow === "bottom" ? items.length - 1 : 0)) { return; }
 78 
 79 			items.eq(selected).blur();
 80 
 81 			if (arrow === "bottom") { selected += 1; } else { selected -= 1; }
 82 			
 83 			items.eq(selected).focus();
 84 		};
 85 		
 86 		// Arrows
 87 		ch.utils.document.bind(ch.events.KEY.UP_ARROW, function (x, event) { selectItem("up", event); });
 88 		ch.utils.document.bind(ch.events.KEY.DOWN_ARROW, function (x, event) { selectItem("bottom", event); });
 89 	};
 90 
 91 
 92 /**
 93 *  Protected Members
 94 */
 95 	/**
 96 	* The component's trigger.
 97 	* @private
 98 	* @name ch.Dropdown#$trigger
 99 	* @type jQuery
100 	*/
101 	that.$trigger = (function () {
102 		
103 		var $el = that.$element.children().eq(0);
104 		
105 		if (!that.$element.hasClass("secondary") || !that.$element.hasClass("ch-dropdown-skin")) { $el.addClass("ch-btn-skin ch-btn-small"); }
106 		
107 		return $el;
108 		
109 	}());
110 
111 	/**
112 	* The component's content.
113 	* @private
114 	* @name ch.Dropdown#$content
115 	* @type jQuery
116 	*/
117 	that.$content = (function () {
118 		
119 		// jQuery Object
120 		var $content = that.$trigger.next()
121 		// Visible
122 			.removeClass("ch-hide")
123 		// Prevent click on content (except links)
124 			.bind("click", function(event) {
125 				if ((event.target || event.srcElement).tagName === "A") {
126 					that.hide();
127 				}
128 				event.stopPropagation();
129 			})
130 		// WAI-ARIA properties
131 			.attr({ "role": "menu", "aria-hidden": "true" });
132 		
133 		// WAI-ARIA for items into content
134 		$content.children("a").attr("role", "menuitem");
135 
136 		// Position
137 		that.position = ch.positioner({
138 			"element": $content,
139 			"context": that.$trigger,
140 			"points": (conf.points || "lt lb"),
141 			"offset": "0 -1",
142 			"reposition": conf.reposition
143 		});
144 		
145 		return $content;
146 	}());
147 
148 
149 	that.show = function (event) {
150 		
151 		// Stop propagation
152 		that.prevent(event);
153 		
154 		// Z-index of content
155 		that.$content.css("z-index", ch.utils.zIndex += 1).attr("aria-hidden", "false");
156 		
157 		// Z-index of trigger over content (secondary / skin dropdown)
158 		if (that.$element.hasClass("secondary") || that.$element.hasClass("ch-dropdown-skin")) { that.$trigger.css("z-index", ch.utils.zIndex += 1); }
159 		
160 		// Inheritance show
161 		that.parent.show(event);
162 		
163 		// Refresh position
164 		that.position("refresh");
165 
166 		// Reset all dropdowns except itself
167 		$.each(ch.instances.dropdown, function (i, e) { 
168 			if (e.uid !== that.uid) { e.hide(); }
169 		});
170 
171 		// Close events
172 		ch.utils.document.one("click " + ch.events.KEY.ESC, function () { that.hide(); });
173 
174 		// Keyboard support
175 		var items = that.$content.find("a");
176 		// Select first anchor child by default
177 			items.eq(0).focus();
178 
179 		if (items.length > 1) { shortcuts(items); };
180 
181 		return that;
182 	};
183 
184 	that.hide = function (event) {
185 
186 		that.parent.hide(event);
187 		
188 		that.$content.attr("aria-hidden", "true");
189 
190 		// Unbind events
191 		ch.utils.document.unbind(ch.events.KEY.ESC + " " + ch.events.KEY.UP_ARROW + " " + ch.events.KEY.DOWN_ARROW);
192 
193 		return that;
194 	};
195

197 *  Public Members
198 */
199  
200 	/**
201 	* The component's instance unique identifier.
202 	* @public
203 	* @name ch.Dropdown#uid
204 	* @type number
205 	*/
206 	
207 	/**
208 	* The element reference.
209 	* @public
210 	* @name ch.Dropdown#element
211 	* @type HTMLElement
212 	*/
213 	
214 	/**
215 	* The component's type.
216 	* @public
217 	* @name ch.Dropdown#type
218 	* @type string
219 	*/	
220 	
221 	/**
222 	* Shows component's content.
223 	* @public
224 	* @function
225 	* @name ch.Dropdown#show
226 	* @returns itself
227 	*/
228 	that["public"].show = function () {
229 		that.show();
230 		
231 		return that["public"];
232 	};
233 
234 	/**
235 	* Hides component's content.
236 	* @public
237 	* @function
238 	* @name ch.Dropdown#hide
239 	* @returns itself
240 	*/ 
241 	that["public"].hide = function () {
242 		that.hide();
243 		
244 		return that["public"];
245 	};
246 	
247 	/**
248 	* Positioning configuration.
249 	* @public
250 	* @function
251 	* @name ch.Dropdown#position
252 	*/
253 	that["public"].position = that.position;
254 
255 /** 
256 *  Default event delegation
257 */			
258 
259 	that.configBehavior();
260 
261 	ch.utils.avoidTextSelection(that.$trigger);
262 	
263 	/**
264 	* Triggers when the component is ready to use (Since 0.8.0).
265 	* @name ch.Dropdown#ready
266 	* @event
267 	* @public
268 	* @since 0.8.0
269 	* @example
270 	* // Following the first example, using 'me' as dropdown's instance controller:
271 	* me.on("ready",function () {
272 	*	this.show();
273 	* });
274 	*/
275 	setTimeout(function(){ that.trigger("ready")}, 50);
276 
277 	return that;
278 };
279 
280 ch.factory("dropdown");
