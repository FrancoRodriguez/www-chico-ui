 51 	/**
 52 	* Closable behavior.
 53 	* @private
 54 	* @function
 55 	* @name ch.Floats#closable
 56 	*/
 57 	// TODO: Create "closable" interface
 58 	var closable = (function () {
 59 		/**
 60 		* Returns any if the component closes automatic. 
 61 		* @public
 62 		* @name ch.Floats#closable
 63 		* @function
 64 		* @example
 65 		* // to get the height
 66 		* me.closable() // true | false | "button"
 67 		* @returns boolean | string
 68 		*/
 69 		that["public"].closable = function () {
 70 			return that.closable;
 71 		};
 72 
 73 		
 74 		return function () {
 75 			
 76 			// Closable Off: don't anything
 77 			if (!that.closable) { return; }
 78 
 79 			// Closable On
 80 
 81 			if (ch.utils.hasOwn(conf, "closeButton") && conf.closeButton || ch.utils.hasOwn(conf, "event") && conf.event === "click") {
 82 				// Append close buttons	
 83 				// It will close with close button
 84 				that.$container
 85 					.prepend("<a class=\"ch-close\" style=\"z-index:" + (ch.utils.zIndex += 1) + "\">×</a>")
 86 					.bind("click", function (event) {
 87 						if ($(event.target || event.srcElement).hasClass("ch-close")) { 
 88 							that.innerHide(event);
 89 						}
 90 					});
 91 			}
 92 
 93 			// It will close only with close button
 94 			if (that.closable === "button") {
 95 				return;
 96 			}
 97 
 98 			// Default Closable behavior
 99 			// It will close with click on document, too
100 			that.on("show", function () {
101 				ch.utils.document.one("click", that.innerHide);
102 			});
103 
104 			// Stop event propatation, if click container.
105 			that.$container.bind("click", function (event) {
106 				event.stopPropagation();
107 			});
108 
109 			// and ESC key support
110 			ch.utils.document.bind(ch.events.KEY.ESC, that.innerHide);
111 		};
112 
113 	})();

143 	/**
144 	* Inner function that resolves the component's layout and returns a static reference.
145 	* @protected
146 	* @name ch.Floats#$container
147 	* @type jQuery
148 	*/
149 	that.$container = (function () { // Create Layout
150 		
151 		// Final jQuery Object
152 		var $container,
153 		
154 		// Component with close button and keyboard binding for close
155 		//	closable = ch.utils.hasOwn(conf, "closeButton") && conf.closeButton,
156 		
157 		// HTML Div Element with role for WAI-ARIA
158 			container = ["<div role=\"" + conf.aria.role + "\""];
159 			
160 		// ID for WAI-ARIA
161 		if (ch.utils.hasOwn(conf.aria, "identifier")) {
162 			
163 			// Generated ID using component name and its instance order
164 			var id = "ch-" + that.type + "-" + (ch.utils.hasOwn(ch.instances, that.type) ? ch.instances[that.type].length + 1 : "1");
165 			
166 			// Add ID to container element
167 			container.push(" id=\"" + id + "\"");
168 			
169 			// Add aria attribute to trigger element
170 			that.$element.attr(conf.aria.identifier, id);
171 		}
172 		
173 		// Classname with component type and extra classes from conf.classes
174 		container.push(" class=\"ch-hide ch-" + that.type + (ch.utils.hasOwn(conf, "classes") ? " " + conf.classes : "") + "\"");
175 		
176 		// Z-index
177 		container.push(" style=\"z-index:" + (ch.utils.zIndex += 1) + ";");
178 		
179 		// Width
180 		if (ch.utils.hasOwn(conf, "width")) {
181 			container.push("width:" + conf.width + ((typeof conf.width === "number") ? "px;" : ";"));
182 		}
183 		
184 		// Height
185 		if (ch.utils.hasOwn(conf, "height")) {
186 			container.push("height:" + conf.height + ((typeof conf.height === "number") ? "px;" : ";"));
187 		}
188 		
189 		// Style and tag close
190 		container.push("\">");
191 		
192 		// Create cone
193 		if (ch.utils.hasOwn(conf, "cone")) { container.push("<div class=\"ch-" + that.type + "-cone\"></div>"); }
194 		
195 		// Create close button
196 		//if (closable) { container.push("<div class=\"btn close\" style=\"z-index:" + (ch.utils.zIndex += 1) + "\"></div>"); }
197 		
198 		// Tag close
199 		container.push("</div>");
200 		
201 		// jQuery Object generated from string
202 		$container = $(container.join(""));
203 		
204 		// Close behavior bindings
205 		/*if (closable) {
206 			// Close button event delegation
207 			$container.bind("click", function (event) {
208 				if ($(event.target || event.srcElement).hasClass("close")) { that.innerHide(event); }
209 			});
210 			
211 			// ESC key support
212 			ch.utils.document.bind(ch.events.KEY.ESC, function (event) { that.innerHide(event); });
213 		}*/
214 		
215 		// Efects configuration
216 		conf.fx = ch.utils.hasOwn(conf, "fx") ? conf.fx : true;
217 
218 		// Position component configuration
219 		conf.position = conf.position || {};
220 		conf.position.element = $container;
221 		conf.position.reposition = ch.utils.hasOwn(conf, "reposition") ? conf.reposition : true;
222

264

266 
267 		// Get the original that.source
268 		var originalSource = that.source;
269 
270 		if (ch.utils.hasOwn(conf, "onContentError")) {
271 			conf.onContentError.call(data.context, data.jqXHR, data.textStatus, data.errorThrown);
272 		}
273 
274 		// Reset content configuration
275 		that.source = originalSource;
276 		that.staticContent = undefined;
277 
278 		if (ch.utils.hasOwn(conf, "position")) {
279 		   ch.positioner(conf.position);
280 		}

284 	/**
285 	* Inner show method. Attach the component layout to the DOM tree.
286 	* @protected
287 	* @function
288 	* @name ch.Floats#innerShow
289 	* @returns itself
290 	*/
291 	that.innerShow = function (event) {
292 		if (event) {
293 			that.prevent(event);
294 		}
295 
296 		// Avoid showing things that are already shown
297 		if (that.active) return;
298 
299 		// Add layout to DOM tree
300 		// Increment zIndex
301 		that.$container
302 			.appendTo("body")
303 			.css("z-index", ch.utils.zIndex++);
304 
305 		// This make a reflow, but we need that the static content appends to DOM
306 		// Get content
307 		that.content();
308 
309 		/**
310 		* Triggers when component is visible.
311 		* @name ch.Floats#show
312 		* @event
313 		* @public
314 		*/
315 		// Show component with effects
316 		if (conf.fx) {
317 			that.$container.fadeIn("fast", function () {
318 				// new callbacks
319 				that.trigger("show");
320 				// Old callback system
321 				that.callbacks('onShow');
322 			});
323 		} else {
324 		// Show component without effects
325 			that.$container.removeClass("ch-hide");
326 			// new callbacks
327 			that.trigger("show");
328 			// Old callback system
329 			that.callbacks('onShow');
330 		}
331 
332 		that.position("refresh");
333 		
334 		that.active = true;
335 
336 		return that;
337 	};
338 
339 	/**
340 	* Inner hide method. Hides the component and detach it from DOM tree.
341 	* @protected
342 	* @function
343 	* @name ch.Floats#innerHide
344 	* @returns itself
345 	*/
346 	that.innerHide = function (event) {
347 		
348 		if (event) {
349 			event.stopPropagation();
350 		}
351 
352 		if (!that.active) {
353 			return;
354 		}
355 
356 		var afterHide = function () {
357 
358 			that.active = false;
359 
360 		/**
361 		* Triggers when component is not longer visible.
362 		* @name ch.Floats#hide
363 		* @event
364 		* @public
365 		*/
366 			// new callbacks
367 			that.trigger("hide");
368 			// Old callback system
369 			that.callbacks('onHide');
370 
371 			that.$container.detach();
372 
373 		};
374 
375 		// Show component with effects
376 		if (conf.fx) {
377 			that.$container.fadeOut("fast", afterHide);
378 
379 		// Show component without effects
380 		} else {
381 			that.$container.addClass("ch-hide");
382 			afterHide();
383 		}
384 
385 		return that;
386 
387 	};

389 	/**
390 	* Getter and setter for size attributes on any float component.
391 	* @protected
392 	* @function
393 	* @name ch.Floats#size
394 	* @param {String} prop Property that will be setted or getted, like "width" or "height".
395 	* @param {String} [data] Only for setter. It's the new value of defined property.
396 	* @returns itself
397 	*/
398 	that.size = function (prop, data) {
399 		// Getter
400 		if (!data) { return that.conf[prop]; }
401 
402 		// Setter
403 		that.conf[prop] = data;
404 
405 		// Container size
406 		that.$container[prop](data);
407 
408 		// Refresh position
409 		that.position("refresh");
410 
411 		return that["public"];
412 	};

445 	/**
446 	* Sets or gets positioning configuration. Use it without arguments to get actual configuration. Pass an argument to define a new positioning configuration.
447 	* @public
448 	* @name ch.Uiobject#position
449 	* @example
450 	* // Change component's position.
451 	* me.position({ 
452 	*	  offset: "0 10",
453 	*	  points: "lt lb"
454 	* });
455 	* @see ch.Uiobject#position
456 	*/
457 	// Create a custom Positioner object to update conf.position data of Float family
458 	that["public"].position = function (o) {
459 
460 		var r = that["public"];
461 
462 		switch (typeof o) {
463 		// Custom Setter: It updates conf.position data
464 		case "object":
465 			// New points
466 			if (ch.utils.hasOwn(o, "points")) { conf.position.points = o.points; }
467 
468 			// New reposition
469 			if (ch.utils.hasOwn(o, "reposition")) { conf.position.reposition = o.reposition; }
470 
471 			// New offset (splitted)
472 			if (ch.utils.hasOwn(o, "offset")) { conf.position.offset = o.offset; }
473 
474 			// New context
475 			if (ch.utils.hasOwn(o, "context")) { conf.position.context = o.context; }
476 
477 			// Original Positioner
478 			that.position(conf.position);
479 
480 			break;
481 
482 		// Refresh
483 		case "string":
484 			that.position("refresh");
485 			
486 			break;
487 
488 		// Getter
489 		case "undefined":
490 		default:
491 			r = that.position();
492 			
493 			break;
494 		}
495 
496 		return r;
497 	};
