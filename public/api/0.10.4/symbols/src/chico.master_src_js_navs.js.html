  2 /**
  3 * Abstract representation of navs components.
  4 * @abstract
  5 * @name Navs
  6 * @class Navs
  7 * @standalone
  8 * @augments ch.Uiobject
  9 * @memberOf ch
 10 * @param {object} conf Object with configuration properties
 11 * @returns itself
 12 * @see ch.Dropdown
 13 * @see ch.Expando
 14 */
 15 
 16 ch.navs = function () {
 17 
 18 	/**
 19 	* Reference to a internal component instance, saves all the information and configuration properties.
 20 	* @private
 21 	* @name ch.Navs#that
 22 	* @type object
 23 	*/ 
 24 	var that = this,
 25 		conf = that.conf;
 26 
 27 	conf.icon = ch.utils.hasOwn(conf, "icon") ? conf.icon : true;
 28 	conf.open = conf.open || false;
 29 	conf.fx = conf.fx || false;
 30 
 31 /**
 32 *	Inheritance
 33 */
 34 
 35 	that = ch.uiobject.call(that);
 36 	that.parent = ch.clon(that);
 37 	
 38 /**
 39 *	Protected Members
 40 */
 41 	/**
 42 	* Status of component
 43 	* @protected
 44 	* @name ch.Navs#active
 45 	* @returns boolean
 46 	*/
 47 	that.active = false;
 48 
 49 	/**
 50 	* The component's trigger.
 51 	* @private
 52 	* @name ch.Navs#$trigger
 53 	* @type jQuery
 54 	*/
 55 	that.$trigger = that.$element.children().eq(0);
 56 	
 57 	/**
 58 	* The component's content.
 59 	* @private
 60 	* @name ch.Navs#$content
 61 	* @type jQuery
 62 	*/
 63 	that.$content = that.$element.children().eq(1);
 64 
 65 	/**
 66 	* Shows component's content.
 67 	* @protected
 68 	* @name ch.Navs#innerShow
 69 	* @returns itself
 70 	*/
 71 	that.innerShow = function (event) {
 72 		that.prevent(event);
 73 
 74 		if (that.active) {
 75 			return that.innerHide(event);
 76 		}
 77 		
 78 		that.active = true;
 79 
 80 		that.$trigger.addClass("ch-" + that.type + "-trigger-on");
 81 		/**
 82 		* onShow callback function
 83 		* @name ch.Navs#onShow
 84 		* @event
 85 		*/
 86 		// Animation
 87 		if (conf.fx) {
 88 			that.$content.slideDown("fast", function () {
 89 				//that.$content.removeClass("ch-hide");
 90 			
 91 				// new callbacks
 92 				that.trigger("show");
 93 				// old callback system
 94 				that.callbacks("onShow");
 95 			});
 96 		} else {
 97 			that.$content.removeClass("ch-hide");
 98 			// new callbacks
 99 			that.trigger("show");
100 			// old callback system
101 			that.callbacks("onShow");
102 		}
103 		
104 		return that;
105 	};
106 
107 	/**
108 	* Hides component's content.
109 	* @protected
110 	* @function
111 	* @name ch.Navs#innerHide
112 	* @returns itself
113 	*/
114 	that.innerHide = function (event) {
115 		that.prevent(event);
116 		
117 		if (!that.active) { return; }
118 		
119 		that.active = false;
120 		
121 		that.$trigger.removeClass("ch-" + that.type + "-trigger-on");
122 		/**
123 		* onHide callback function
124 		* @name ch.Navs#onHide
125 		* @event
126 		*/
127 		// Animation
128 		if (conf.fx) {
129 			that.$content.slideUp("fast", function () {
130 				//that.$content.addClass("ch-hide");
131 				that.callbacks("onHide");
132 			});
133 		} else {
134 			that.$content.addClass("ch-hide");
135 			// new callbacks
136 			that.trigger("hide");
137 			// old callback system
138 			that.callbacks("onHide");
139 		}
140 		
141 		return that;
142 	};
143 
144 	/**
145 	* Create component's layout
146 	* @protected
147 	* @function
148 	* @name ch.Navs#configBehavior
149 	*/
150 	that.configBehavior = function () {
151 		that.$trigger
152 			.addClass("ch-" + that.type + "-trigger")
153 			.bind("click", function (event) { that.innerShow(event); });
154 
155 		that.$content.addClass("ch-" + that.type + "-content ch-hide");
156 
157 		// Visual configuration
158 		if (conf.icon) { $("<span class=\"ch-" + that.type + "-ico\">Drop</span>").appendTo(that.$trigger); }
159 		if (conf.open) { that.innerShow(); }
160 
161 	};
162 
163 /**
164 * Public Members
165 */
166 	/**
167 	* Shows component's content.
168 	* @public
169 	* @function
170 	* @name ch.Navs#show
171 	* @returns itself
172 	*/
173 	that["public"].show = function(){
174 		that.innerShow();
175 		return that["public"];
176 	};
177 
178 	/**
179 	* Hides component's content.
180 	* @public
181 	* @function
182 	* @name ch.Navs#hide
183 	* @returns itself
184 	*/	
185 	that["public"].hide = function(){
186 		that.innerHide();
187 		return that["public"];
188 	};
189 
190 	/**
191 	* Returns a Boolean if the component's core behavior is active. That means it will return 'true' if the component is on and it will return false otherwise.
192 	* @public
193 	* @function
194 	* @name ch.Navs#isActive
195 	* @returns boolean
196 	*/
197 	that["public"].isActive = function () {
198 		return that.active;
199 	};
200 	
201 /**
202 *	Default event delegation
203 */
204 
205 	that.configBehavior();
206 	that.$element.addClass("ch-" + that.type);
207 
208 	/**
209 	* Triggers when component is visible.
210 	* @name ch.Navs#show
211 	* @event
212 	* @public
213 	* @example
214 	* me.on("show",function () {
215 	*	otherComponent.hide();
216 	* });
217 	* @see ch.Navs#event:show
218 	*/
219 
220 	/**
221 	* Triggers when component is not longer visible.
222 	* @name ch.Navs#hide
223 	* @event
224 	* @public
225 	* @example
226 	* me.on("hide",function () {
227 	*	otherComponent.show();
228 	* });
229 	* @see ch.Navs#event:hide
230 	*/
231 
232 	return that;
233 }
