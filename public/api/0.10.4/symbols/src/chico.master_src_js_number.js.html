  2 /**
  3 * Validate numbers.
  4 * @name Number
  5 * @class Number
  6 * @interface
  7 * @augments ch.Controls
  8 * @requires ch.Validation
  9 * @memberOf ch
 10 * @param {Object} [conf] Object with configuration properties.
 11 * @param {String} [conf.message] Validation message.
 12 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
 13 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 14 * @param {String} [conf.context] It's a reference to position the validation-bubble
 15 * @returns itself
 16 * @see ch.Validation
 17 * @see ch.Required
 18 * @see ch.Custom
 19 * @see ch.String
 20 * @see ch.Validator
 21 * @see ch.Condition
 22 * @example
 23 * // Create a number validation
 24 * $("input").number("This field must be a number.");
 25 */
 26 ch.extend("validation").as("number", function(conf) {
 27 	
 28 	// Define the conditions of this interface
 29 	conf.condition = {
 30 		name: "number",
 31 		patt: /^(-?[0-9\s]+)$/,
 32 		message: conf.msg || conf.message
 33 	};
 34 
 35 	return conf;
 36 
 37 });

 39 /**
 40 * Validate a number with a minimun value.
 41 * @name Min
 42 * @class Min
 43 * @interface
 44 * @augments ch.Controls
 45 * @requires ch.Validation
 46 * @memberOf ch
 47 * @param {Object} [conf] Object with configuration properties.
 48 * @param {String} [conf.message] Validation message.
 49 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
 50 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 51 * @param {String} [conf.context] It's a reference to position the validation-bubble
 52 * @param {Number} value Minimun number value.
 53 * @returns itself
 54 * @see ch.Validation
 55 * @see ch.Required
 56 * @see ch.Custom
 57 * @see ch.String
 58 * @see ch.Validator
 59 * @see ch.Condition
 60 * @example
 61 * $("input").min(10, "Write a number bigger than 10");
 62 */
 63 ch.extend("validation").as("min", function (conf) {
 64 
 65 	conf.condition = {
 66 		name: "min",
 67 		expr: function(a,b) { return a >= b },
 68 		message: conf.msg || conf.message,
 69 		value: conf.value
 70 	};
 71 
 72 	return conf;
 73 
 74 });
