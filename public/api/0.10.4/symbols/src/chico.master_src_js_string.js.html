1 /**
  2 * Validate strings.
  3 * @name String 
  4 * @class String
  5 * @interface
  6 * @augments ch.Controls
  7 * @requires ch.Validation
  8 * @memberOf ch
  9 * @param {Object} [conf] Object with configuration properties.
 10 * @param {String} [conf.message] Validation message.
 11 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
 12 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 13 * @param {String} [conf.context] It's a reference to position the validation-bubble
 14 * @returns itself
 15 * @see ch.Validation
 16 * @see ch.Required
 17 * @see ch.Custom
 18 * @see ch.Number
 19 * @see ch.Validator
 20 * @see ch.Condition
 21 * @example
 22 * // Create a string validation
 23 * $("input").string("This field must be a string.");
 24 */
 25 ch.extend("validation").as("string", function (conf) {
 26 
 27 	conf.condition = {
 28 		name: "string",
 29 		// the following regular expression has the utf code for the lating characters
 30 		// the ranges are A,EI,O,U,a,ei,o,u,รง,ร please for reference see http://www.fileformat.info/info/charset/UTF-8/list.htm
 31 		patt: /^([a-zA-Z\u00C0-\u00C4\u00C8-\u00CF\u00D2-\u00D6\u00D9-\u00DC\u00E0-\u00E4\u00E8-\u00EF\u00F2-\u00F6\u00E9-\u00FC\u00C7\u00E7\s]*)$/,
 32 		message: conf.msg || conf.message
 33 	};
 34 
 35 	return conf;
 36

 51 * @param {String} [conf.context] It's a reference to position the validation-bubble
 52 * @returns itself
 53 * @see ch.Validation
 54 * @see ch.Required
 55 * @see ch.Custom
 56 * @see ch.Number
 57 * @see ch.Validator
 58 * @see ch.Condition
 59 * @example
 60 * // Create a email validation
 61 * $("input").email("This field must be a valid email.");
 62 */
 63 ch.extend("validation").as("email", function (conf) {
 64 
 65 	conf.condition = {
 66 		name: "email",
 67 		patt: /^(("[\w-\s]+")|([\w-]+(?:\.[\w-]+)*)|("[\w-\s]+")([\w-]+(?:\.[\w-]+)*))(@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$)|(@\[?((25[0-5]\.|2[0-4][0-9]\.|1[0-9]{2}\.|[0-9]{1,2}\.))((25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\.){2}(25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\]?$)/,
 68 		message: conf.msg || conf.message
 69 	};
 70 
 71 	return conf;
 72

112 /**
113 * Validate a minimun amount of characters.
114 * @name MinLength
115 * @class MinLength
116 * @interface
117 * @augments ch.Controls
118 * @requires ch.Validation
119 * @memberOf ch
120 * @param {Object} [conf] Object with configuration properties.
121 * @param {String} [conf.message] Validation message.
122 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
123 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
124 * @param {String} [conf.context] It's a reference to position the validation-bubble
125 * @param {Number} value Minimun number value.
126 * @returns itself
127 * @see ch.Validation
128 * @see ch.Required
129 * @see ch.Custom
130 * @see ch.Number
131 * @see ch.Validator
132 * @see ch.Condition
133 * @example
134 * // Create a minLength validation
135 * $("input").minLength(10, "At least 10 characters..");
136 */
137 ch.extend("validation").as("minLength", function (conf) {
138 
139 	conf.condition = {
140 		name: "minLength",
141 		expr: function(a,b) { return a.length >= b },
142 		message: conf.msg || conf.message,
143 		value: conf.value
144 	};
145 
146 	return conf;
147 
148 });
