1 /**
  2 * Validation is a validation engine for html forms elements.
  3 * @name Validation
  4 * @class Validation
  5 * @augments ch.Controls
  6 * @requires ch.Form
  7 * @requires ch.Validator
  8 * @requires ch.Required
  9 * @requires ch.String
 10 * @requires ch.Number
 11 * @requires ch.Custom
 12 * @memberOf ch
 13 * @param {Object} [conf] Object with configuration properties.
 14 * @param {String} [conf.message] Validation message.
 15 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
 16 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 17 * @param {String} [conf.context] It's a reference to position the validation-bubble.
 18 * @returns itself
 19 */
 20 
 21 ch.validation = function (conf) {
 22 
 23 	/**
 24 	* Reference to a internal component instance, saves all the information and configuration properties.
 25 	* @protected
 26 	* @name ch.Validation#that
 27 	* @type itself
 28 	*/
 29 	var that = this;
 30 
 31 	conf = ch.clon(conf);
 32 	
 33 	// Configuration by default
 34 	conf.offset = conf.offset || "15 0";
 35 	conf.points = conf.points || "lt rt";
 36 
 37 	that.conf = conf;
 38 
 39 /**
 40 * Inheritance
 41 */
 42 
 43 	that = ch.controls.call(that);
 44 	that.parent = ch.clon(that);
 45 
 46 /**
 47 * Private Members
 48 */
 49 
 50 	// Reference to a Validator instance. If there isn't any, the Validation instance will create one.
 51 	var validator = that.validator = (function(){
 52 		var c = {};
 53 			c.condition = conf.condition
 54 	 	return that.$element.validator(c);
 55 	})();
 56 
 57 	/**
 58 	* Search for instances of Validation with the same trigger, and then merge it's properties with it.
 59 	* @private
 60 	* @name ch.Validation#checkInstance
 61 	* @function
 62 	* @returns Object
 63 	*/
 64 	var checkInstance;
 65 	if (checkInstance = function() {
 66 
 67 		var instance, instances = ch.instances.validation;
 68 		if ( instances && instances.length > 0 ) {
 69 			for (var i = 0, j = instances.length; i < j; i+=1) {
 70 				instance = instances[i];
 71 
 72 				if (instance.element !== that.element) {
 73 					continue;
 74 				}
 75 
 76 				return {
 77 					exists: true,
 78 					object: instance
 79 				}
 80 			}
 81 		}
 82 	}()){
 83 		return checkInstance;
 84 	};
 85 
 86 	// Reference to a Form instance. If there isn't any, the Validation instance will create one.
 87 	var form = that.form = (function() {
 88 		if (ch.utils.hasOwn(ch.instances, "form") && ch.instances.form.length > 0) {
 89 			var i = 0, j = ch.instances.form.length;
 90 			for (i; i < j; i+=1) {
 91 				if (ch.instances.form[i].element === that.$element.parents("form")[0]) {
 92 					return ch.instances.form[i]; // Get my parent
 93 				};
 94 			};
 95 		} else {
 96 			that.$element.parents("form").form();
 97 			var last = (ch.instances.form.length - 1);
 98 			return ch.instances.form[last]; // Set my parent
 99 		}
100 	})();
101 	form.children.push(that["public"]);
102 
103 	/**
104 	* Validation event
105 	* @private
106 	* @name ch.Validation#validationEvent
107 	*/
108 	var validationEvent = (that.$element.hasClass("options") || that.$element.hasClass("ch-form-options") || that.element.tagName == "SELECT") ? "change" : "blur";
109 
110 	var clear = function() {
111 
112 		that.$element.removeClass("error ch-form-error");
113 		that["float"].innerHide();
114 
115 		validator.clear();
116 
117 		/**
118 		* Triggers when al validations are cleared.
119 		* @name ch.Validation#clear
120 		* @event
121 		* @public
122 		* @example
123 		* me.on("clear",function(){
124 		*	submitButton.enable();
125 		* });
126 		*/
127 		// old callback system
128 		that.callbacks('onClear');
129 		// new callback
130 		that.trigger("clear");
131 	};
132 
133 /**
134 * Protected Members
135 */
136 
137 	/**
138 	* Flag that let you know if the validations is enabled or not.
139 	* @protected
140 	* @name ch.Validation#enabled
141 	* @type boolean
142 	*/
143 	that.enabled = true;
144 	
145 	/**
146 	* Reference to the Float component instanced.
147 	* @protected
148 	* @type Object
149 	* @name ch.Validation#float
150 	*/
151 	that["float"] = that.createFloat({
152 		"$element": (function() {
153 			var reference;
154 			// CHECKBOX, RADIO
155 			if (that.$element.hasClass("options") || that.$element.hasClass("ch-form-options")) {
156 				// Helper reference from will be fired
157 				// H4
158 				if (that.$element.find("h4").length > 0) {
159 					var h4 = that.$element.find("h4"); // Find h4
160 						h4.wrapInner("<span>"); // Wrap content with inline element
161 					reference = h4.children(); // Inline element in h4 like helper reference
162 				// Legend
163 				} else if (that.$element.prev().prop("tagName") == "LEGEND") {
164 					reference = that.$element.prev(); // Legend like helper reference
165 				} else {
166 					reference = $(that.$element.find("label")[0]);
167 				}
168 			// INPUT, SELECT, TEXTAREA
169 			} else {
170 				reference = that.$element;
171 			}
172 			return reference;
173 		})(),
174 		"type": "validation",
175 		"content": "<p class=\"ch-message-error\">Error.</p>",
176 		"cone": true,
177 		"cache": false,
178 		"closable": false,
179 		"aria": {
180 			"role": "alert"
181 		},
182 		"offset": conf.offset,
183 		"points": conf.points
184 	});
185 
186 	/**
187 	* Runs all validations to check if it has an error.
188 	* @protected
189 	* @type function
190 	* @returns boolean
191 	* @name ch.Validation#process
192 	*/
193 	that.process = function () {
194 
195 		// Pre-validation: Don't validate disabled
196 		if (that.$element.attr('disabled') || !that.enabled) { return false; }
197 
198 		/**
199 		* Triggers before start validation process.
200 		* @name ch.Validation#beforeValidate
201 		* @event
202 		* @public
203 		* @example
204 		* me.on("beforeValidate",function(event) {
205 		*	submitButton.disable();
206 		* });
207 		*/
208 		// old callback system
209 		that.callbacks('beforeValidate');
210 		// new callback
211 		that.trigger("beforeValidate");
212 		
213 		// Executes the validators engine with a specific value and returns an object.
214 		// Context is the validation
215 		var gotError = validator.validate.call(that["public"], that.element.value);
216 
217 		// Save the validator's status.
218 		var status = !gotError.status;
219 
220 		// If has Error...
221 		if (status) {
222 
223 			if (that.$element.prop("tagName") === "INPUT" || that.$element.prop("tagName") === "TEXTAREA") {
224 				that.$element.addClass("error ch-form-error");
225 			}
226 
227 			that["float"]["public"].show("<p class=\"ch-message-error\">" + (gotError.msg || form.messages[gotError.condition] || "Error") + "</p>");
228 
229 			// Add blur or change event only one time
230 			if (!that.$element.data("events")) { that.$element.one(validationEvent, that.process); }
231 
232 			/**
233 			* Triggers when an error occurs on the validation process.
234 			* @name ch.Validation#error
235 			* @event
236 			* @public
237 			* @example
238 			* me.on("error",function(event, condition) {
239 			*	if (condition === "required") {
240 			* 		errorModal.show();
241 			* 	}
242 			* });
243 			*/
244 			// old callback system
245 			that.callbacks('onError', gotError.condition);
246 			// new callback
247 			that.trigger("error", gotError.condition);
248 
249 		// else NOT Error!
250 		} else {
251 			that.$element.removeClass("error ch-form-error");
252 			that["float"].innerHide();
253 		}
254 
255 		/**
256 		* Triggers when the validation process ends.
257 		* @name ch.Validation#afterValidate
258 		* @event
259 		* @public
260 		* @example
261 		* me.on("afterValidate",function(){
262 		*	submitButton.disable();
263 		* });
264 		*/
265 		// old callback system
266 		that.callbacks('afterValidate');
267 		// new callback
268 		that.trigger("afterValidate");
269 
270 		return status;
271 
272 	};

344

346 	};

356 		return that.$element;

397 	/**
398 	* Turn on Validation and Validator engine or an specific condition.
399 	* @public
400 	* @name ch.Validation#enable
401 	* @function
402 	* @returns itself
403 	* @see ch.Validator
404 	*/
405 	that["public"].enable = function(condition){
406 		validator.enable(condition);
407 
408 		if (!condition) {
409 			that.enabled = true;
410 		}
411 
412 		return that["public"];
413 	};

415 	/**
416 	* Turn off Validation and Validator engine or an specific condition.
417 	* @public
418 	* @name ch.Validation#disable
419 	* @function
420 	* @returns itself
421 	* @see ch.Validator
422 	*/
423 	that["public"].disable = function (condition) {
424 		// Clean the validation if is active;
425 		clear();
426 
427 		// Turn off validator
428 		validator.disable(condition);
429 
430 		// Turn off validation, if all conditions are disabled
431 		if (!condition){
432 			that.enabled = false;
433 		}
434 
435 		return that["public"];
436 	};

439 	* Turn on/off the Validation and Validator engine.
440 	* @public
441 	* @since 0.10.4
442 	* @name ch.Validation#toggleEnable
443 	* @function
444 	* @returns itself
445 	* @see ch.Validator
446 	*/
447 	that["public"].toggleEnable = function () {
448 
449 		if (that.enabled) {
450 			that["public"].disable();
451 		} else {
452 			that["public"].enable();
453 		}
454 
455 		return that["public"];
456 	};

494 	/**
495 	* Sets or gets conditions messages
496 	* @public
497 	* @since 0.10.4
498 	* @name ch.Validation#message
499 	* @function
500 	* @returns itself
501 	* @example
502 	* validation.message(condition, message);
503 	* // Sets a new message
504 	* validation.message("required", "New message for required validation");
505 	* // Gets a message from a condition
506 	* validation.message("required");
507 	*/
508 	that["public"].message = function (condition, msg) {
509 		if (condition === undefined) {
510 			throw "validation.message(condition, message): Please, give me a condition as parameter.";
511 		}
512 
513 		// Get a new message from a condition
514 		if (msg === undefined) {
515 			return validator.conditions[condition].message;
516 		}
517 
518 		// Sets a new message
519 		validator.conditions[condition].message = msg;
520 
521 		if (validator.isActive()) {
522 			that["public"]["float"].content("<p class=\"ch-message-error\">" + msg + "</p>");
523 		}
524 
525 		return that["public"];
526 	}
