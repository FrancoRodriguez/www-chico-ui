1 /**
  2 * Carousel is a large list of elements. Some elements will be shown in a preset area, and others will be hidden waiting for the user interaction to show it.
  3 * @name Carousel
  4 * @class Carousel
  5 * @augments ch.Uiobject
  6 * @requires ch.List
  7 * @standalone
  8 * @memberOf ch
  9 * @param {Object} [conf] Object with configuration properties.
 10 * @param {Number || String} [conf.width] Sets width property of the component's layout. By default, the width is elastic.
 11 * @param {Number || String} [conf.height] Sets height property of the component's layout. By default, the value is the <li> element height.
 12 * @param {Boolean} [conf.pagination] Shows a pagination. By default, the value is false.
 13 * @param {Boolean} [conf.arrows] Shows arrows icons to move over the pages. By default, the value is true.
 14 * @param {Array} [conf.asyncData] Defines the content of each item that will be load asnchronously as array.
 15 * @param {Function} [conf.asyncRender] The function that receives asyncData content and must return a string with result of manipulate that content.
 16 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are enable.
 17 * @returns itself
 18 * @example
 19 * // Create a new carousel with some configuration.
 20 * var me = $(".example").carousel({
 21 *     "width": 500,
 22 *     "height": "200px",
 23 *     "pagination": true,
 24 *     "arrows": false,
 25 *     "asyncData": [
 26 *         {src: 'a.png', alt: 'A'},
 27 *         {src: 'b.png', alt: 'B'},
 28 *         {src: 'c.png', alt: 'C'}
 29 *     ],
 30 *     "asyncRender": function (data) {
 31 *         return '<img src="' + data.src + '" alt="' + data.alt + '"/>';
 32 *     },
 33 *     "fx": false
 34 * });
 35 * @example
 36 * // Create a new carousel without configuration.
 37 * var me = $(".example").carousel();
 38 */
 39

 79 	/**
 80 	* Does what is necessary to make ready the component structure.
 81 	* @private
 82 	* @name ch.Carousel#createLayout
 83 	* @function
 84 	*/
 85 	var createLayout = function () {
 86 		
 87 		// Add class to component to support old HTML snippet
 88 		that.$element.addClass("ch-carousel");
 89 		
 90 		// Calculate extra width for content
 91 		extraWidth = (ch.utils.html.hasClass("ie6")) ? that.itemsWidth : 0;
 92 		
 93 		// Set width to Carousel if exists a width in configuration
 94 		if (ch.utils.hasOwn(conf, "width")) { that.$element.css("width", conf.width); }
 95 		
 96 		// Set height to Carousel if exists a height in configuration
 97 		if (ch.utils.hasOwn(conf, "height")) { that.$element.css("height", conf.height); }
 98 		
 99 		// Disable CSS transition if it's specified
100 		if (!conf.fx && ch.features.transition) { that.$content.addClass("ch-carousel-nofx"); }
101 		
102 		// Set container size based on items size
103 		that.$mask.css("height", that.$items.outerHeight());
104 		
105 		// WAI-ARIA for items
106 		$.each(that.$items, function (i, e) {
107 			
108 			// Page where this item is in
109 			var page = ~~(i / that.itemsPerPage) + 1;
110 			
111 			$(e).attr({
112 				"aria-hidden": page !== that.currentPage,
113 				"aria-setsize": that.itemsTotal,
114 				"aria-posinset": i + 1,
115 				"aria-label": "page" + page
116 			});
117 		});
118 		
119 		// Total amount of items (Widthout include queue items)
120 		var itemsAmount = that.$items.length;
121 		
122 		// At the begin, add items from queue if page is incomplete
123 		if (ch.utils.hasOwn(conf, "asyncData") && itemsAmount < that.itemsPerPage) {
124 			that.addItems(that.itemsPerPage - itemsAmount);
125 		}
126 	},

128 	/**
129 	* Creates Previous and Next arrows.
130 	* @private
131 	* @function
132 	* @name ch.Carousel#createArrows
133 	*/
134 		createArrows = function () {
135 			
136 			// Previous arrow
137 			var $prev = $("<p class=\"ch-carousel-prev" + (conf.rolling ? "" : " ch-hide") + "\" role=\"button\" aria-hidden=\"" + (!conf.rolling) + "\"><span>Previous</span></p>")
138 				.bind("click", that.prev)
139 				.prependTo(that.$element),
140 			
141 			// Next arrow
142 				$next = $("<p class=\"ch-carousel-next\" role=\"button\" aria-hidden=\"false\"><span>Next</span></p>")
143 				.bind("click", that.next)
144 				.appendTo(that.$element);
145 			
146 			// Positions arrows vertically in middle of Carousel
147 			$prev[0].style.top = $next[0].style.top = (that.$element.outerHeight() - $prev.outerHeight()) / 2 + "px";
148 			
149 			/**
150 			* Manages arrows turning it on and off when non-continue Carousel is moving.
151 			* @protected
152 			* @function
153 			* @name ch.Carousel#manageArrows
154 			* @param {Number} page Page to be moved.
155 			*/
156 			that.manageArrows = function (page) {
157 				// Case 1: Both arrows shown on Carousel's middle
158 				if (page > 1 && page < that.pages) {
159 					$prev.attr("aria-hidden", "false").removeClass("ch-hide");
160 					$next.attr("aria-hidden", "false").removeClass("ch-hide");
161 				} else {
162 				// Case 2: Previous arrow hidden on first page
163 					if (page === 1) {
164 						$prev.addClass("ch-hide").attr("aria-hidden", "true");
165 						$next.attr("aria-hidden", "false").removeClass("ch-hide");
166 				// Case 3: Next arrow hidden on last page
167 					} else if (page === that.pages) {
168 						$prev.attr("aria-hidden", "false").removeClass("ch-hide");
169 						$next.addClass("ch-hide").attr("aria-hidden", "true");
170 					}
171 				}
172 			};
173 		},

175 	/**
176 	* Creates Carousel pagination.
177 	* @private
178 	* @function
179 	* @name ch.Carousel#createPagination
180 	*/
181 	// TODO: Re-create pagination only if amount of pages change. Else, re-position it.
182 		createPagination = function () {
183 			
184 			// Create a element List for new pagination
185 			var $pagination = $("<ul class=\"ch-carousel-pages ch-hide\" role=\"tablist\">"),
186 			
187 			// Each page into list element
188 				$thumbnails;
189 	
190 			// Create each mini thumbnail an append to list
191 			for (var i = 1; i <= that.pages; i += 1) {
192 				
193 				// Mark as active if thumbnail is the same that current page
194 				var status = (i === that.currentPage) ? " class=\"ch-carousel-pages-on\" aria-selected=\"true\"" : " aria-selected=\"false\"",
195 				
196 				// Thumbnail with closure
197 					$thumb = $("<li" + status + " role=\"tab\" aria-controls=\"page" + i + "\">" + i + "</li>")
198 						.bind("click", function (i) {
199 							return function () {
200 								that.goTo(i);
201 							};
202 						}(i));
203 				
204 				$pagination.append($thumb);
205 			};
206 			
207 			// Append list to Carousel
208 			that.$element.append($pagination);
209 			
210 			// Positions list
211 			$pagination.css("left", (that.$element.outerWidth() - $pagination.outerWidth()) / 2).removeClass("ch-hide");
212 			
213 			// Set pagination children as thumbnails
214 			$thumbnails = $pagination.children();
215 			
216 			/**
217 			* Removes the active status classname of last selected page and adds it to the selected page.
218 			* @protected
219 			* @function
220 			* @name ch.Carousel#managePagination
221 			* @param {Number} page Page to be moved.
222 			*/
223 			that.managePagination = function (page) {
224 				$thumbnails.eq(that.currentPage - 1).removeClass("ch-carousel-pages-on").attr("aria-selected", "false");
225 				$thumbnails.eq(page - 1).addClass("ch-carousel-pages-on").attr("aria-selected", "true");
226 			};
227 		},

229 	/**
230 	* Calculates items amount on each page.
231 	* @protected
232 	* @function
233 	*/
234 		getItemsPerPage = function () {
235 			// Space to be distributed among all items
236 			var widthDiff = that.$mask.outerWidth() - that.itemsWidth;
237 			
238 			// If there are space to be distributed, calculate pages
239 			return that.itemsPerPage = (widthDiff > that.itemsWidth) ? ~~(widthDiff / that.itemsWidth) : 1;
240 		},
241 	
242 	/**
243 	* Calculates total amount of pages.
244 	* @private
245 	* @function
246 	* @name ch.Carousel#getPages
247 	* @returns {Number} Total amount of pages
248 	*/
249 		getPages = function () {
250 			// (Total amount of items) / (items amount on each page)
251 			// TODO: $coll.children =? that.$items.length
252 			return that.pages = Math.ceil(that.itemsTotal / that.itemsPerPage);
253 		},

255 	/**
256 	* Calculates all necesary data to draw Carousel correctly.
257 	* @private
258 	* @function
259 	* @name ch.Carousel#draw
260 	*/
261 		draw = function () {
262 			
263 			// Reset size of carousel mask
264 			maskWidth = that.$mask.outerWidth();
265 			
266 			// Recalculate items amount on each page
267 			getItemsPerPage();
268 			
269 			// Recalculate total amount of pages
270 			getPages();
271 			
272 			// Calculate variable margin between each item
273 			that.itemsMargin = Math.ceil(((maskWidth - (that.itemsWidth * that.itemsPerPage)) / that.itemsPerPage) / 2);
274 			
275 			// Modify sizes only if new items margin are positive numbers
276 			if (that.itemsMargin < 0) { return; }
277 			
278 			// Detach content from DOM for make a few changes
279 			that.$content.detach();
280 			
281 			// Move Carousel to first page for reset initial position
282 			that.goTo(1);
283 			
284 			// Save rendered items amount
285 			var i = that.$items.length;
286 			
287 			// Set new margin to all items
288 			while (i) {
289 				that.$items[i -= 1].style.marginLeft = that.$items[i].style.marginRight = that.itemsMargin + "px";
290 			}
291 			
292 			// Change content size and append it to DOM again
293 			// TODO: Use "width:-moz-max-content;" once instead .css("width"). Maybe add support to ch.features
294 			that.$content
295 				.css("width", (that.itemsWidth + that.itemsMargin * 2) * that.$items.length + extraWidth)
296 				.appendTo(that.$mask);
297 			
298 			// Manage Previous and Next arrows
299 			if (conf.arrows) {
300 				// Deletes pagination if already exists
301 				that.$element.find(".ch-carousel-next, .ch-carousel-prev").remove();
302 				
303 				// Creates updated pagination
304 				if (that.pages > 1) { createArrows(); }
305 			}
306 			
307 			// Manage pagination
308 			if (conf.pagination) {
309 				// Deletes pagination if already exists
310 				that.$element.find(".ch-carousel-pages").remove();
311 				
312 				// Creates updated pagination
313 				if (that.pages > 1) { createPagination(); }
314 			}
315 		},

434 	/**
435 	* Move items from queue to collection.
436 	* @protected
437 	* @name ch.Carousel#addItems
438 	* @function
439 	* @param {Number} amount Amount of items that will be added.
440 	*/
441 	that.addItems = function (amount) {
442 		
443 		// Take the sample from queue
444 		var sample = that.queue.splice(0, amount),
445 		
446 		// Condition if exists a render function on component configuration object
447 			hasRender = ch.utils.hasOwn(conf, "asyncRender"),
448 		
449 		// Position where new items will be added
450 			itemIndex = that.$items.length;
451 		
452 		// Append asynchronous items to collection
453 		// HTML Li Element with classname and styles and content from conf.async with or without render function
454 		for (var i = 0; i < amount; i += 1) {
455 			
456 			// Page where this item is in
457 			var page = ~~(itemIndex / that.itemsPerPage) + 1;
458 			
459 			sample[i] = "<li class=\"ch-carousel-item\" role=\"listitem\" aria-hidden=\"" + (page !== that.currentPage) + "\" aria-setsize=\"" + that.itemsTotal + "\" aria-posinset=\"" + (itemIndex += 1) + "\" aria-label=\"page" + page + "\" style=\"margin-right: " + that.itemsMargin + "px; margin-left: " + that.itemsMargin + "px;\">" + (hasRender ? conf.asyncRender(sample[i]) : sample[i]) + "</li>";
460 		};
461 		
462 		// Expand content width for include new items (item width and margin) * (total amount of items) + extra width
463 		// TODO: Use "width:-moz-max-content;" once instead .css("width"). Maybe add support to ch.features
464 		that.$content.css("width", (that.itemsWidth + that.itemsMargin * 2) * that.itemsTotal + extraWidth);
465 		
466 		// Append collection again
467 		that.$collection.append(sample.join(""));
468 		
469 		// Update items collection
470 		that.$items = that.$collection.children();
471 		
472 		that.callbacks("onItemsAdded");
473 		that.trigger("itemsAdded");
474 		
475 	};
476 	
477 	
478 	/**
479 	* Analizes if next page needs to load items from queue and execute addItems() method.
480 	* @protected
481 	* @name ch.Carousel#asyncItemsLoad
482 	* @function
483 	*/
484 	that.asyncItemsLoad = function () {
485 		
486 		// Load only when there are items in queue
487 		if (that.queue.length === 0) { return; }
488 		
489 		// Amount of items from the beginning to current page
490 		var itemsHere = that.currentPage * that.itemsPerPage,
491 		
492 		// Items rendered
493 			itemsRendered = that.$items.length;
494 		
495 		// Load only when there are more visible items than items rendered
496 		if (itemsHere < itemsRendered) { return; }
497 		
498 		// How many items needs to add for complete next page
499 		var amount = itemsHere % itemsRendered;
500 		
501 		// If isn't needed items to complete a page, then add an entire page
502 		amount = (amount === 0) ? that.itemsPerPage : amount;
503 		
504 		// If next page needs less items than it support, then add that amount
505 		amount = (that.queue.length < amount) ? that.queue.length : amount;
506 		
507 		// Add these
508 		that.addItems(amount);
509 		
510 	};
511 	
512 	// Moves to a defined page
513 	that.goTo = function (page) {
514 		
515 		// Validation of page parameter
516 		if (page === that.currentPage || page > that.pages || page < 1 || isNaN(page)) { return that; }
517 		
518 		// Manage arrows
519 		if (!conf.rolling && conf.arrows) { that.manageArrows(page); }
520 		
521 		// Select thumbnail on pagination
522 		if (conf.pagination) { that.managePagination(page); }
523 		
524 		// Coordinates of next movement
525 		var movement = -(maskWidth * (page - 1));
526 
527 		// TODO: review this conditional
528 		// Case 1: Movement with CSS transition
529 		if (conf.fx && ch.features.transition) {
530 			that.$content.css("left", movement);
531 		// Case 2: Movement with jQuery animate
532 		} else if (conf.fx) {
533 			that.$content.animate({ left: movement });
534 		// Case 3: Movement without transition or jQuery
535 		} else {
536 			that.$content.css("left", movement);
537 		}
538 		
539 		// Refresh selected page
540 		that.currentPage = page;
541 		
542 		// WAI-ARIA to set items as "hide"
543 		$.each(that.$items, function (i, e) {
544 			$(e).attr("aria-hidden", ~~(i / that.itemsPerPage) + 1 !== page);
545 		});
546 		
547 		that.callbacks("onSelect");
548 		that.trigger("select");
549 		
550 		return that;
551 	};
552 
553 	// Move to the previous page.

564 	// Move to the next page.

568 		
569 		// Asynchronous item load feature
570 		if (ch.utils.hasOwn(conf, "asyncData")) { that.asyncItemsLoad(); }
571 		
572 		that.callbacks("onNext");
573 		that.trigger("next");
574 		
575 		return that;
576 	};
