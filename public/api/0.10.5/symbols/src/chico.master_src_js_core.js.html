322 /** 
323 * Class to create UI Components
324 * @abstract
325 * @name Factory
326 * @class Factory
327 * @param o Configuration Object
328 * @example
329 *	o {
330 *		component: "chat",
331 *		callback: function(){},
332 *		[script]: "http://..",
333 *		[style]: "http://..",
334 *		[callback]: function(){}	
335 *	}
336 * @returns collection
337 * @memberOf ch
338 */
339 // TODO: Always it should receive a conf object as parameter (see Multiple component)
340 // TODO: Try to deprecate .and() method on Validator
341 ch.factory = function(o) {
342 
343 	var x = o.component || o;
344 	
345 	var create = function(x) { 
346 
347 		// Send configuration to a component trough options object
348 		$.fn[x] = function( options ) {
349 
350 			var results = [];
351 			var that = this;
352 
353 			// Could be more than one argument
354 			var _arguments = arguments;
355 			
356 			that.each( function(i, e) {
357 				
358 				var conf = options || {};
359 
360 				var context = {};
361 					context.type = x;
362 					context.element = e;
363 					context.$element = $(e);
364 					context.uid = ch.utils.index += 1; // Global instantiation index
365 			
366 				switch(typeof conf) {
367 					// If argument is a number, join with the conf
368 					case "number":
369 						var num = conf;
370 						conf = {};
371 						conf.value = num;
372 						
373 						// Could come a messages as a second argument
374 						if (_arguments[1]) {
375 							conf.msg = _arguments[1];
376 						};
377 					break;
378 					
379 					// This could be a message
380 					case "string":
381 						var msg = conf;
382 						conf = {};
383 						conf.msg = msg;
384 					break;
385 					
386 					// This is a condition for custom validation
387 					case "function":
388 						var func = conf;
389 						conf = {};
390 						conf.lambda = func;
391 						
392 						// Could come a messages as a second argument
393 						if (_arguments[1]) {
394 							conf.msg = _arguments[1];
395 						};
396 					break;
397 				};
398 
399 				// Create a component from his constructor
400 				var created = ch[x].call( context, conf );
401 
402 				/*
403 					MAPPING INSTANCES
404 					Internal interface for avoid mapping objects
405 					{
406 						exists:true,
407 						object: {}
408 					}
409 				*/
410 
411 				created = ( ch.utils.hasOwn(created, "public") ) ? created["public"] : created;
412 
413 				if (created.type) {
414 					var type = created.type;
415 					// If component don't exists in the instances map create an empty array
416 					if (!ch.instances[type]) { ch.instances[type] = []; }
417 						ch.instances[type].push( created );
418 				}
419 
420 				// Avoid mapping objects that already exists
421 				if (created.exists) {
422 					// Return the inner object
423 					created = created.object;
424 				}
425 
426 				results.push( created );
427 
428 			});
429 
430 			// return the created components collection or single component
431 			return ( results.length > 1 ) ? results : results[0];
432 		};
433 
434 		// if a callback is defined 
435 		if ( o.callback ) { o.callback(); }
436 
437 	} // end create function
438 
439 	if ( ch[x] ) {
440 		// script already here, just create it
441 		create(x);
442 
443 	} else {
444 		// get resurces and call create later
445 		ch.get({
446 			"method":"component",
447 			"component": x,
448 			"script": (o.script)? o.script : "src/js/"+x+".js",
449 			"styles": (o.style)? o.style : "src/css/"+x+".css",
450 			"callback":create
451 		});
452 	}
453 }

455 /**
456 * Load components or content
457 * @abstract
458 * @name Get
459 * @class Get
460 * @param {object} o Configuration object 
461 * @example
462 *	o {
463 *		component: "chat",
464 *		[script]: "http://..",
465 *		[style]: "http://..",
466 *		[callback]: function(){}
467 *	}
468 * @memberOf ch
469 */
470 ch.get = function(o) {
471 	
472 	// ch.get: "Should I get a style?"
473 	if ( o.style ) {
474 		var style = document.createElement('link');
475 			style.href = o.style;
476 			style.rel = 'stylesheet';
477 			style.type = 'text/css';
478 	}
479 	// ch.get: "Should I get a script?"		
480 	if ( o.script ) {
481 		var script = document.createElement("script");
482 			script.src = o.script;
483 	}
484 
485 	var head = document.getElementsByTagName("head")[0] || document.documentElement;
486 	// Handle Script loading
487 	var done = false;
488 
489 	// Attach handlers for all browsers
490 	script.onload = script.onreadystatechange = function() {
491 
492 		if ( !done && (!this.readyState || 
493 			this.readyState === "loaded" || this.readyState === "complete") ) {
494 			done = true;
495 			// if callback is defined call it
496 			if ( o.callback ) { o.callback( o.component ); }
497 			// Handle memory leak in IE
498 			script.onload = script.onreadystatechange = null;
499 			if ( head && script.parentNode ) { head.removeChild( script ); }
500 		}
501 	};
502 
503 	// Use insertBefore instead of appendChild to circumvent an IE6 bug.
504 	// This arises when a base node is used.
505 	if ( o.script ) { head.insertBefore( script, head.firstChild ); }
506 	if ( o.style ) { head.appendChild( style ); }
507 
508 }

587 /**
588 * Extend is a utility that resolve creating interfaces problem for all UI-Objects.
589 * @abstract
590 * @name Extend
591 * @class Extend
592 * @memberOf ch
593 * @param {string} name Interface's name.
594 * @param {function} klass Class to inherit from.
595 * @param {function} [process] Optional function to pre-process configuration, recieves a 'conf' param and must return the configration object.
596 * @returns class
597 * @example
598 * // Create an URL interface type based on String component.
599 * ch.extend("string").as("url");
600 * @example
601 * // Create an Accordion interface type based on Menu component.
602 * ch.extend("menu").as("accordion"); 
603 * @example
604 * // And the coolest one...
605 * // Create an Transition interface type based on his Modal component, with some conf manipulations:
606 * ch.extend("modal").as("transition", function(conf) {
607 *	conf.closeButton = false;
608 *	conf.msg = conf.msg || conf.content || "Please wait...";
609 *	conf.content = $("<div>").addClass("loading").after( $("<p>").html(conf.msg) );
610 *	return conf;
611 * });
612 */
613 
614 ch.extend = function (klass) {
615 
616 	"use strict";
617 
618 	return {
619 		as: function (name, process) {
620 			// Create the component in Chico-UI namespace
621 			ch[name] = function (conf) {
622 				// Some interfaces need a data value,
623 				// others simply need to be 'true'.
624 				conf[name] = conf.value || true;
625 	
626 				// Invoke pre-proccess if is defined,
627 				// or grab the raw conf argument,
628 				// or just create an empty object.
629 				conf = (process) ? process(conf) : conf || {};
630 	
631 				// Here we recieve messages,
632 				// or create an empty object.
633 				conf.messages = conf.messages || {};
634 	
635 				// If the interface recieve a 'msg' argument,
636 				// store it in the message map.
637 				if (ch.utils.hasOwn(conf, "msg")) {
638 					conf.messages[name] = conf.msg;
639 					conf.msg = null;
640 					delete conf.msg;
641 				}
642 				// Here is where the magic happen,
643 				// invoke the class with the new conf,
644 				// and return the instance to the namespace.
645 				return ch[klass].call(this, conf);
646 			};
647 			// Almost done, now we need expose the new component,
648 			// let's ask the factory to do it for us.
649 			ch.factory(name);
650 		} // end as method
651 	} // end return
652 };
