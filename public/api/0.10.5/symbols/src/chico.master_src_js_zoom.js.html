1 /**
  2 * Zoom shows a contextual reference to an augmented version of main declared image.
  3 * @name Zoom
  4 * @class Zoom
  5 * @augments ch.Floats
  6 * @requires ch.Positioner
  7 * @requires ch.onImagesLoads
  8 * @memberOf ch
  9 * @param {Object} [conf] Object with configuration properties.
 10 * @param {Boolean} [conf.fx] Enable or disable fade effect on show. By default, the effect are disabled.
 11 * @param {Boolean} [conf.context] Sets a reference to position of component that will be considered to carry out the position. By default is the anchor of HTML snippet.
 12 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or "lt rt" by default.
 13 * @param {String} [conf.offset] Sets the offset in pixels that component will be displaced from original position determined by points. It's specified by configuration or "20 0" by default.
 14 * @param {String} [conf.message] This message will be shown when component needs to communicate that is in process of load. It's "Loading zoom..." by default.
 15 * @param {Number} [conf.width] Width of floated area of zoomed image. Example: 500, "500px", "50%". Default: 350.
 16 * @param {Number} [conf.height] Height of floated area of zoomed image. Example: 500, "500px", "50%". Default: 350.
 17 * @returns itself
 18 * @see ch.Modal
 19 * @see ch.Tooltip
 20 * @see ch.Layer
 21 */

126 	* Reference to main image declared on HTML code snippet.

128 	* @name ch.Zoom#original

221 	/**
222 	* Reference to the augmented version of image, that will be displayed into a floated element.
223 	* @private
224 	* @name ch.Zoom#zoomed
225 	* @type Object
226 	*/
227 		zoomed = (function () {
228 			// Define the content source
229 			var $img = that.source = $("<img src=\"" + that.element.href + "\">");
230 
231 			// Grab some data when zoomed image loads
232 			$img.onImagesLoads(function () {
233 
234 				// Save the zoomed image size
235 				zoomed.width = $img.prop("width");
236 				zoomed.height = $img.prop("height");
237 
238 				// Save the zoom ratio
239 				ratio.width = zoomed.width / original.width;
240 				ratio.height = zoomed.height / original.height;
241 
242 				// Seeker: Size relative to zoomed image respect zoomed area
243 				var w = ~~(conf.width / ratio.width),
244 					h = ~~(conf.height / ratio.height);
245 
246 				// Seeker: Save half width and half height
247 				seeker.width = w / 2;
248 				seeker.height = h / 2;
249 
250 				// Seeker: Set size and append it
251 				seeker.$shape.css({"width": w, "height": h}).appendTo(that.$element);
252 
253 				// Remove loading
254 				$loading.remove();
255 
256 				// Change zoomed image status to Ready
257 				zoomed.ready = true;
258 
259 				// TODO: MAGIC here! if mouse is over image show seeker and make all that innerShow do
260 			});
261 
262 			return {
263 				/**
264 				* Reference to HTML Element of augmented image.
265 				* @private
266 				* @name img
267 				* @memberOf ch.Zoom#zoomed
268 				* @type Object
269 				*/
270 				"$image": $img,
271 
272 				/**
273 				* Status of augmented image. When it's load, the status is "true".
274 				* @private

279 				"ready": false,

281 				/**

288 				"width": 0,
289 
290 				/**

296 				*/
297 				"height": 0

301 	/**
302 	* Calculates movement limits and sets it to seeker and augmented image.
303 	* @private
304 	* @function
305 	* @name ch.Zoom#move
306 	* @param {Event} event Mouse event to take the cursor position.
307 	*/
308 		move = function (event) {
309 
310 			var x, y;
311 
312 			// Left side of seeker LESS THAN left side of image
313 			if (event.pageX - seeker.width < offset.left) {
314 				x = 0;
315 			// Right side of seeker GREATER THAN right side of image
316 			} else if (event.pageX + seeker.width > original.width + offset.left) {
317 				x = original.width - (seeker.width * 2) - 2;
318 			// Free move
319 			} else {
320 				x = event.pageX - offset.left - seeker.width;
321 			}
322 
323 			// Top side of seeker LESS THAN top side of image
324 			if (event.pageY - seeker.height < offset.top) {
325 				y = 0;
326 			// Bottom side of seeker GREATER THAN bottom side of image
327 			} else if (event.pageY + seeker.height > original.height + offset.top) {
328 				y = original.height - (seeker.height * 2) - 2;
329 			// Free move
330 			} else {
331 				y = event.pageY - offset.top - seeker.height;
332 			}
333 
334 			// Move seeker
335 			seeker.$shape.css({"left": x, "top": y});
336 
337 			// Move zoomed image
338 			zoomed.$image.css({"left": (-ratio.width * x), "top": (-ratio.height * y)});
339 
340 		};

346 	/**
347 	* Inner show method. Attach the component's layout to the DOM tree and load defined content.
348 	* @protected
349 	* @name ch.Zoom#innerShow
350 	* @function
351 	* @returns itself
352 	*/
353 	that.innerShow = function () {
354 
355 		// If the component isn't loaded, show loading transition
356 		if (!zoomed.ready) {
357 			$loading.removeClass("ch-hide");
358 			return that;
359 		}
360 
361 		// Update position of anchor here because Zoom can be inside a Carousel and its position updates
362 		offset = that.$element.offset();
363 
364 		// Bind move calculations
365 		that.$element.bind("mousemove", function (event) { move(event); });
366 
367 		// Show seeker
368 		seeker.$shape.removeClass("ch-hide");
369 
370 		// Show float
371 		that.parent.innerShow();
372 
373 		return that;
374 
375 	};
376 
377 	/**
378 	* Inner hide method. Hides the component's layout and detach it from DOM tree.
379 	* @protected
380 	* @name ch.Zoom#innerHide
381 	* @function
382 	* @returns itself
383 	*/
384 	that.innerHide = function () {
385 
386 		// If the component isn't loaded, hide loading transition
387 		if (!zoomed.ready) {
388 			$loading.addClass("ch-hide");
389 			return that;
390 		}
391 
392 		// Unbind move calculations
393 		that.$element.unbind("mousemove");
394 
395 		// Hide seeker
396 		seeker.$shape.addClass("ch-hide");
397 
398 		// Hide float
399 		that.parent.innerHide();
400 
401 		return that;
402 
403 	};

405 	/**
406 	* Getter and setter for size attributes of float that contains the zoomed image.
407 	* @protected
408 	* @function
409 	* @name ch.Zoom#size
410 	* @param {string} prop Property that will be setted or getted, like "width" or "height".
411 	* @param {string} [data] Only for setter. It's the new value of defined property.
412 	* @returns itself
413 	*/
414 	that.size = function (prop, data) {
415 
416 		// Seeker: Updates styles and size value
417 		if (data) {
418 			// Seeker: Size relative to zoomed image respect zoomed area
419 			var size = ~~(data / ratio[prop]);
420 
421 			// Seeker: Save half width and half height
422 			seeker[prop] = size / 2;
423 
424 			// Seeker: Set size
425 			seeker.$shape.css(prop, size);
426 		}
427 
428 		// Change float size
429 		return that.parent.size(prop, data);
430 	};
