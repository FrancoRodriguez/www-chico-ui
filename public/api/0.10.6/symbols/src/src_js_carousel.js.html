1 /**
  2 * Carousel is a large list of elements. Some elements will be shown in a preset area, and others will be hidden waiting for the user interaction to show it.
  3 * @name Carousel
  4 * @class Carousel
  5 * @augments ch.Uiobject
  6 * @memberOf ch
  7 * @param {Object} [conf] Object with configuration properties.
  8 * @param {Number || String} [conf.width] Sets width property of the component's layout. By default, the width is elastic.
  9 * @param {Boolean} [conf.pagination] Shows a pagination. By default, the value is false.
 10 * @param {Boolean} [conf.arrows] Shows arrows icons over or outside the mask. By default, the value is "outside".
 11 * @param {Array} [conf.asyncData] Defines the content of each item that will be load asnchronously as array.
 12 * @param {Function} [conf.asyncRender] The function that receives asyncData content and must return a string with result of manipulate that content.
 13 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are enabled.
 14 * @param {Number} [conf.maxItems] (Since 0.10.6) Set the max amount of items to show in each page.
 15 * @param {Number} [conf.page] (Since 0.10.6) Initialize the Carousel in a specified page.
 16 * @returns itself
 17 * @exampleDescription Create a Carousel without configuration.
 18 * @example
 19 * var foo = $("#example").carousel();
 20 * @exampleDescription Create a Carousel with configuration parameters.
 21 * @example
 22 * var foo = $("#example").carousel({
 23 *     "width": 500,
 24 *     "pagination": true,
 25 *     "arrows": "over"
 26 * });
 27 * @exampleDescription Create a Carousel with items asynchronously loaded.
 28 * @example
 29 * var foo = $("#example").carousel({
 30 *     "asyncData": [
 31 *         {"src": "a.png", "alt": "A"},
 32 *         {"src": "b.png", "alt": "B"},
 33 *         {"src": "c.png", "alt": "C"}
 34 *     ],
 35 *     "asyncRender": function (data) {
 36 *         return "<img src=\"" + data.src + "\" alt=\"" + data.alt + "\" />";
 37 *     }
 38 * });
 39 */

185 	/**
186 	* Calculates and set the size of items and its margin to get an adaptive Carousel.
187 	* @private
188 	* @name ch.Carousel#updateResponsiveness
189 	* @function
190 	*/
191 		updateResponsiveness = function () {
192 
193 			// Grabs if there are MORE THAN ONE item in a page or just one
194 			var moreThanOne = itemsPerPage > 1,
195 
196 			// Total space to use as margin into mask
197 			// It's the difference between mask width and total width of all items
198 				freeSpace = maskWidth - (itemOuterWidth * itemsPerPage),
199 
200 			// Amount of spaces to distribute the free space
201 			// When there are 6 items on a page, there are 5 spaces between them
202 			// Except when there are only one page that NO exist spaces
203 				spaces = moreThanOne ? itemsPerPage - 1 : 0,
204 
205 			// Free space for each space between items
206 			// Ceil to delete float numbers (not Floor, because next page is seen)
207 			// There is no margin when there are only one item in a page
208 				margin = moreThanOne ? Math.ceil(freeSpace / spaces / 2) : 0,
209 
210 			// Width to add to each item to get responsivity
211 			// When there are more than one item, get extra width for each one
212 			// When there are only one item, extraWidth must be just the freeSpace
213 				extraWidth = moreThanOne ? Math.ceil(freeSpace / itemsPerPage / 2) : Math.ceil(freeSpace);
214 
215 			// Update ONLY IF margin changed from last redraw
216 			if (itemExtraWidth === extraWidth) { return; }
217 
218 			// Update global values
219 			itemMargin = margin;
220 			itemExtraWidth = extraWidth;
221 
222 			// Update distance needed to move ONLY ONE page
223 			// The width of all items on a page, plus the width of all margins of items
224 			pageWidth = (itemOuterWidth + extraWidth + margin) * itemsPerPage;
225 
226 			// Update list width
227 			// Consider one more page to get an error margin
228 			$list.css("width", pageWidth * (pages + 1));
229 
230 			// Update element styles
231 			$items.css({
232 				"width": itemWidth + extraWidth,
233 				"margin-right": margin
234 			});
235 		},

237 	/**
238 	* Trigger all recalculations to get the functionality measures.
239 	* @private
240 	* @name ch.Carousel#draw
241 	* @function
242 	*/
243 		draw = function () {
244 
245 			// Avoid wrong calculations going to first page
246 			goToPage(1);
247 
248 			// Trigger Draw event as deferred
249 			that.trigger("redraw");
250 
251 			// Trigger Draw event as configuration
252 			that.callbacks("onRedraw");
253 
254 			// Update the width of the mask
255 			maskWidth = $mask.outerWidth();
256 
257 			// Update amount of items into a single page (from conf or auto calculations)
258 			itemsPerPage = (function () {
259 				// The width of each item into the width of the mask
260 				var i = ~~(maskWidth / itemOuterWidth);
261 
262 				// Avoid zero items in a page
263 				if (i === 0) { return 1; }
264 
265 				// Limit amount of items when user set a maxItems amount
266 				if (ch.utils.hasOwn(conf, "maxItems") && i > conf.maxItems) {
267 					return conf.maxItems;
268 				}
269 
270 				// Default calculation
271 				return i;
272 			}());
273 
274 			// Update amount of total pages
275 			// The ratio between total amount of items and items in each page
276 			totalPages = Math.ceil(($items.length + queue.length) / itemsPerPage);
277 
278 			// Update only if pages amount changed from last redraw
279 			if (pages !== totalPages) {
280 				// Update value
281 				pages = totalPages;
282 				// Set WAI-ARIA properties to each item
283 				updateARIA();
284 				// Update arrows (when pages === 1, there is no arrows)
285 				updateArrows();
286 				// Update pagination
287 				updatePagination();
288 			}
289 
290 			// Update the margin between items and its size
291 			if (!ch.utils.hasOwn(conf, "width")) {
292 				updateResponsiveness();
293 			}
294 		},

445 		updateArrows = function () {
446 
447 			// Check arrows existency
448 			if (!arrowsCreated) { return; }
449 
450 			// Case 1: Disable both arrows if there are ony one page
451 			if (pages === 1) {
452 				$prevArrow.attr("aria-hidden", "true").addClass("ch-carousel-disabled");

455 			} else if (currentPage === 1) {

460 				$prevArrow.attr("aria-hidden", "false").removeClass("ch-carousel-disabled");
461 				$nextArrow.attr("aria-hidden", "true").addClass("ch-carousel-disabled");
462 			// Case 4: Enable both arrows on Carousel's middle
463 			} else {
464 				$prevArrow.attr("aria-hidden", "false").removeClass("ch-carousel-disabled");
465 				$nextArrow.attr("aria-hidden", "false").removeClass("ch-carousel-disabled");

469 	/**
470 	* Allows to render the arrows over the mask or not.
471 	* @private
472 	* @name ch.Carousel#arrowsFlow
473 	* @function
474 	* @param {String || Boolean} config Defines the arrows behavior. It can be "outside", "over", "none", true or false. By default it's "outside".
475 	*/
476 		arrowsFlow = function (config) {
477 
478 			// Getter
479 			if (config === undefined) { return conf.arrows; }
480 
481 			// Setter
482 			switch (conf.arrows = config) {
483 			// The arrows are on the sides of the mask
484 			case "outside":
485 			default:
486 				// Add the adaptive class to mask
487 				$mask.addClass("ch-carousel-adaptive");
488 				// Append arrows if previously were deleted
489 				addArrows();
490 				break;
491 
492 			// The arrows are over the mask
493 			case "over":
494 			case true:
495 				// Remove the adaptive class to mask
496 				$mask.removeClass("ch-carousel-adaptive");
497 				// Append arrows if previously were deleted
498 				addArrows();
499 				break;
500 
501 			// No arrows
502 			case "none":
503 			case false:
504 				// Remove the adaptive class to mask
505 				$mask.removeClass("ch-carousel-adaptive");
506 				// Detach arrows from DOM and continue to remove adaptive class
507 				removeArrows();
508 				break;
509 			}
510 		},

543 			// Get all thumbnails of pagination element

671 	/**
672 	* Moves the list corresponding to specified displacement.
673 	* @private
674 	* @name ch.Carousel#translate
675 	* @function
676 	* @param {Number} displacement Distance to move the list.
677 	*/
678 		translate = (function () {
679 
680 			// CSS property written as string to use on CSS movement
681 			var transform = "-" + ch.utils.VENDOR_PREFIX + "-transform";
682 
683 			// Translate list using CSS translate transform
684 			function CSSMove(displacement) {
685 				$list.css(transform, "translateX(" + displacement + "px)");
686 			}
687

702 	/**
703 	* Updates all necessary data to move to a specified page.
704 	* @private
705 	* @name ch.Carousel#goToPage
706 	* @function
707 	* @param {Number || String} page Reference of page to go. It can be specified as number or "first" or "last" string.
708 	*/
709 		goToPage = function (page) {
710 
711 			// Page getter
712 			if (!page) { return currentPage; }
713 
714 			// Page setter
715 			// Change to number the text pages ("first" and "last")
716 			if (page === "first") { page = 1; }
717 			else if (page === "last") { page = pages; }
718 
719 			// Avoid strings from here
720 			page = parseInt(page);
721 
722 			// Avoid to select:
723 			// - The same page that is selected yet
724 			// - A page less than 1
725 			// - A page greater than total amount of pages
726 			if (page === currentPage || page < 1 || page > pages) { return; }
727 
728 			// Perform these tasks in the following order:
729 			// Task 1: Move the list!!!
730 			// Position from 0 (zero), to page to move (page number beginning in zero)
731 			translate(-pageWidth * (page - 1));
732 
733 			// Task 2: Update selected thumbnail on pagination
734 			switchPagination(currentPage, page);
735 
736 			// Task 3: Update value of current page
737 			currentPage = page;
738 
739 			// Task 4: Check for arrows behavior on first, last and middle pages
740 			updateArrows();
741 
742 			// Task 5: Get items from queue to the list, if it's necessary
743 			loadAsyncItems();
744 
745 			// Task 6: Set WAI-ARIA properties to each item
746 			updateARIA();
747 
748 			// Trigger Select event as configuration
749 			that.trigger("select");
750 
751 			// Trigger Select event as deferred
752 			that.callbacks("onSelect");
753 		},

755 	/**
756 	* Move items from queue to collection.
757 	* @private
758 	* @name ch.Carousel#addItems
759 	* @function
760 	* @param {Number} amount Amount of items that will be added.
761 	*/
762 		addItems = function (amount) {
763 
764 			// Take the sample from queue
765 			var sample = queue.splice(0, amount);
766 
767 			// Replace sample items with Carousel item template)
768 			for (var i = 0; i < amount; i += 1) {
769 				// Replace sample item
770 				sample[i] = [
771 					// Open tag with ARIA role
772 					"<li role=\"listitem\"",
773 					// Add the same margin than all siblings items
774 					" style=\"width: " + (itemWidth + itemExtraWidth) + "px; margin-right: " + itemMargin + "px\"",
775 					// Add content (executing a template, if user specify it) and close the tag
776 					">" + (conf.asyncRender(sample[i]) || sample[i]) + "</li>"
777 				// Get it as string
778 				].join("");
779 			};
780 
781 			// Add sample items to the list
782 			$list.append(sample.join(""));
783 
784 			// Update items collection
785 			$items = $list.children();
786 
787 			// Trigger item addition event as deferred
788 			that.trigger("itemsAdded");
789 
790 			// Trigger item addition event as configuration
791 			that.callbacks("onItemsAdded");
792 		},
793 
794 	/**
795 	* Analizes if next page needs to load items from queue and execute addItems() method.
796 	* @private
797 	* @name ch.Carousel#loadAsyncItems
798 	* @function
799 	*/
800 		loadAsyncItems = function () {
801 
802 			// Load only when there are items in queue
803 			if (queue.length === 0) { return; }
804 
805 			// Amount of items from the beginning to current page
806 			var total = currentPage * itemsPerPage,
807 
808 			// How many items needs to add to items rendered to complete to this page
809 				amount = total - $items.length;
810 
811 			// Load only when there are items to add
812 			if (amount < 1) { return; }
813 
814 			// If next page needs less items than it support, then add that amount
815 			amount = (queue.length < amount) ? queue.length : amount;
816 
817 			// Add these
818 			addItems(amount);
819 		};
