1 /**
  2 * Controls brings the functionality of all form controls.
  3 * @abstract
  4 * @name Controls
  5 * @class Controls
  6 * @augments ch.Uiobject
  7 * @requires ch.Floats
  8 * @memberOf ch
  9 * @returns itself
 10 * @see ch.Countdown
 11 * @see ch.Validation
 12 * @see ch.AutoComplete
 13 * @see ch.DatePicker
 14 */
 15 
 16 ch.controls = function () {
 17 
 18 	/**
 19 	* Reference to a internal component instance, saves all the information and configuration properties.
 20 	* @name ch.Controls#that
 21 	* @type Object
 22 	*/
 23 	var that = this,
 24 
 25 		conf = that.conf;
 26 
 27 /**
 28 *  Inheritance
 29 */
 30 	that = ch.uiobject.call(that);
 31 	that.parent = ch.clon(that);
 32 
 33 /**
 34 *  Protected Members
 35 */
 36 
 37 	/**
 38 	* Creates a reference to the Float component instanced.
 39 	* @protected
 40 	* @type Object
 41 	* @name ch.Controls#createFloat
 42 	*/
 43 	that.createFloat = function (c) {
 44 		c.position = {
 45 			"context": conf.context || c.context || c.$element || that.$element,
 46 			"offset": c.offset,
 47 			"points": c.points
 48 		};
 49 
 50 		return ch.floats.call({
 51 			"element": (ch.utils.hasOwn(c, "$element")) ? c.$element[0] : that.element,
 52 			"$element": c.$element || that.$element,
 53 			"uid": (ch.utils.index += 1),
 54 			"type": c.type || that.type,
 55 			"conf": c
 56 		});
 57 
 58 		
 59 	};
 60 
 61 /**
 62 *  Public Members
 63 */
 64 
 65 	return that;
 66 };
