1 /**
  2 * Navs is a representation of navs components.
  3 * @abstract
  4 * @name Navs
  5 * @class Navs
  6 * @standalone
  7 * @augments ch.Uiobject
  8 * @memberOf ch
  9 * @param {object} conf Object with configuration properties
 10 * @returns itself
 11 * @see ch.Dropdown
 12 * @see ch.Expando
 13 */
 14 
 15 ch.navs = function () {
 16 
 17 	/**
 18 	* Reference to a internal component instance, saves all the information and configuration properties.
 19 	* @private
 20 	* @name ch.Navs#that
 21 	* @type object
 22 	*/ 
 23 	var that = this,
 24 		conf = that.conf;
 25 
 26 	conf.icon = ch.utils.hasOwn(conf, "icon") ? conf.icon : true;
 27 	conf.open = conf.open || false;
 28 	conf.fx = conf.fx || false;
 29 
 30 /**
 31 *	Inheritance
 32 */
 33 
 34 	that = ch.uiobject.call(that);
 35 	that.parent = ch.clon(that);
 36 	
 37 /**
 38 *	Protected Members
 39 */
 40 	/**
 41 	* Status of component
 42 	* @protected
 43 	* @name ch.Navs#active
 44 	* @returns boolean
 45 	*/
 46 	that.active = false;
 47 
 48 	/**
 49 	* The component's trigger.
 50 	* @private
 51 	* @name ch.Navs#$trigger
 52 	* @type jQuery
 53 	*/
 54 	that.$trigger = that.$element.children().eq(0);
 55 	
 56 	/**
 57 	* The component's content.
 58 	* @private
 59 	* @name ch.Navs#$content
 60 	* @type jQuery
 61 	*/
 62 	that.$content = that.$element.children().eq(1);
 63 
 64 	/**
 65 	* Shows component's content.
 66 	* @protected
 67 	* @name ch.Navs#innerShow
 68 	* @returns itself
 69 	*/
 70 	that.innerShow = function (event) {
 71 		that.prevent(event);
 72 
 73 		if (that.active) {
 74 			return that.innerHide(event);
 75 		}
 76 		
 77 		that.active = true;
 78 
 79 		that.$trigger.addClass("ch-" + that.type + "-trigger-on");
 80 		/**
 81 		* onShow callback function
 82 		* @name ch.Navs#onShow
 83 		* @event
 84 		*/
 85 		// Animation
 86 		if (conf.fx) {
 87 			that.$content.slideDown("fast", function () {
 88 				//that.$content.removeClass("ch-hide");
 89 			
 90 				// new callbacks
 91 				that.trigger("show");
 92 				// old callback system
 93 				that.callbacks("onShow");
 94 			});
 95 		} else {
 96 			that.$content.removeClass("ch-hide");
 97 			// new callbacks
 98 			that.trigger("show");
 99 			// old callback system
100 			that.callbacks("onShow");
101 		}
102 		
103 		return that;
104 	};
105 
106 	/**
107 	* Hides component's content.
108 	* @protected
109 	* @function
110 	* @name ch.Navs#innerHide
111 	* @returns itself
112 	*/
113 	that.innerHide = function (event) {
114 		that.prevent(event);
115 		
116 		if (!that.active) { return; }
117 		
118 		that.active = false;
119 		
120 		that.$trigger.removeClass("ch-" + that.type + "-trigger-on");
121 		/**
122 		* onHide callback function
123 		* @name ch.Navs#onHide
124 		* @event
125 		*/
126 		// Animation
127 		if (conf.fx) {
128 			that.$content.slideUp("fast", function () {
129 				//that.$content.addClass("ch-hide");
130 				that.callbacks("onHide");
131 			});
132 		} else {
133 			that.$content.addClass("ch-hide");
134 			// new callbacks
135 			that.trigger("hide");
136 			// old callback system
137 			that.callbacks("onHide");
138 		}
139 		
140 		return that;
141 	};
142 
143 	/**
144 	* Create component's layout
145 	* @protected
146 	* @function
147 	* @name ch.Navs#configBehavior
148 	*/
149 	that.configBehavior = function () {
150 		that.$trigger
151 			.addClass("ch-" + that.type + "-trigger")
152 			.bind("click", function (event) { that.innerShow(event); });
153 
154 		that.$content.addClass("ch-" + that.type + "-content ch-hide");
155 
156 		// Visual configuration
157 		if (conf.icon) { $("<span class=\"ch-" + that.type + "-ico\">Drop</span>").appendTo(that.$trigger); }
158 		if (conf.open) { that.innerShow(); }
159 
160 	};
161 
162 /**
163 * Public Members
164 */
165 	/**
166 	* Shows component's content.
167 	* @public
168 	* @function
169 	* @name ch.Navs#show
170 	* @returns itself
171 	*/
172 	that["public"].show = function(){
173 		that.innerShow();
174 		return that["public"];
175 	};
176 
177 	/**
178 	* Hides component's content.
179 	* @public
180 	* @function
181 	* @name ch.Navs#hide
182 	* @returns itself
183 	*/	
184 	that["public"].hide = function(){
185 		that.innerHide();
186 		return that["public"];
187 	};
188 
189 	/**
190 	* Returns a Boolean if the component's core behavior is active. That means it will return 'true' if the component is on and it will return false otherwise.
191 	* @public
192 	* @function
193 	* @name ch.Navs#isActive
194 	* @returns boolean
195 	*/
196 	that["public"].isActive = function () {
197 		return that.active;
198 	};
199 	
200 /**
201 *	Default event delegation
202 */
203 
204 	that.configBehavior();
205 	that.$element.addClass("ch-" + that.type);
206 
207 	/**
208 	* Triggers when component is visible.
209 	* @name ch.Navs#show
210 	* @event
211 	* @public
212 	* @example
213 	* widget.on("show",function () {
214 	*	otherComponent.hide();
215 	* });
216 	* @see ch.Navs#event:show
217 	*/
218 
219 	/**
220 	* Triggers when component is not longer visible.
221 	* @name ch.Navs#hide
222 	* @event
223 	* @public
224 	* @example
225 	* widget.on("hide",function () {
226 	*	otherComponent.show();
227 	* });
228 	* @see ch.Navs#event:hide
229 	*/
230 
231 	return that;
232 }
