1 /**
  2 * String validates a given text as string.
  3 * @name String 
  4 * @class String
  5 * @interface
  6 * @augments ch.Controls
  7 * @augments ch.Validation
  8 * @requires ch.Validation
  9 * @memberOf ch
 10 * @param {Object} [conf] Object with configuration properties.
 11 * @param {String} [conf.message] Validation message.
 12 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
 13 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 14 * @param {String} [conf.context] It's a reference to position the validation-bubble
 15 * @returns itself
 16 * @factorized
 17 * @see ch.Validation
 18 * @see ch.Required
 19 * @see ch.Custom
 20 * @see ch.Number
 21 * @see ch.Validator
 22 * @see ch.Condition
 23 * @exampleDescription Create a string validation
 24 * @example
 25 * $("input").string("This field must be a string.");
 26 */
 27 ch.extend("validation").as("string", function (conf) {
 28 
 29 	conf.condition = {
 30 		name: "string",
 31 		// the following regular expression has the utf code for the lating characters
 32 		// the ranges are A,EI,O,U,a,ei,o,u,รง,ร please for reference see http://www.fileformat.info/info/charset/UTF-8/list.htm
 33 		patt: /^([a-zA-Z\u00C0-\u00C4\u00C8-\u00CF\u00D2-\u00D6\u00D9-\u00DC\u00E0-\u00E4\u00E8-\u00EF\u00F2-\u00F6\u00E9-\u00FC\u00C7\u00E7\s]*)$/,
 34 		message: conf.msg || conf.message
 35 	};
 36 
 37 	return conf;
 38 
 39 });

 58 * @see ch.Required
 59 * @see ch.Custom
 60 * @see ch.Number
 61 * @see ch.Validator
 62 * @see ch.Condition
 63 * @exampleDescription Create a email validation
 64 * @example

 66 */
 67 ch.extend("validation").as("email", function (conf) {
 68 
 69 	conf.condition = {
 70 		name: "email",
 71 		patt: /^(("[\w-\s]+")|([\w-]+(?:\.[\w-]+)*)|("[\w-\s]+")([\w-]+(?:\.[\w-]+)*))(@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$)|(@\[?((25[0-5]\.|2[0-4][0-9]\.|1[0-9]{2}\.|[0-9]{1,2}\.))((25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\.){2}(25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\]?$)/,
 72 		message: conf.msg || conf.message
 73 	};
 74 
 75 	return conf;
 76

121 /**
122 * MinLength validates a minimun amount of characters.
123 * @name MinLength
124 * @class MinLength
125 * @interface
126 * @augments ch.Controls
127 * @augments ch.Validation
128 * @requires ch.Validation
129 * @memberOf ch
130 * @param {Object} [conf] Object with configuration properties.
131 * @param {String} [conf.message] Validation message.
132 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
133 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
134 * @param {String} [conf.context] It's a reference to position the validation-bubble
135 * @param {Number} value Minimun number value.
136 * @returns itself
137 * @factorized
138 * @see ch.Validation
139 * @see ch.Required
140 * @see ch.Custom
141 * @see ch.Number
142 * @see ch.Validator
143 * @see ch.Condition
144 * @exampleDescription Create a minLength validation
145 * @example
146 * $("input").minLength(10, "At least 10 characters..");
147 */
148 ch.extend("validation").as("minLength", function (conf) {
149 
150 	conf.condition = {
151 		name: "minLength",
152 		expr: function(a,b) { return a.length >= b },
153 		message: conf.msg || conf.message,
154 		value: conf.value
155 	};
156 
157 	return conf;
158 
159 });
