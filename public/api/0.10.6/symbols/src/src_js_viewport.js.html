1 /**
  2 * Viewport is a reference to position and size of the visible area of browser.
  3 * @name Viewport
  4 * @class Viewport
  5 * @standalone
  6 * @memberOf ch
  7 */
  8 ch.viewport = {
  9 
 10 	/**
 11 	* Width of the visible area.
 12 	* @public
 13 	* @name ch.Viewport#width
 14 	* @type Number
 15 	*/
 16 	"width": ch.utils.window.width(),
 17 
 18 	/**
 19 	* Height of the visible area.
 20 	* @public
 21 	* @name ch.Viewport#height
 22 	* @type Number
 23 	*/
 24 	"height": ch.utils.window.height(),
 25 
 26 	/**
 27 	* Left offset of the visible area.
 28 	* @public
 29 	* @name ch.Viewport#left
 30 	* @type Number
 31 	*/
 32 	"left": ch.utils.window.scrollLeft(),
 33 
 34 	/**
 35 	* Top offset of the visible area.
 36 	* @public
 37 	* @name ch.Viewport#top
 38 	* @type Number
 39 	*/
 40 	"top": ch.utils.window.scrollTop(),
 41 
 42 	/**
 43 	* Right offset of the visible area.
 44 	* @public
 45 	* @name ch.Viewport#right
 46 	* @type Number
 47 	*/
 48 	"right": ch.utils.window.scrollLeft() + ch.utils.window.width(),
 49 
 50 	/**
 51 	* Bottom offset of the visible area.
 52 	* @public
 53 	* @name ch.Viewport#bottom
 54 	* @type Number
 55 	*/
 56 	"bottom": ch.utils.window.scrollTop() + ch.utils.window.height(),
 57 
 58 	/**
 59 	* Element representing the visible area.
 60 	* @public
 61 	* @name ch.Viewport#element
 62 	* @type Object
 63 	*/
 64 	"element": ch.utils.window,
 65 
 66 	/**
 67 	* Updates width and height of the visible area and updates ch.viewport.width and ch.viewport.height
 68 	* @public
 69 	* @function
 70 	* @name ch.Viewport#getSize
 71 	* @returns Object
 72 	*/
 73 	"getSize": function () {
 74 
 75 		return {
 76 			"width": this.width = ch.utils.window.width(),
 77 			"height": this.height = ch.utils.window.height()
 78 		};
 79 
 80 	},
 81 
 82 	/**
 83 	* Updates left, top, right and bottom coordinates of the visible area, relative to the window.
 84 	* @public
 85 	* @function
 86 	* @name ch.Viewport#getPosition
 87 	* @returns Object
 88 	*/
 89 	"getPosition": function () {
 90 
 91 		var size = this.getSize();
 92 
 93 		return {
 94 			"left": 0,
 95 			"top": 0,
 96 			"right": size.width,
 97 			"bottom": size.height,
 98 			// Size is for use as context on Positioner
 99 			// (see getCoordinates method on Positioner)
100 			"width": size.width,
101 			"height": size.height
102 		};
103 		
104 	},
105 	
106 	/**
107 	* Updates left, top, right and bottom coordinates of the visible area, relative to the document.
108 	* @public
109 	* @function
110 	* @name ch.Viewport#getOffset
111 	* @returns Object
112 	*/
113 	"getOffset": function () {
114 
115 		var position = this.getPosition(),
116 			scrollLeft = ch.utils.window.scrollLeft(),
117 			scrollTop = ch.utils.window.scrollTop();
118 
119 		return {
120 			"left": this.left = scrollLeft,
121 			"top": this.top = scrollTop,
122 			"right": this.right = scrollLeft + position.right,
123 			"bottom": this.bottom = scrollTop + position.bottom
124 		};
125 		
126 	}
127 };
