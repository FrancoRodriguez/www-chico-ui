1 /**
  2 * Expando lets you show or hide the content. Expando needs a pair: the title and the content related to that title.
  3 * @name Expando
  4 * @class Expando
  5 * @augments ch.Navs
  6 * @standalone
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {Boolean} [conf.open] Shows the expando open when component was loaded. By default, the value is false.
 10 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are disable.
 11 * @returns itself
 12 * @factorized
 13 * @exampleDescription Create a new expando without configuration.
 14 * @example
 15 * var widget = $(".example").expando();
 16 * @exampleDescription Create a new expando with configuration.
 17 * @example
 18 * var widget = $(".example").expando({
 19 *     "open": true,
 20 *     "fx": true
 21 * });
 22 */
 23  
 24 ch.expando = function (conf) {
 25 
 26 	/**
 27 	* Reference to a internal component instance, saves all the information and configuration properties.
 28 	* @private
 29 	* @name ch.Expando#that
 30 	* @type object
 31 	*/
 32 	var that = this;
 33 		
 34 	conf = ch.clon(conf);
 35 	that.conf = conf;
 36 	
 37 /**
 38 *	Inheritance
 39 */
 40 
 41 	that = ch.navs.call(that);
 42 	that.parent = ch.clon(that);
 43 
 44 /**
 45 *  Protected Members
 46 */ 
 47 	var $nav = that.$element.children(),
 48 		triggerAttr = {
 49 			"aria-expanded":conf.open,
 50 			"aria-controls":"ch-expando-" + that.uid
 51 		},
 52 		contentAttr = {
 53 			id:triggerAttr["aria-controls"],
 54 			"aria-hidden":!triggerAttr["aria-expanded"]
 55 		};
 56 
 57 	/**
 58 	* The component's trigger.
 59 	* @protected
 60 	* @name ch.Expando#$trigger
 61 	* @type jQuery
 62 	*/
 63 	that.$trigger = that.$trigger.attr(triggerAttr);
 64 	
 65 	/**
 66 	* The component's trigger.
 67 	* @protected
 68 	* @name ch.Expando#$content
 69 	* @type jQuery
 70 	*/
 71 	that.$content = $nav.eq(1).attr(contentAttr);
 72 	
 73 	/**
 74 	* Shows component's content.
 75 	* @protected
 76 	* @function
 77 	* @name ch.Expando#innerShow
 78 	* @returns itself
 79 	*/
 80 	that.innerShow = function(event){
 81 		that.$trigger.attr("aria-expanded","true");
 82 		that.$content.attr("aria-hidden","false");
 83 		that.parent.innerShow();
 84 		return that;
 85 	}
 86 
 87 	/**
 88 	* Hides component's content.
 89 	* @protected
 90 	* @function
 91 	* @name ch.Expando#innerHide
 92 	* @returns itself
 93 	*/
 94 	that.innerHide = function(event){
 95 		that.$trigger.attr("aria-expanded","false");
 96 		that.$content.attr("aria-hidden","true");
 97 		that.parent.innerHide();
 98 		return that;
 99 	}
100 	
101 	
102 /**
103 *  Public Members
104 */
105  
106 	/**
107 	* @borrows ch.Object#uid as ch.Expando#uid
108 	*/	
109 	
110 	/**
111 	* @borrows ch.Object#element as ch.Expando#element
112 	*/
113 
114 	/**
115 	* @borrows ch.Object#type as ch.Expando#type
116 	*/
117 
118 	/**
119 	* @borrows ch.Navs#show as ch.Expando#show
120 	*/
121 
122 	/**
123 	* @borrows ch.Navs#hide as ch.Expando#hide
124 	*/
125 
126 /**
127 *  Default event delegation
128 */
129 
130 	that.$trigger.children().attr("role","presentation");
131 	ch.utils.avoidTextSelection(that.$trigger);
132 	
133 	/**
134 	* Triggers when the component is ready to use (Since 0.8.0).
135 	* @name ch.Expando#ready
136 	* @event
137 	* @public
138 	* @since 0.8.0
139 	* @exampleDescription Following the first example, using <code>widget</code> as expando's instance controller:
140 	* @example
141 	* widget.on("ready",function () {
142 	*	this.show();
143 	* });
144 	*/
145 	setTimeout(function(){ that.trigger("ready") }, 50);
146 
147 	return that;
148 };
149 
150 ch.factory("expando");
