1 /**
  2 * TabNavigator lets you create tabs for static and dynamic content.
  3 * @name TabNavigator
  4 * @class TabNavigator
  5 * @augments ch.Controllers
  6 * @memberOf ch
  7 * @param {Object} [conf] Object with configuration properties.
  8 * @param {Number} [conf.selected] Selects a child that will be open when component was loaded. By default, the value is 1.
  9 * @returns itself
 10 * @factorized
 11 * @exampleDescription Create a new Tab Navigator without configuration.
 12 * @example
 13 * var widget = $(".example").tabNavigator();
 14 * @exampleDescription Create a new Tab Navigator with configuration.
 15 * @example
 16 * var widget = $(".example").tabNavigator({
 17 *     "selected": 2
 18 * });
 19 */
 20 
 21 ch.tabNavigator = function (conf) {
 22 
 23 	/**
 24 	* Reference to a internal component instance, saves all the information and configuration properties.
 25 	* @private
 26 	* @name ch.TabNavigator#that
 27 	* @type object
 28 	*/
 29 	var that = this;
 30 
 31 	conf = ch.clon(conf);
 32 
 33 	that.conf = conf;
 34 
 35 /**
 36 *	Inheritance
 37 */
 38 
 39 	that = ch.controllers.call(that);
 40 	that.parent = ch.clon(that);
 41 
 42 /**
 43 *	Private Members
 44 */
 45 
 46 	// Add CSS class to the main element 
 47 	that.$element.addClass("ch-tabNavigator");
 48 
 49 	/**
 50 	* The actual location hash, is used to know if there's a specific tab selected.
 51 	* @private
 52 	* @name ch.TabNavigator#hash
 53 	* @type string
 54 	*/
 55 	var hash = window.location.hash.replace("#!/", ""),
 56 
 57 	/**
 58 	* A boolean property to know if the some tag should be selected.
 59 	* @private
 60 	* @name ch.TabNavigator#hashed
 61 	* @type boolean
 62 	* @default false
 63 	*/
 64 		hashed = false,
 65 
 66 	/**
 67 	* Get wich tab is selected.
 68 	* @private
 69 	* @name ch.TabNavigator#selected
 70 	* @type number
 71 	*/
 72 		//selected = conf.selected - 1 || conf.value - 1 || 0,
 73 		selected = conf.selected - 1 || conf.value - 1 || -1,
 74 
 75 	/**
 76 	* Create controller's children.
 77 	* @private
 78 	* @name ch.TabNavigator#createTabs
 79 	* @function
 80 	*/
 81 		createTabs = function () {
 82 	
 83 			// Children
 84 			that.$triggers.find("a").each(function (i, e) {
 85 	
 86 				// Tab context
 87 				var tab = {};
 88 					tab.uid = that.uid + "#" + i;
 89 					tab.type = "tab";
 90 					tab.element = e;
 91 					tab.$element = $(e);
 92 					tab.controller = that["public"];
 93 	
 94 				// Tab configuration
 95 				var config = {};
 96 					config.open = (selected === i);
 97 					config.onShow = function () { selected = i; };
 98 				
 99 				if (ch.utils.hasOwn(that.conf, "cache")) { config.cache = that.conf.cache; }
100 	
101 				/**
102 				* Fired when the content of one dynamic tab loads.
103 				* @name ch.TabNavigator#onContentLoad
104 				* @event
105 				* @public
106 				*/
107 				if (ch.utils.hasOwn(that.conf, "onContentLoad")) { config.onContentLoad = that.conf.onContentLoad; }
108 				
109 				/**
110 				* Fired when the content of one dynamic tab did not load.
111 				* @name ch.TabNavigator#onContentError
112 				* @event
113 				* @public
114 				*/
115 				if (ch.utils.hasOwn(that.conf, "onContentError")) { config.onContentError = that.conf.onContentError; }
116 	
117 				// Create Tabs
118 				that.children.push(ch.tab.call(tab, config));
119 	
120 				// Bind new click to have control
121 				$(e).on("click", function (event) {
122 					that.prevent(event);
123 					select(i + 1);
124 				});
125 	
126 			});
127 	
128 			return;
129 	
130 		},
131 
132 	/**
133 	* Select a child to show its content.
134 	* @name ch.TabNavigator#select
135 	* @private
136 	* @function
137 	*/
138 		select = function (tab) {
139 			
140 			// If select a tab that don't exist do nothing
141 			if(!that.children[tab - 1]){ return; }
142 
143 			tab = that.children[tab - 1];
144 
145 			// Don't click me if I'm open
146 			if (tab === that.children[selected]) { return; }
147 	
148 			// Hide my open bro
149 			$.each(that.children, function (i, e) {
150 				if (tab !== e) { e.innerHide(); }
151 			});
152 			
153 			// shows the tab
154 			tab.innerShow();
155 	
156 			//Change location hash
157 			window.location.hash = "#!/" + tab.$content.attr("id");
158 	
159 			/**
160 			* Fired when a tab is selected.
161 			* @name ch.TabNavigator#onSelect
162 			* @event
163 			* @public
164 			*/
165 			that.callbacks("onSelect");
166 
167 			// new callback
168 			that.trigger("select");
169 	
170 			return that;
171 			
172 		};
173 
174 /**
175 *	Protected Members
176 */
177 
178 	/**
179 	* The component's triggers container.
180 	* @protected
181 	* @name ch.TabNavigator#$triggers
182 	* @type jQuery
183 	*/
184 	that.$triggers = that.$element.children(":first").addClass("ch-tabNavigator-triggers").attr("role", "tablist");
185 
186 	/**
187 	* The component's content.
188 	* @protected
189 	* @name ch.TabNavigator#$content
190 	* @type jQuery
191 	*/
192 	that.$content = that.$triggers.next().addClass("ch-tabNavigator-content ch-box").attr("role", "presentation");
193 
194 /**
195 *	Public Members
196 */
197 
198 	/**
199 	* @borrows ch.Object#uid as ch.TabNavigator#uid
200 	*/	
201 	
202 	/**
203 	* @borrows ch.Object#element as ch.TabNavigator#element
204 	*/
205 
206 	/**
207 	* @borrows ch.Object#type as ch.TabNavigator#type
208 	*/
209 
210 	/**
211 	* Children instances associated to this controller.
212 	* @public
213 	* @name ch.TabNavigator#children
214 	* @type collection
215 	*/
216 	that["public"].children = that.children;
217 
218 	/**
219 	* Select a specific child or get the selected tab.
220 	* @public
221 	* @name ch.TabNavigator#select
222 	* @function
223 	* @param {number} tab Tab's index.
224 	* @exampleDescription Create a TabNavigator
225 	* @example
226 	* widget.select(2);
227 	* 
228 	* // Get selected page
229 	* var selected = widget.select();
230 	*/
231 	that["public"].select = function (tab) {
232 		// Returns selectd tab instead set it
233 		if(!tab){
234 			return (selected + 1);
235 		}
236 
237 		select(tab);
238 		return that["public"];
239 
240 	};
241 
242 	/**
243 	* Deprecated - Returns the selected child's index.
244 	* @public
245 	* @name ch.TabNavigator#getSelected
246 	* @function
247 	* @returns {number} selected Tab's index.
248 	* @deprecated
249 	*/
250 	that["public"].getSelected = function () { return (selected + 1); };
251 
252 /**
253 *	Default event delegation
254 */
255 
256 	createTabs();
257 
258 	// If hash open that tab
259 	for(var i = that.children.length; i--;) {
260 		if (that.children[i].$content.attr("id") === hash) {
261 			select(i+1);
262 			hashed = true;
263 			break;
264 		}
265 	};
266 
267 	// Shows the first tab if not hash or it's hash and it isn't from the current tab
268 	if( !hash || ( hash && !hashed ) ){
269 		that.children[0].innerShow();
270 	}
271 	
272 	/**
273 	* Triggers when the component is ready to use (Since 0.8.0).
274 	* @name ch.TabNavigator#ready
275 	* @event
276 	* @public
277 	* @since 0.8.0
278 	* @example
279 	* // Following the first example, using <code>widget</code> as tabNavigator's instance controller:
280 	* widget.on("ready",function () {
281 	*	this.show();
282 	* });
283 	*/
284 	//This avoit to trigger execute after the component was instanciated
285 	setTimeout(function(){that.trigger("ready")}, 50);
286 
287 	return that;
288 
289 };

293 /**
294 * Tab is a simple unit of content for TabNavigators.
295 * @abstract
296 * @name Tab
297 * @class Tab
298 * @augments ch.Navs
299 * @memberOf ch
300 * @param {object} conf Object with configuration properties
301 * @returns itself
302 * @ignore
303 */
304 
305 ch.tab = function (conf) {
306 	/**
307 	* Reference to a internal component instance, saves all the information and configuration properties.
308 	* @private
309 	* @name ch.Tab#that
310 	* @type object
311 	* @ignore
312 	*/
313 	var that = this;
314 
315 	conf = ch.clon(conf);
316 	conf.icon = false;
317 
318 	that.conf = conf;
319 
320 /**
321 *	Inheritance
322 */
323 
324 	that = ch.uiobject.call(that);
325 	that.parent = ch.clon(that);
326 
327 /**
328 *	Private Members
329 */
330 	/**
331 	* Creates the basic structure for the tab's content.
332 	* @private
333 	* @name ch.Tab#createContent
334 	* @function
335 	* @ignore
336 	*/
337 	var createContent = function () {
338 		
339 		var href = that.element.href.split("#"),
340 			controller = that.$element.parents(".ch-tabNavigator"),
341 			content = controller.find("#" + href[1]);
342 
343 		// If there are a tabContent...
344 		if (content.length > 0) {
345 			return content;
346 
347 		// If tabContent doesn't exists
348 		} else {
349 			/**
350 			* Content configuration property.
351 			* @public
352 			* @name ch.Tab#source
353 			* @type string
354 			* @ignore
355 			*/
356 			that.source = that.element.href;
357 
358 			var id = (href.length === 2) ? href[1] : "ch-tab" + that.uid.replace("#", "-");
359 
360 			// Create tabContent
361 			return $("<div id=\"" + id + "\" role=\"tabpanel\" class=\"ch-hide\">").appendTo(controller.children().eq(1));
362 		}
363 
364 	};
365 
366 /**
367 *	Protected Members
368 */
369 	/**
370 	* Reference to the trigger element.
371 	* @protected
372 	* @name ch.Tab#$trigger
373 	* @type jQuery
374 	* @ignore
375 	*/
376 	that.$trigger = that.$element;
377 
378 	/**
379 	* The component's content.
380 	* @protected
381 	* @name ch.Tab#$content
382 	* @type jQuery
383 	* @ignore
384 	*/
385 	that.$content = createContent();
386 
387 	/**
388 	* Shows component's content.
389 	* @protected
390 	* @function
391 	* @name ch.Tab#innerShow
392 	* @returns itself
393 	* @ignore
394 	*/
395 	that.innerShow = function (event) {
396 		that.prevent(event);
397 
398 		that.active = true;
399 
400 		// Load my content if I'need an ajax request 
401 		if (ch.utils.hasOwn(that, "source")) { that.content(); }
402 
403 		// Show me
404 		that.$trigger.addClass("ch-" + that["type"] + "-trigger-on");
405 
406 		// Set me as hidden false
407 		that.$content
408 				.attr("aria-hidden", "false")
409 				.removeClass("ch-hide");
410 
411 		return that;
412 	};
413 
414 	/**
415 	* Hides component's content.
416 	* @protected
417 	* @function
418 	* @name ch.Tab#innerHide
419 	* @returns itself
420 	* @ignore
421 	*/
422 	that.innerHide = function (event) {
423 		that.prevent(event);
424 
425 		if (!that.active) { return; }
426 		
427 		that.active = false;
428 
429 		// Hide me
430 		that.$trigger.removeClass("ch-" + that["type"] + "-trigger-on");
431 
432 		// Set all inactive tabs as hidden
433 		that.$content
434 				.attr("aria-hidden", "true")
435 				.addClass("ch-hide");
436 
437 		return that;
438 	};
439 
440 	/**
441 	* This callback is triggered when async data is loaded into component's content, when ajax content comes back.
442 	* @protected
443 	* @name ch.Tab#contentCallback
444 	* @ignore
445 	*/
446 	that["public"].on("contentLoad", function (event, context) {
447 
448 		that.$content.html(that.staticContent);
449 
450 		if (ch.utils.hasOwn(conf, "onContentLoad")) {
451 			conf.onContentLoad.call(context, that.staticContent);
452 		}
453 
454 	});
455 
456 	/**
457 	* This callback is triggered when async request fails.
458 	* @public
459 	* @name contentCallback
460 	* @returns {Chico-UI Object}
461 	* @memberOf ch.TabNavigator
462 	* @ignore
463 	*/
464 	that["public"].on("contentError", function (event, data) {
465 
466 		that.$content.html(that.staticContent);
467 
468 		// Get the original that.source
469 		var originalSource = that.source;
470 
471 		if (ch.utils.hasOwn(conf, "onContentError")) {
472 			conf.onContentError.call(data.context, data.jqXHR, data.textStatus, data.errorThrown);
473 		}
474 
475 		// Reset content configuration
476 		that.source = originalSource;
477 		
478 		that.staticContent = undefined;
479 
480 	});
481 
482 /**
483 *	Public Members
484 */
485 
486 /**
487 *	Default event delegation
488 */
489 	
490 	// Add the attributes for WAI-ARIA to the tabs and tabpanel
491 	that.$content.attr({
492 		"role": "tabpanel",
493 		"aria-hidden": that.$content.hasClass("ch-hide"),
494 		"class": "ch-hide"
495 	});
496 
497 	that.$trigger.attr({
498 		"role": "tab",
499 		"arial-controls": that.$content.attr("id"),
500 		"class": "ch-tab-trigger"
501 	});
502 		
503 	return that;
504 }
