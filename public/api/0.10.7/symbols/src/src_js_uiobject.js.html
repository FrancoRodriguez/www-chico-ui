 20 ch.uiobject = function(){
 21 
 22 	/**
 23 	* Reference to a internal component instance, saves all the information and configuration properties.
 24 	* @private
 25 	* @name ch.Uiobject#that
 26 	* @type object
 27 	*/
 28 	var that = this;
 29 
 30 	var conf = that.conf;
 31 	
 32 
 33 /**

 35 */
 36

 38 	that.parent = ch.clon(that);

 44 */

 46 	/**
 47 	* Component static content.
 48 	* @protected
 49 	* @name ch.Uiobject#staticContent
 50 	* @type string
 51 	*/
 52 	that.staticContent;
 53 
 54 	/**
 55 	* DOM Parent of content, this is useful to attach DOM Content when float is hidding.
 56 	* @protected
 57 	* @name ch.Uiobject#DOMParent
 58 	* @type HTMLElement
 59 	*/
 60 	that.DOMParent;
 61 
 62 	/**
 63 	* Component original content.
 64 	* @protected
 65 	* @name ch.Uiobject#originalContent
 66 	* @type HTMLElement
 67 	*/
 68 	that.originalContent;
 69 
 70 	/**
 71 	* Set and get the content of a component. With no arguments will behave as a getter function. Send any kind of content and will be a setter function. Use a valid URL for AJAX content, use a CSS selector for a DOM content or just send a static content like HTML or Text.
 72 	* @ignore
 73 	* @name ch.Uiobject#content
 74 	* @protected
 75 	* @function
 76 	* @param {string} [content] Could be a simple text, html or a url to get the content with ajax.
 77 	* @returns {string} content
 78 	* @requires ch.Cache
 79 	* @exampleDescription Simple static content
 80 	* @example
 81 	* $(element).layer().content("Some static content");
 82 	* @exampleDescription Get DOM content
 83 	* @example
 84 	* $(element).layer().content("#hiddenContent");
 85 	* @exampleDescription Get AJAX content
 86 	* @example
 87 	* $(element).layer().content("http://chico.com/content/layer.html");
 88 	*/
 89 	that.content = function(content) {
 90 
 91 		var _get = (content) ? false : true,
 92 
 93 			// Local argument
 94 			content = content,
 95 			// Local isURL
 96 			sourceIsUrl = ch.utils.isUrl(that.source),
 97 			// Local isSelector
 98 			sourceIsSelector = ch.utils.isSelector(that.source),
 99 			// Local inDom
100 			sourceInDom = (!sourceIsUrl) ? ch.utils.inDom(that.source) : false,
101 			// Get context, could be a single component or a controller
102 			context = ( ch.utils.hasOwn(that, "controller") ) ? that.controller : that["public"],
103 			// undefined, for comparison.
104 			undefined,
105 			// Save cache configuration
106 			cache = ( ch.utils.hasOwn(conf, "cache") ) ? conf.cache : true;
107 
108 		/**
109 		* Get content
110 		*/
111 
112 		// return defined content
113 		if (_get) {
114 
115 			// no source, no content
116 			if (that.source === undefined) {
117 				that.staticContent = "<p>No content defined for this component</p>";
118 				that.trigger("contentLoad");
119 
120 				return;
121 			}
122 
123 			// First time we need to get the content.
124 			// Is cache is off, go and get content again.
125 			// Yeap, recursive.
126 			if (!cache || that.staticContent === undefined) {
127 				that.content(that.source);
128 				return;
129 			}
130 
131 			// Get data from cache if the staticContent was defined
132 			if (cache && that.staticContent) {
133 				var fromCache = ch.cache.get(that.source);
134 
135 				// Load content from cache
136 				if (fromCache && that.staticContent != fromCache) {
137 					that.staticContent = fromCache;
138 
139 					that.trigger("contentLoad");
140 
141 					// Return and load content from cache
142 					return;
143 				}
144 
145 				// Return and show the latest content that was loaded
146 				return;
147 			}
148 		}
149 
150 		/**
151 		* Set content
152 		*/
153 
154 		// Reset cache to overwrite content
155 		conf.cache = false;
156 
157 		// Local isURL
158 		var isUrl = ch.utils.isUrl(content),
159 			// Local isSelector
160 			isSelector = ch.utils.isSelector(content),
161 			// Local inDom
162 			inDom = (!isUrl) ? ch.utils.inDom(content) : false;
163 
164 		/* Evaluate static content*/
165 
166 		// Set 'that.staticContent' and overwrite 'that.source'
167 		// just in case you want to update DOM or AJAX Content.
168 
169 		that.staticContent = that.source = content;
170 
171 		/* Evaluate AJAX content*/
172 
173 		if (isUrl) {
174 
175 			// First check Cache
176 			// Check if this source is in our cache
177 			if (cache) {
178 				var fromCache = ch.cache.get(that.source);
179 				if (fromCache) {
180 					conf.cache = cache;
181 					that.staticContent = fromCache;
182 					that.trigger("contentLoad", context);
183 					return;
184 				}
185 			}
186 
187 			var _method, _serialized, _params = "x=x";
188 
189 			// If the trigger is a form button, serialize its parent to send data to the server.
190 			if (that.$element.attr('type') == 'submit') {
191 				_method = that.$element.parents('form').attr('method') || 'GET';
192 				_serialized = that.$element.parents('form').serialize();
193 				_params = _params + ((_serialized != '') ? '&' + _serialized : '');
194 			};
195 
196 			// Set ajax config
197 			// On IE (6-7) "that" reference losts for second time
198 			// Why?? I don't know... but with a setTimeOut() works fine!
199 			setTimeout(function(){
200 
201 				$.ajax({
202 					url: that.source,
203 					type: _method || 'GET',
204 					data: _params,
205 					// each component could have a different cache configuration
206 					cache: cache,
207 					async: true,
208 					beforeSend: function(jqXHR){
209 						// Ajax default HTTP headers
210 						jqXHR.setRequestHeader("X-Requested-With", "XMLHttpRequest");
211 					},
212 					success: function(data, textStatus, jqXHR){
213 						// Save data as staticContent
214 						that.staticContent = data;
215 
216 						// Use the contentLoad callback.
217 						that.trigger("contentLoad", context);
218 
219 						// Save new staticContent to the cache
220 						if (cache) {
221 							ch.cache.set(that.source, that.staticContent);
222 						}
223 
224 					},
225 					error: function(jqXHR, textStatus, errorThrown){
226 						that.staticContent = "<p>Error on ajax call.</p>";
227 
228 						var data = {
229 							"context": context,
230 							"jqXHR": jqXHR,
231 							"textStatus": textStatus,
232 							"errorThrown": errorThrown
233 						};
234 
235 						// Use the contentError callback.
236 						that.trigger("contentError", data);
237 					}
238 				});
239 
240 			}, 0);
241 
242 			// Return Spinner and wait for async callback
243 			that.$content.html("<div class=\"ch-loading\"></div>");
244 			that.staticContent = undefined;
245 
246 		} else {
247 
248 			/* Evaluate DOM content*/
249 
250 			if (isSelector && inDom) {
251 
252 				// Save original DOMFragment.
253 				that.originalContent = $(content);
254 
255 				// Save DOMParent, so we know where to re-insert the content.
256 				that.DOMParent = that.originalContent.parent();
257 
258 				// Save a clone to original DOM content
259 				that.staticContent = that.originalContent.clone().removeClass("ch-hide");
260 
261 			}
262 
263 			// Save new data to the cache
264 			if (cache) {
265 				ch.cache.set(that.source, that.staticContent);
266 			}
267 
268 			// First time we need to set the callbacks that append and remove the original content.
269 			if (that.originalContent && that.originalContent.selector == that.source) {
270 
271 				// Remove DOM content from DOM to avoid ID duplications
272 				that["public"].on("show", function() {
273 					that.originalContent.detach();
274 				});
275 
276 				// Returns DOMelement to DOM
277 				that["public"].on("hide", function(){
278 					that.originalContent.appendTo(that.DOMParent||"body");
279 				});
280 			}
281 		}
282 
283 		/* Set previous cache configuration*/
284 		conf.cache = cache;
285 
286 		// trigger contentLoad callback for DOM and Static content.
287 		if (that.staticContent !== undefined) {
288 			that.trigger("contentLoad", context);
289 		}
290 
291 	};
292 
293 	/**
294 	* Prevent propagation and default actions.
295 	* @name ch.Uiobject#prevent
296 	* @function
297 	* @protected
298 	* @param {event} event Recieves a event object
299 	*/
300 	that.prevent = function(event) {
301 
302 		if (event && typeof event == "object") {
303 			event.preventDefault();
304 			event.stopPropagation();
305 		};
306 
307 		return that;
308 	};

318 	/**
319 	* Sets and gets component content. To get the defined content just use the method without arguments, like 'widget.content()'. To define a new content pass an argument to it, like 'widget.content("new content")'. Use a valid URL to get content using AJAX. Use a CSS selector to get content from a DOM Element. Or just use a String with HTML code.
320 	* @ignore
321 	* @public
322 	* @name ch.Uiobject#content
323 	* @function
324 	* @param {string} content Static content, DOM selector or URL. If argument is empty then will return the content.
325 	* @exampleDescription Get the defined content
326 	* @example
327 	* widget.content();
328 	* @exampleDescription Set static content
329 	* @example
330 	* widget.content("Some static content");
331 	* @exampleDescription Set DOM content
332 	* @example
333 	* widget.content("#hiddenContent");
334 	* @exampleDescription Set AJAX content
335 	* @example
336 	* widget.content("http://chico.com/some/content.html");
337 	*/
338 	that["public"].content = function(content){
339 		if (content) { // sets
340 			// Reset content data
341 			that.source = content;
342 			that.staticContent = undefined;
343 
344 			if (that.active) {
345 				that.content(content);
346 			}
347 
348 			return that["public"];
349 
350 		} else { // gets
351 			return that.staticContent;
352 		}
353 	};

369 	*/
370 	
371 
372 	return that;
