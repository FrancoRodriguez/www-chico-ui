176 	/**
177 	* Parse string to YYYY/MM/DD or DD/MM/YYYY format date.
178 	* @private
179 	* @function
180 	* @name ch.DatePicker#parseDate
181 	* @param value {String} The date to be parsed.
182 	*/
183 		parseDate = function (value) {
184 
185 			// Splitted string
186 			value = value.split("/");
187 
188 			// Date to be returned
189 			var result = [];
190 
191 			// Parse date
192 			switch (conf.format) {
193 				case "DD/MM/YYYY":
194 					result.push(value[2], value[1], value[0]);
195 					break;
196 				case "MM/DD/YYYY":
197 					result.push(value[2], value[0], value[1]);
198 					break;
199 			}
200 
201 			return result.join("/");
202 		},

260 	/**
261 	* Creates a complete month in a table.
262 	* @private
263 	* @function
264 	* @name ch.DatePicker#createTable
265 	* @param date {Object} Date from will be created the entire month.
266 	* @return jQuery Object
267 	*/
268 		createTable = function (date) {
269 
270 			// Total amount of days into month
271 			var cells = (function () {
272 
273 				// Amount of days of current month
274 				var currentMonth = new Date(date.year, date.month, 0).getDate(),
275 
276 				// Amount of days of previous month
277 					prevMonth = new Date([date.year, date.month, "01"].join("/")).getDay(),
278 
279 				// Merge amount of previous and current month
280 					subtotal = prevMonth + currentMonth,
281 
282 				// Amount of days into last week of month
283 					latest = subtotal % 7,
284 
285 				// Amount of days of next month
286 					nextMonth = (latest > 0) ? 7 - latest : 0;
287 
288 				return {
289 					"previous": prevMonth,
290 					"subtotal": subtotal,
291 					"total": subtotal + nextMonth
292 				};
293 
294 			}()),
295 
296 			// Final array with month table structure
297 				r = [
298 					"<table class=\"ch-calendar ch-datagrid\" role=\"grid\" id=\"ch-calendar-grid-" + that.uid + "\">",
299 					"<caption>" + MONTHS_NAMES[date.month - 1] + " - " + date.year + "</caption>",
300 					thead,
301 					"<tbody>",
302 					"<tr class=\"week\" role=\"row\">"
303 				];
304 
305 			// Iteration of weekdays
306 			for (var i = 0; i < cells.total; i += 1) {
307 
308 				// Push an empty cell on previous and next month
309 				if (i < cells.previous || i > cells.subtotal - 1) {
310 					r.push("<td role=\"gridcell\" class=\"ch-disabled\">X</td>");
311 					continue;
312 				}
313 
314 				// Positive number of iteration
315 				var positive = i + 1,
316 
317 				// Day number
318 					day = positive - cells.previous,
319 
320 				// Define if it's the day selected
321 					isSelected = selected && date.year === selected.year && date.month === selected.month && day === selected.day;
322 
323 				// Create cell
324 				r.push(
325 					// Open cell structure including WAI-ARIA and classnames space opening
326 					"<td role=\"gridcell\"" + (isSelected ? " aria-selected=\"true\"" : "") + " class=\"ch-calendar-day",
327 
328 					// Add Today classname if it's necesary
329 					(date.year === today.year && date.month === today.month && day === today.day) ? " ch-calendar-today" : null,
330 
331 					// Add Selected classname if it's necesary
332 					(isSelected ? " ch-calendar-selected" : null),
333 
334 					// From/to range. Disabling cells
335 					(
336 						// Disable cell if it's out of FROM range
337 						(from && day < from.day && date.month === from.month && date.year === from.year) ||
338 
339 						// Disable cell if it's out of TO range
340 						(to && day > to.day && date.month === to.month && date.year === to.year)
341 
342 					) ? " ch-disabled" : null,
343 
344 					// Close classnames attribute and print content closing cell structure
345 					"\">" + day + "</td>"
346 				);
347 
348 				// Cut week if there are seven days
349 				if (positive % 7 === 0) {
350 					r.push("</tr><tr class=\"ch-calendar-week\" role=\"row\">");
351 				}
352 
353 			};
354 
355 			// Return table object
356 			return $(r.join(""));
357 
358 		},

386 			/**
387 			* Refresh arrows visibility depending on "from" and "to" limits.
388 			* @private
389 			* @name update
390 			* @memberOf ch.DatePicker#arrows
391 			* @function
392 			*/
393 			"update": function () {
394 
395 				// "From" limit
396 				if (from) {
397 					// Hide previous arrow when it's out of limit
398 					if (from.month >= currentDate.month && from.year >= currentDate.year) {
399 						arrows.$prev.addClass("ch-hide").attr("aria-hidden", "true");
400 					// Show previous arrow when it's out of limit
401 					} else {
402 						arrows.$prev.removeClass("ch-hide").attr("aria-hidden", "false");
403 					}
404 				}
405 
406 				// "To" limit
407 				if (to) {
408 					// Hide next arrow when it's out of limit
409 					if (to.month <= currentDate.month && to.year <= currentDate.year) {
410 						arrows.$next.addClass("ch-hide").attr("aria-hidden", "true");
411 					// Show next arrow when it's out of limit
412 					} else {
413 						arrows.$next.removeClass("ch-hide").attr("aria-hidden", "false");
414 					}
415 				}
416 			}
