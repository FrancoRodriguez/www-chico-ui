1 /**
  2 * AutoComplete lets you suggest anything from an input element. Use a suggestion service or use a collection with the suggestions.
  3 * @name AutoComplete
  4 * @class AutoComplete
  5 * @augments ch.Controls
  6 * @see ch.Controls
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {String} conf.url The url pointing to the suggestions's service.
 10 * @param {String} [conf.message] It represent the text when no options are shown.
 11 * @param {Array} [conf.suggestions] The suggestions's collection. If a URL is set at conf.url parametter this will be omitted.
 12 * @returns itself
 13 * @factorized
 14 * @exampleDescription Create a new autoComplete with configuration.
 15 * @example
 16 * var widget = $(".example").autoComplete({
 17 *     "url": "http://site.com/mySuggestions?q=",
 18 *     "message": "Write..."
 19 * });
 20 */
 21 
 22 ch.autoComplete = function(conf){
 23 
 24 	/**
 25 	* Reference to a internal component instance, saves all the information and configuration properties.
 26 	* @private
 27 	* @name ch.AutoComplete#that
 28 	* @type object
 29 	*/
 30 	var that = this;
 31 
 32 	conf = ch.clon(conf);
 33 	conf.icon = false;
 34 	conf.type = "autoComplete";
 35 	conf.message = conf.message || "Please write to be suggested";
 36 	conf.suggestions = conf.suggestions;
 37 	conf.jsonpCallback = conf.jsonpCallback || "autoComplete";
 38 
 39 	that.conf = conf;
 40 
 41 /**
 42 *	Inheritance
 43 */
 44 
 45 	that = ch.controls.call(that);
 46 	that.parent = ch.clon(that);

 48 /**
 49 *  Private Members
 50 */
 51 
 52 	/**
 53 	* Select an item.
 54 	* @private
 55 	* @type Function
 56 	* @name ch.AutoComplete#selectItem
 57 	*/
 58 	var selectItem = function (arrow, event) {
 59 		that.prevent(event);
 60 
 61 		if (that.selected === (arrow === "bottom" ? that.items.length - 1 : 0)) { return; }
 62 		$(that.items[that.selected]).removeClass("ch-autoComplete-selected");
 63 
 64 		if (arrow === "bottom") { that.selected += 1; } else { that.selected -= 1; }
 65 		$(that.items[that.selected]).addClass("ch-autoComplete-selected");
 66 	};
 67 
 68 /**
 69 *  Protected Members
 70 */
 71 
 72 	/**
 73 	* The number of the selected item.
 74 	* @protected
 75 	* @type Number
 76 	* @name ch.AutoComplete#selected
 77 	*/
 78 	that.selected = -1;
 79 
 80 	/**
 81 	* List of the shown suggestions.
 82 	* @protected
 83 	* @type Array
 84 	* @name ch.AutoComplete#suggestions
 85 	*/
 86 	that.suggestions = that.conf.suggestions;
 87 
 88 	/**
 89 	* The input where the AutoComplete works.
 90 	* @protected
 91 	* @type jQuery
 92 	* @name ch.AutoComplete#$trigger
 93 	*/
 94 	//that.$trigger = that.$element.addClass("ch-" + that.type + "-trigger");
 95 
 96 	/**
 97 	* Inner reference to content container. Here is where the content will be added.
 98 	* @protected
 99 	* @type jQuery
100 	* @name ch.AutoComplete#$content
101 	*/
102 	that.$content = $("<ul class=\"ch-autoComplete-list\"></ul>");
103 
104 	/**
105 	* It has the items loaded.
106 	* @protected
107 	* @type Boolean
108 	* @name ch.AutoComplete#behaviorActived
109 	*/
110 	that.behaviorActived = false;
111 
112 	/**
113 	* It has the items loaded.
114 	* @protected
115 	* @type Array
116 	* @name ch.AutoComplete#items
117 	*/
118 	that.items = [];
119 
120 	/**
121 	* Reference to the Float component instanced.
122 	* @protected
123 	* @type Object
124 	* @name ch.AutoComplete#float
125 	*/
126 	that["float"] = that.createFloat({
127 		"content": that.$content,
128 		"points": conf.points,
129 		"points": "lt lb",
130 		"cache": false,
131 		"closable": false,
132 		"aria": {
133 			"role": "tooltip",
134 			"identifier": "aria-describedby"
135 		},
136 		"width": that.$element.outerWidth() + "px"
137 	});
138 
139 	/**
140 	* It sets On/Off the loading icon.
141 	* @protected
142 	* @function
143 	* @name ch.AutoComplete#loading
144 	*/
145 	that.loading = function(show){
146 		if(show){
147 			that.$element.addClass("ch-autoComplete-loading");
148 		} else {
149 			that.$element.removeClass("ch-autoComplete-loading");
150 		}
151 	}
152 
153 	/**
154 	* It fills the content inside the element represented by the float.
155 	* @protected
156 	* @function
157 	* @name ch.AutoComplete#populateContent
158 	*/
159 	that.populateContent = function (event,result) {
160 		// No results doesn't anything
161 		if (result.length===0 || that.element.value==="") {
162 			that.loading(false);
163 			that["float"].innerHide();
164 			return that;
165 		}
166 
167 		// Only one result and the same as the input hide float and doesn't anything
168 		if (result.length===1 && result[0]===that.element.value) {
169 			that.loading(false);
170 			that["float"].innerHide();
171 			return that;
172 		}
173 
174 		var list = "";
175 		$.each(result, function (i, e) {
176 			list+="<li data-index=\""+i+"\">"+e+"</li>";
177 		})
178 
179 		that.trigger("contentUnload");
180 		that.$content.html(list);
181 		that.selected = -1;
182 
183 		that["float"].content(that.$content);
184 
185 		that.trigger("contentLoaded");
186 
187 		that.items = that.$content.children();
188 
189 		// Adds only once the behavior
190 		if (!that.behaviorActived) {
191 			that.suggestionsBehavior(event);
192 			that.behaviorActived = true;
193 		}
194 
195 		that["float"].innerShow();
196 		that.loading(false);
197 		return that;
198 	}
199 
200 	/**
201 	* It does the query to the server if configured an URL, or it does the query inside the array given.
202 	* @protected
203 	* @function
204 	* @name ch.AutoComplete#doQuery
205 	*/
206 	that.doQuery = function(event){
207 		var q = that.$element.val().toLowerCase();
208 		// When URL is configured it will execute an ajax request.
209 		if (that.element.value !== "" && event.keyCode !== 38 && event.keyCode !== 40  && event.keyCode !== 13  && event.keyCode !== 27) {
210 			if (that.conf.url !== undefined) {
211 				that.loading(true);
212 				var result = $.ajax({
213 					url: that.conf.url + q + "&callback=" + that.conf.jsonpCallback,
214 					dataType:"jsonp",
215 					cache:false,
216 					global:true,
217 					context: window,
218 					jsonp:that.conf.jsonpCallback,
219 					crossDomain:true
220 				});
221 			// When not URL configured and suggestions array were configured it search inside the suggestions array.
222 			} else if (that.conf.url === undefined) {
223 				var result = [];
224 				for(var a=(that.suggestions.length-1);(a+1);a--){
225 					var word = that.suggestions[a].toLowerCase();
226 					var exist = word.search(q);
227 					if(!exist){
228 						result.push(that.suggestions[a]);
229 					}
230 				};
231 				that.populateContent(event,result);
232 			}
233 		}
234 		return that;
235 	}
236 
237 	/**
238 	* Binds the behavior related to the list.
239 	* @protected
240 	* @function
241 	* @name ch.AutoComplete#suggestionsBehavior
242 	*/
243 	that.suggestionsBehavior = function(event){
244 		// BACKSPACE key bheavior. When backspace go to the start show the message
245 		ch.utils.document.on(ch.events.KEY.BACKSPACE + ".autoComplete", function (x, event) {
246 
247 			// When the user make backspace with empty input autocomplete is shutting off
248 			if(that.element.value.length===0){
249 				that.prevent(event);
250 				that.$element.trigger("blur");
251 			}
252 
253 			// When isn't any letter it hides the float
254 			if(that.element.value.length<=1){
255 				that["float"].innerHide();
256 				that.loading(false);
257 			}
258 
259 		})
260 		// ESC key behavior, it closes the suggestions's list
261 		.on(ch.events.KEY.ESC + ".autoComplete", function (x, event) { that.$element.trigger("blur"); })
262 		// ENTER key behavior, it selects the item who is selected
263 		.on(ch.events.KEY.ENTER + ".autoComplete", function (x, event) { that.$element.val($(that.items[that.selected]).text()); that.$element.trigger("blur"); })
264 		// UP ARROW key behavior, it selects the previous item
265 		.on(ch.events.KEY.UP_ARROW + ".autoComplete", function (x, event) { selectItem("up", event); })
266 		// DOWN ARROW key behavior, it selects the next item
267 		.on(ch.events.KEY.DOWN_ARROW + ".autoComplete", function (x, event) { selectItem("bottom", event); });
268 		// MouseOver & MouseDown Behavior
269 		that["float"].$content.on("mouseover mousedown",function(evt){
270 			var event = evt || window.event;
271 			var target = event.target || event.srcElement;
272 			var type = event.type;
273 			if(target.tagName === "LI"){
274 				// mouse over behavior
275 				if(type === "mouseover"){
276 					// removes the class if one is selected
277 					$(that.items[that.selected]).removeClass("ch-autoComplete-selected");
278 					// selects the correct item
279 					that.selected = parseInt(target.getAttribute("data-index"));
280 					// adds the class to highlight the item
281 					$(that.items[that.selected]).addClass("ch-autoComplete-selected");
282 				}
283 				// mouse down behavior
284 				if(type === "mousedown") {
285 					that.prevent(event);
286 					that.$element.val($(that.items[that.selected]).text());
287 					that.$element.trigger("blur");
288 				}
289 			}
290 		});
291 	}

293 	/**
294 	* Internal show method. It adds the behavior.
295 	* @protected
296 	* @function
297 	* @name ch.AutoComplete#show
298 	*/
299 	that.show = function(event){
300 		// new callbacks
301 		that.trigger("show");
302 		var query = that.element.value;
303 		that.doQuery(event);
304 		// Global keyup behavior
305 		ch.utils.document.on("keyup.autoComplete", function (event) {that.doQuery(event);  });
306 		//that.$content.html("");
307 
308 		return that;
309 	}

311 	/**
312 	* Internal hide method. It removes the behavior.
313 	* @protected
314 	* @function
315 	* @name ch.AutoComplete-hide
316 	*/
317 	that.hide = function(event){
318 		that.trigger("hide");
319 		that.behaviorActived = false;
320 		that.$content.off("mouseover mousedown");
321 		ch.utils.document.off(".autoComplete");
322 		that["float"].innerHide();
323 		return that;
324 	}

326 	/**
327 	* It gives the main behavior(focus, blur and turn off autocomplete attribute) to the $trigger.
328 	* @protected
329 	* @function
330 	* @name ch.AutoComplete#configBehavior
331 	*/
332 	that.configBehavior = function () {
333 		that.$element
334 			.bind("focus", function (event) {
335 				that.show(event);
336 			})
337 			.bind("blur", function (event) {
338 				that.hide(event);
339 			})
340 			.attr("autocomplete","off")
341 			.addClass("ch-" + that.type + "-trigger");
342 		return that;
343 	};
