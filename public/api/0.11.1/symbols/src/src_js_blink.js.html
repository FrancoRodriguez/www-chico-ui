1 /**
  2 * Blink lets you give visual feedback to the user. Blink can be used when the user performs some action that changes some data at the screen. Blink creates a visual highlight changing background color from yellow to white.
  3 * @name Blink
  4 * @class Blink
  5 * @memberOf ch
  6 * @param {Object} conf Configuration object
  7 * @param {number} [conf.time] Amount of time to blink in milliseconds
  8 * @returns jQuery Object
  9 * @factorized
 10 * @exampleDescription Blinks a element.
 11 * @example
 12 * var widget = $(".example").blink();
 13 */
 14 ch.blink = function (conf) {
 15 
 16 	var that = this,
 17 		// Hex start level toString(16).
 18 		level = 1,
 19 		// Time, 200 miliseconds by default.
 20 		t = conf.time || 200;
 21 
 22 	// Inner highlighter.
 23 	function highlight () {
 24 		// Let know everyone we are active.
 25 		that.$element.addClass("ch-blink").attr("role","alert").attr("aria-live","polite");
 26 
 27 		// Begin steps.
 28 		setTimeout(step, t);
 29 	};
 30 
 31 	// Color iteration.
 32 	function step () {
 33 		// New hex level.
 34 		var h = level.toString(16);
 35 		// Change background-color, redraw().
 36 		that.element.style.backgroundColor = '#FFFF' + h + h;
 37 		// Itearate for all hex levels.
 38 		if (level < 15) {
 39 			// Increment hex level.
 40 			level += 1;
 41 			// Inner recursion.
 42 			setTimeout(step, t);
 43 		} else {
 44 			// Stop right there...
 45 			that.$element.removeClass("ch-blink").attr("aria-live","off").removeAttr("role");
 46 		}
 47 	};
 48 
 49 	// Start a blink if the element isn't active.
 50 	if (!that.$element.hasClass("ch-blink")) {
 51 		highlight();
 52 	}
 53 	// Return the element so keep chaining things.
 54 	return that.$element;
 55 }
 56 ch.factory("blink");
