1 /**
  2 * Cache control utility.
  3 * @name Cache
  4 * @class Cache
  5 * @memberOf ch
  6 */
  7 
  8 ch.cache = {
  9 
 10 	/**
 11 	* Map of cached resources
 12 	* @public
 13 	* @name ch.Cache#map
 14 	* @type object
 15 	*/
 16 	map: {},
 17 
 18 	/**
 19 	* Set a resource to the cache control
 20 	* @public
 21 	* @function
 22 	* @name ch.Cache#set
 23 	* @param {string} url Resource location
 24 	* @param {string} data Resource information
 25 	*/
 26 	set: function(url, data) {
 27 		ch.cache.map[url] = data;
 28 	},
 29 
 30 	/**
 31 	* Get a resource from the cache
 32 	* @public
 33 	* @function
 34 	* @name ch.Cache#get
 35 	* @param {string} url Resource location
 36 	* @returns data Resource information
 37 	*/
 38 	get: function(url) {
 39 		return ch.cache.map[url];
 40 	},
 41 
 42 	/**
 43 	* Remove a resource from the cache
 44 	* @public
 45 	* @function
 46 	* @name ch.Cache#rem
 47 	* @param {string} url Resource location
 48 	*/
 49 	rem: function(url) {
 50 		ch.cache.map[url] = null;
 51 		delete ch.cache.map[url];
 52 	},
 53 
 54 	/**
 55 	* Clears the cache map
 56 	* @public
 57 	* @function
 58 	* @name ch.Cache#flush
 59 	*/
 60 	flush: function() {
 61 		delete ch.cache.map;
 62 		ch.cache.map = {};
 63 	}
 64 };
