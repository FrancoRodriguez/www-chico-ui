413 /**
414 * Class to create UI Components
415 * @name Factory
416 * @class Factory
417 * @param o Configuration Object
418 * @example
419 *	o {
420 *		component: "chat",
421 *		callback: function(){},
422 *		[script]: "http://..",
423 *		[style]: "http://..",
424 *		[callback]: function(){}
425 *	}
426 * @returns collection
427 * @memberOf ch
428 */
429 // TODO: Always it should receive a conf object as parameter (see Multiple component)
430 // TODO: Try to deprecate .and() method on Validator
431 ch.factory = function(o) {
432 
433 	var x = o.component || o;
434 
435 	var create = function(x) {
436 
437 		// Send configuration to a component trough options object
438 		$.fn[x] = function( options ) {
439 
440 			var results = [];
441 			var that = this;
442 
443 			// Could be more than one argument
444 			var _arguments = arguments;
445 
446 			that.each( function(i, e) {
447 
448 				var conf = options || {};
449 
450 				var context = {};
451 					context.type = x;
452 					context.element = e;
453 					context.$element = $(e);
454 					context.uid = ch.utils.index += 1; // Global instantiation index
455 
456 				switch(typeof conf) {
457 					// If argument is a number, join with the conf
458 					case "number":
459 						var num = conf;
460 						conf = {};
461 						conf.value = num;
462 
463 						// Could come a messages as a second argument
464 						if (_arguments[1]) {
465 							conf.msg = _arguments[1];
466 						};
467 					break;
468 
469 					// This could be a message
470 					case "string":
471 						var msg = conf;
472 						conf = {};
473 						conf.msg = msg;
474 					break;
475 
476 					// This is a condition for custom validation
477 					case "function":
478 						var func = conf;
479 						conf = {};
480 						conf.lambda = func;
481 
482 						// Could come a messages as a second argument
483 						if (_arguments[1]) {
484 							conf.msg = _arguments[1];
485 						};
486 					break;
487 				};
488 
489 				// Create a component from his constructor
490 				var created = ch[x].call( context, conf );
491 
492 				/*
493 					MAPPING INSTANCES
494 					Internal interface for avoid mapping objects
495 					{
496 						exists:true,
497 						object: {}
498 					}
499 				*/
500 
501 				created = ( ch.utils.hasOwn(created, "public") ) ? created["public"] : created;
502 
503 				if (created.type) {
504 					var type = created.type;
505 					// If component don't exists in the instances map create an empty array
506 					if (!ch.instances[type]) { ch.instances[type] = []; }
507 						ch.instances[type].push( created );
508 				}
509 
510 				// Avoid mapping objects that already exists
511 				if (created.exists) {
512 					// Return the inner object
513 					created = created.object;
514 				}
515 
516 				results.push( created );
517 
518 			});
519 
520 			// return the created components collection or single component
521 			return ( results.length > 1 ) ? results : results[0];
522 		};
523 
524 		// if a callback is defined
525 		if ( o.callback ) { o.callback(); }
526 
527 	} // end create function
528 
529 	if ( ch[x] ) {
530 		// script already here, just create it
531 		create(x);
532 
533 	} else {
534 		// get resurces and call create later
535 		ch.get({
536 			"method":"component",
537 			"component": x,
538 			"script": (o.script)? o.script : "src/js/"+x+".js",
539 			"styles": (o.style)? o.style : "src/css/"+x+".css",
540 			"callback":create
541 		});
542 	}
543 }

545 /**
546 * Load components or content
547 * @name Get
548 * @class Get
549 * @param {object} o Configuration object
550 * @example
551 *	o {
552 *		component: "chat",
553 *		[script]: "http://..",
554 *		[style]: "http://..",
555 *		[callback]: function(){}
556 *	}
557 * @memberOf ch
558 */
559 ch.get = function(o) {
560 
561 	// ch.get: "Should I get a style?"
562 	if ( o.style ) {
563 		var style = document.createElement('link');
564 			style.href = o.style;
565 			style.rel = 'stylesheet';
566 			style.type = 'text/css';
567 	}
568 	// ch.get: "Should I get a script?"
569 	if ( o.script ) {
570 		var script = document.createElement("script");
571 			script.src = o.script;
572 	}
573 
574 	var head = document.getElementsByTagName("head")[0] || document.documentElement;
575 	// Handle Script loading
576 	var done = false;
577 
578 	// Attach handlers for all browsers
579 	script.onload = script.onreadystatechange = function() {
580 
581 		if ( !done && (!this.readyState ||
582 			this.readyState === "loaded" || this.readyState === "complete") ) {
583 			done = true;
584 			// if callback is defined call it
585 			if ( o.callback ) { o.callback( o.component ); }
586 			// Handle memory leak in IE
587 			script.onload = script.onreadystatechange = null;
588 			if ( head && script.parentNode ) { head.removeChild( script ); }
589 		}
590 	};
591 
592 	// Use insertBefore instead of appendChild to circumvent an IE6 bug.
593 	// This arises when a base node is used.
594 	if ( o.script ) { head.insertBefore( script, head.firstChild ); }
595 	if ( o.style ) { head.appendChild( style ); }
596 
597 }

696 * @example

702 * @example

711 ch.extend = function (klass) {
712 
713 	"use strict";
714 
715 	return {
716 		as: function (name, process) {
717 			// Create the component in Chico-UI namespace
718 			ch[name] = function (conf) {
719 				// Some interfaces need a data value,
720 				// others simply need to be 'true'.
721 				conf[name] = conf.value || true;
722 
723 				// Invoke pre-proccess if is defined,
724 				// or grab the raw conf argument,
725 				// or just create an empty object.
726 				conf = (process) ? process(conf) : conf || {};
727 
728 				// Here we recieve messages,
729 				// or create an empty object.
730 				conf.messages = conf.messages || {};
731 
732 				// If the interface recieve a 'msg' argument,
733 				// store it in the message map.
734 				if (ch.utils.hasOwn(conf, "msg")) {
735 					conf.messages[name] = conf.msg;
736 					conf.msg = null;
737 					delete conf.msg;
738 				}
739 				// Here is where the magic happen,
740 				// invoke the class with the new conf,
741 				// and return the instance to the namespace.
742 				return ch[klass].call(this, conf);
743 			};
744 			// Almost done, now we need expose the new component,
745 			// let's ask the factory to do it for us.
746 			ch.factory(name);
747 		} // end as method
748 	} // end return
749 };
