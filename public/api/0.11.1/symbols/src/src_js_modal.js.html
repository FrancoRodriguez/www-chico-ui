1 /**
  2 * Modal is a centered floated window with a dark gray dimmer background. Modal lets you handle its size, positioning and content.
  3 * @name Modal
  4 * @class Modal
  5 * @augments ch.Floats
  6 * @memberOf ch
  7 * @param {Object} [conf] Object with configuration properties.
  8 * @param {String} [conf.content] Sets content by: static content, DOM selector or URL. By default, the content is the href attribute value  or form's action attribute.
  9 * @param {Number || String} [conf.width] Sets width property of the component's layout. By default, the width is "500px".
 10 * @param {Number || String} [conf.height] Sets height property of the component's layout. By default, the height is elastic.
 11 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are enable.
 12 * @param {Boolean} [conf.cache] Enable or disable the content cache. By default, the cache is enable.
 13 * @param {String} [conf.closable] Sets the way (true, "button" or false) the Modal close. By default, the modal close true.
 14 * @returns itself
 15 * @factorized
 16 * @see ch.Floats
 17 * @see ch.Tooltip
 18 * @see ch.Layer
 19 * @see ch.Zoom
 20 * @exampleDescription Create a new modal window triggered by an anchor with a class name 'example'.
 21 * @example
 22 * var widget = $("a.example").modal();
 23 * @exampleDescription Create a new modal window triggered by form.
 24 * @example
 25 * var widget = $("form").modal();
 26 * @exampleDescription Create a new modal window with configuration.
 27 * @example
 28 * var widget = $("a.example").modal({
 29 *     "content": "Some content here!",
 30 *     "width": "500px",
 31 *     "height": 350,
 32 *     "cache": false,
 33 *     "fx": false
 34 * });
 35 * @exampleDescription Now <code>widget</code> is a reference to the modal instance controller. You can set a new content by using <code>widget</code> like this:
 36 * @example
 37 * widget.content("http://content.com/new/content");
 38 */
 39 
 40 ch.modal = function (conf) {
 41 
 42 	/**
 43 	* Reference to a internal component instance, saves all the information and configuration properties.
 44 	* @private
 45 	* @name ch.Modal#that
 46 	* @type object
 47 	*/
 48 	var that = this;
 49 	conf = ch.clon(conf);
 50 
 51 	conf.classes = conf.classes || "ch-box";
 52 	conf.reposition = false;
 53 
 54 	// Closable configuration
 55 	conf.closeButton = ch.utils.hasOwn(conf, "closeButton") ? conf.closeButton : true;
 56 	conf.closable = ch.utils.hasOwn(conf, "closable") ? conf.closable : "button";
 57 
 58 	conf.aria = {};
 59 
 60 	if (conf.closeButton) {
 61 		conf.aria.role = "dialog";
 62 		conf.aria.identifier = "aria-label";
 63 	} else {
 64 		conf.aria.role = "alert";
 65 	}
 66 
 67 	that.conf = conf;
 68 
 69 /**
 70 *	Inheritance
 71 */
 72 
 73 	that = ch.floats.call(that);
 74 	that.parent = ch.clon(that);
 75 
 76 /**
 77 *	Private Members
 78 */
 79 
 80 	/**
 81 	* Reference to the dimmer object, the gray background element.
 82 	* @private
 83 	* @name ch.Modal#$dimmer
 84 	* @type jQuery
 85 	*/
 86 	var $dimmer = $("<div class=\"ch-dimmer\">");
 87 
 88 	// Set dimmer height for IE6
 89 	if (ch.utils.html.hasClass("ie6")) { $dimmer.height(parseInt(document.documentElement.clientHeight, 10) * 3); }
 90 
 91 	/**
 92 	* Reference to dimmer control, turn on/off the dimmer object.
 93 	* @private
 94 	* @name ch.Modal#dimmer
 95 	* @type object
 96 	*/
 97 	var dimmer = {
 98 		on: function () {
 99 
100 			if (that.active) { return; }
101 
102 			$dimmer
103 				.css("z-index", ch.utils.zIndex += 1)
104 				.appendTo(ch.utils.body)
105 				.fadeIn();
106 
107 			if (conf.closable) {
108 				$dimmer.one("click", function (event) { that.innerHide(event) });
109 			}
110 
111 			// TODO: position dimmer with Positioner
112 			if (!ch.features.fixed) {
113 			 	ch.positioner({ element: $dimmer });
114 			}
115 
116 			if (ch.utils.html.hasClass("ie6")) {
117 				$("select, object").css("visibility", "hidden");
118 			}
119 		},
120 		off: function () {
121 			$dimmer.fadeOut("normal", function () {
122 				$dimmer.detach();
123 				if (ch.utils.html.hasClass("ie6")) {
124 					$("select, object").css("visibility", "visible");
125 				}
126 			});
127 		}
128 	};
129 
130 /**
131 *	Protected Members
132 */
133 
134 	/**
135 	* Inner show method. Attach the component's layout to the DOM tree and load defined content.
136 	* @protected
137 	* @name ch.Modal#innerShow
138 	* @function
139 	* @returns itself
140 	*/
141 	that.innerShow = function (event) {
142 		dimmer.on();
143 		that.parent.innerShow(event);
144 		that.$element.blur();
145 		return that;
146 	};
147 
148 	/**
149 	* Inner hide method. Hides the component's layout and detach it from DOM tree.
150 	* @protected
151 	* @name ch.Modal#innerHide
152 	* @function
153 	* @returns itself
154 	*/
155 	that.innerHide = function (event) {
156 		dimmer.off();
157 		that.parent.innerHide(event);
158 		return that;
159 	};
160 
161 	/**
162 	* Returns any if the component closes automatic.
163 	* @protected
164 	* @name ch.Modal#closable
165 	* @function
166 	* @returns boolean
167 	*/
168 
169 /**
170 *	Public Members
171 */
172 
173 	/**
174 	* @borrows ch.Object#uid as ch.Modal#uid
175 	*/
176 
177 	/**
178 	* @borrows ch.Object#element as ch.Modal#element
179 	*/
180 
181 	/**
182 	* @borrows ch.Object#type as ch.Modal#type
183 	*/
184 
185 	/**
186 	* @borrows ch.Uiobject#content as ch.Modal#content
187 	*/
188 
189 	/**
190 	* @borrows ch.Floats#isActive as ch.Modal#isActive
191 	*/
192 
193 	/**
194 	* @borrows ch.Floats#show as ch.Modal#show
195 	*/
196 
197 	/**
198 	* @borrows ch.Floats#hide as ch.Modal#hide
199 	*/
200 
201 	/**
202 	* @borrows ch.Floats#width as ch.Modal#width
203 	*/
204 
205 	/**
206 	* @borrows ch.Floats#height as ch.Modal#height
207 	*/
208 
209 	/**
210 	* @borrows ch.Floats#position as ch.Modal#position
211 	*/
212 
213 	/**
214 	* @borrows ch.Floats#closable as ch.Modal#closable
215 	*/
216 
217 /**
218 *	Default event delegation
219 */
220 
221 	if (that.element.tagName === "INPUT" && that.element.type === "submit") {
222 		that.$element.parents("form").on("submit", function (event) { that.innerShow(event); });
223 	} else {
224 		that.$element.on("click", function (event) { that.innerShow(event); });
225 	}
226 
227 	/**
228 	* Triggers when the component is ready to use.
229 	* @name ch.Modal#ready
230 	* @event
231 	* @public
232 	* @example
233 	* // Following the first example, using <code>widget</code> as modal's instance controller:
234 	* widget.on("ready",function () {
235 	*	this.show();
236 	* });
237 	*/
238 	setTimeout(function(){ that.trigger("ready")}, 50);
239 
240 	return that;
241 };

246 /**
247 * Transition lets you give feedback to the users when their have to wait for an action.
248 * @name Transition
249 * @class Transition
250 * @interface
251 * @augments ch.Floats
252 * @requires ch.Modal
253 * @memberOf ch
254 * @param {Object} [conf] Object with configuration properties.
255 * @param {String} [conf.content] Sets content by: static content, DOM selector or URL. By default, the content is the href attribute value  or form's action attribute.
256 * @param {Number || String} [conf.width] Sets width property of the component's layout. By default, the width is "500px".
257 * @param {Number || String} [conf.height] Sets height property of the component's layout. By default, the height is elastic.
258 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are enable.
259 * @param {Boolean} [conf.cache] Enable or disable the content cache. By default, the cache is enable.
260 * @param {String} [conf.closable] Sets the way (true, "button" or false) the Transition close. By default, the transition close true.
261 * @returns itself
262 * @factorized
263 * @see ch.Tooltip
264 * @see ch.Layer
265 * @see ch.Zoom
266 * @see ch.Modal
267 * @see ch.Floats
268 * @exampleDescription Create a transition.
269 * @example
270 * var widget = $("a.example").transition();
271 * @exampleDescription Create a transition with configuration.
272 * @example
273 * var widget = $("a.example").transition({
274 *     "content": "Some content here!",
275 *     "width": "500px",
276 *     "height": 350,
277 *     "cache": false,
278 *     "fx": false
279 * });
280 */
281 
282 ch.extend("modal").as("transition", function (conf) {
283 
284 	conf.closable = false;
285 
286 	conf.msg = conf.msg || conf.content || "Please wait...";
287 
288 	conf.content = $("<div class=\"ch-loading-big\"></div><p>" + conf.msg + "</p>");
289 
290 	return conf;
291 });
