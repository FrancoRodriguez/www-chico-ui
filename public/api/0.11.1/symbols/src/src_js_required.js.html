1 /**
  2 * Required validates that a must be filled.
  3 * @name Required
  4 * @class Required
  5 * @interface
  6 * @augments ch.Controls
  7 * @augments ch.Validation
  8 * @memberOf ch
  9 * @param {Object} [conf] Object with configuration properties.
 10 * @param {String} [conf.message] Validation message.
 11 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
 12 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 13 * @param {String} [conf.context] It's a reference to position the validation-bubble
 14 * @returns itself
 15 * @factorized
 16 * @see ch.Validation
 17 * @see ch.Custom
 18 * @see ch.Number
 19 * @see ch.String
 20 * @see ch.Validator
 21 * @see ch.Condition
 22 * @exampleDescription Simple validation
 23 * @example
 24 * $("input").required("This field is required");
 25 */
 26 ch.extend("validation").as("required", function(conf) {
 27 
 28 	conf.condition = {
 29 		name: "required",
 30 		expr: function(e) {
 31 
 32 			var $e = $(e);
 33 
 34 			var tag = ( $e.hasClass("options") || $e.hasClass("ch-form-options")) ? "OPTIONS" : e.tagName;
 35 			switch (tag) {
 36 				case 'OPTIONS':
 37 					return $e.find('input:checked').length !== 0;
 38 				break;
 39 
 40 				case 'SELECT':
 41 					var val = $e.val();
 42 					return (val != "-1" && val != "");
 43 				break;
 44 
 45 				case 'INPUT':
 46 				case 'TEXTAREA':
 47 					return $.trim($e.val()).length !== 0;
 48 				break;
 49 			};
 50 		},
 51 		message: conf.msg || conf.message,
 52 		value: conf.value
 53 	};
 54 
 55 	return conf;
 56 
 57 });
