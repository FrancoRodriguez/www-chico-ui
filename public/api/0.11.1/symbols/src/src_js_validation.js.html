  4 * @class Validation

 13 * @param {Object} [conf] Object with configuration properties.
 14 * @param {String} [conf.message] Validation message.
 15 * @param {String} [conf.points] Sets the points where validation-bubble will be positioned.
 16 * @param {String} [conf.offset] Sets the offset in pixels that validation-bubble will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 17 * @param {String} [conf.context] It's a reference to position the validation-bubble.
 18 * @returns itself
 19 * @factorized
 20 * @see ch.Controls

 22 * @see ch.Validator
 23 * @see ch.Required

 25 * @see ch.Number

 29 ch.validation = function (conf) {
 30 
 31 	/**
 32 	* Reference to a internal component instance, saves all the information and configuration properties.
 33 	* @protected
 34 	* @name ch.Validation#that
 35 	* @type itself
 36 	*/
 37 	var that = this;
 38 
 39 	conf = ch.clon(conf);
 40 	
 41 	// Configuration by default
 42 	conf.offset = conf.offset || "10 0";
 43 	conf.points = conf.points || "lt rt";
 44 
 45 	that.conf = conf;
 46 
 47 /**
 48 * Inheritance
 49 */
 50 
 51 	that = ch.controls.call(that);
 52 	that.parent = ch.clon(that);
 53 
 54 /**
 55 * Private Members
 56 */
 57 
 58 	// Reference to a Validator instance. If there isn't any, the Validation instance will create one.
 59 	var validator = that.validator = (function(){
 60 		var c = {};
 61 			c.condition = conf.condition
 62 	 	return that.$element.validator(c);
 63 	})();
 64 
 65 	/**
 66 	* Search for instances of Validation with the same trigger, and then merge it's properties with it.
 67 	* @private
 68 	* @name ch.Validation#checkInstance
 69 	* @function
 70 	* @returns Object
 71 	*/
 72 	var checkInstance;
 73 	if (checkInstance = function() {
 74 
 75 		var instance, instances = ch.instances.validation;
 76 		if ( instances && instances.length > 0 ) {
 77 			for (var i = 0, j = instances.length; i < j; i+=1) {
 78 				instance = instances[i];
 79 
 80 				if (instance.element !== that.element) {
 81 					continue;
 82 				}
 83 
 84 				return {
 85 					exists: true,
 86 					object: instance
 87 				}
 88 			}
 89 		}
 90 	}()){
 91 		return checkInstance;
 92 	};
 93 
 94 	// Reference to a Form instance. If there isn't any, the Validation instance will create one.
 95 	var form = that.form = (function() {
 96 
 97 		if (ch.utils.hasOwn(ch.instances, "form") && ch.instances.form.length > 0) {
 98 			var i = 0, j = ch.instances.form.length;
 99 			for (i; i < j; i+=1) {
100 				if (ch.instances.form[i].element === that.$element.parents("form")[0]) {
101 					return ch.instances.form[i]; // Get my parent
102 				} 
103 			};
104 		} 
105 
106 		that.$element.parents("form").form();
107 		var last = (ch.instances.form.length - 1);
108 		return ch.instances.form[last]; // Set my parent
109 
110 	})();
111 
112 	form.children.push(that["public"]);
113 
114 	/**
115 	* Validation event
116 	* @private
117 	* @name ch.Validation#validationEvent
118 	*/
119 	var validationEvent = (that.$element.hasClass("options") || that.$element.hasClass("ch-form-options") || that.element.tagName == "SELECT") ? "change" : "blur";
120 
121 	var clear = function() {
122 
123 		that.$element.removeClass("error ch-form-error");
124 		that["float"].innerHide();
125 
126 		validator.clear();
127 
128 
129 		/**
130 		* Triggers when al validations are cleared.
131 		* @name ch.Validation#clear
132 		* @event
133 		* @public
134 		* @exampleDescription Title
135 		* @example
136 		* widget.on("clear",function(){
137 		*	submitButton.enable();
138 		* });
139 		*/
140 		// old callback system
141 		that.callbacks('onClear');
142 		// new callback
143 		that.trigger("clear");
144 	};

146 /**
147 * Protected Members
148 */
149 
150 	/**
151 	* Flag that let you know if the validations is enabled or not.
152 	* @protected
153 	* @name ch.Validation#enabled
154 	* @type boolean
155 	*/
156 	that.enabled = true;
157 	
158 	/**
159 	* Reference to the Float component instanced.
160 	* @protected
161 	* @type Object
162 	* @name ch.Validation#float
163 	*/
164 	that["float"] = that.createFloat({
165 		"$element": (function() {
166 			var reference;
167 			// CHECKBOX, RADIO
168 			// TODO: when old forms be deprecated we must only support ch-form-options class
169 			if (that.$element.hasClass("options") || that.$element.hasClass("ch-form-options")) {
170 				// Helper reference from will be fired
171 				// H4
172 				if (that.$element.find("h4").length > 0) {
173 					var h4 = that.$element.find("h4"); // Find h4
174 						h4.wrapInner("<span>"); // Wrap content with inline element
175 					reference = h4.children(); // Inline element in h4 like helper reference
176 				// Legend
177 				} else if (that.$element.prev().prop("tagName") == "LEGEND") {
178 					reference = that.$element.prev(); // Legend like helper reference
179 				} else {
180 					reference = $(that.$element.find("label")[0]);
181 				}
182 			// INPUT, SELECT, TEXTAREA
183 			} else {
184 				reference = that.$element;
185 			}
186 			return reference;
187 		})(),
188 		"type": "validation",
189 		"content": "Error.",
190 		"classes": conf.classes || "ch-box-error",
191 		"cone": true,
192 		"cache": false,
193 		"closable": false,
194 		"aria": {
195 			"role": "alert"
196 		},
197 		"offset": conf.offset,
198 		"points": conf.points,
199 		"reposition": false
200 	});
201 	
202 
203 	/**
204 	* Stores the error object
205 	* @protected
206 	* @type Object
207 	* @name ch.Validation#error
208 	*/
209 	that.error = { 
210 		"condition": false,
211 		"msg": ""
212 	}
213 
214 	/**
215 	* Runs all validations to check if it has an error.
216 	* @protected
217 	* @type function
218 	* @returns boolean
219 	* @name ch.Validation#process
220 	*/
221 	that.process = function (evt) {
222 
223 		// Pre-validation: Don't validate disabled
224 		if (that.$element.attr("disabled") || !that.enabled) { return false; }
225 
226 		/**
227 		* Triggers before start validation process.
228 		* @name ch.Validation#beforeValidate
229 		* @event
230 		* @public
231 		* @exampleDescription
232 		* @example
233 		* widget.on("beforeValidate",function(event) {
234 		*	submitButton.disable();
235 		* });
236 		*/
237 		// old callback system
238 		that.callbacks('beforeValidate');
239 		// new callback
240 		that.trigger("beforeValidate");
241 		
242 		// Executes the validators engine with a specific value and returns an object.
243 		// Context is the validation
244 		var gotError = validator.validate.call(that["public"], that.element.value);
245 
246 		// Save the validator's status.
247 		var status = !gotError.status;
248 
249 		// If has Error...
250 		if (status) {
251 
252 			if (that.$element.prop("tagName") === "INPUT" || that.$element.prop("tagName") === "TEXTAREA") {
253 				// TODO: remove error class when deprecate old forms only ch-form error must be.
254 				that.$element.addClass("error ch-form-error");
255 			}
256 
257 			// to avoid reload the same content
258 			if (!that["float"]["public"].isActive() || !that.error.condition || that.error.condition !== gotError.condition) {
259 				that["float"]["public"].show((gotError.msg || form.messages[gotError.condition] || "Error"));
260 			} 
261 
262 			// Add blur or change event only one time to the element or to the elements's group
263 			if (!that.$element.data("events")) { that.$element.one(validationEvent, function(evt){that.process(evt);}); }
264 
265 			/**
266 			* Triggers when an error occurs on the validation process.
267 			* @name ch.Validation#error
268 			* @event
269 			* @public
270 			* @exampleDescription
271 			* @example
272 			* widget.on("error",function(event, condition) {
273 			*	if (condition === "required") {
274 			* 		errorModal.show();
275 			* 	}
276 			* });
277 			*/
278 			// old callback system
279 			that.callbacks('onError', gotError.condition);
280 			// new callback
281 			that.trigger("error", gotError.condition);
282 
283 			// Saves gotError
284 			that.error = gotError;
285 
286 		// else NOT Error!
287 		} else {
288 			that.$element.removeClass("error ch-form-error");
289 			that["float"].innerHide();
290 		}
291 
292 		/**
293 		* Triggers when the validation process ends.
294 		* @name ch.Validation#afterValidate
295 		* @event
296 		* @public
297 		* @exampleDescription
298 		* @example
299 		* widget.on("afterValidate",function(){
300 		*	submitButton.disable();
301 		* });
302 		*/
303 		// old callback system
304 		that.callbacks('afterValidate');
305 		// new callback
306 		that.trigger("afterValidate");
307 
308 		return status;
309 
310 	};

437 	* Turn on Validation and Validator engine or an specific condition.

441 	* @returns itself
442 	* @see ch.Validator

444 	that["public"].enable = function(condition){
445 		validator.enable(condition);
446 
447 		if (!condition) {
448 			that.enabled = true;
449 		}
450 
451 		return that["public"];
452 	};

454 	/**
455 	* Turn off Validation and Validator engine or an specific condition.
456 	* @public
457 	* @name ch.Validation#disable
458 	* @function
459 	* @returns itself
460 	* @see ch.Validator
461 	*/
462 	that["public"].disable = function (condition) {
463 		// Clean the validation if is active;
464 		clear();
465 
466 		// Turn off validator
467 		validator.disable(condition);
468 
469 		// Turn off validation, if all conditions are disabled
470 		if (!condition){
471 			that.enabled = false;
472 		}
473 
474 		return that["public"];
475 	};

533 	/**
534 	* Sets or gets conditions messages
535 	* @public
536 	* @since 0.10.4
537 	* @name ch.Validation#message
538 	* @function
539 	* @returns itself
540 	* @exampleDescription Sets a new message 
541 	* @example
542 	* validation.message("required", "New message for required validation");
543 	* @exampleDescription Gets a message from a condition
544 	* @example	
545 	* validation.message("required");
546 	*/
547 	that["public"].message = function (condition, msg) {
548 		if (condition === undefined) {
549 			throw "validation.message(condition, message): Please, give me a condition as parameter.";
550 		}
551 
552 		// Get a new message from a condition
553 		if (msg === undefined) {
554 			return validator.conditions[condition].message;
555 		}
556 
557 		// Sets a new message
558 		validator.conditions[condition].message = msg;
559 
560 		if (validator.isActive()) {
561 			that["public"]["float"].content(msg);
562 		}
563 
564 		return that["public"];
565 	}
