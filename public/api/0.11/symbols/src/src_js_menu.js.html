1 /**
  2 * Menu lets you organize the links by categories.
  3 * @name Menu
  4 * @class Menu
  5 * @augments ch.Controllers
  6 * @requires ch.Expando
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {Number} [conf.selected] Selects a child that will be open when component was loaded.
 10 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are disable.
 11 * @returns itself
 12 * @factorized
 13 * @see ch.Expando
 14 * @see ch.Controllers
 15 * @exampleDescription Create a new menu without configuration.
 16 * @example
 17 * var widget = $(".example").menu();
 18 * @exampleDescription Create a new menu with configuration.
 19 * @example
 20 * var widget = $(".example").menu({
 21 *     "selected": 2,
 22 *     "fx": true
 23 * });
 24 */
 25 
 26 ch.menu = function(conf){
 27 
 28 	/**
 29 	* Reference to a internal component instance, saves all the information and configuration properties.
 30 	* @private
 31 	* @name ch.Menu#that
 32 	* @type object
 33 	*/
 34 	var that = this;
 35 	
 36 	conf = ch.clon(conf);
 37 	conf.icon = ch.utils.hasOwn(conf, "icon") ? conf.icon : true;
 38 
 39 	that.conf = conf;
 40 	
 41 /**
 42 *	Inheritance
 43 */
 44 
 45 	that = ch.controllers.call(that);
 46 	that.parent = ch.clon(that);
 47 
 48 /**
 49 *	Private Members
 50 */
 51 
 52 	/**
 53 	* Indicates witch child is opened
 54 	* @private
 55 	* @name ch.Menu#selected
 56 	* @type number
 57 	*/
 58 	var selected = conf.selected - 1;
 59 
 60 	/**
 61 	* Inits an Expando component on each list inside main HTML code snippet
 62 	* @private
 63 	* @name ch.Menu#createLayout
 64 	* @function
 65 	*/
 66 	var createLayout = function(){
 67 
 68 		// No slide efects for IE6 and IE7
 69 		var efects = (ch.utils.html.hasClass("ie6") || ch.utils.html.hasClass("ie7")) ? false : true;
 70 
 71 		// List elements
 72 		that.$element.children().each(function(i, e){
 73 			// List element
 74 			var $li = $(e);
 75 									  
 76 			// Children of list elements
 77 			var $child = $li.children();
 78 
 79 			// Anchor inside list
 80 			if($child.eq(0).prop("tagName") == "A") {
 81 
 82 				// Add attr role to match wai-aria
 83 				$li.attr("role","presentation");
 84 
 85 				// Add class to list and anchor
 86 				$li.addClass("ch-bellows").children().addClass("ch-bellows-trigger");
 87 
 88 				// Add anchor to that.children
 89 				that.children.push( $child[0] );
 90 
 91 				return;
 92 			};
 93 
 94 			// List inside list, inits an Expando
 95 			var expando = $li.expando({
 96 				icon: conf.icon,
 97 				// Show/hide on IE6/7 instead slideUp/slideDown
 98 				fx: efects,
 99 				onShow: function(){
100 					// Updates selected tab when it's opened
101 					selected = i;
102 
103 					/**
104 					* Callback function
105 					* @name onSelect
106 					* @type {Function}
107 					* @memberOf ch.Menu
108 					*/
109 					that.callbacks.call(that, "onSelect");
110 					
111 					// new callback
112 					/**
113 					* It is triggered when the a fold is selected by the user.
114 					* @name ch.Menu#select
115 					* @event
116 					* @public
117 					* @exampleDescription When the user select
118 					* @example
119 					* widget.on("select",function(){
120 					*     app.off();
121 					* });
122 					*/
123 					that.trigger("select");
124 				}
125 			});
126 			
127 			var childs = $li.children(),
128 				$triggerCont = $(childs[0]),
129 				$menu = $(childs[1]);
130 				if (!conf.accordion) {
131 					$menu.attr("role","menu");
132 					$menu.children().children().attr("role", "menuitem");
133 					$menu.children().attr("role", "presentation");
134 				} 
135 				$triggerCont.attr("role","presentation");
136 
137 			// Add expando to that.children
138 			that.children.push( expando );
139 
140 		});
141 	};
142 	
143 	/**
144 	* Opens specific Expando child and optionally grandson
145 	* @private
146 	* @function
147 	* @name ch.Menu#select
148 	* @ignore
149 	*/
150 	var select = function (item) {
151 
152 		var child, grandson;
153 
154 		// Split item parameter, if it's a string with hash
155 		if (typeof item === "string") {
156 			var sliced = item.split("#");
157 			child = sliced[0] - 1;
158 			grandson = sliced[1];
159 
160 		// Set child when item is a Number
161 		} else {
162 			child = item - 1;
163 		}
164 
165 		// Specific item of that.children list
166 		var itemObject = that.children[ child ];
167 
168 		// Item as object
169 		if (ch.utils.hasOwn(itemObject, "uid")) {
170 
171 			// Show this list
172 			itemObject.show();
173 
174 			// Select grandson if splited parameter got a specific grandson
175 			if (grandson) $(itemObject.element).find("a").eq(grandson - 1).addClass("ch-menu-on");
176 
177 			// Accordion behavior
178 			if (conf.accordion) {
179 				// Hides every that.children list that don't be this specific list item
180 				$.each(that.children, function(i, e){
181 					if(
182 						// If it isn't an anchor...
183 						(e.tagName != "A") &&
184 						// If there are an unique id...
185 						(ch.utils.hasOwn(e, "uid")) &&
186 						// If unique id is different to unique id on that.children list...
187 						(that.children[ child ].uid != that.children[i].uid)
188 					){
189 						// ...hide it
190 						e.hide();
191 					};
192 				});
193 				
194 			};
195 
196 		// Item as anchor
197 		} else{
198 			// Just selects it
199 			that.children[ child ].addClass("ch-menu-on");
200 		};
201 
202 		return that;
203 	};
204 
205 	/**
206 	* Binds controller's own click to expando triggers
207 	* @private
208 	* @name ch.Menu#configureAccordion
209 	* @function
210 	*/
211 	var configureAccordion = function(){
212 
213 		$.each(that.children, function(i, e){
214 			$(e.element).find(".ch-expando-trigger").unbind("click").bind("click", function(){
215 				select(i + 1);
216 			});
217 		});
218 
219 		return;
220 	};
221 
222 /**
223 *	Protected Members
224 */
225 
226 /**
227 *	Public Members
228 */
229 	/**
230 	* @borrows ch.Object#uid as ch.Menu#uid
231 	*/	
232 	
233 	/**
234 	* @borrows ch.Object#element as ch.Menu#element
235 	*/
236 
237 	/**
238 	* @borrows ch.Object#type as ch.Menu#type
239 	*/
240 	
241 	/**
242 	* Select a specific children.
243 	* @public
244 	* @name select
245 	* @name ch.Menu
246 	* @param item The number of the item to be selected
247 	* @returns 
248 	*/
249 	that["public"].select = function (item) {
250 		// Getter
251 		if (!item) {
252 			if (isNaN(selected)) {
253 				return "";
254 			}
255 			return selected + 1;
256 		}
257 
258 		// Setter
259 		select(item);
260 		return that["public"];
261 	};
262 
263 /**
264 *	Default event delegation
265 */
266 
267 	// Sets component main class name
268 
269 	// Inits an Expando component on each list inside main HTML code snippet
270 	createLayout();
271 
272 	// Accordion behavior
273 	if (conf.accordion) {
274 		// Sets the interface main class name for avoid
275 		configureAccordion();
276 		that.$element.addClass('ch-accordion')
277 	} else {
278 		that.$element.addClass('ch-menu');
279 		// Set the wai-aria for Menu
280 		that.$element.attr("role","navigation");
281 	}
282 
283 	// Select specific item if there are a "selected" parameter on component configuration object
284 	if (ch.utils.hasOwn(conf, "selected")) select(conf.selected);
285 	
286 	/**
287 	* Triggers when the component is ready to use (Since 0.8.0).
288 	* @name ch.Menu#ready
289 	* @event
290 	* @public
291 	* @since 0.8.0
292 	* @exampleDescription Following the first example, using <code>widget</code> as menu's instance controller:
293 	* @example
294 	* widget.on("ready",function () {
295 	*	this.select();
296 	* });
297 	*/
298 	setTimeout(function(){ that.trigger("ready")}, 50);
299 
300 	return that;
301 
302 };
