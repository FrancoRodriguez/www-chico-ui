  2 /**
  3 * Validate numbers.
  4 * @name Number
  5 * @class Number
  6 * @interface
  7 * @augments ch.Watcher
  8 * @memberOf ch
  9 * @param {object} conf Object with configuration properties
 10 * @returns itself
 11 * @see ch.Watcher
 12 * @see ch.Required
 13 * @see ch.Custom
 14 * @see ch.String
 15 * @example
 16 * // Create a number validation
 17 * $("input").number("This field must be a number.");
 18 */
 19 
 20 ch.extend("watcher").as("number", function(conf) {
 21 
 22 	// $.number("message"); support
 23 	if ( !conf.number && !conf.min && !conf.max && !conf.price ) {
 24 		conf.number = true;
 25 	};
 26 	
 27 	// Define the conditions of this interface
 28 	conf.conditions = [{
 29 			name: "number",
 30 			patt: /^([0-9\s]+)$/ 
 31 		},{
 32 			name: "min",
 33 			expr: function(a,b) { return a >= b } 
 34 		},{
 35 			name: "max",
 36 			expr: function(a,b) { return a <= b } 
 37 		}];
 38 
 39 	return conf;
 40 
 41 });

 78 /**
 79 * Validate a number with a price format.
 80 * @name Price
 81 * @class Price
 82 * @interface
 83 * @augments ch.Watcher
 84 * @memberOf ch
 85 * @param {string} [message] Validation message.
 86 * @returns itself
 87 * @see ch.Watcher
 88 * @example
 89 * $("input").price("Write valid price.");
 90 */
 91 
 92 ch.extend("watcher").as("price",function(conf){
 93 
 94 	conf.price = true;	
 95 
 96 	// Define the conditions of this interface
 97 	conf.conditions = [{
 98 			name: "price",
 99 			patt: /^(\d+)[.,]?(\d?\d?)$/ 
100 		}];
101 
102 	return conf;
103 
104 });
