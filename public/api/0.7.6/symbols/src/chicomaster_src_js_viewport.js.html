1 /**
  2 * Viewport is a reference to position and size of the visible area of browser.
  3 * @abstract
  4 * @name Viewport
  5 * @class Viewport
  6 * @memberOf ch
  7 */
  8 ch.viewport = {
  9 
 10 	/**
 11 	* Width of the visible area.
 12 	* @public
 13 	* @name ch.Viewport#width
 14 	* @type Number
 15 	*/
 16 	"width": ch.utils.window.width(),
 17 
 18 	/**
 19 	* Height of the visible area.
 20 	* @public
 21 	* @name ch.Viewport#height
 22 	* @type Number
 23 	*/
 24 	"height": ch.utils.window.height(),
 25 
 26 	/**
 27 	* Left offset of the visible area.
 28 	* @public
 29 	* @name ch.Viewport#left
 30 	* @type Number
 31 	*/
 32 	"left": ch.utils.window.scrollLeft(),
 33 
 34 	/**
 35 	* Top offset of the visible area.
 36 	* @public
 37 	* @name ch.Viewport#top
 38 	* @type Number
 39 	*/
 40 	"top": ch.utils.window.scrollTop(),
 41 
 42 	/**
 43 	* Right offset of the visible area.
 44 	* @public
 45 	* @name ch.Viewport#right
 46 	* @type Number
 47 	*/
 48 	"right": ch.utils.window.scrollLeft() + ch.utils.window.width(),
 49 
 50 	/**
 51 	* Bottom offset of the visible area.
 52 	* @public
 53 	* @name ch.Viewport#bottom
 54 	* @type Number
 55 	*/
 56 	"bottom": ch.utils.window.scrollTop() + ch.utils.window.height(),
 57 
 58 	/**
 59 	* Updates width and height of the visible area and updates ch.viewport.width and ch.viewport.height
 60 	* @public
 61 	* @function
 62 	* @name ch.Viewport#getSize
 63 	* @returns Size Object
 64 	*/
 65 	"getSize": function () {
 66 
 67 		return {
 68 			"width": this.width = ch.utils.window.width(),
 69 			"height": this.height = ch.utils.window.height()
 70 		};
 71 
 72 	},
 73 
 74 	/**
 75 	* Updates left, top, right and bottom offset of the visible area and updates respective ch.viewport properties.
 76 	* @public
 77 	* @function
 78 	* @name ch.Viewport#getOffset
 79 	* @returns Offset Object
 80 	*/
 81 	"getOffset": function () {
 82 
 83 		var size = this.getSize(),
 84 			scrollLeft = ch.utils.window.scrollLeft(),
 85 			scrollTop = ch.utils.window.scrollTop();
 86 
 87 		return {
 88 			"left": this.left = scrollLeft,
 89 			"top": this.top = scrollTop,
 90 			"right": this.right = scrollLeft + size.width,
 91 			"bottom": this.bottom = scrollTop + size.height
 92 		};
 93 		
 94 	}
 95 };
