1 /**
  2 * Watcher is a validation engine for html forms elements.
  3 * @abstract
  4 * @name Watcher
  5 * @class Watcher
  6 * @augments ch.Uiobject
  7 * @memberOf ch
  8 * @requires ch.Form
  9 * @requires ch.Validator
 10 * @requires ch.Helper
 11 * @requires ch.Positioner
 12 * @requires ch.Events
 13 * @param {object} conf Object with configuration properties
 14 * @returns itself
 15 * @see ch.Required
 16 * @see ch.String
 17 * @see ch.Number
 18 * @see ch.Custom
 19 */
 20 
 21 ch.watcher = function (conf) {
 22 
 23 	/**
 24 	* Reference to a internal component instance, saves all the information and configuration properties.
 25 	* @protected
 26 	* @name ch.Watcher#that
 27 	* @type itself
 28 	*/
 29 	var that = this;
 30 
 31 	conf = ch.clon(conf);
 32 	that.conf = conf;
 33 
 34 /**
 35 * Inheritance
 36 */
 37 
 38 	that = ch.object.call(that);
 39 	that.parent = ch.clon(that);
 40 
 41 /**
 42 * Private Members
 43 */
 44 	
 45 	// Reference to a Validator instance. If there isn't any, the Watcher instance will create one.
 46 	var validator = that.validator = (function(){
 47 		var c = {};
 48 			c.condition = conf.condition
 49 	 	return that.$element.validator(c);
 50 	})();
 51 
 52 	/**
 53 	* Search for instances of Watcher with the same trigger, and then merge it's properties with it.
 54 	* @private
 55 	* @name ch.Watcher#checkInstance
 56 	* @function
 57 	* @returns Object
 58 	*/
 59 	var checkInstance;
 60 	if (checkInstance = function() {
 61 
 62 		var instance, instances = ch.instances.watcher;
 63 		if ( instances && instances.length > 0 ) {
 64 			for (var i = 0, j = instances.length; i < j; i+=1) {
 65 				instance = instances[i];
 66 
 67 				if (instance.element !== that.element) {
 68 					continue;
 69 				}
 70 
 71 				return {
 72 					exists: true,
 73 					object: instance
 74 				}
 75 			}
 76 		}
 77 	}()){
 78 		return checkInstance;
 79 	};
 80 
 81 	// Reference to a Form instance. If there isn't any, the Watcher instance will create one.
 82 	var form = that.form = (function() {
 83 		if (ch.utils.hasOwn(ch.instances, "form") && ch.instances.form.length > 0) {
 84 			var i = 0, j = ch.instances.form.length;
 85 			for (i; i < j; i+=1) {
 86 				if (ch.instances.form[i].element === that.$element.parents("form")[0]) {
 87 					return ch.instances.form[i]; // Get my parent
 88 				};
 89 			};
 90 		} else {
 91 			that.$element.parents("form").form();
 92 			var last = (ch.instances.form.length - 1);
 93 			return ch.instances.form[last]; // Set my parent
 94 		}
 95 	})();
 96 	form.children.push(that["public"]);
 97 
 98 	/**
 99 	* Validation event
100 	* @private
101 	* @name ch.Watcher#validationEvent
102 	*/
103 	var validationEvent = (that.$element.hasClass("options") || that.element.tagName == "SELECT") ? "change" : "blur";
104 
105 	var hasError = function () {
106 
107 		// Pre-validation: Don't validate disabled
108 		if (that.$element.attr('disabled') || !that.enabled) { return false; }
109 
110 		/**
111 		* Triggers before start validation process.
112 		* @name ch.Watcher#beforeValidate
113 		* @event
114 		* @public
115 		* @example
116 		* me.on("beforeValidate",function(){
117 		*	submitButton.disable();
118 		* });
119 		*/
120 		// old callback system
121 		that.callbacks('beforeValidate');
122 		// new callback
123 		that.trigger("beforeValidate");
124 		
125 		var gotError = validator.validate(value());
126 
127 		var status = !gotError.status;
128 
129 		if (status) {
130 			that.$element.addClass("error");
131 			that.helper.show(gotError.msg || form.messages[gotError.condition] || "Error.");
132 
133 			// Add blur or change event only one time
134 			if (!that.$element.data("events")) { that.$element.one(validationEvent, function(event){ hasError(); }); }
135 
136 			/**
137 			* Triggers when an error occurs on the validation process.
138 			* @name ch.Watcher#error
139 			* @event
140 			* @public
141 			* @example
142 			* me.on("error",function(){
143 			*	errorModal.show();
144 			* });
145 			*/
146 			// old callback system
147 			that.callbacks('onError');
148 			// new callback
149 			that.trigger("error");
150 
151 		} else {
152 			that.$element.removeClass("error");
153 			that.helper.hide();
154 		}
155 
156 		/**
157 		* Triggers when the validation process ends.
158 		* @name ch.Watcher#afterValidate
159 		* @event
160 		* @public
161 		* @example
162 		* me.on("afterValidate",function(){
163 		*	submitButton.disable();
164 		* });
165 		*/
166 		// old callback system
167 		that.callbacks('afterValidate');
168 		// new callback
169 		that.trigger("afterValidate");
170 
171 		return status;
172 
173 	};
174 
175 	var clear = function() {
176 
177 		that.$element.removeClass("error");
178 		that.helper.hide();
179 
180 		// Don't work
181 		//that.$element.unbind(validationEvent, hasError); // Remove blur and change event
182 		
183 		validator.clear();
184 
185 		/**
186 		* Triggers when al validations are cleared.
187 		* @name ch.Watcher#clear
188 		* @event
189 		* @public
190 		* @example
191 		* me.on("clear",function(){
192 		*	submitButton.enable();
193 		* });
194 		*/
195 		// old callback system
196 		that.callbacks('onClear');
197 		// new callback
198 		that.trigger("clear");
199 	};
200 
201 	/**
202 	* Returns a value of element
203 	* @private
204 	* @name ch.Watcher#value
205 	* @function
206 	* @returns string
207 	*/
208 	var value = function(){
209 		return that.element.value;
210 	};
211 
212 
213 /**
214 * Protected Members
215 */
216 
217 	/**
218 	* Flag that let you know if the watchers is enabled or not.
219 	* @protected
220 	* @name ch.Watcher#enabled
221 	* @type boolean
222 	*/
223 	that.enabled = true;
224 
225 	that.$reference = (function() {
226 		var reference;
227 		// CHECKBOX, RADIO
228 		if (that.$element.hasClass("options")) {
229 			// Helper reference from will be fired
230 			// H4
231 			if (that.$element.find('h4').length > 0) {
232 				var h4 = that.$element.find('h4'); // Find h4
233 					h4.wrapInner('<span>'); // Wrap content with inline element
234 				reference = h4.children(); // Inline element in h4 like helper reference
235 			// Legend
236 			} else if (that.$element.prev().prop('tagName') == 'LEGEND') {
237 				reference = that.$element.prev(); // Legend like helper reference
238 			}
239 		// INPUT, SELECT, TEXTAREA
240 		} else {
241 			reference = that.$element;
242 		}
243 		return reference;
244 	})();
245 
246 	that.helper = that.$reference.helper();
247 
248 /**
249 *	Public Members
250 */
251 
252 	/**
253 	* The 'uid' is the Chico's unique instance identifier. Every instance has a different 'uid' property. You can see its value by reading the 'uid' property on any public instance.
254 	* @public
255 	* @name ch.Watcher#uid
256 	* @type number
257 	*/
258 
259 	/**
260 	* Reference to a DOM Element. This binding between the component and the HTMLElement, defines context where the component will be executed. Also is usual that this element triggers the component default behavior.
261 	* @public
262 	* @name ch.Watcher#element
263 	* @type HTMLElement
264 	*/
265 
266 	/**
267 	* This public property defines the component type. All instances are saved into a 'map', grouped by its type. You can reach for any or all of the components from a specific type with 'ch.instances'.
268 	* @public
269 	* @name ch.Watcher#type
270 	* @type string
271 	*/
272 	that["public"].type = "watcher"; // Everything is a "watcher" type, no matter what interface is used
273 
274 	/**
275 	* Used by the helper's positioner to do his magic.
276 	* @public
277 	* @name ch.Watcher#reference
278 	* @type jQuery Object
279 	* @TODO: remove 'reference' from public scope
280 	*/
281 	that["public"].reference = that.$reference;
282 
283 	/**
284 	* Run all configured validations.
285 	* @public
286 	* @function
287 	* @name ch.Watcher#hasError
288 	* @returns boolean
289 	*/
290 	that["public"].hasError = function(){
291 		return hasError();
292 	}
293 	
294 	/**
295 	* Run all configured validations.
296 	* @public
297 	* @function
298 	* @name ch.Watcher#validate
299 	* @returns boolean
300 	*/
301 	that["public"].validate = function(){
302 		hasError();
303 
304 		return that["public"];
305 	}
306 
307 	/**
308 	* Clear all active validations.
309 	* @public
310 	* @name ch.Watcher#clear
311 	* @function
312 	* @returns itself
313 	*/
314 	that["public"].clear = function() {
315 		clear();
316 
317 		return that["public"];
318 	};
319 
320 	/**
321 	* Clear all active validations.
322 	* @public
323 	* @name ch.Watcher#resset
324 	* @function
325 	* @returns itself
326 	*/
327 	that["public"].reset = function() {
328 		clear();
329 
330 		return that["public"];
331 	};
332 
333 	/**
334 	* Let you keep chaining methods.
335 	* @public
336 	* @name ch.Watcher#and
337 	* @function
338 	* @returns jQuery Object
339 	*/
340 	that["public"].and = function(){
341 		return that.$element;
342 	};
343 
344 	/**
345 	* Is the little sign that floats showing the validation message. Is a Float component, so you can change it's content, width or height and change its visibility state.
346 	* @public
347 	* @name ch.Watcher#form
348 	* @type ch.Form
349 	* @see ch.Form
350 	*/
351 	that["public"].form = form;
352 
353 	/**
354 	* Is the little sign that floats showing the validation message. Is a Float component, so you can change it's content, width or height and change its visibility state.
355 	* @public
356 	* @name ch.Watcher#validator
357 	* @type ch.Validator
358 	* @see ch.Validator
359 	*/
360 	that["public"].validator = validator;
361 	
362 	/**
363 	* Is the little sign that floats showing the validation message. Is a Float component, so you can change it's content, width or height and change its visibility state.
364 	* @public
365 	* @name ch.Watcher#helper
366 	* @type ch.Helper
367 	* @see ch.Floats
368 	* @see ch.Helper
369 	*/
370 	that["public"].helper = that.helper;
371 
372 	/**
373 	* Turn on Watcher and Validator engine or an specific condition.
374 	* @public
375 	* @name ch.Watcher#enable
376 	* @function
377 	* @returns itself
378 	* @see ch.Validator
379 	*/
380 	that["public"].enable = function(condition){
381 		validator.enable(condition);
382 
383 		if (!condition) {
384 			that.enabled = true;
385 		}
386 
387 		return that["public"];
388 	}
389 
390 	/**
391 	* Turn off Watcher and Validator engine or an specific condition.
392 	* @public
393 	* @name ch.Watcher#disable
394 	* @function
395 	* @returns itself
396 	* @see ch.Validator
397 	*/
398 	that["public"].disable = function(condition){
399 		validator.disable(condition);
400 
401 		if (!condition){
402 			that.enabled = false;
403 		}
404 
405 		return that["public"];
406 	}
407 
408 
409 	/**
410 	* Turn off Watcher and Validator engine or an specific condition.
411 	* @public
412 	* @name ch.Watcher#isActive
413 	* @function
414 	* @returns boolean
415 	* @see ch.Validator
416 	*/
417 	that["public"].isActive = function(){
418 		return validator.isActive();
419 	}
420 
421 /**
422 *	Default event delegation
423 */
424 
425 	/**
426 	* Triggers when the component is ready to use.
427 	* @name ch.Watcher#ready
428 	* @event
429 	* @public
430 	* @example
431 	* // Following the first example, using 'me' as modal's instance controller:
432 	* me.on("ready",function(){
433 	*	this.show();
434 	* });
435 	*/
436 	that.trigger("ready");
437 
438 	return that;
439 };
440 ch.factory("watcher");
