1 /**
  2 * Description
  3 * @abstract
  4 * @name Condition
  5 * @class Condition
  6 * @memberOf ch
  7 */
  8 ch.condition = function(condition) {
  9 
 10 /**
 11 * Reference to a internal component instance, saves all the information and configuration properties.
 12 * @protected
 13 * @name ch.Condition#that
 14 * @type itself
 15 */
 16 	var that = this;
 17 
 18 /**
 19 * Private Members
 20 */
 21 
 22 	/**
 23 	* Flag that let you know if the condition is enabled or not.
 24 	* @private
 25 	* @name ch.Condition#enabled
 26 	* @type boolean
 27 	*/
 28 	var	enabled = true,
 29 
 30 		test = function (value) {
 31 
 32 			if (!enabled) {
 33 				return true;
 34 			}
 35 	
 36 			if (condition.patt){
 37 				return condition.patt.test(value);
 38 			}
 39 	
 40 			if (condition.expr){
 41 				return condition.expr(value, condition.value);
 42 			}
 43 	
 44 			if (condition.func){
 45 				// Call validation function with 'this' as scope.
 46 				return condition.func.call(that, value);
 47 			}
 48 	
 49 		},
 50 
 51 		enable = function() {
 52 			enabled = true;
 53 			
 54 			return condition;
 55 		},
 56 	
 57 		disable = function() {
 58 			enabled = false;
 59 			
 60 			return condition;
 61 		};
 62 
 63 /**
 64 * Protected Members
 65 */
 66 
 67 /**
 68 * Public Members
 69 */
 70 
 71 	/**
 72 	* Flag that let you know if the all conditions are enabled or not.
 73 	* @public
 74 	* @name ch.Condition#name
 75 	* @type string
 76 	*/
 77 
 78 	/**
 79 	* Message defined for this condition
 80 	* @public
 81 	* @name ch.Condition#message
 82 	* @type string
 83 	*/
 84 
 85 	/**
 86 	* Run configured condition
 87 	* @public
 88 	* @name ch.Condition#test
 89 	* @function
 90 	* @returns boolean
 91 	*/
 92 
 93 	/**
 94 	* Turn on condition.
 95 	* @public
 96 	* @function
 97 	* @name ch.Condition#enable
 98 	* @returns itself
 99 	*/
100 
101 	/**
102 	* Turn off condition.
103 	* @public
104 	* @function
105 	* @name ch.Condition#disable
106 	* @returns itself
107 	*/
108 
109 	condition = $.extend(condition, {
110 		test: test,
111 		enable: enable,
112 		disable: disable
113 	});
114 
115 	return condition;
116 
117 };
