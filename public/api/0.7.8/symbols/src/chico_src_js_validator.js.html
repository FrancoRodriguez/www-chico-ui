1 /**
  2 * Validator is a validation engine for html forms elements.
  3 * @abstract
  4 * @name Validator
  5 * @class Validator
  6 * @augments ch.Object
  7 * @memberOf ch
  8 * @requires ch.Condition
  9 * @param {Object} conf Object with configuration properties
 10 * @returns itself
 11 * @see ch.Condition
 12 */
 13 
 14 ch.validator = function(conf) {
 15 	/**
 16 	* Reference to a internal component instance, saves all the information and configuration properties.
 17 	* @protected
 18 	* @name ch.Validator#that
 19 	* @type Object
 20 	*/
 21 	var that = this;
 22 	conf = ch.clon(conf);
 23 	that.conf = conf;	
 24 
 25 /**
 26 * Inheritance
 27 */
 28 
 29 	that = ch.object.call(that);
 30 	that.parent = ch.clon(that);
 31 
 32 /**
 33 * Private Members
 34 */
 35 	var conditions = (function(){
 36 		var c = {}; // temp collection
 37 		var condition = ch.condition.call(that["public"], conf.condition);
 38 
 39 		c[condition.name] = condition;
 40 
 41 		// return all the configured conditions
 42 		return c;
 43 	})(); // Love this ;)
 44 
 45 	/**
 46 	* Search for instances of Validators with the same trigger, and then merge it's properties with it.
 47 	* @private
 48 	* @name ch.Validator#checkInstance
 49 	* @function
 50 	* @returns Object
 51 	*/
 52 	var checkInstance;
 53 	if (checkInstance = function() {
 54 
 55 		var instance, instances = ch.instances.validator;
 56 		if ( instances && instances.length > 0 ) {
 57 			for (var i = 0, j = instances.length; i < j; i+=1) {
 58 				instance = instances[i];
 59 
 60 				if (instance.element !== that.element) {
 61 
 62 					continue;
 63 				}
 64 
 65 				// Extend instance's conditions
 66 				instance.extend(conditions);
 67 
 68 				// To let know the ch.Factory that already exists,
 69 				// this way we avoid to have duplicated references.
 70 				return {
 71 					exists: true,
 72 					object: instance
 73 				}
 74 			}
 75 		}
 76 	}()){
 77 		return checkInstance;
 78 	};
 79 
 80 	var validate = function(value) {
 81 
 82 		if (!that.enabled) { return true; }
 83 
 84 		var condition, tested, empty, val, message, required = conditions["required"];
 85 
 86 		// Avoid fields that aren't required when they are empty or de-activated
 87 		if (!required && value === "" && that.active === false) { return {"status": true}; }
 88 
 89 		if (that.enabled && (!that.active || value !== "" || required)) {
 90 			/**
 91 			* Triggers before start validation process.
 92 			* @name ch.Validator#beforeValidate
 93 			* @event
 94 			* @public
 95 			* @example
 96 			* me.on("beforeValidate",function(){
 97 			*	submitButton.disable();
 98 			* });
 99 			*/
100 			// old callback system
101 			that.callbacks('beforeValidate');
102 			// new callback
103 			that.trigger("beforeValidate");
104 
105 			// for each condition
106 			for (condition in conditions){
107 
108 				val = ((condition === "required") ? that.element : value);
109 				tested = test(condition, val);
110 
111 				// return false if any test fails,
112 				if (!tested) {
113 
114 					/**
115 					* Triggers when an error occurs on the validation process.
116 					* @name ch.Validator#error
117 					* @event
118 					* @public
119 					* @example
120 					* me.on("error",function(event, condition){
121 					*	errorModal.show();
122 					* });
123 					*/
124 					// old callback system
125 					that.callbacks('onError', condition);
126 					// new callback
127 					that.trigger("error", condition);
128 
129 					that.active = true;
130 
131 					// stops the proccess
132 					//return false;
133 					return {
134 						"status": false,
135 						"condition": condition,
136 						"msg": conditions[condition].message
137 					}
138 				};
139 			}
140 		}
141 
142 		// Status OK (with previous error)
143 		if (that.active || !that.enabled) {
144 			// Public status OK
145 			that.active = false;
146 		}
147 
148 		/**
149 		* Triggers when the validation process ends.
150 		* @name ch.Validator#afterValidate
151 		* @event
152 		* @public
153 		* @example
154 		* me.on("afterValidate",function(){
155 		*	submitButton.disable();
156 		* });
157 		*/
158 		// old callback system
159 		that.callbacks('afterValidate');
160 		// new callback
161 		that.trigger("afterValidate");
162 
163 		// It's all good ;)
164 		//return true;
165 		return {
166 			"status": true
167 		}
168 	}
169 
170 	/**
171 	* Test a condition looking for error.
172 	* @private
173 	* @name ch.Validator#test
174 	* @see ch.Condition
175 	*/
176 	var test = function(condition, value){
177 		
178 		if (value === "" && condition !== "required") { return true };
179 		
180 		var isOk = false,
181 			condition = conditions[condition];
182 
183 		isOk = condition.test(value);
184 
185 		return isOk;
186 		
187 	};
188 
189 /**
190 * Protected Members
191 */
192 
193 	/**
194 	* Flag that let you know if there's a validation going on.
195 	* @protected
196 	* @name ch.Validator#active
197 	* @type boolean
198 	*/
199 	that.active = false;
200 
201 	/**
202 	* Flag that let you know if the all conditions are enabled or not.
203 	* @protected
204 	* @name ch.Validator#enabled
205 	* @type boolean
206 	*/
207 	that.enabled = true;
208 
209 /**
210 *	Public Members
211 */
212 
213 	/**
214 	* The 'uid' is the Chico's unique instance identifier. Every instance has a different 'uid' property. You can see its value by reading the 'uid' property on any public instance.
215 	* @public
216 	* @name ch.Validator#uid
217 	* @type Number
218 	*/
219 	
220 	/**
221 	* This public property defines the component type. All instances are saved into a 'map', grouped by its type. You can reach for any or all of the components from a specific type with 'ch.instances'.
222 	* @public
223 	* @name ch.Validator#type
224 	* @type String
225 	*/
226 	that["public"].type = "validator";
227 
228 	/**
229 	* This public Map saves all the validation configurations from this instance.
230 	* @public
231 	* @name ch.Validator#conditions
232 	* @type object
233 	*/
234 	that["public"].conditions = conditions;
235 
236 	/**
237 	* Active is a boolean property that let you know if there's a validation going on.
238 	* @public
239 	* @name ch.Validator#isActive
240 	* @function
241 	* @returns itself
242 	*/
243 	that["public"].isActive = function() {
244 		return that.active;
245 	};
246 
247 	/**
248 	* Let you keep chaining methods.
249 	* @public
250 	* @name ch.Validator#and
251 	* @function
252 	* @returns itself
253 	*/
254 	that["public"].and = function(){
255 		return that.$element;
256 	};
257 
258 	/**
259 	* Merge its conditions with a new conditions of another instance with the same trigger.
260 	* @public
261 	* @name ch.Validator#extend
262 	* @function
263 	* @returns itself
264 	*/
265 	that["public"].extend = function(input){
266 		$.extend(conditions, input);
267 
268 		return that["public"];
269 	};
270 
271 	/**
272 	* Clear all active validations.
273 	* @public
274 	* @name ch.Validator#clear
275 	* @function
276 	* @returns itself
277 	*/
278 	that["public"].clear = function() {
279 		that.active = false;
280 
281 		return that["public"];
282 	};
283 
284 	/**
285 	* Runs all configured conditions and returns an object with a status value, condition name and a message.
286 	* @public
287 	* @function
288 	* @name ch.Validator#validate
289 	* @returns Status Object
290 	*/
291 	that["public"].validate = function(value){
292 		return validate(value);
293 	}
294 
295 	/**
296 	* Turn on Validator engine or an specific condition.
297 	* @public
298 	* @name ch.Validator#enable
299 	* @function
300 	* @returns itself
301 	*/
302 	that["public"].enable = function(condition){
303 		if (condition && conditions[condition]){
304 			// Enable specific condition
305 			conditions[condition].enable();
306 		} else {
307 			// enable all
308 			that.enabled = true;
309 			for (condition in conditions){
310 				conditions[condition].enable();
311 			}
312 		}
313 		return that["public"];
314 	}
315 
316 	/**
317 	* Turn on Validator engine or an specific condition.
318 	* @public
319 	* @name ch.Validator#disable
320 	* @function
321 	* @returns itself
322 	*/
323 	that["public"].disable = function(condition){
324 		if (condition && conditions[condition]){
325 			// disable specific condition
326 			conditions[condition].disable();
327 		} else {
328 			// disable all
329 			that.enabled = false;
330 			for (condition in conditions){
331 				conditions[condition].disable();
332 			}
333 		}
334 		return that["public"];
335 	}
336 
337 /**
338 *	Default event delegation
339 */
340 	/**
341 	* Triggers when the component is ready to use.
342 	* @name ch.Validator#ready
343 	* @event
344 	* @public
345 	* @example
346 	* // Following the first example, using 'me' as modal's instance controller:
347 	* me.on("ready",function(){
348 	*	this.show();
349 	* });
350 	*/
351 	that.trigger("ready");
352 
353 	return that;
354 };
355 ch.factory("validator");
