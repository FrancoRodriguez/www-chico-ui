1 /**
  2 * Shows messages on the screen with a contextual floated UI-Component.
  3 * @name Helper
  4 * @class Helper
  5 * @augments ch.Floats
  6 * @memberOf ch
  7 * @param {object} conf Object with configuration properties
  8 * @returns itself
  9 */
 10 ch.helper = function(conf){
 11 	/**
 12 	* Reference to a internal component instance, saves all the information and configuration properties.
 13 	* @private
 14 	* @name ch.Helper#that
 15 	* @type object
 16 	*/
 17 	var that = this,
 18 
 19 		conf = {};
 20 		
 21 	conf.cone = true;
 22 	conf.cache = false;
 23 	
 24 	conf.aria = {};
 25 	conf.aria.role = "alert";
 26 
 27 	conf.position = {};
 28 	conf.position.context = that.$element;
 29 	conf.position.offset = "15 0";
 30 	conf.position.points = "lt rt";
 31 	conf.position.reposition = true;
 32 
 33 	that.conf = conf;
 34 
 35 /**
 36 *	Inheritance
 37 */
 38 
 39 	that = ch.floats.call(that);
 40 	that.parent = ch.clon(that);
 41 
 42 /**
 43 *  Private Members
 44 */
 45 
 46 
 47 /**
 48 *  Protected Members
 49 */
 50 
 51 	that.content("<span class=\"ico error\">Error: </span><p>Error.</p>");
 52 
 53 	/**
 54 	* Inner show method. Attach the component layout to the DOM tree.
 55 	* @protected
 56 	* @function
 57 	* @name ch.Helper#innerShow
 58 	* @returns itself
 59 	*/
 60 	that.innerShow = function () {
 61 
 62 		if (!that.active) {
 63 			// Load content and show!
 64 			that.parent.innerShow();
 65 		}
 66 
 67 		return that;
 68 	};
 69 
 70 /**
 71 *  Public Members
 72 */
 73 
 74 	/**
 75 	* The 'uid' is the Chico's unique instance identifier. Every instance has a different 'uid' property. You can see its value by reading the 'uid' property on any public instance.
 76 	* @public
 77 	* @name ch.Helper#uid
 78 	* @type number
 79 	*/
 80 
 81 	/**
 82 	* Reference to a DOM Element. This binding between the component and the HTMLElement, defines context where the component will be executed. Also is usual that this element triggers the component default behavior.
 83 	* @public
 84 	* @name ch.Helper#element
 85 	* @type HTMLElement
 86 	*/
 87 
 88 	/**
 89 	* This public property defines the component type. All instances are saved into a 'map', grouped by its type. You can reach for any or all of the components from a specific type with 'ch.instances'.
 90 	* @public
 91 	* @name ch.Helper#type
 92 	* @type string
 93 	*/
 94 
 95 	/**
 96 	* Sets and gets component content. To get the defined content just use the method without arguments, like 'me.content()'. To define a new content pass an argument to it, like 'me.content("new content")'. Use a valid URL to get content using AJAX. Use a CSS selector to get content from a DOM Element. Or just use a String with HTML code.
 97 	* @public
 98 	* @function
 99 	* @name ch.Helper#content
100 	* @param {String} content Static content, DOM selector or URL. If argument is empty then will return the content.
101 	* @example
102 	* // Get the defined content
103 	* me.content();
104 	* @example
105 	* // Set static content
106 	* me.content("Some static content");
107 	* @example
108 	* // Set DOM content
109 	* me.content("#hiddenContent");
110 	* @example
111 	* // Set AJAX content
112 	* me.content("http://chico.com/some/content.html");
113 	* @see ch.Object#content
114 	*/
115 	that["public"].content = function (message) {
116 
117 		if (message) {
118 			that.content("<span class=\"ico error\">Error: </span><p>" + message + "</p>");
119 		} else {
120 			return that.content();
121 		}
122 
123 		return that["public"];
124 	};
125 
126 	/**
127 	* Returns a Boolean if the component's core behavior is active. That means it will return 'true' if the component is on and it will return false otherwise.
128 	* @public
129 	* @function
130 	* @name ch.Helper#isActive
131 	* @returns boolean
132 	*/
133 
134 	/**
135 	* Triggers the innerShow method and returns the public scope to keep method chaining.
136 	* @public
137 	* @function
138 	* @name ch.Helper#show
139 	* @returns itself
140 	* @see ch.Floats#show
141 	* @example
142 	* // Following the first example, using 'me' as modal's instance controller:
143 	* me.show();
144 	*/
145 
146 	/**
147 	* Triggers the innerHide method and returns the public scope to keep method chaining.
148 	* @public
149 	* @function
150 	* @name ch.Helper#hide
151 	* @returns itself
152 	* @see ch.Floats#hide
153 	* @example
154 	* // Following the first example, using 'me' as modal's instance controller:
155 	* me.hide();
156 	*/
157 
158 	/**
159 	* Sets or gets the width property of the component's layout. Use it without arguments to get the value. To set a new value pass an argument, could be a Number or CSS value like '300' or '300px'.
160 	* @public
161 	* @function
162 	* @name ch.Helper#width
163 	* @returns itself
164 	* @see ch.Floats#size
165 	* @example
166 	* // to set the width
167 	* me.width(700);
168 	* @example
169 	* // to get the width
170 	* me.width // 700
171 	*/
172 
173 	/**
174 	* Sets or gets the height property of the component's layout. Use it without arguments to get the value. To set a new value pass an argument, could be a Number or CSS value like '100' or '100px'.
175 	* @public
176 	* @function
177 	* @name ch.Helper#height
178 	* @returns itself
179 	* @see ch.Floats#size
180 	* @example
181 	* // to set the heigth
182 	* me.height(300);
183 	* @example
184 	* // to get the heigth
185 	* me.height // 300
186 	*/
187 
188 	/**
189 	* Sets or gets positioning configuration. Use it without arguments to get actual configuration. Pass an argument to define a new positioning configuration.
190 	* @public
191 	* @function
192 	* @name ch.Helper#position
193 	* @example
194 	* // Change component's position.
195 	* me.position({ 
196 	*	offset: "0 10",
197 	*	points: "lt lb"
198 	* });
199 	* @see ch.Object#position
200 	*/
201 
202 /**
203 *  Default event delegation
204 */
205 
206 	ch.utils.body.bind(ch.events.LAYOUT.CHANGE, function () { that.position("refresh"); });
207 
208 	/**
209 	* Triggers when the component is ready to use.
210 	* @name ch.Helper#ready
211 	* @event
212 	* @public
213 	* @example
214 	* // Following the first example, using 'me' as Layer's instance controller:
215 	* me.on("ready",function () {
216 	*	this.show();
217 	* });
218 	*/
219 	that.trigger("ready");
220 
221 	return that;
222 };
223 
224 ch.factory("helper");
