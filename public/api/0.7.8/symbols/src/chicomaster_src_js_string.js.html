  5 * @interface

  9 * @returns itself

 14 * @example
 15 * // Create a string validation
 16 * $("input").string("This field must be a string.");
 17 */
 18 ch.extend("watcher").as("string", function (conf) {
 19 
 20 	conf.condition = {
 21 		name: "string",

 23 		// the ranges are A,EI,O,U,a,ei,o,u,รง,ร please for reference see http://www.fileformat.info/info/charset/UTF-8/list.htm
 24 		patt: /^([a-zA-Z\u00C0-\u00C4\u00C8-\u00CF\u00D2-\u00D6\u00D9-\u00DC\u00E0-\u00E4\u00E8-\u00EF\u00F2-\u00F6\u00E9-\u00FC\u00C7\u00E7\s]*)$/,
 25 		message: conf.msg || conf.message
 26 	};
 27 
 28 	return conf;

 32 /**
 33 * Validate email sintaxis.
 34 * @name Email
 35 * @class Email
 36 * @interface
 37 * @augments ch.Validator
 38 * @memberOf ch
 39 * @param {string} [message] Validation message.
 40 * @returns itself
 41 * @see ch.Validator
 42 * @example
 43 * // Create a email validation
 44 * $("input").email("This field must be a valid email.");
 45 */
 46 ch.extend("watcher").as("email", function (conf) {
 47 
 48 	conf.condition = {
 49 		name: "email",
 50 		patt: /^(("[\w-\s]+")|([\w-]+(?:\.[\w-]+)*)|("[\w-\s]+")([\w-]+(?:\.[\w-]+)*))(@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$)|(@\[?((25[0-5]\.|2[0-4][0-9]\.|1[0-9]{2}\.|[0-9]{1,2}\.))((25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\.){2}(25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\]?$)/,
 51 		message: conf.msg || conf.message
 52 	};
 53 
 54 	return conf;
 55 
 56 });

 85 /**
 86 * Validate a minimun amount of characters.
 87 * @name MinLength

 89 * @interface

 95 * @see ch.Validator
 96 * @example
 97 * // Create a minLength validation
 98 * $("input").minLength(10, "At least 10 characters..");

100 ch.extend("watcher").as("minLength", function (conf) {
101 
102 	conf.condition = {
103 		name: "minLength",
104 		expr: function(a,b) { return a.length >= b },

106 		value: conf.value
107 	};
108 
109 	return conf;
