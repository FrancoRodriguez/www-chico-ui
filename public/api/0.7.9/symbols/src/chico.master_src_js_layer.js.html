1 /**
  2 * Is a contextual floated UI-Object.
  3 * @name Layer
  4 * @class Layer
  5 * @augments ch.Floats
  6 * @standalone
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {String} [conf.content] Sets content by: static content, DOM selector or URL. By default, the content is empty.
 10 * @param {Number || String} [conf.width] Sets width property of the component's layout. By default, the width is "500px".
 11 * @param {Number || String} [conf.height] Sets height property of the component's layout. By default, the height is elastic.
 12 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are enable.
 13 * @param {String} [conf.event] Sets the event ("click" or "hover") that trigger show method. By default, the event is "hover".
 14 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or centered by default: "cm cm".
 15 * @param {String} [conf.offset] Sets the offset in pixels that component will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 16 * @param {Number} [conf.showTime] Sets a delay time to show component's contents. By default, the value is 400ms.
 17 * @param {Number} [conf.hideTime] Sets a delay time to hide component's contents. By default, the value is 400ms.
 18 * @param {Boolean} [conf.cache] Enable or disable the content cache. By default, the cache is enable.
 19 * @returns itself
 20 * @see ch.Tooltip
 21 * @see ch.Modal
 22 * @see ch.Zoom
 23 * @example
 24 * // Create a new contextual layer with configuration.
 25 * var me = $(".some-element").layer({
 26 *     "content": "Some content here!",
 27 *     "width": "200px",
 28 *     "height": 50,
 29 *     "event": "click",
 30 *     "showTime": 600,
 31 *     "hideTime": 200,
 32 *     "offset": "10 -10",
 33 *     "cache": false,
 34 *     "points": "lt rt"
 35 * });
 36 * @example
 37 * // Create a simple contextual layer
 38 * var me = $(".some-element").layer("<tag>Some content.</tag>");
 39 * @example
 40 * // Now 'me' is a reference to the layer instance controller.
 41 * // You can set a new content by using 'me' like this: 
 42 * me.content("http://content.com/new/content");
 43 */
 44 
 45 ch.layer = function (conf) {
 46 
 47 	/**
 48 	* Reference to a internal component instance, saves all the information and configuration properties.
 49 	* @private
 50 	* @name ch.Layer#that
 51 	* @type object
 52 	*/
 53 	var that = this;
 54 	conf = ch.clon(conf);
 55 	
 56 	conf.cone = true;
 57 	conf.closeButton = conf.event === "click";
 58 	conf.classes = "box";
 59 	
 60 	conf.aria = {};
 61 	conf.aria.role = "tooltip";
 62 	conf.aria.identifier = "aria-describedby";
 63 	
 64 	conf.position = {};
 65 	conf.position.context = that.$element;
 66 	conf.position.offset = conf.offset || "0 10";
 67 	conf.position.points = conf.points || "lt lb";
 68 
 69 	that.conf = conf;
 70 
 71 /**
 72 *	Inheritance
 73 */
 74 
 75 	that = ch.floats.call(that);
 76 	that.parent = ch.clon(that);

 88 	*/
 89 	var showTime = conf.showTime || 400,
 90 
 91 	/**
 92 	* Delay time to hide component's contents.
 93 	* @private
 94 	* @name ch.Layer#hideTime
 95 	* @type number
 96 	* @default 400
 97 	*/
 98 		hideTime = conf.hideTime || 400,
 99

102 	* @private
103 	* @name ch.Layer#st
104 	* @type timer
105 	*/
106 		st,
107

109 	* Hide timer instance.
110 	* @private
111 	* @name ch.Layer#ht
112 	* @type timer
113 	*/
114 		ht,
115 
116 	/**
117 	* Starts show timer.
118 	* @private
119 	* @function
120 	* @name ch.Layer#showTimer
121 	*/
122 		showTimer = function () { st = setTimeout(that.innerShow, showTime); },
123 
124 	/**
125 	* Starts hide timer.
126 	* @private
127 	* @function
128 	* @name ch.Layer#hideTimer
129 	*/
130 		hideTimer = function (event) {
131 			if (conf.event !== "click") {
132 				var target = event.srcElement || event.target;
133 				
134 				var relatedTarget = event.relatedTarget || event.toElement;
135 				
136 				if (target === relatedTarget || relatedTarget === undefined || relatedTarget.parentNode === null || target.nodeName === "SELECT") { return; }
137 			}
138 	
139 			ht = setTimeout(that.innerHide, hideTime);
140 		},
141 
142 	/**
143 	* Clear all timers.
144 	* @private
145 	* @function
146 	* @name ch.Layer#clearTimers
147 	*/
148 		clearTimers = function () { clearTimeout(st); clearTimeout(ht); },
149 
150 	/**
151 	* Stop event bubble propagation to avoid hiding the layer by click on his own layout.
152 	* @private
153 	* @function
154 	* @name ch.Layer#stopBubble
155 	*/
156 		stopBubble = function (event) { event.stopPropagation(); };
157 
158 	/**
159 	* Stop event bubble propagation to avoid hiding the layer by click on his own layout.
160 	* @private
161 	* @name ch.Layer#stopBubble
162 	* @function

164 /*	stopBubble = function (event) {

166 		var relatedTarget = event.relatedTarget || event.toElement;
167 		if (target === relatedTarget || relatedTarget === undefined || relatedTarget.parentNode === null || target.nodeName === "SELECT") { return; };
168 		hideTimer();

171 /**
172 *	Protected Members
173 */
174 
175 	/**
176 	* Inner show method. Attach the component layout to the DOM tree.
177 	* @protected
178 	* @function
179 	* @name ch.Layer#innerShow
180 	* @returns itself
181 	*/
182 	that.innerShow = function (event) {
183 
184 		// Reset all layers
185 		$.each(ch.instances.layer, function (i, e) { e.hide(); });
186 		
187 		// conf.position.context = that.$element;
188 		that.parent.innerShow(event);
189 
190 		// Click
191 		if (conf.event === "click") {
192 			// Document events
193 			ch.utils.document.one("click", that.innerHide);
194 			that.$container.bind("click", stopBubble);
195 		// Hover
196 		} else { 		
197 			clearTimers();
198 			that.$container.one("mouseenter", clearTimers).bind("mouseleave", hideTimer);
199 		}
200 
201 		return that;
202 	};
203 
204 	/**
205 	* Inner hide method. Hides the component and detach it from DOM tree.
206 	* @protected
207 	* @function
208 	* @name ch.Layer#innerHide
209 	* @returns itself
210 	*/
211 	that.innerHide = function (event) {
212 		that.$container.unbind("click", stopBubble).unbind("mouseleave", hideTimer);
213 
214 		that.parent.innerHide(event);
215 	}
