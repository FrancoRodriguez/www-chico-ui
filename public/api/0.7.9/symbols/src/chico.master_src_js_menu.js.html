1 /**
  2 * Menu is a UI-Component.
  3 * @name Menu
  4 * @class Menu
  5 * @augments ch.Controllers
  6 * @requires ch.Expando
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {Number} [conf.selected] Selects a child that will be open when component was loaded.
 10 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are disable.
 11 * @returns itself
 12 * @example
 13 * // Create a new menu with configuration.
 14 * var me = $(".example").menu({
 15 *     "selected": 2,
 16 *     "fx": true
 17 * });
 18 * @example
 19 * // Create a new menu without configuration.
 20 * var me = $(".example").menu();
 21 */
 22 
 23 ch.menu = function(conf){
 24 
 25 	/**
 26 	* Reference to a internal component instance, saves all the information and configuration properties.
 27 	* @private
 28 	* @name ch.Menu#that
 29 	* @type object
 30 	*/
 31 	var that = this;
 32 	
 33 	conf = ch.clon(conf);
 34 	
 35 	that.conf = conf;
 36 	
 37 /**
 38 *	Inheritance
 39 */
 40 
 41 	that = ch.controllers.call(that);
 42 	that.parent = ch.clon(that);
 43 
 44 /**
 45 *	Private Members
 46 */
 47 
 48 	/**
 49 	* Indicates witch child is opened
 50 	* @private
 51 	* @name ch.Menu#selected
 52 	* @type number
 53 	*/
 54 	var selected = conf.selected - 1;
 55 
 56 	/**
 57 	* Inits an Expando component on each list inside main HTML code snippet
 58 	* @private
 59 	* @name ch.Menu#createLayout
 60 	* @function
 61 	*/
 62 	var createLayout = function(){
 63 		
 64 		// No slide efects for IE6 and IE7
 65 		var efects = (ch.utils.html.hasClass("ie6") || ch.utils.html.hasClass("ie7")) ? false : true;
 66 		
 67 		// List elements
 68 		that.$element.children().each(function(i, e){
 69 			// List element
 70 			var $li = $(e);
 71 									  
 72 			// Children of list elements
 73 			var $child = $li.children();
 74 		
 75 			// Anchor inside list
 76 			if($child.eq(0).prop("tagName") == "A") {
 77 				
 78 				// Add attr role to match wai-aria
 79 				$li.attr("role","presentation");
 80 				
 81 				// Add class to list and anchor
 82 				$li.addClass("ch-bellows").children().addClass("ch-bellows-trigger");
 83 				
 84 				// Add anchor to that.children
 85 				that.children.push( $child[0] );
 86 				
 87 				return;
 88 			};
 89 		
 90 			// List inside list, inits an Expando
 91 			var expando = $li.expando({
 92 				// Show/hide on IE6/7 instead slideUp/slideDown
 93 				fx: efects,
 94 				onShow: function(){
 95 					// Updates selected tab when it's opened
 96 					selected = i;
 97 
 98 					/**
 99 					* Callback function
100 					* @name onSelect
101 					* @type {Function}
102 					* @memberOf ch.Menu
103 					*/
104 					that.callbacks.call(that, "onSelect");
105 					// new callback
106 					that.trigger("select");
107 				}
108 			});
109 			
110 			var childs = $li.children(),
111 				$triggerCont = $(childs[0]),
112 				$menu = $(childs[1]);
113 				if (!conf.accordion) {
114 					$menu.attr("role","menu");
115 					$menu.children().children().attr("role","menuitem");
116 				}
117 				$menu.children().attr("role","presentation");
118 				$triggerCont.attr("role","presentation");
119 			
120 			// Add expando to that.children
121 			that.children.push( expando );
122 
123 		});
124 	};
125 	
126 	/**
127 	* Opens specific Expando child and optionally grandson
128 	* @private
129 	* @function
130 	* @name ch.Menu#select
131 	*/
132 	var select = function(item){
133 
134 		var child, grandson;
135 		
136 		// Split item parameter, if it's a string with hash
137 		if (typeof item === "string") {
138 			var sliced = item.split("#");
139 			child = sliced[0] - 1;
140 			grandson = sliced[1];
141 		
142 		// Set child when item is a Number
143 		} else {
144 			child = item - 1;
145 		}
146 
147 		// Specific item of that.children list
148 		var itemObject = that.children[ child ];
149 		
150 		// Item as object
151 		if (ch.utils.hasOwn(itemObject, "uid")) {
152 			
153 			// Show this list
154 			itemObject.show();
155 			
156 			// Select grandson if splited parameter got a specific grandson
157 			if (grandson) $(itemObject.element).find("a").eq(grandson - 1).addClass("ch-menu-on");
158 			
159 			// Accordion behavior
160 			if (conf.accordion) {
161 				// Hides every that.children list that don't be this specific list item
162 				$.each(that.children, function(i, e){
163 					if(
164 						// If it isn't an anchor...
165 						(e.tagName != "A") &&
166 						// If there are an unique id...
167 						(ch.utils.hasOwn(e, "uid")) &&
168 						// If unique id is different to unique id on that.children list...
169 						(that.children[ child ].uid != that.children[i].uid)
170 					){
171 						// ...hide it
172 						e.hide();
173 					};
174 				});
175 				
176 			};
177 		
178 		// Item as anchor
179 		} else{
180 			// Just selects it
181 			that.children[ child ].addClass("ch-menu-on");
182 		};
183 
184 		return that;
185 	};
186 	
187 	/**
188 	* Binds controller's own click to expando triggers
189 	* @private
190 	* @name ch.Menu#configureAccordion
191 	* @function
192 	*/
193 	var configureAccordion = function(){
194 
195 		$.each(that.children, function(i, e){
196 			$(e.element).find(".ch-expando-trigger").unbind("click").bind("click", function(){
197 				select(i + 1);
198 			});
199 		});
200 		
201 		return;
202 	};
203 
204 /**
205 *	Protected Members
206 */
207 
208 /**
209 *	Public Members
210 */
211 	/**
212 	* The component's instance unique identifier.
213 	* @public
214 	* @name ch.Menu#uid
215 	* @type number
216 	*/	
217 	
218 	/**
219 	* The element reference.
220 	* @public
221 	* @name ch.Menu#element
222 	* @type HTMLElement
223 	*/
224 	
225 	/**
226 	* The component's type.
227 	* @public
228 	* @name ch.Menu#type
229 	* @type string
230 	*/
231 	
232 	/**
233 	* Select a specific children.
234 	* @public
235 	* @name ch.Menu#select
236 	* @function
237 	*/
238 	that["public"].select = function(item){
239 		select(item);
240 
241 		return that["public"];
242 	};
243 
244 /**
245 *	Default event delegation
246 */
247 	
248 	// Sets component main class name
249 	
250 	
251 	
252 	
253 
254 	// Inits an Expando component on each list inside main HTML code snippet
255 	createLayout();
256 
257 	// Accordion behavior
258 	if (conf.accordion) {
259 		// Sets the interface main class name for avoid
260 		configureAccordion();
261 		that.$element.addClass('ch-accordion')
262 	} else {
263 		that.$element.addClass('ch-menu');
264 		// Set the wai-aria for Menu
265 		that.$element.attr("role","navigation");
266 	}
267 
268 	// Select specific item if there are a "selected" parameter on component configuration object
269 	if (ch.utils.hasOwn(conf, "selected")) select(conf.selected);
270 
271 	return that;
272 
273 };
