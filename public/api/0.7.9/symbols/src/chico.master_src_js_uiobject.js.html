 64 	/**
 65 	* Set and get the content of a component. With no arguments will behave as a getter function. Send any kind of content and will be a setter function. Use a valid URL for AJAX content, use a CSS selector for a DOM content or just send a static content like HTML or Text.
 66 	* @name ch.Uiobject#content
 67 	* @protected
 68 	* @function
 69 	* @param {string} [content] Could be a simple text, html or a url to get the content with ajax.
 70 	* @returns {string} content
 71 	* @requires ch.Cache
 72 	* @example
 73 	* // Simple static content
 74 	* $(element).layer().content("Some static content");
 75 	* @example
 76 	* // Get DOM content
 77 	* $(element).layer().content("#hiddenContent");
 78 	* @example
 79 	* // Get AJAX content
 80 	* $(element).layer().content("http://chico.com/content/layer.html");
 81 	*/
 82 	that.content = function(content) {
 83 
 84 		var _get = (content) ? false : true,
 85 
 86 			// Local argument
 87 			content = content,
 88 			// Local isURL
 89 			sourceIsUrl = ch.utils.isUrl(that.source),
 90 			// Local isSelector
 91 			sourceIsSelector = ch.utils.isSelector(that.source),
 92 			// Local inDom
 93 			sourceInDom = (!sourceIsUrl) ? ch.utils.inDom(that.source) : false,
 94 			// Get context, could be a single component or a controller
 95 			context = ( ch.utils.hasOwn(that, "controller") ) ? that.controller : that["public"],
 96 			// undefined, for comparison.
 97 			undefined,
 98 			// Save cache configuration
 99 			cache = ( ch.utils.hasOwn(conf, "cache") ) ? conf.cache : true;
100 
101 		/**
102 		* Get content
103 		*/
104 
105 		// return defined content
106 		if (_get) {
107 
108 			// no source, no content
109 			if (that.source === undefined) {
110 				that.staticContent = "<p>No content defined for this component</p>";
111 				that.trigger("contentLoad");
112 
113 				return;
114 			}
115 
116 			// First time we need to get the content.
117 			// Is cache is off, go and get content again.
118 			// Yeap, recursive.
119 			if (!cache || that.staticContent === undefined) {
120 				that.content(that.source);
121 				return;
122 			}
123 
124 			// Get data from cache if the staticContent was defined
125 			if (cache && that.staticContent) {
126 				var fromCache = ch.cache.get(that.source);
127 
128 				// Load content from cache
129 				if (fromCache && that.staticContent != fromCache) {
130 					that.staticContent = fromCache;
131 
132 					that.trigger("contentLoad");
133 
134 					// Return and load content from cache
135 					return;
136 				}
137 
138 				// Return and show the latest content that was loaded
139 				return;
140 			}
141 		}
142 
143 		/**
144 		* Set content
145 		*/
146 
147 		// Reset cache to overwrite content
148 		conf.cache = false;
149 
150 		// Local isURL
151 		var isUrl = ch.utils.isUrl(content),
152 			// Local isSelector
153 			isSelector = ch.utils.isSelector(content),
154 			// Local inDom
155 			inDom = (!isUrl) ? ch.utils.inDom(content) : false;
156 
157 		/* Evaluate static content*/
158 
159 		// Set 'that.staticContent' and overwrite 'that.source'
160 		// just in case you want to update DOM or AJAX Content.
161 
162 		that.staticContent = that.source = content;
163 
164 		/* Evaluate AJAX content*/
165 
166 		if (isUrl) {
167 
168 			// First check Cache
169 			// Check if this source is in our cache
170 			if (cache) {
171 				var fromCache = ch.cache.get(that.source);
172 				if (fromCache) {
173 					conf.cache = cache;
174 					that.staticContent = fromCache;
175 					that.trigger("contentLoad", context);
176 					return;
177 				}
178 			}
179 
180 			var _method, _serialized, _params = "x=x";
181 
182 			// If the trigger is a form button, serialize its parent to send data to the server.
183 			if (that.$element.attr('type') == 'submit') {
184 				_method = that.$element.parents('form').attr('method') || 'GET';
185 				_serialized = that.$element.parents('form').serialize();
186 				_params = _params + ((_serialized != '') ? '&' + _serialized : '');
187 			};
188 
189 			// Set ajax config
190 			// On IE (6-7) "that" reference losts for second time
191 			// Why?? I don't know... but with a setTimeOut() works fine!
192 			setTimeout(function(){
193 
194 				$.ajax({
195 					url: that.source,
196 					type: _method || 'GET',
197 					data: _params,
198 					// each component could have a different cache configuration
199 					cache: cache,
200 					async: true,
201 					beforeSend: function(jqXHR){
202 						// Ajax default HTTP headers
203 						jqXHR.setRequestHeader("X-Requested-With", "XMLHttpRequest");
204 					},
205 					success: function(data, textStatus, jqXHR){
206 						// Save data as staticContent
207 						that.staticContent = data;
208 
209 						// Use the contentLoad callback.
210 						that.trigger("contentLoad", context);
211 
212 						// Save new staticContent to the cache
213 						if (cache) {
214 							ch.cache.set(that.source, that.staticContent);
215 						}
216 
217 					},
218 					error: function(jqXHR, textStatus, errorThrown){
219 						that.staticContent = "<p>Error on ajax call.</p>";
220 
221 						var data = {
222 							"context": context,
223 							"jqXHR": jqXHR,
224 							"textStatus": textStatus,
225 							"errorThrown": errorThrown
226 						};
227 
228 						// Use the contentError callback.
229 						that.trigger("contentError", data);
230 					}
231 				});
232 
233 			}, 0);
234 
235 			// Return Spinner and wait for async callback
236 			that.$content.html("<div class=\"loading\"></div>");
237 			that.staticContent = undefined;
238 
239 		} else {
240 
241 			/* Evaluate DOM content*/
242 
243 			if (isSelector && inDom) {
244 
245 				// Save original DOMFragment.
246 				that.originalContent = $(content);
247 
248 				// Save DOMParent, so we know where to re-insert the content.
249 				that.DOMParent = that.originalContent.parent();
250 
251 				// Save a clone to original DOM content
252 				that.staticContent = that.originalContent.clone().removeClass("ch-hide");
253 
254 			}
255 
256 			// Save new data to the cache
257 			if (cache) {
258 				ch.cache.set(that.source, that.staticContent);
259 			}
260 
261 			// First time we need to set the callbacks that append and remove the original content.
262 			if (that.originalContent && that.originalContent.selector == that.source) {
263 
264 				// Remove DOM content from DOM to avoid ID duplications
265 				that["public"].on("show", function() {
266 					that.originalContent.detach();
267 				});
268 
269 				// Returns DOMelement to DOM
270 				that["public"].on("hide", function(){
271 					that.originalContent.appendTo(that.DOMParent||"body");
272 				});
273 			}
274 		}
275 
276 		/* Set previous cache configuration*/
277 		conf.cache = cache;
278 
279 		// trigger contentLoad callback for DOM and Static content.
280 		if (that.staticContent !== undefined) {
281 			that.trigger("contentLoad", context);
282 		}
283 
284 	};

286 	/**
287 	* Change component's position configuration. If a "refresh" {string} is recived, will refresh component's positioning with the same configuration. You can send an {object} with a new configuration.
288 	* @name ch.Uiobject#position
289 	* @function
290 	* @protected
291 	* @param {string} ["refresh"] Refresh
292 	* @returns {object} Configuration object if no arguments are sended.
293 	* @see ch.Positioner
294 	*/
295 	// TODO: Add examples!!!
296 	that.position = function(o) {
297 
298 		switch(typeof o) {
299 
300 			case "object":
301 				conf.position.context = o.context || conf.position.context;
302 				conf.position.points = o.points || conf.position.points;
303 				conf.position.offset = o.offset || conf.position.offset;
304 				conf.position.fixed = o.fixed || conf.position.fixed;
305 
306 				ch.positioner(conf.position);
307 				return that["public"];
308 				break;
309 
310 			case "string":
311 				if(o != "refresh") {
312 					alert("Chico UI error: position() expected to find \"refresh\" parameter.");
313 				}
314 
315 				ch.positioner(conf.position);
316 
317 				return that["public"];
318 				break;
319 
320 			case "undefined":
321 				return conf.position;
322 				break;
323 		};
324 
325 	};

328 	* Prevent propagation and default actions.
329 	* @name ch.Uiobject#prevent
330 	* @function
331 	* @protected
332 	* @param {event} event Recieves a event object
333 	*/
334 	that.prevent = function(event) {
335 
336 		if (event && typeof event == "object") {
337 			event.preventDefault();
338 			event.stopPropagation();
339 		};
340 
341 		return that;
342 	};

390 	/**
391 	* Sets and gets component content. To get the defined content just use the method without arguments, like 'me.content()'. To define a new content pass an argument to it, like 'me.content("new content")'. Use a valid URL to get content using AJAX. Use a CSS selector to get content from a DOM Element. Or just use a String with HTML code.
392 	* @public
393 	* @name ch.Uiobject#content
394 	* @function
395 	* @param {string} content Static content, DOM selector or URL. If argument is empty then will return the content.
396 	* @example
397 	* // Get the defined content
398 	* me.content();
399 	* @example
400 	* // Set static content
401 	* me.content("Some static content");
402 	* @example
403 	* // Set DOM content
404 	* me.content("#hiddenContent");
405 	* @example
406 	* // Set AJAX content
407 	* me.content("http://chico.com/some/content.html");
408 	*/
409 	that["public"].content = function(content){
410 		if (content) { // sets
411 			// Reset content data
412 			that.source = content;
413 			that.staticContent = undefined;
414 
415 			if (that.active) {
416 				that.content(content);
417 			}
418 
419 			return that["public"];
420 
421 		} else { // gets
422 			return that.staticContent;
423 		}
424 	};
