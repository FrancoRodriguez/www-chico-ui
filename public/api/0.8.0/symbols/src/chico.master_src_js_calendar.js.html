1 /**

  3 * @name Calendar

  5 * @augments ch.Controllers
  6 * @requires ch.Dropdown
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {String} [conf.format] Sets the date format. By default is "DD/MM/YYYY".
 10 * @param {String} [conf.selected] Sets a date that should be selected by default. By default is the date of today.
 11 * @param {Array} [conf.monthsNames] By default is ["Enero", ... , "Diciembre"].
 12 * @param {Array} [conf.weekdays] By default is ["Dom", ... , "Sab"].
 13 * @returns itself
 14 * @see ch.Dropdown
 15 * @example
 16 * // Create a new calendar with configuration.
 17 * var me = $(".example").calendar({
 18 *     "format": "YYYY/MM/DD",
 19 *     "selected": "2012/12/25",
 20 *     "monthsNames": ["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"],
 21 *     "weekdays": ["Su", "Mo", "Tu", "We", "Thu", "Fr", "Sa"]
 22 * });

 24 * // Create a new calendar with a class name 'example'.
 25 * var me = $(".example").calendar();

123 	/**
124 	* HTML Template to months
125 	* @private
126 	* @name ch.Calendar#templateMonth
127 	* @type jQuery
128 	*/
129 	var templateMonth = $("<table class=\"ch-calendar-month datagrid\" role=\"grid\" id=\"ch-calendar-grid-"+that.uid+"\">" + weekdays + "</table>")
130 		.bind("click", function (event) {
131 
132 			event = event || window.event;
133 			
134 			var src = event.target || event.srcElement;
135 
136 			if (src.nodeName !== "TD" || src.className.indexOf("day")) {
137 				that.prevent(event);
138 				return;
139 			}
140 
141 			select(that.currentDate.getFullYear() + "/" + (that.currentDate.getMonth() + 1) + "/" + src.innerHTML);
142 		});

144 	/**
145 	* Creates a complete month and returns it in a table
146 	* @private
147 	* @function
148 	* @name ch.Calendar#createMonth
149 	* @return string
150 	*/
151 	var createMonth = function (date) {
152 
153 		var date = new Date(date);
154 
155 		var tableMonth = templateMonth.clone(true);
156 
157 		var currentMonth = {};
158 			currentMonth.fullDate = new Date(date.getFullYear(), date.getMonth(), 1);
159 			currentMonth.date = currentMonth.fullDate.getDate();
160 			currentMonth.day = currentMonth.fullDate.getDay();
161 			currentMonth.month = currentMonth.fullDate.getMonth();
162 			currentMonth.year = currentMonth.fullDate.getFullYear();
163 
164 		var currentDate = {};
165 			currentDate.fullDate = new Date(date.getFullYear(), date.getMonth(), 1);
166 			currentDate.date = currentDate.fullDate.getDate();
167 			currentDate.day = currentDate.fullDate.getDay();
168 			currentDate.month = currentDate.fullDate.getMonth();
169 			currentDate.year = currentDate.fullDate.getFullYear();
170 
171 		var firstWeekday = currentMonth.day;
172 
173 		var classToday,
174 			classSelected,
175 			weeks = ["<tbody>"];
176 
177 		do {
178 			
179 			weeks.push("<tr class=\"week\" role=\"row\">");
180 
181 			for (var i = 0; i < 7; i += 1) {
182 
183 				if (currentDate.date === 1) {
184 					for (var i = 0; i < firstWeekday; i += 1) {
185 						weeks.push("<td class=\"disable\"></td>");
186 					};
187 				}
188 				
189 				classToday = (currentDate.date === today.getDate() && currentDate.month === today.getMonth() && currentDate.year === today.getFullYear()) ? " today" : "";
190 
191 				classSelected = (selected && currentDate.date === selected.getDate() && currentDate.month === selected.getMonth() && currentDate.year === selected.getFullYear()) ? " selected\" aria-selected=\"true\"" : "";
192 				
193 				weeks.push("<td class=\"day" + classToday + classSelected + "\"  role=\"gridcell\">" + currentDate.date + "</td>");
194 				
195 				currentDate.fullDate.setDate(currentDate.date+1);
196 				currentDate.date = currentDate.fullDate.getDate();
197 				currentDate.day = currentDate.fullDate.getDay();
198 				currentDate.month = currentDate.fullDate.getMonth();
199 				currentDate.year = currentDate.fullDate.getFullYear();
200 
201 				if (currentDate.month != currentMonth.month) { break; }
202 
203 			};
204 
205 			weeks.push("</tr>");
206 			
207 		} while (currentDate.month === currentMonth.month);
208 
209 		weeks.push("</tbody>");
210 
211 		tableMonth
212 			.prepend("<caption>" + MONTHS_NAMES[currentMonth.month] + " - " + currentMonth.year + "</caption>")
213 			.append(weeks.join(""));
214 
215 		return tableMonth;
216 	};

287 	/**
288 	* Parse string to YY/MM/DD format date
289 	* @private
290 	* @function
291 	* @name ch.Calendar#parseDate 	
292 	*/
293 	var parseDate = function (value) {
294 		var date = value.split("/");
295 		
296 		switch (conf.format) {
297 			case "DD/MM/YYYY":
298 				return date[2] + "/" + date[1] + "/" + date[0];
299 			break;
300 			
301 			case "MM/DD/YYYY":
302 				return date[2] + "/" + date[0] + "/" + date[1];
303 			break;
304 		};
305 	};

308 	/**
309 	* Map of formart's date
310 	* @private
311 	* @name ch.Calendar#FORMAT_DATE
312 	* @type object
313 	*/
314 	var FORMAT_DATE = {
315 		
316 		"YYYY/MM/DD": function (date) {
317 			return date.getFullYear() + "/" + (parseInt(date.getMonth(), 10) + 1 < 10 ? "0" : "") + (parseInt(date.getMonth(), 10) + 1) + "/" + (parseInt(date.getDate(), 10) < 10 ? "0" : "") + date.getDate();
318 		},
319 		
320 		"DD/MM/YYYY": function (date) {
321 			return (parseInt(date.getDate(), 10) < 10 ? "0" : "") + date.getDate() + "/" + (parseInt(date.getMonth(), 10) + 1 < 10 ? "0" : "") + (parseInt(date.getMonth(), 10) + 1) + "/" + date.getFullYear();
322 		},
323 		
324 		"MM/DD/YYYY": function (date) {
325 			return (parseInt(date.getMonth(), 10) + 1 < 10 ? "0" : "") + "/" + (parseInt(date.getMonth(), 10) + 1) + "/" + date.getFullYear();
326 		}
327 	};
