222 /** 
223 * Utility to clone objects
224 * @function
225 * @name clon
226 * @param o Object to clone
227 * @returns object
228 * @memberOf ch
229 */
230 ch.clon = function(o) {
231 
232 	obj = {};
233 
234 	for (x in o) {
235 		obj[x] = o[x]; 
236 	};
237 	
238 	return obj;
239 };

242 /** 
243 * Class to create UI Components
244 * @abstract
245 * @name Factory
246 * @class Factory
247 * @param o Configuration Object
248 * @example
249 *	o {
250 *		component: "chat",
251 *		callback: function(){},
252 *		[script]: "http://..",
253 *		[style]: "http://..",
254 *		[callback]: function(){}	
255 *	}
256 * @returns collection
257 * @memberOf ch
258 */
259 
260 ch.factory = function(o) {
261 
262 	var x = o.component || o;
263 	
264 	var create = function(x) { 
265 
266 		// Send configuration to a component trough options object
267 		$.fn[x] = function( options ) {
268 
269 			var results = [];
270 			var that = this;
271 
272 			// Could be more than one argument
273 			var _arguments = arguments;
274 			
275 			that.each( function(i, e) {
276 				
277 				var conf = options || {};
278 
279 				var context = {};
280 					context.type = x;
281 					context.element = e;
282 					context.$element = $(e);
283 					context.uid = ch.utils.index += 1; // Global instantiation index
284 			
285 				switch(typeof conf) {
286 					// If argument is a number, join with the conf
287 					case "number":
288 						var num = conf;
289 						conf = {};
290 						conf.value = num;
291 						
292 						// Could come a messages as a second argument
293 						if (_arguments[1]) {
294 							conf.msg = _arguments[1];
295 						};
296 					break;
297 					
298 					// This could be a message
299 					case "string":
300 						var msg = conf;
301 						conf = {};
302 						conf.msg = msg;
303 					break;
304 					
305 					// This is a condition for custom validation
306 					case "function":
307 						var func = conf;
308 						conf = {};
309 						conf.lambda = func;
310 						
311 						// Could come a messages as a second argument
312 						if (_arguments[1]) {
313 							conf.msg = _arguments[1];
314 						};
315 					break;
316 				};
317 
318 				// Create a component from his constructor
319 				var created = ch[x].call( context, conf );
320 
321 				/*
322 					MAPPING INSTANCES
323 					Internal interface for avoid mapping objects
324 					{
325 						exists:true,
326 						object: {}
327 					}
328 				*/
329 
330 				created = ( ch.utils.hasOwn(created, "public") ) ? created["public"] : created;
331 
332 				if (created.type) {
333 					var type = created.type;
334 					// If component don't exists in the instances map create an empty array
335 					if (!ch.instances[type]) { ch.instances[type] = []; }
336 						ch.instances[type].push( created );
337 				}
338 
339 				// Avoid mapping objects that already exists
340 				if (created.exists) {
341 					// Return the inner object
342 					created = created.object;
343 				}
344 
345 				results.push( created );
346 
347 			});
348 
349 			// return the created components collection or single component
350 			return ( results.length > 1 ) ? results : results[0];
351 		};
352 
353 		// if a callback is defined 
354 		if ( o.callback ) { o.callback(); }
355 
356 	} // end create function
357 
358 	if ( ch[x] ) {
359 		// script already here, just create it
360 		create(x);
361 
362 	} else {
363 		// get resurces and call create later
364 		ch.get({
365 			"method":"component",
366 			"component": x,
367 			"script": (o.script)? o.script : "src/js/"+x+".js",
368 			"styles": (o.style)? o.style : "src/css/"+x+".css",
369 			"callback":create
370 		});
371 	}
372 }

374 /**
375 * Load components or content
376 * @abstract
377 * @name Get
378 * @class Get
379 * @param {object} o Configuration object 
380 * @example
381 *	o {
382 *		component: "chat",
383 *		[script]: "http://..",
384 *		[style]: "http://..",
385 *		[callback]: function(){}
386 *	}
387 * @memberOf ch
388 */
389 ch.get = function(o) {
390 	
391 	// ch.get: "Should I get a style?"
392 	if ( o.style ) {
393 		var style = document.createElement('link');
394 			style.href = o.style;
395 			style.rel = 'stylesheet';
396 			style.type = 'text/css';
397 	}
398 	// ch.get: "Should I get a script?"		
399 	if ( o.script ) {
400 		var script = document.createElement("script");
401 			script.src = o.script;
402 	}
403 
404 	var head = document.getElementsByTagName("head")[0] || document.documentElement;
405 	// Handle Script loading
406 	var done = false;
407 
408 	// Attach handlers for all browsers
409 	script.onload = script.onreadystatechange = function() {
410 
411 		if ( !done && (!this.readyState || 
412 			this.readyState === "loaded" || this.readyState === "complete") ) {
413 			done = true;
414 			// if callback is defined call it
415 			if ( o.callback ) { o.callback( o.component ); }
416 			// Handle memory leak in IE
417 			script.onload = script.onreadystatechange = null;
418 			if ( head && script.parentNode ) { head.removeChild( script ); }
419 		}
420 	};
421 
422 	// Use insertBefore instead of appendChild to circumvent an IE6 bug.
423 	// This arises when a base node is used.
424 	if ( o.script ) { head.insertBefore( script, head.firstChild ); }
425 	if ( o.style ) { head.appendChild( style ); }
426 
427 }

430 /**
431 * Returns a data object with features supported by the device
432 * @abstract
433 * @name Support
434 * @class Support
435 * @returns object
436 * @memberOf ch 
437 */
438 ch.support = function() {
439 	
440 	/**
441 	* Private reference to the <body> element
442 	* @private
443 	* @name thisBody
444 	* @type HTMLBodyElement
445 	* @memberOf ch.Support
446 	*/
447 	var thisBody = document.body || document.documentElement;
448 	
449 	/**
450 	* Based on: http://gist.github.com/373874
451 	* Verify that CSS3 transition is supported (or any of its browser-specific implementations)
452 	*
453 	* @private
454 	* @returns boolean
455 	* @memberOf ch.Support
456 	*/
457 	var transition = (function(){
458 		var thisStyle = thisBody.style;
459 		return thisStyle.WebkitTransition !== undefined || thisStyle.MozTransition !== undefined || thisStyle.OTransition !== undefined || thisStyle.transition !== undefined;
460 	})();
461 
462 	/**
463 	* Based on: http://kangax.github.com/cft/#IS_POSITION_FIXED_SUPPORTED
464 	* Verify that position fixed is supported
465 	* 
466 	* @private
467 	* @returns boolean
468 	* @memberOf ch.Support
469 	*/	
470 	var fixed = (function(){
471 		var isSupported = false;
472 		var e = document.createElement("div");
473 			e.style.position = "fixed";
474 			e.style.top = "10px";
475 			
476 		thisBody.appendChild(e);
477 		if (e.offsetTop === 10) { isSupported = true; };
478 		thisBody.removeChild(e);
479 		
480 		return isSupported;
481 		
482 	})();
483 
484 	return {
485 		/**
486 		* Boolean property that indicates if CSS3 Transitions are supported by the device.
487 		* @public
488 		* @name transition
489 		* @type boolean
490 		* @memberOf ch.Support
491 		*/
492 		transition: transition,
493 		/**
494 		* Boolean property that indicates if Fixed positioning are supported by the device.
495 		* @public
496 		* @name fixed
497 		* @type boolean
498 		* @memberOf ch.Support
499 		*/
500 		fixed: fixed
501 	};
502 
503 };

506 /**
507 * Extend is a utility that resolve creating interfaces problem for all UI-Objects.
508 * @abstract
509 * @name Extend
510 * @class Extend
511 * @memberOf ch
512 * @param {string} name Interface's name.
513 * @param {function} klass Class to inherit from.
514 * @param {function} [process] Optional function to pre-process configuration, recieves a 'conf' param and must return the configration object.
515 * @returns class
516 * @example
517 * // Create an URL interface type based on String component.
518 * ch.extend("string").as("url");
519 * @example
520 * // Create an Accordion interface type based on Menu component.
521 * ch.extend("menu").as("accordion"); 
522 * @example
523 * // And the coolest one...
524 * // Create an Transition interface type based on his Modal component, with some conf manipulations:
525 * ch.extend("modal").as("transition", function(conf) {
526 *	conf.closeButton = false;
527 *	conf.msg = conf.msg || conf.content || "Please wait...";
528 *	conf.content = $("<div>").addClass("loading").after( $("<p>").html(conf.msg) );
529 *	return conf;
530 * });
531 */
532 
533 ch.extend = function (klass) {
534 
535 	"use strict";
536 
537 	return {
538 	as: function (name, process) {
539 		// Create the component in Chico-UI namespace
540 		ch[name] = function (conf) {
541 			// Some interfaces need a data value,
542 			// others simply need to be 'true'.
543 			//conf[name] = conf.value || true;
544 
545 			// Invoke pre-proccess if is defined,
546 			// or grab the raw conf argument,
547 			// or just create an empty object.
548 			conf = (process) ? process(conf) : conf || {};
549 
550 			// Here we recieve messages,
551 			// or create an empty object.
552 			conf.messages = conf.messages || {};
553 
554 			// If the interface recieve a 'msg' argument,
555 			// store it in the message map.
556 			if (ch.utils.hasOwn(conf, "msg")) {
557 				conf.messages[name] = conf.msg;
558 				conf.msg = null;
559 				delete conf.msg;
560 			}
561 			// Here is where the magic happen,
562 			// invoke the class with the new conf,
563 			// and return the instance to the namespace.
564 			return ch[klass].call(this, conf);
565 		};
566 		// Almost done, now we need expose the new component,
567 		// let's ask the factory to do it for us.
568 		ch.factory(name);
569 	} // end as method
570 	} // end return
571 };
