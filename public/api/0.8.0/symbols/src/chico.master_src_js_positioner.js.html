 25 *     context: "#context2",
 26 *     points: "lt rt"

 32 *     context: "#context3",
 33 *     points: "lt rt",

422 		/**
423 		* Calculates left and top position from specific points.
424 		* @private
425 		* @name ch.Positioner#getCoordinates
426 		* @function
427 		* @param {String} points String with points to be calculated.
428 		* @returns Offset measures
429 		* @example
430 		* var foo = getCoordinates("lt rt");
431 		* 
432 		* foo = {
433 		*     left: Number,
434 		*     top: Number
435 		* };
436 		*/
437 			getCoordinates = function (pts) {
438 
439 				// Calculates left or top position from points related to specific axis (X or Y).
440 				// TODO: Complete cases: X -> lc, cl, rc, cr. Y -> tm, mt, bm, mb.
441 				var calculate = function (reference) {
442 
443 					var r;
444 
445 					switch (reference) {
446 					// X references
447 					case "ll": r = context.left; break;
448 					case "lr": r = context.right; break;
449 					case "rl": r = context.left - $element.outerWidth(); break;
450 					case "rr": r = context.right - $element.outerWidth(); break;
451 					case "cc": r = context.left + (context.width / 2) - ($element.outerWidth() / 2); break;
452 					// Y references
453 					case "tt": r = context.top; break;
454 					case "tb": r = context.bottom; break;
455 					case "bt": r = context.top - $element.outerHeight(); break;
456 					case "bb": r = context.bottom - $element.outerHeight(); break;
457 					case "mm": r = context.top + (context.height / 2) - ($element.outerHeight() / 2); break;
458 					}
459 
460 					return r;
461 				},
462 
463 				// Splitted points
464 					splittedPoints = pts.split(" ");
465 
466 				// Calculates left and top with references to X and Y axis points (crossed points)
467 				return {
468 					"left": calculate(splittedPoints[0].charAt(0) + splittedPoints[1].charAt(0)),
469 					"top": calculate(splittedPoints[0].charAt(1) + splittedPoints[1].charAt(1))
470 				};
471 			},

487 				friendly = CLASS_MAP[points];

490 				if (!conf.reposition) { return coordinates; }

492 				if (points !== "lt lb" && points !== "rt rb" && points !== "lt rt") { return coordinates; }
493 
494 				// Intelligence
495 				// TODO: Improve and unify intelligence code
496 				var newData,
497 					newPoints = points,
498 					offsetX = /*relativeParent.left + */offset[0],

507 				if (coordinates.top + offsetY + $element.outerHeight() > ch.viewport.bottom && points !== "lt rt") {

516 						friendly = newData.friendly;
517 					}
518 				}
519

521 				if (coordinates.left + offsetX + $element.outerWidth() > ch.viewport.right) {
522 					// TODO: Improve this
523 					var orientation = (newPoints.charAt(0) === "r") ? "l" : "r";

526

528 					newData.friendly = CLASS_MAP[newPoints];
529 
530 					if (newData.left + offsetX > ch.viewport.left) {
531 						coordinates.top = newData.top;
532 						coordinates.left = newData.left - (2 * offset[0]);
533 						friendly = newData.friendly;
534 					}
535 				}
536 
537 				// Returns left and top offset related to modified points
538 				return coordinates;
