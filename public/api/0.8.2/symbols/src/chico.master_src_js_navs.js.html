  2 /**
  3 * Abstract representation of navs components.
  4 * @abstract
  5 * @name Navs
  6 * @class Navs
  7 * @standalone
  8 * @augments ch.Uiobject
  9 * @memberOf ch
 10 * @param {object} conf Object with configuration properties
 11 * @returns itself
 12 * @see ch.Dropdown
 13 * @see ch.Expando
 14 */
 15 
 16 ch.navs = function () {
 17 
 18 	/**
 19 	* Reference to a internal component instance, saves all the information and configuration properties.
 20 	* @private
 21 	* @name ch.Navs#that
 22 	* @type object
 23 	*/ 
 24 	var that = this,
 25 		conf = that.conf;
 26 	
 27 	conf.icon = ch.utils.hasOwn(conf, "icon") ? conf.icon : true;
 28 	conf.open = conf.open || false;
 29 	conf.fx = conf.fx || false;
 30 
 31 /**
 32 *	Inheritance
 33 */
 34 
 35 	that = ch.uiobject.call(that);
 36 	that.parent = ch.clon(that);
 37 	
 38 /**
 39 *	Protected Members
 40 */
 41 	/**
 42 	* Status of component
 43 	* @protected
 44 	* @name ch.Navs#active
 45 	* @returns boolean
 46 	*/
 47 	that.active = false;
 48 
 49 	/**
 50 	* Shows component's content.
 51 	* @protected
 52 	* @name ch.Navs#show
 53 	* @returns itself
 54 	*/
 55 	that.show = function (event) {
 56 		that.prevent(event);
 57 
 58 		if (that.active) {
 59 			return that.hide(event);
 60 		}
 61 		
 62 		that.active = true;
 63 
 64 		that.$trigger.addClass("ch-" + that.type + "-trigger-on");
 65 		/**
 66 		* onShow callback function
 67 		* @name ch.Navs#onShow
 68 		* @event
 69 		*/
 70 		// Animation
 71 		if (conf.fx) {
 72 			that.$content.slideDown("fast", function () {
 73 				//that.$content.removeClass("ch-hide");
 74 			
 75 				// new callbacks
 76 				that.trigger("show");
 77 				// old callback system
 78 				that.callbacks("onShow");
 79 			});
 80 		} else {
 81 			that.$content.removeClass("ch-hide");
 82 			// new callbacks
 83 			that.trigger("show");
 84 			// old callback system
 85 			that.callbacks("onShow");
 86 		}
 87 		
 88 		return that;
 89 	};
 90 	/**
 91 	* Hides component's content.
 92 	* @protected
 93 	* @name ch.Navs#hide
 94 	* @returns itself
 95 	*/
 96 	that.hide = function (event) {
 97 		that.prevent(event);
 98 		
 99 		if (!that.active) { return; }
100 		
101 		that.active = false;
102 		
103 		that.$trigger.removeClass("ch-" + that.type + "-trigger-on");
104 		/**
105 		* onHide callback function
106 		* @name ch.Navs#onHide
107 		* @event
108 		*/
109 		// Animation
110 		if (conf.fx) {
111 			that.$content.slideUp("fast", function () {
112 				//that.$content.addClass("ch-hide");
113 				that.callbacks("onHide");
114 			});
115 		} else {
116 			that.$content.addClass("ch-hide");
117 			// new callbacks
118 			that.trigger("hide");
119 			// old callback system
120 			that.callbacks("onHide");
121 		}
122 		
123 		return that;
124 	};
125 
126 	/**
127 	* Create component's layout
128 	* @protected
129 	* @name ch.Navs#createLayout
130 	*/
131 	that.configBehavior = function () {
132 		that.$trigger
133 			.addClass("ch-" + that.type + "-trigger")
134 			.bind("click", function (event) { that.show(event); });
135 
136 		that.$content.addClass("ch-" + that.type + "-content ch-hide");
137 
138 		// Visual configuration
139 		if (conf.icon) { $("<span class=\"ico\">Drop</span>").appendTo(that.$trigger); }
140 		if (conf.open) { that.show(); }
141 
142 	};
143 	
144 /**
145 *	Default event delegation
146 */
147 	that.$element.addClass("ch-" + that.type);
148 
149 	/**
150 	* Triggers when component is visible.
151 	* @name ch.Navs#show
152 	* @event
153 	* @public
154 	* @example
155 	* me.on("show",function () {
156 	*	this.content("Some new content");
157 	* });
158 	* @see ch.Floats#event:show
159 	*/
160 
161 	/**
162 	* Triggers when component is not longer visible.
163 	* @name ch.Navs#hide
164 	* @event
165 	* @public
166 	* @example
167 	* me.on("hide",function () {
168 	*	otherComponent.show();
169 	* });
170 	* @see ch.Floats#event:hide
171 	*/
172 
173 	return that;
174 }
