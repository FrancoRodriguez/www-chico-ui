1 /**
  2 * Object represent the abstract class of all Objects.
  3 * @abstract
  4 * @name Object
  5 * @class Object
  6 * @memberOf ch
  7 * @see ch.Controllers
  8 * @see ch.Floats
  9 * @see ch.Navs
 10 * @see ch.Watcher
 11 */
 12 
 13 ch.object = function(){
 14 
 15 	/**
 16 	* Reference to a internal component instance, saves all the information and configuration properties.
 17 	* @private
 18 	* @name ch.Object#that
 19 	* @type object
 20 	*/
 21 	var that = this;
 22 
 23 	var conf = that.conf;
 24 
 25 /**
 26 *	Public Members
 27 */
 28 
 29 	/**
 30 	* This method will be deprecated soon. Triggers a specific callback inside component's context.
 31 	* @name ch.Object#callbacks
 32 	* @function
 33 	* @protected
 34 	*/
 35 	// TODO: Add examples!!!
 36 	that.callbacks = function(when, data) {
 37 		if( ch.utils.hasOwn(conf, when) ) {
 38 			var context = ( that.controller ) ? that.controller["public"] : that["public"];
 39 			return conf[when].call( context, data );
 40 		};
 41 	};
 42 
 43 	/**
 44 	* Triggers a specific event within the component public context.
 45 	* @name ch.Object#trigger
 46 	* @function
 47 	* @protected
 48 	* @param {string} event The event name you want to trigger.
 49 	* @since version 0.7.1
 50 	*/
 51 	that.trigger = function(event, data) {
 52 		$(that["public"]).trigger("ch-"+event, data);
 53 	};
 54 
 55 	/**
 56 	* Component's public scope. In this scope you will find all public members.
 57 	*/
 58 	that["public"] = {};
 59 
 60 	/**
 61 	* The 'uid' is the Chico's unique instance identifier. Every instance has a different 'uid' property. You can see its value by reading the 'uid' property on any public instance.
 62 	* @public
 63 	* @name ch.Object#uid
 64 	* @type number
 65 	* @ignore
 66 	*/
 67 	that["public"].uid = that.uid;
 68 
 69 	/**
 70 	* Reference to a DOM Element. This binding between the component and the HTMLElement, defines context where the component will be executed. Also is usual that this element triggers the component default behavior.
 71 	* @public
 72 	* @name ch.Object#element
 73 	* @type HTMLElement
 74 	* @ignore
 75 	*/
 76 	that["public"].element = that.element;
 77 
 78 	/**
 79 	* This public property defines the component type. All instances are saved into a 'map', grouped by its type. You can reach for any or all of the components from a specific type with 'ch.instances'.
 80 	* @public
 81 	* @name ch.Object#type
 82 	* @type string
 83 	* @ignore
 84 	*/
 85 	that["public"].type = that.type;
 86 	
 87 	/**
 88 	* Triggers a specific event within the component public context.
 89 	* @name ch.Object#trigger
 90 	* @function
 91 	* @public
 92 	* @param {string} event The event name you want to trigger.
 93 	* @since version 0.7.1
 94 	*/
 95 	that["public"].trigger = function(event, data) {
 96 		$(that["public"]).trigger("ch-"+event, data);
 97 	};
 98 	
 99 	/**
100 	* Triggers a specific event within the component public context.
101 	* @name ch.Object#trigger
102 	* @function
103 	* @protected
104 	* @param {string} event The event name you want to trigger.
105 	* @since version 0.7.1
106 	*/
107 	that["public"].trigger = that.trigger;
108 
109 	/**
110 	* Add a callback function from specific event.
111 	* @public
112 	* @function
113 	* @name ch.Object#on
114 	* @param {string} event Event name.
115 	* @param {function} handler Handler function.
116 	* @returns itself
117 	* @since version 0.7.1
118 	* @example
119 	* // Will add a event handler to the "ready" event
120 	* me.on("ready", startDoingStuff);
121 	*/
122 
123 	that["public"].on = function(event, handler) {
124 		if (event && handler) {
125 			$(that["public"]).bind("ch-"+event, handler);
126 		}
127 		return that["public"];
128 	};
129 
130 	/**
131 	* Add a callback function from specific event.
132 	* @public
133 	* @function
134 	* @name ch.Object#once
135 	* @param {string} event Event name.
136 	* @param {function} handler Handler function.
137 	* @returns itself
138 	* @since version 0.8.0
139 	* @example
140 	* // Will add a event handler to the "contentLoad" event once
141 	* me.once("contentLoad", startDoingStuff);
142 	*/
143 	that["public"].once = function(event, handler) {
144 
145 		if (event && handler) {
146 			$(that["public"]).one("ch-"+event, handler);
147 		}
148 
149 		return that["public"];
150 	};
151 
152 	/**
153 	* Removes a callback function from specific event.
154 	* @public
155 	* @function
156 	* @name ch.Object#off
157 	* @param {string} event Event name.
158 	* @param {function} handler Handler function.
159 	* @returns itself
160 	* @since version 0.7.1
161 	* @example
162 	* // Will remove event handler to the "ready" event
163 	* var startDoingStuff = function () {
164 	*     // Some code here!
165 	* };
166 	*
167 	* me.off("ready", startDoingStuff);
168 	*/
169 	that["public"].off = function(event, handler) {
170 		if (event && handler) {
171 			$(that["public"]).unbind("ch-"+event, handler);
172 		} else if (event) {
173 			$(that["public"]).unbind("ch-"+event);
174 		}
175 		return that["public"];
176 	};
177 
178 	return that;
179 };
