123 	* @private

126 	*/
127 	var parseDate = function (value) {
128 		var date = value.split("/");
129 		
130 		switch (conf.format) {
131 			case "DD/MM/YYYY":
132 				return date[2] + "/" + date[1] + "/" + date[0];
133 			break;
134 			
135 			case "MM/DD/YYYY":
136 				return date[2] + "/" + date[0] + "/" + date[1];
137 			break;
138 		};
139 	};

204 	var createMonth = function (date) {
205 
206 		var date = new Date(date);
207 
208 		var tableMonth = templateMonth.clone(true);
209 
210 		var currentMonth = {};
211 			currentMonth.fullDate = new Date(date.getFullYear(), date.getMonth(), 1);
212 			currentMonth.date = currentMonth.fullDate.getDate();
213 			currentMonth.day = currentMonth.fullDate.getDay();
214 			currentMonth.month = currentMonth.fullDate.getMonth();
215 			currentMonth.year = currentMonth.fullDate.getFullYear();
216 
217 		// Get previous month
218 		/*
219 		var prevMonth = {};
220 			prevMonth.fullDate = new Date(currentMonth.fullDate.getFullYear(), currentMonth.fullDate.getMonth() - 1, 1);
221 
222 			// Get the last day of the previous month
223 			prevMonth.fullDate.setTime(prevMonth.fullDate.getTime() + ((32 - prevMonth.fullDate.getDate()) * 86400000) );
224 			prevMonth.fullDate.setTime(prevMonth.fullDate.getTime() - (prevMonth.fullDate.getDate() * 86400000) );
225 			prevMonth.date = prevMonth.fullDate.getDate();
226 		*/
227 
228 		var currentDate = ch.clon(currentMonth);
229 
230 		var firstWeekday = currentMonth.day;
231 
232 		var classToday,
233 			classSelected,
234 			classDisabled,
235 			waiSelected,
236 			weeks = ["<tbody>"];
237 
238 		do {
239 			
240 			weeks.push("<tr class=\"week\" role=\"row\">");
241 
242 			for (var i = 0; i < 7; i += 1) {
243 
244 				if (currentDate.date === 1) {
245 					/*var last = (prevMonth.date - (firstWeekday - 1));
246 					for (var i = 0; i < firstWeekday; i += 1) {
247 						weeks.push("<td class=\"otherMonth\" role=\"gridcell\">" + (last + i) + "</td>");
248 					};*/
249 					
250 					for (i; i < firstWeekday; i += 1) {
251 						weeks.push("<td class=\"otherMonth\" role=\"gridcell\"></td>");
252 					};
253 				}
254 				
255 				classToday = (currentDate.date === today.getDate() && currentDate.month === today.getMonth() && currentDate.year === today.getFullYear()) ? " today" : "";

257 				classSelected = (selected && currentDate.date === selected.getDate() && currentDate.month === selected.getMonth() && currentDate.year === selected.getFullYear()) ? " selected" : "";
258 				
259 				classDisabled = ((from && from.getDate() > currentDate.date && from.getMonth() === currentDate.month && from.getFullYear() === currentDate.year) || (to && to.getDate() < currentDate.date && to.getMonth() === currentDate.month && to.getFullYear() === currentDate.year)) ? " disabled" : "";
260 
261 				waiSelected = (classSelected !== "") ? "aria-selected=\"true\"" : "";
262 				
263 				weeks.push("<td class=\"day" + classToday + classSelected + classDisabled +"\" " + waiSelected + "  role=\"gridcell\">" + currentDate.date + "</td>");
264 				
265 
266 				currentDate.fullDate.setDate(currentDate.date+1);
267 				currentDate.date = currentDate.fullDate.getDate();

269 				currentDate.month = currentDate.fullDate.getMonth();

271 
272 				if (currentDate.month != currentMonth.month) {
273 					for (var j = 1; j < (7-i); j += 1) {
274 						weeks.push("<td class=\"otherMonth\" role=\"gridcell\"></td>");//" + j + "
275 					};
276 					break;
277 				}
278 
279 			};

281 			weeks.push("</tr>");
282 			
283 		} while (currentDate.month === currentMonth.month);
284 
285 		weeks.push("</tbody>");
286 
287 		tableMonth
288 			.prepend("<caption>" + MONTHS_NAMES[currentMonth.month] + " - " + currentMonth.year + "</caption>")

367 	* Map of formart's date
368 	* @private
369 	* @name ch.Calendar#FORMAT_DATE
370 	* @type object
371 	*/
372 	var FORMAT_DATE = {
373 		
374 		"YYYY/MM/DD": function (date) {
375 			return date.getFullYear() + "/" + (parseInt(date.getMonth(), 10) + 1 < 10 ? "0" : "") + (parseInt(date.getMonth(), 10) + 1) + "/" + (parseInt(date.getDate(), 10) < 10 ? "0" : "") + date.getDate();
376 		},
377 		
378 		"DD/MM/YYYY": function (date) {
379 			return (parseInt(date.getDate(), 10) < 10 ? "0" : "") + date.getDate() + "/" + (parseInt(date.getMonth(), 10) + 1 < 10 ? "0" : "") + (parseInt(date.getMonth(), 10) + 1) + "/" + date.getFullYear();
380 		},
381 		
382 		"MM/DD/YYYY": function (date) {
383 			return (parseInt(date.getMonth(), 10) + 1 < 10 ? "0" : "") + "/" + (parseInt(date.getMonth(), 10) + 1) + "/" + date.getFullYear();
384 		}
385 	};
