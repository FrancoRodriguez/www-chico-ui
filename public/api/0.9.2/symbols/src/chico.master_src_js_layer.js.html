1 /**
  2 * Is a contextual floated UI-Object.
  3 * @name Layer
  4 * @class Layer
  5 * @augments ch.Floats
  6 * @standalone
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {String} [conf.content] Sets content by: static content, DOM selector or URL. By default, the content is empty.
 10 * @param {Number || String} [conf.width] Sets width property of the component's layout. By default, the width is "500px".
 11 * @param {Number || String} [conf.height] Sets height property of the component's layout. By default, the height is elastic.
 12 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are enable.
 13 * @param {String} [conf.event] Sets the event ("click" or "hover") that trigger show method. By default, the event is "hover".
 14 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or centered by default: "cm cm".
 15 * @param {String} [conf.offset] Sets the offset in pixels that component will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 16 * @param {Number} [conf.showTime] Sets a delay time to show component's contents. By default, the value is 400ms.
 17 * @param {Number} [conf.hideTime] Sets a delay time to hide component's contents. By default, the value is 400ms.
 18 * @param {Boolean} [conf.cache] Enable or disable the content cache. By default, the cache is enable.
 19 * @param {String} [conf.closeHandler] Sets the way ("any" or "button") the Layer close when conf.event is set as "click". By default, the layer close "any".
 20 * @returns itself
 21 * @see ch.Tooltip
 22 * @see ch.Modal
 23 * @see ch.Zoom
 24 * @example
 25 * // Create a new contextual layer with configuration.
 26 * var me = $(".some-element").layer({
 27 *     "content": "Some content here!",
 28 *     "width": "200px",
 29 *     "height": 50,
 30 *     "event": "click",
 31 *     "showTime": 600,
 32 *     "hideTime": 200,
 33 *     "offset": "10 -10",
 34 *     "cache": false,
 35 *     "points": "lt rt"
 36 * });
 37 * @example
 38 * // Create a simple contextual layer
 39 * var me = $(".some-element").layer("<tag>Some content.</tag>");
 40 * @example
 41 * // Now 'me' is a reference to the layer instance controller.
 42 * // You can set a new content by using 'me' like this: 
 43 * me.content("http://content.com/new/content");
 44 */
 45 
 46 ch.layer = function (conf) {
 47 
 48 	/**
 49 	* Reference to a internal component instance, saves all the information and configuration properties.
 50 	* @private
 51 	* @name ch.Layer#that
 52 	* @type object
 53 	*/
 54 	var that = this;
 55 	conf = ch.clon(conf);
 56 	
 57 	conf.cone = true;
 58 	conf.closeButton = conf.event === "click";
 59 	conf.classes = "box";
 60 	conf.closeHandler = conf.closeHandler || "any";
 61 	
 62 	conf.aria = {};
 63 	conf.aria.role = "tooltip";
 64 	conf.aria.identifier = "aria-describedby";
 65 	
 66 	conf.position = {};
 67 	conf.position.context = that.$element;
 68 	conf.position.offset = conf.offset || "0 10";
 69 	conf.position.points = conf.points || "lt lb";
 70 
 71 	that.conf = conf;
 72

 74 /**
 75 *	Inheritance
 76 */
 77 
 78 	that = ch.floats.call(that);
 79 	that.parent = ch.clon(that);

 97 	* @name ch.Layer#hideTime

100 	*/

109 		st,

121 	* @private
122 	* @function
123 	* @name ch.Layer#showTimer
124 	*/
125 		showTimer = function () { st = setTimeout(that.innerShow, showTime); },
126 
127 	/**
128 	* Starts hide timer.
129 	* @private
130 	* @function
131 	* @name ch.Layer#hideTimer
132 	*/
133 		hideTimer = function (event) {
134 			if (conf.event !== "click") {
135 				var target = event.srcElement || event.target;
136 				
137 				var relatedTarget = event.relatedTarget || event.toElement;
138 				
139 				if (target === relatedTarget || relatedTarget === undefined || relatedTarget.parentNode === null || target.nodeName === "SELECT") { return; }
140 			}
141 
142 			ht = setTimeout(that.innerHide, hideTime);
143 		},
144 
145 	/**
146 	* Clear all timers.
147 	* @private
148 	* @function
149 	* @name ch.Layer#clearTimers
150 	*/
151 		clearTimers = function () { clearTimeout(st); clearTimeout(ht); },
152

155 	* @private
156 	* @function
157 	* @name ch.Layer#stopBubble
158 	*/
159 		stopBubble = function (event) { event.stopPropagation(); };

178 	/**
179 	* It sets the hablity of auto close the component or indicate who closes the component.
180 	* @protected
181 	* @function
182 	* @name ch.Layer#closeHandler
183 	* @returns itself
184 	*/
185 	that.closeHandler = conf.closeHandler;
186 
187 	/**
188 	* Inner show method. Attach the component layout to the DOM tree.
189 	* @protected
190 	* @function
191 	* @name ch.Layer#innerShow
192 	* @returns itself
193 	*/
194 	that.innerShow = function (event) {
195 		// Reset all layers, except me and not auto closable layers
196 		$.each(ch.instances.layer, function (i, e) {
197 			if (e !== that["public"] && e.closable()==="any") {
198 				e.hide();
199 			}
200 		});
201 		
202 		// conf.position.context = that.$element;
203 		that.parent.innerShow(event);
204 
205 		// Click in the button
206 		if (conf.event === "click" && conf.close === "button") {
207 			// Document events
208 			that.$container.find(".close").one("click", that.innerHide);
209 		// Click anywhere
210 		} else if (conf.event === "click") {
211 			// Document events
212 			ch.utils.document.one("click", that.innerHide);
213 			that.$container.bind("click", stopBubble);
214 		// Hover
215 		} else { 		
216 			clearTimers();
217 			that.$container.one("mouseenter", clearTimers).bind("mouseleave", hideTimer);
218 		}
219 
220 		return that;
221 	};
222 
223 	/**
224 	* Inner hide method. Hides the component and detach it from DOM tree.
225 	* @protected
226 	* @function
227 	* @name ch.Layer#innerHide
228 	* @returns itself
229 	*/
230 	that.innerHide = function (event) {
231 		that.$container.unbind("click", stopBubble).unbind("mouseleave", hideTimer);
232 		
233 		that.parent.innerHide(event);
234 	}

295 		}
