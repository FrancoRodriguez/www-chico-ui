1 /**
  2 * Event Emitter Class for the browser.
  3 * @name EventEmitter
  4 * @class EventEmitter
  5 * @memberOf ch
  6 */
  7 ch.EventEmitter = function () {
  8 	var collection = {},
  9 		maxListeners = 10;
 10 
 11 	/**
 12 	* Adds a listener to the collection for a specified event.
 13 	* @public
 14 	* @function
 15 	* @name ch.EventEmitter#addListener
 16 	* @param {string} event Event name.
 17 	* @param {function} listener Listener function.
 18 	* @exampleDescription Adds a new listener.
 19 	* @example
 20 	* // Will add a event listener to the "ready" event
 21 	* var startDoingStuff = function () {
 22 	*     // Some code here!
 23 	* };
 24 	*
 25 	* me.addListener("ready", startDoingStuff);
 26 	* // or
 27 	* me.on("ready", startDoingStuff);
 28 	*/
 29 	this.addListener = this.on = function (event, listener) { // Event: 'newListener'
 30 		if (typeof collection[event] === "undefined") {
 31 			collection[event] = [];
 32 		}
 33 
 34 		if (collection[event].length + 1 > maxListeners) {
 35 			throw "Warning: So many listeners for an event.";
 36 		}
 37 		
 38 		collection[event].push(listener);
 39 
 40 		if (event !== "newListener") {
 41 			this.emit("newListener");
 42 		}
 43 
 44 		return this;
 45 	};
 46 
 47 	/**
 48 	* Adds a one time listener to the collection for a specified event. It will execute only once.
 49 	* @public
 50 	* @function
 51 	* @name ch.EventEmitter#once
 52 	* @param {string} event Event name.
 53 	* @param {function} listener Listener function.
 54 	* @returns itself
 55 	* @exampleDescription Adds a new listener that will execute only once.
 56 	* @example
 57 	* // Will add a event handler to the "contentLoad" event once
 58 	* me.once("contentLoad", startDoingStuff);
 59 	*/
 60 	this.once = function (event, listener) {
 61 
 62 		var fn = function (event, data) {
 63 			listener.call(this, event, data);
 64 			this.off(event.type, fn);
 65 		};
 66 
 67 		this.on(event, fn);
 68 
 69 		return this;
 70 	};
 71 
 72 	/**
 73 	* Removes a listener from the collection for a specified event.
 74 	* @public
 75 	* @function
 76 	* @name ch.EventEmitter#removeListener
 77 	* @param {string} event Event name.
 78 	* @param {function} listener Listener function.
 79 	* @returns itself
 80 	* @exampleDescription Removes a listener.
 81 	* @example
 82 	* // Will remove event handler to the "ready" event
 83 	* var startDoingStuff = function () {
 84 	*     // Some code here!
 85 	* };
 86 	*
 87 	* me.removeListener("ready", startDoingStuff);
 88 	* // or
 89 	* me.off("ready", startDoingStuff);
 90 	*/
 91 	this.removeListener = this.off = function (event, listener) {
 92 		if (collection[event] instanceof Array) {
 93 
 94 			if (listener) {
 95 				var listeners = collection[event],
 96 					j = 0,
 97 					len = listeners.length;
 98 
 99 				for (j; j < len; j += 1) {
100 					if (listeners[j] === listener) {
101 						listeners.splice(j, 1);
102 						break;
103 					}
104 				}
105 			}
106 		}
107 
108 		return this;
109 	};
110 
111 	/**
112 	* Removes all listeners from the collection for a specified event.
113 	* @public
114 	* @function
115 	* @name ch.EventEmitter#removeAllListeners
116 	* @param {string} event Event name.
117 	* @returns itself
118 	* @exampleDescription Removes all listeners.
119 	* @example
120 	* me.removeAllListeners("ready");
121 	*/
122 	this.removeAllListeners = function (event) {
123 		delete collection[event];
124 
125 		return this;
126 	};
127 
128 	/**
129 	* Increases the number of listeners. Set to zero for unlimited.
130 	* @public
131 	* @function
132 	* @name ch.EventEmitter#setMaxListeners
133 	* @param {number} n Number of max listeners.
134 	* @returns itself
135 	* @exampleDescription Increases the number of listeners.
136 	* @example
137 	* me.setMaxListeners(20);
138 	*/
139 	this.setMaxListeners = function (n) {
140 		maxListeners = n;
141 
142 		return this;
143 	};
144 
145 	/**
146 	* Returns all listeners from the collection for a specified event.
147 	* @public
148 	* @function
149 	* @name ch.EventEmitter#listeners
150 	* @param {string} event The name of the Event.
151 	* @returns Array
152 	* @exampleDescription Gets all listeners.
153 	* @example
154 	* me.listeners("ready");
155 	*/
156 	this.listeners = function (event) {
157 		return collection[event];
158 	};
159 
160 	/**
161 	* Execute each of the listener collection in order with the data object.
162 	* @name ch.EventEmitter#emit
163 	* @public
164 	* @function
165 	* @param {string} event The event name you want to emit.
166 	* @param {object} data Optionl data
167 	* @exampleDescription Emits a new custom event.
168 	* @example
169 	* // Will add a event handler to the "ready" event
170 	* me.emit("ready", {});
171 	*/
172 	this.emit = function (event, data) {
173 
174 		if (typeof event === "string") {
175 			event = { "type": event };
176 		}
177 
178 		if (!event.target) {
179 			event.target = this;
180 		}
181 
182 		if (!event.type) {
183 			throw new Error("Event object missing 'type' property.");
184 		}
185 
186 		if (collection[event.type] instanceof Array) {
187 			var listeners = collection[event.type],
188 				i = 0,
189 				len = listeners.length;
190 
191 			for (i; i < len; i += 1) {
192 				listeners[i].call(this, event, data);
193 			}
194 		}
195 
196 		return this;
197 
198 	};
199 
200 	return this;
201 };
