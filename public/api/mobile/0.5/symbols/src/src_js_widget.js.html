1 /**
  2 * Represents the abstract class of all Widgets.
  3 * @abstract
  4 * @name Widget
  5 * @class Widget
  6 * @memberOf ch
  7 * @see ch.Expando
  8 * @see ch.Menu
  9 * @see ch.Layer
 10 * @see ch.Modal
 11 */
 12 ch.Widget = function () {
 13 	/**
 14 	* Reference to a internal component instance, saves all the information and configuration properties.
 15 	* @private
 16 	* @name ch.Widget-that
 17 	* @type object
 18 	*/
 19 	var that = this;
 20 
 21 	// Use ch.EventEmitter Class
 22 	ch.EventEmitter.call(that);
 23 
 24 	/**
 25 	* Status of component
 26 	* @protected
 27 	* @name ch.Widget#active
 28 	* @type boolean
 29 	*/
 30 	that.active = false;
 31 
 32 	/**
 33 	* Component's public scope. In this scope you will find all public members.
 34 	*/
 35 	that["public"] = {};
 36 
 37 	/**
 38 	* The 'uid' is the Chico's unique instance identifier. Every instance has a different 'uid' property. You can see its value by reading the 'uid' property on any public instance.
 39 	* @public
 40 	* @name ch.Widget#uid
 41 	* @type number
 42 	*/
 43 	that["public"].uid = that.uid;
 44 
 45 	/**
 46 	* Reference to a DOM Element. This binding between the component and the HTMLElement, defines context where the component will be executed. Also is usual that this element triggers the component default behavior.
 47 	* @public
 48 	* @name ch.Widget#el
 49 	* @type HTMLElement
 50 	*/
 51 	that["public"].el = that.el;
 52 
 53 	/**
 54 	* This public property defines the component type. All instances are saved into a 'map', grouped by its type. You can reach for any or all of the components from a specific type with 'ch.instances'.
 55 	* @public
 56 	* @name ch.Widget#type
 57 	* @type string
 58 	*/
 59 	that["public"].type = that.type;
 60 	
 61 	/**
 62 	* Execute each of the listener collection in order with the data object.
 63 	* @name ch.Widget#Emit
 64 	* @public
 65 	* @param {string} event The event name you want to emit.
 66 	* @param {object} data Optionl data
 67 	* @example
 68 	* // Will add a event handler to the "ready" event
 69 	* widget.emit("ready", {});
 70 	*/
 71 	that["public"].emit = that.emit;
 72 
 73 	/**
 74 	* Adds a listener to the collection for a specified event.
 75 	* @public
 76 	* @function
 77 	* @name ch.Widget#on
 78 	* @param {string} event Event name.
 79 	* @param {function} listener Listener function.
 80 	* @example
 81 	* // Will add a event listener to the "ready" event
 82 	* var startDoingStuff = function () {
 83 	*     // Some code here!
 84 	* };
 85 	*
 86 	* widget.on("ready", startDoingStuff);
 87 	*/
 88 	that["public"].on = that.on;
 89 
 90 	/**
 91 	* Adds a one time listener to the collection for a specified event. It will execute only once.
 92 	* @public
 93 	* @function
 94 	* @name ch.Widget#once
 95 	* @param {string} event Event name.
 96 	* @param {function} listener Listener function.
 97 	* @returns itself
 98 	* @example
 99 	* // Will add a event handler to the "contentLoad" event once
100 	* widget.once("contentLoad", startDoingStuff);
101 	*/
102 	that["public"].once = that.once;
103 
104 	/**
105 	* Removes a listener from the collection for a specified event.
106 	* @public
107 	* @function
108 	* @name ch.Widget#off
109 	* @param {string} event Event name.
110 	* @param {function} listener Listener function.
111 	* @returns itself
112 	* @example
113 	* // Will remove event handler to the "ready" event
114 	* var startDoingStuff = function () {
115 	*     // Some code here!
116 	* };
117 	*
118 	* widget.off("ready", startDoingStuff);
119 	*/
120 	that["public"].off = that.off;
121 
122 	/**
123 	* Removes all listeners from the collection for a specified event.
124 	* @protected
125 	* @function
126 	* @name ch.Widget#offAll
127 	* @param {string} event Event name.
128 	* @returns itself
129 	* @example
130 	* widget.offAll("ready");
131 	*/
132 	that["public"].offAll = that.offAll;
133 
134 	/**
135 	* Increases the number of listeners. Set to zero for unlimited.
136 	* @public
137 	* @function
138 	* @name ch.Widget#setMaxListeners
139 	* @param {number} n Number of max listeners.
140 	* @returns itself
141 	* @example
142 	* widget.setMaxListeners(20);
143 	*/
144 	that["public"].setMaxListeners = that.setMaxListeners;
145 
146 	return that;
147 };
