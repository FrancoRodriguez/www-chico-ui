1 /**
  2 * Expandable lets you show or hide the content. Expandable needs a pair: the title and the content related to that title.
  3 * @name Expandable
  4 * @class Expandable
  5 * @augments ch.Widget
  6 * @memberOf ch
  7 * @param {Object} [conf] Object with configuration properties.
  8 * @param {Boolean} [conf.open] Shows the Expandable open when component was loaded. By default, the value is false.
  9 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are disable.
 10 * @returns itself
 11 * @factorized
 12 * @see ch.Widget
 13 * @exampleDescription Create a new Expandable without configuration.
 14 * @example
 15 * var widget = $(".example").expandable();
 16 * @exampleDescription Create a new Expandable with configuration.
 17 * @example
 18 * var widget = $(".example").expandable({
 19 *     "open": true
 20 * });
 21 */
 22 (function (window, $, ch) {
 23 	'use strict';
 24 
 25 	if (ch === undefined) {
 26 		throw new window.Error('Expected ch namespace defined.');
 27 	}
 28 
 29 	function Expandable($el, conf) {
 30 		/**
 31 		* Reference to a internal component instance, saves all the information and configuration properties.
 32 		* @private
 33 		* @name ch.Expandable-that
 34 		* @type object
 35 		*/
 36 		var that = this,
 37 
 38 			/**
 39 			* Reference to Parent Class.
 40 			* @private
 41 			* @name ch.Expandable-parent
 42 			* @type object
 43 			*/
 44 			parent;
 45 
 46 		that.$element = $el;
 47 		that.element = $el[0];
 48 		that.type = that.type || 'expandable';
 49 		conf = conf || {};
 50 
 51 		conf = ch.util.clone(conf);
 52 		conf.open = conf.open || false;
 53 		conf.classes = conf.classes || "";
 54 		that.conf = conf;
 55 
 56 	/**
 57 	*	Inheritance
 58 	*/
 59 		that = ch.Widget.call(that);
 60 		parent = ch.util.clone(that);
 61 
 62 	/**
 63 	*  Private Members
 64 	*/
 65 
 66 		/**
 67 		* Private reference to the Zepto element
 68 		* @privated
 69 		* @name ch.Expandable-$el
 70 		* @type Zepto Object
 71 		*/
 72 		var $el = that.$element,
 73 
 74 			/**
 75 			* Private reference to the element
 76 			* @privated
 77 			* @name ch.Expandable-el
 78 			* @type HTMLElement
 79 
 80 			*/
 81 			el = that.element,
 82 
 83 			/**
 84 			* The component's toggle.
 85 			* @privated
 86 			* @function
 87 			* @name ch.Expandable-$toggle
 88 			* @returns itself
 89 			*/
 90 			toggle = function () {
 91 				that.$trigger.toggleClass("ch-" + that["type"] + "-trigger-on");
 92 				that.$content.toggleClass("ch-hide");
 93 
 94 				// Arrows icons
 95 				/*if (conf.icon) { }*/
 96 
 97 				return that;
 98 			};
 99 
100 	/**
101 	*  Protected Members
102 	*/
103 
104 		/**
105 		* The component's trigger.
106 		* @protected
107 		* @name ch.Expandable#trigger
108 		* @type HTMLElement
109 		*/
110 		that.trigger = el.firstElementChild;
111 
112 		/**
113 		* The component's trigger.
114 		* @protected
115 		* @name ch.Expandable#$trigger
116 		* @type Zepto Object
117 		*/
118 		that.$trigger = $(that.trigger);
119 
120 		/**
121 		* The component's content.
122 		* @protected
123 		* @name ch.Expandable#content
124 		* @type HTMLElement
125 		*/
126 		that.content = el.lastElementChild;
127 
128 		/**
129 		* The component's content.
130 		* @protected
131 		* @name ch.Expandable#$content
132 		* @type Zepto Object
133 		*/
134 		that.$content = $(that.content);
135 
136 		/**
137 		* Shows component's content.
138 		* @protected
139 		* @function
140 		* @name ch.Expandable#innerShow
141 		* @returns itself
142 		*/
143 		that.innerShow = function (event) {
144 
145 			if (that.active) { return that.innerHide(event); }
146 
147 			that.active = true;
148 
149 			toggle();
150 
151 			// ARIA attr
152 			that.trigger.setAttribute("aria-expanded", "true");
153 			that.content.setAttribute("aria-hidden", "false");
154 
155 
156 			/**
157 			* Triggers when component is visible.
158 			* @name ch.Expandable#show
159 			* @event
160 			* @public
161 			* @exampleDescription It change the content when the component was shown.
162 			* @example
163 			* widget.on("show",function () {
164 			*	this.content("Some new content");
165 			* });
166 			*/
167 			that.emit("show");
168 
169 			return that;
170 		};
171 
172 		/**
173 		* Hides component's content.
174 		* @protected
175 		* @function
176 		* @name ch.Expandable#innerHide
177 		* @returns itself
178 		*/
179 		that.innerHide = function (event) {
180 
181 			if (!that.active) { return; }
182 
183 			that.active = false;
184 
185 			toggle();
186 
187 			// ARIA attr
188 			that.trigger.setAttribute("aria-expanded", "false");
189 			that.content.setAttribute("aria-hidden", "true");
190 
191 
192 			/**
193 			* Triggers when component is not longer visible.
194 			* @name ch.Expandable#hide
195 			* @event
196 			* @public
197 			* @exampleDescription When the component hides show other component.
198 			* @example
199 			* widget.on("hide",function () {
200 			*	otherComponent.show();
201 			* });
202 			*/
203 			that.emit("hide");
204 
205 			return that;
206 		};
207 
208 		/**
209 		* Create component's behaivor
210 		* @protected
211 		* @function
212 		* @name ch.Expandable#configBehavior
213 		*/
214 		that.configBehavior = function () {
215 
216 			$el.addClass("ch-" + that.type);
217 
218 			// ARIA
219 			el.setAttribute("role", "presentation");
220 
221 			that.trigger.setAttribute("aria-expanded", false);
222 			that.trigger.setAttribute("aria-controls", "ch-" + that["type"] + "-" + that.uid);
223 
224 			that.content.setAttribute("id", "ch-" + that["type"] + "-" + that.uid);
225 			that.content.setAttribute("aria-hidden", true);
226 
227 			// Trigger behaivor
228 			// ClassNames
229 			that.$trigger.addClass("ch-" + that.type + "-trigger");
230 
231 
232 			/*if (conf.icon) { }*/
233 
234 			// Events
235 			that.$trigger.bind(ch.events.TAP, function (event) { event.preventDefault(); that.innerShow(event); });
236 
237 			// Content behaivor
238 
239 			// ClassNames
240 			that.$content.addClass("ch-" + that.type + "-content ch-hide " + conf.classes);
241 
242 			// Visual configuration
243 			if (conf.open) { that.innerShow(); }
244 
245 		};
246 
247 
248 	/**
249 	*  Public Members
250 	*/
251 
252 		/**
253 		* @borrows ch.Widget#uid as ch.Expandable#uid
254 		*/
255

257 		* @borrows ch.Widget#el as ch.Expandable#el
258 		*/
259

261 		* @borrows ch.Widget#type as ch.Expandable#type
262 		*/
263

267

270 		*/
271 
272 		/**
273 		* @borrows ch.Widget#once as ch.Expandable#once
274 		*/
275 
276 		/**
277 		* @borrows ch.Widget#off as ch.Expandable#off
278 		*/
279 
280 		/**
281 		* @borrows ch.Widget#offAll as ch.Expandable#offAll
282 		*/
283 
284 		/**
285 		* @borrows ch.Widget#setMaxListeners as ch.Expandable#setMaxListeners
286 		*/
287 
288 
289 		/**
290 		* Shows component's content.
291 		* @public
292 		* @function
293 		* @name ch.Expandable#show
294 		* @returns itself
295 		*/
296 		that["public"].show = function(){
297 			that.innerShow();
298 			return that["public"];
299 		};
300 
301 		/**
302 		* Hides component's content.
303 		* @public
304 		* @function
305 		* @name ch.Expandable#hide
306 		* @returns itself
307 		*/
308 		that["public"].hide = function(){
309 			that.innerHide();
310 			return that["public"];
311 		};
312 
313 	/**
314 	*  Default behaivor
315 	*/
316 
317 		that.configBehavior();
318 
319 		/**
320 		* Emits an event when the component is ready to use.
321 		* @name ch.Expandable#ready
322 		* @event
323 		* @public
324 		* @example
325 		* // Following the first example, using 'me' as Expandable's instance controller:
326 		* widget.on("ready",function () {
327 		*	this.show();
328 		* });
329 		*/
330 		window.setTimeout(function(){ that.emit("ready")}, 50);
331 
332 		return that['public'];
333 	}
