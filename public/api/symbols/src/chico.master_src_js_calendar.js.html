 98 	/**
 99 	* HTML Template to months
100 	* @private
101 	* @name ch.Calendar#templateMonth
102 	* @type jQuery
103 	*/
104 	var templateMonth = $("<table class=\"datagrid\">")
105 		.addClass("ch-calendar-month")
106 		.append(weekdays)
107 		.bind("click", function(event){
108 
109 			event = event || window.event;
110 			src = event.target || event.srcElement;
111 
112 			if (src.nodeName !== "TD" || src.className.indexOf("day")) {
113 				that.prevent(event);

115 			};

117 			select( that.currentDate.getFullYear() + "/" + (that.currentDate.getMonth() + 1) + "/" + src.innerHTML );
118 		});

121 	/**
122 	* Creates a complete month and returns it in a table
123 	* @private
124 	* @function
125 	* @name ch.Calendar#createMonth
126 	* @return string
127 	*/
128 	var createMonth = function(date){
129 
130 		var date = new Date(date);
131 
132 		var tableMonth = templateMonth.clone(true);
133 
134 		var currentMonth = {};
135 			currentMonth.fullDate = new Date(date.getFullYear(), date.getMonth(), 1);
136 			currentMonth.date = currentMonth.fullDate.getDate();
137 			currentMonth.day = currentMonth.fullDate.getDay();
138 			currentMonth.month = currentMonth.fullDate.getMonth();
139 			currentMonth.year = currentMonth.fullDate.getFullYear();
140 
141 
142 		var currentDate = {};
143 			currentDate.fullDate = new Date(date.getFullYear(), date.getMonth(), 1);
144 			currentDate.date = currentDate.fullDate.getDate();
145 			currentDate.day = currentDate.fullDate.getDay();
146 			currentDate.month = currentDate.fullDate.getMonth();
147 			currentDate.year = currentDate.fullDate.getFullYear();
148 
149 		var firstWeekday = currentMonth.day;
150 
151 		var weeks, classToday, classSelected;
152 
153 		weeks = "<tbody>";
154 
155 		do {
156 			
157 			weeks += "<tr class=\"week\">";
158 
159 			for (var i = 0; i < 7; i += 1) {
160 
161 				if (currentDate.date == 1) {
162 					for (var i = 0; i < firstWeekday; i += 1) {
163 						weeks += "<td class=\"disable\"></td>";
164 					};
165 				};
166 				
167 				classToday = (currentDate.date == today.getDate() && currentDate.month == today.getMonth() && currentDate.year == today.getFullYear()) ? " today" : "";
168 
169 				classSelected = (selected && currentDate.date == selected.getDate() && currentDate.month == selected.getMonth() && currentDate.year == selected.getFullYear()) ? " selected" : "";
170 				
171 				weeks += "<td class=\"day" + classToday +  classSelected + "\">" + currentDate.date + "</td>";
172 				
173 				currentDate.fullDate.setDate(currentDate.date+1);
174 				currentDate.date = currentDate.fullDate.getDate();
175 				currentDate.day = currentDate.fullDate.getDay();
176 				currentDate.month = currentDate.fullDate.getMonth();
177 				currentDate.year = currentDate.fullDate.getFullYear();
178 
179 				if ( currentDate.month != currentMonth.month ) { break; };
180 
181 			};
182 
183 			weeks += "</tr>";
184 			
185 		} while (currentDate.month == currentMonth.month);
186 
187 		weeks += "</tbody>";
188 
189 		tableMonth
190 			.prepend("<caption>"+MONTHS_NAMES[currentMonth.month] + " - " + currentMonth.year+"</caption>")
191 			.append(weeks);
192 
193 		return tableMonth;
194 	};

266 	/**
267 	* Parse string to YY/MM/DD format date
268 	* @private
269 	* @function
270 	* @name ch.Calendar#parseDate 	
271 	*/
272 	var parseDate = function(value){
273 		var date = value.split("/");
274 		
275 		switch (conf.format) {
276 			case "DD/MM/YYYY":
277 				return date[2] + "/" + date[1] + "/" + date[0];
278 			break;
279 			
280 			case "MM/DD/YYYY":
281 				return date[2] + "/" + date[0] + "/" + date[1];
282 			break;
283 		};
284 	};

287 	/**
288 	* Map of formart's date
289 	* @private
290 	* @name ch.Calendar#FORMAT_DATE
291 	* @type object
292 	*/
293 	var FORMAT_DATE = {
294 		"YYYY/MM/DD": function(date){ return  date.getFullYear() + "/" + (parseInt(date.getMonth(), 10) + 1 < 10 ? '0' : '') + (parseInt(date.getMonth(), 10) + 1) + "/" + (parseInt(date.getDate(), 10) < 10 ? '0' : '') + date.getDate(); },
295 		"DD/MM/YYYY": function(date){ return (parseInt(date.getDate(), 10) < 10 ? '0' : '') + date.getDate() + "/" + (parseInt(date.getMonth(), 10) + 1 < 10 ? '0' : '') + (parseInt(date.getMonth(), 10) + 1) + "/" + date.getFullYear()},
296 		"MM/DD/YYYY": function(date){ return (parseInt(date.getMonth(), 10) + 1 < 10 ? '0' : '') + "/" + (parseInt(date.getMonth(), 10) + 1) + "/" + date.getFullYear()}
297 	};
