  2 /**
  3 * A navegable list of items, UI-Object.
  4 * @name Dropdown
  5 * @class Dropdown
  6 * @augments ch.Navs
  7 * @memberOf ch
  8 * @param {object} conf Object with configuration properties
  9 * @returns itself
 10 */
 11 
 12 ch.dropdown = function(conf){
 13 
 14 
 15 	/**
 16 	* Reference to a internal component instance, saves all the information and configuration properties.
 17 	* @private
 18 	* @name ch.Dropdown#that
 19 	* @type object
 20 	*/
 21 	var that = this;
 22 
 23 	conf = ch.clon(conf);
 24 	that.conf = conf;
 25 	
 26 /**
 27 *	Inheritance
 28 */
 29 
 30 	that = ch.navs.call(that);
 31 	that.parent = ch.clon(that);
 32 
 33 /**
 34 *  Private Members
 35 */
 36 	/**
 37 	* Adds keyboard events.
 38 	* @private
 39 	* @function
 40 	* @name ch.Dropdown#shortcuts
 41 	*/
 42 	var shortcuts = function(items){
 43 		
 44 		// Keyboard support
 45 		var selected = 0;
 46 		
 47 		// Item selected by mouseover
 48 		// TODO: It's over keyboard selection and it is generating double selection.
 49 		$.each(items, function(i, e){
 50 			$(e).bind("mouseenter", function(){
 51 				selected = i;
 52 				items.eq( selected ).focus();
 53 			});
 54 		});
 55 		
 56 		var selectItem = function(arrow, event){
 57 			that.prevent(event);
 58 			
 59 			if(selected == ((arrow == "bottom") ? items.length - 1 : 0)) return;
 60 			
 61 			items.eq( selected ).blur();
 62 			
 63 			if(arrow == "bottom") selected += 1; else selected -= 1;
 64 			
 65 			items.eq( selected ).focus();
 66 		};
 67 		
 68 		// Arrows
 69 		ch.utils.document.bind(ch.events.KEY.UP_ARROW, function(x, event){ selectItem("up", event); });
 70 		ch.utils.document.bind(ch.events.KEY.DOWN_ARROW, function(x, event){ selectItem("bottom", event); });
 71 	};
 72 
 73 
 74 /**
 75 *  Protected Members
 76 */
 77 	/**
 78 	* The component's trigger.
 79 	* @private
 80 	* @name ch.Dropdown#$trigger
 81 	* @type jQuery
 82 	*/
 83 	that.$trigger = that.$element.children().eq(0);
 84 	/**
 85 	* The component's content.
 86 	* @private
 87 	* @name ch.Dropdown#$content
 88 	* @type jQuery
 89 	*/
 90 	that.$content = that.$trigger.next().detach(); // Save on memory;
 91 
 92 	that.show = function(event){
 93 		that.prevent(event);
 94 
 95 		that.$content.css('z-index', ch.utils.zIndex ++);
 96 		
 97 		if (that.$element.hasClass("secondary")) { // Z-index of trigger over conten 
 98 			that.$trigger.css('z-index', ch.utils.zIndex ++);
 99 		}; 
100 
101 		
102 		that.$element
103 			.addClass("ch-dropdown-on")
104 			.css('z-index', ch.utils.zIndex ++);
105 
106 		that.parent.show(event);
107 		that.position("refresh");
108 
109 		// Reset all dropdowns
110 		$.each(ch.instances.dropdown, function(i, e){ 
111 			if (e.uid !== that.uid) e.hide();
112 		});
113 
114 		// Close events
115 		ch.utils.document.one("click " + ch.events.KEY.ESC, function(event){ that.hide(event); });
116 		// Close dropdown after click an option (link)
117 		that.$content.find("a").one("click", function(){ that.hide(); });
118 
119 		// Keyboard support
120 		var items = that.$content.find("a");
121 			items.eq(0).focus(); // Select first anchor child by default
122 
123 		if (items.length > 1){ shortcuts(items); };
124 
125 		return that;
126 	};
127 
128 	that.hide = function(event){
129 		that.prevent(event);
130 
131 		that.parent.hide(event);
132 			that.$element.removeClass("ch-dropdown-on");
133 
134 		// Unbind events
135 		ch.utils.document.unbind(ch.events.KEY.ESC + " " + ch.events.KEY.UP_ARROW + " " + ch.events.KEY.DOWN_ARROW);
136 
137 		return that;
138 	};
139 	
140 /**
141 *  Public Members
142 */
143  
144 	/**
145 	* The component's instance unique identifier.
146 	* @public
147 	* @name ch.Dropdown#uid
148 	* @type number
149 	*/
150 	
151 	/**
152 	* The element reference.
153 	* @public
154 	* @name ch.Dropdown#element
155 	* @type HTMLElement
156 	*/
157 	
158 	/**
159 	* The component's type.
160 	* @public
161 	* @name ch.Dropdown#type
162 	* @type string
163 	*/	
164 	
165 	/**
166 	* Shows component's content.
167 	* @public
168 	* @function
169 	* @name ch.Dropdown#show
170 	* @returns itself
171 	*/
172 	that["public"].show = function(){
173 		that.show();
174 		
175 		return that["public"];
176 	};
177 
178 	/**
179 	* Hides component's content.
180 	* @public
181 	* @function
182 	* @name ch.Dropdown#hide
183 	* @returns itself
184 	*/ 
185 	that["public"].hide = function(){
186 		that.hide();
187 		
188 		return that["public"];
189 	};
190 	/**
191 	* Positioning configuration.
192 	* @public
193 	* @function
194 	* @name ch.Dropdown#position
195 	*/
196 
197 
198 /**
199 *  Default event delegation
200 */			
201 
202 	that.configBehavior();
203 	
204 	that.$element.after( that.$content ); // Put content out of element
205 	ch.utils.avoidTextSelection(that.$trigger);
206 	
207 	if (that.$element.hasClass("secondary")) that.$content.addClass("secondary");
208 	
209 	// Prevent click on content (except links)
210 	that.$content.bind("click", function(event){ event.stopPropagation(); });
211 	
212 	// Position
213 	that.conf.position = {};
214 	that.conf.position.element = that.$content;
215 	that.conf.position.context = that.$trigger;
216 	that.conf.position.points = conf.points || "lt lb";
217 	that.conf.position.offset = "0 -1";
218 
219 	return that;
220 
221 };
222 
223 ch.factory("dropdown");
