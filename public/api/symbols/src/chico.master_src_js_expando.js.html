  2 /**
  3 * Expando is a UI-Component.
  4 * @name Expando
  5 * @class Expando
  6 * @augments ch.Navs
  7 * @memberOf ch
  8 * @param {object} conf Object with configuration properties
  9 * @returns itself
 10 */
 11  
 12 ch.expando = function(conf){
 13 
 14 	/**
 15 	* Reference to a internal component instance, saves all the information and configuration properties.
 16 	* @private
 17 	* @name ch.Expando#that
 18 	* @type object
 19 	*/
 20 	var that = this;
 21 		
 22 	conf = ch.clon(conf);
 23 	that.conf = conf;
 24 	
 25 /**
 26 *	Inheritance
 27 */
 28 
 29 	that = ch.navs.call(that);
 30 	that.parent = ch.clon(that);
 31 
 32 /**
 33 *  Protected Members
 34 */ 
 35 
 36 	that.$trigger = that.$element.children().eq(0).wrapInner("<span>").children();
 37 
 38 	that.$content = that.$element.children().eq(1);
 39 
 40 /**
 41 *  Public Members
 42 */
 43  
 44 	/**
 45 	* The component's instance unique identifier.
 46 	* @public
 47 	* @name ch.Expando#uid
 48 	* @type number
 49 	*/
 50 	
 51 	/**
 52 	* The element reference.
 53 	* @public
 54 	* @name ch.Expando#element
 55 	* @type HTMLElement
 56 	*/
 57 	
 58 	/**
 59 	* The component's type.
 60 	* @public
 61 	* @name ch.Expando#type
 62 	* @type string
 63 	*/
 64 	
 65 	/**
 66 	* Shows component's content.
 67 	* @public
 68 	* @function
 69 	* @name ch.Expando#show
 70 	* @returns itself
 71 	*/
 72 	that["public"].show = function(){
 73 		that.show();
 74 		
 75 		return that["public"];
 76 	};
 77 
 78 	/**
 79 	* Hides component's content.
 80 	* @public
 81 	* @function
 82 	* @name ch.Expando#hide
 83 	* @returns itself
 84 	*/	
 85 	that["public"].hide = function(){
 86 		that.hide();
 87 		
 88 		return that["public"];
 89 	};
 90 	
 91 
 92 /**
 93 *  Default event delegation
 94 */		
 95 	
 96 	that.configBehavior();
 97 	ch.utils.avoidTextSelection(that.$trigger);
 98 
 99 	return that;
100 
101 };
102 
103 ch.factory("expando");
