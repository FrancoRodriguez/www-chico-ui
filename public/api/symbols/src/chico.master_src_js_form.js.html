  2 /**
  3 * Forms is a Controller of DOM's HTMLFormElement.
  4 * @name Form
  5 * @class Form
  6 * @augments ch.Controllers
  7 * @memberOf ch
  8 * @param {object} conf Object with configuration properties
  9 * @returns itself
 10 */
 11 
 12 ch.form = function(conf) {
 13 
 14 /**
 15 * Validation
 16 */
 17 	// Are there action and submit type?
 18 	if ( this.$element.find(":submit").length == 0 || this.$element.attr("action") == "" ){ 
 19 		alert("Form fatal error: The <input type=submit> is missing, or need to define a action attribute on the form tag.");
 20 		return;
 21 	};
 22 
 23 	// Is there form in map instances?	
 24 	if ( ch.utils.hasOwn(ch.instances, "form") && ch.instances.form.length > 0 ){
 25 		for(var i = 0, j = ch.instances.form.length; i < j; i++){
 26 			if(ch.instances.form[i].element === this.element){
 27 				return { 
 28 					exists: true, 
 29 					object: ch.instances.form[i]
 30 				};
 31 			};
 32 		};
 33 	};
 34 
 35 	/**
 36 	* Reference to a internal component instance, saves all the information and configuration properties.
 37 	* @private
 38 	* @name ch.Form#that
 39 	* @type object
 40 	*/ 
 41 	var that = this;
 42 	
 43 	conf = ch.clon(conf);
 44 	// Disable HTML5 browser-native validations
 45 	that.$element.attr("novalidate", "novalidate");
 46 	// Grab submit button
 47 	that.$submit = that.$element.find("input:submit");
 48 	
 49 	that.conf = conf;
 50 
 51 /**
 52 *  Inheritance
 53 */
 54 
 55 	that = ch.controllers.call(that);
 56 	that.parent = ch.clon(that);
 57 	
 58 	
 59 /**
 60 *  Private Members
 61 */
 62 
 63 	/**
 64 	* A Boolean property that indicates is exists errors in the form.
 65 	* @private
 66 	* @name ch.Form#status
 67 	* @type boolean
 68 	*/ 
 69 	var status = true;
 70 	
 71 	/**
 72 	* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
 73 	*/
 74 	var validate = function(){
 75 
 76 		/**
 77 		* Callback function
 78 		* @name ch.Form#beforeValidate
 79 		* @event
 80 		* @public
 81 		*/
 82 		that.callbacks("beforeValidate");
 83 		// new callback
 84 		that.trigger("beforeValidate");
 85 		
 86 		// Status OK (with previous error)
 87 		if ( !status ) {
 88 			status = true;
 89 		};
 90 		
 91 		var i = 0, j = that.children.length, toFocus, childrenError = [];
 92 		// Shoot validations
 93 		for ( i; i < j; i++ ) {
 94 			var child = that.children[i];
 95 			 // Validate
 96 			child.validate();
 97 			// Save children with errors
 98 			if ( child.active() ) {
 99 				childrenError.push( child );
100 			}
101 		};
102 		
103 		// Is there's an error
104 		if ( childrenError.length > 0 ) {
105 			status = false;
106 			// Issue UI-332: On validation must focus the first field with errors.
107 			// Doc: http://wiki.ml.com/display/ux/Mensajes+de+error
108 			if (childrenError[0].element.tagName === "DIV") {
109 				$(childrenError[0].element).find("input:first").focus();
110 			} else {
111 				childrenError[0].element.focus();
112 			}
113 		} else {
114 			status = true;	
115 		}
116 		/**
117 		* Callback function
118 		* @name ch.Form#onValidate
119 		* @event
120 		* @public
121 		*/
122 		/**
123 		* Callback function
124 		* @name ch.Form#onError
125 		* @event
126 		* @public
127 		*/
128 		//status ? that.callbacks("onValidate") : that.callbacks("onError");  
129 		if (status) {
130 			that.callbacks("onValidate");
131 			// new callback
132 			that.trigger("onValidate");	
133 		} else {
134 			that.callbacks("onError");
135 			// new callback
136 			that.trigger("onError");
137 		}
138 
139 		/**
140 		* Callback function
141 		* @name ch.Form#afterValidate
142 		* @event
143 		* @public
144 		*/
145 		that.callbacks("afterValidate");
146 		// new callback
147 		that.trigger("afterValidate");
148 		
149 		return that;
150 	};
151 
152 	/**
153 	* This methods triggers the 'beforSubmit' callback, then will execute validate() method, 
154 	* and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
155 	*/
156 	var submit = function(event) {
157 
158 		/**
159 		* Callback function
160 		* @name ch.Form#beforeSubmit
161 		* @event
162 		* @public
163 		*/
164 		that.callbacks("beforeSubmit");
165 		// new callback
166 		that.trigger("beforeSubmit");
167 		
168 		// re-asign submit event   
169 		that.$element.one("submit", submit);
170 
171 		// Execute all validations
172 		validate();
173 		
174 		// If an error ocurs prevent default actions
175 		if ( !status ) {
176 			that.prevent(event);
177 		}
178 
179 		/**
180 		* Callback function
181 		* @name ch.Form#onSubmit
182 		* @event
183 		* @public
184 		*/
185 
186 		// Is there's no error but there's a onSubmit callback
187 		if ( status && ch.utils.hasOwn(conf, "onSubmit")) {
188 			// Avoid default actions
189 			that.prevent(event);
190 			// To execute defined onSubmit callback
191 			that.callbacks("onSubmit");
192 			// new callback
193 			that.trigger("submit");
194 		}
195 
196 		/**
197 		* Callback function
198 		* @name ch.Form#afterSubmit
199 		* @event
200 		* @public
201 		*/
202 
203 		that.callbacks("afterSubmit");
204 		// new callback
205 		that.trigger("afterSubmit");
206 			
207 		// Return that to chain methods
208 		return that;
209 	};
210 
211 	/**
212 	* Use this method to clear al validations.
213 	*/
214 	var clear = function(event){		
215 		
216 		that.prevent(event);		
217 		
218 		var i = 0, j = that.children.length;
219 		for(i; i < j; i += 1) {
220 		  that.children[i].reset();
221 		}
222 
223 		/**
224 		* Callback function
225 		* @name ch.Form#onClear
226 		* @event
227 		* @public
228 		*/
229 		that.callbacks("onClear");
230 		// new callback
231 		that.trigger("clear");
232 
233 		return that;
234 	};
235 
236 	/**
237 	* Use this method to reset the form's input elements.
238 	*/	
239 	var reset = function(event){
240 		clear();
241 		that.element.reset(); // Reset html form native
242 		/**
243 		* Callback function
244 		* @name ch.Form#onReset
245 		* @event
246 		* @public
247 		*/
248 		that.callbacks("onReset");
249 		// new callback
250 		that.trigger("reset");
251 		
252 		return that;
253 	};
254

256 /**
257 *  Public Members
258 */	
259 	/**
260 	* The component's instance unique identifier.
261 	* @public
262 	* @name ch.Form#uid
263 	* @type number
264 	*/ 
265 
266 	/**
267 	* The element reference.
268 	* @public
269 	* @name ch.Form#element
270 	* @type HTMLElement
271 	*/
272 
273 	/**
274 	* The component's type.
275 	* @public
276 	* @name ch.Form#type
277 	* @type string
278 	*/
279 
280 	/**
281 	* Watcher instances associated to this controller.
282 	* @public
283 	* @name ch.Form#children
284 	* @type collection
285 	*/
286 	that["public"].children = that.children;
287 	
288 	/**
289 	* Collection of messages defined.
290 	* @public
291 	* @name ch.Form#messages
292 	* @type string
293 	*/
294 	that["public"].messages = conf.messages || {};
295 	
296 	/**
297 	* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
298 	* @public
299 	* @function
300 	* @name ch.Form#validate
301 	* @returns itself
302 	*/
303 	that["public"].validate = function() { 
304 		validate(); 
305 		
306 		return that["public"]; 
307 	};
308 	
309 	/**
310 	* This methods triggers the 'beforSubmit' callback, then will execute validate() method, and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
311 	* @public
312 	* @function
313 	* @name ch.Form#submit
314 	* @returns itself
315 	*/
316 	that["public"].submit = function() { 
317 		submit(); 
318 		
319 		return that["public"]; 
320 	};
321 
322 	/**
323 	* Return the status value.
324 	* @public
325 	* @function
326 	* @name ch.Form#getStatus
327 	* @returns itself
328 	*/	
329 	that["public"].getStatus = function(){
330 		return status;	
331 	};
332 
333 	/**
334 	* Use this method to clear al validations.
335 	* @public
336 	* @function
337 	* @name ch.Form#clear
338 	* @returns itself
339 	*/ 
340 	that["public"].clear = function() { 
341 		clear(); 
342 		
343 		return that["public"]; 
344 	};
345 	/**
346 	* Use this method to clear al validations.
347 	* @public
348 	* @function
349 	* @name ch.Form#reset
350 	* @returns itself
351 	*/ 
352 	that["public"].reset = function() { 
353 		reset(); 
354 		
355 		return that["public"]; 
356 	};
357 
358 
359 /**
360 *  Default event delegation
361 */	
362 
363 	// patch exists because the components need a trigger
364 	if (ch.utils.hasOwn(conf, "onSubmit")) {
365 		that.$element.bind('submit', function(event){ that.prevent(event); });
366 		// Delete all click handlers asociated to submit button >NATAN: Why?
367 			//Because if you want do something on submit, you need that the trigger (submit button) 
368 			//don't have events associates. You can add funcionality on onSubmit callback
369 		that.$element.find(":submit").unbind('click');
370 	};
371 
372 	// Bind the submit
373 	that.$element.one("submit", submit);
374 	
375 	// Bind the reset
376 	that.$element.find(":reset, .resetForm").bind("click", function(event){ reset(event); });
377 
378 	return that;
379 };
