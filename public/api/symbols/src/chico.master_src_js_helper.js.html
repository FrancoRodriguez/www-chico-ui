  2 /**
  3 * Shows messages on the screen with a contextual floated UI-Component.
  4 * @name Helper
  5 * @class Helper
  6 * @augments ch.Floats
  7 * @memberOf ch
  8 * @param {object} conf Object with configuration properties
  9 * @returns itself
 10 */
 11 
 12 ch.helper = function(controller){
 13 
 14 	/**
 15 	* Reference to a internal component instance, saves all the information and configuration properties.
 16 	* @private
 17 	* @name ch.Helper#that
 18 	* @type object
 19 	*/
 20 	var that = this;
 21 
 22 	var conf = {};		
 23 		conf.cone = true;
 24 		conf.position = {};
 25 		conf.position.context = controller.reference;
 26 		conf.position.offset = "15 0";
 27 		conf.position.points = "lt rt";
 28 		conf.cache = false;
 29 	
 30 	that.conf = conf;
 31 
 32 /**
 33 *	Inheritance
 34 */
 35 
 36 	that = ch.floats.call(that);
 37 	that.parent = ch.clon(that);
 38 
 39 /**
 40 *  Private Members
 41 */
 42 
 43 
 44 
 45 /**
 46 *  Protected Members
 47 */ 
 48  
 49 	that.content("I'm not sure what just happened, this field might have some problems. Can you take a look?")
 50 
 51  	that.$trigger = that.$element;
 52 
 53 	/**
 54 	* Inner show method. Attach the component layout to the DOM tree.
 55 	* @protected
 56 	* @function
 57 	* @name ch.Helper#innerShow
 58 	* @returns itself
 59 	*/ 
 60 		
 61 	that.innerShow = function() {
 62 
 63 		if ( !that.active ) {
 64 			// Load content and show!
 65 			that.parent.innerShow();
 66 		};
 67 
 68 		// Just Reload content!
 69 		that.$content.html("<span class=\"ico error\">Error: </span><p>" + that.content() + "</p>");
 70 
 71 		return that;
 72 	};
 73 
 74 /**
 75 *  Public Members
 76 */
 77  
 78 	/**
 79 	* The 'uid' is the Chico's unique instance identifier. Every instance has a different 'uid' property. You can see its value by reading the 'uid' property on any public instance.
 80 	* @public
 81 	* @name ch.Helper#uid
 82 	* @type number
 83 	*/
 84 
 85 	/**
 86 	* Reference to a DOM Element. This binding between the component and the HTMLElement, defines context where the component will be executed. Also is usual that this element triggers the component default behavior.
 87 	* @public
 88 	* @name ch.Helper#element
 89 	* @type HTMLElement
 90 	*/
 91 
 92 	/**
 93 	* This public property defines the component type. All instances are saved into a 'map', grouped by its type. You can reach for any or all of the components from a specific type with 'ch.instances'.
 94 	* @public
 95 	* @name ch.Helper#type
 96 	* @type string
 97 	*/
 98 
 99 	/**
100 	* Sets and gets component content. To get the defined content just use the method without arguments, like 'me.content()'. To define a new content pass an argument to it, like 'me.content("new content")'. Use a valid URL to get content using AJAX. Use a CSS selector to get content from a DOM Element. Or just use a String with HTML code.
101 	* @public
102 	* @function
103 	* @name ch.Helper#content
104 	* @param {String} content Static content, DOM selector or URL. If argument is empty then will return the content.
105 	* @example
106 	* // Get the defined content
107 	* me.content();
108 	* @example
109 	* // Set static content
110 	* me.content("Some static content");
111 	* @example
112 	* // Set DOM content
113 	* me.content("#hiddenContent");
114 	* @example
115 	* // Set AJAX content
116 	* me.content("http://chico.com/some/content.html");
117 	* @see ch.Object#content
118 	*/
119 
120 	/**
121 	* Returns a Boolean if the component's core behavior is active. That means it will return 'true' if the component is on and it will return false otherwise.
122 	* @public
123 	* @function
124 	* @name ch.Helper#isActive
125 	* @returns boolean
126 	*/
127 
128 	/**
129 	* Triggers the innerShow method and returns the public scope to keep method chaining.
130 	* @public
131 	* @function
132 	* @name ch.Helper#show
133 	* @returns itself
134 	* @see ch.Floats#show
135 	* @example
136 	* // Following the first example, using 'me' as modal's instance controller:
137 	* me.show();
138 	*/
139 
140 	/**
141 	* Triggers the innerHide method and returns the public scope to keep method chaining.
142 	* @public
143 	* @function
144 	* @name ch.Helper#hide
145 	* @returns itself
146 	* @see ch.Floats#hide
147 	* @example
148 	* // Following the first example, using 'me' as modal's instance controller:
149 	* me.hide();
150 	*/
151 	
152 	/**
153 	* Sets or gets the width property of the component's layout. Use it without arguments to get the value. To set a new value pass an argument, could be a Number or CSS value like '300' or '300px'.
154 	* @public
155 	* @function
156 	* @name ch.Helper#width
157 	* @returns itself
158 	* @see ch.Floats#size
159 	* @example
160 	* // to set the width
161 	* me.width(700);

163 	* // to get the width
164 	* me.width // 700

166

168 	* Sets or gets the height property of the component's layout. Use it without arguments to get the value. To set a new value pass an argument, could be a Number or CSS value like '100' or '100px'.
169 	* @public
170 	* @function
171 	* @name ch.Helper#height
172 	* @returns itself
173 	* @see ch.Floats#size
174 	* @example
175 	* // to set the heigth
176 	* me.height(300);
177 	* @example
178 	* // to get the heigth
179 	* me.height // 300
180 	*/
181 	
182 	/**
183 	* Sets or gets positioning configuration. Use it without arguments to get actual configuration. Pass an argument to define a new positioning configuration.
184 	* @public
185 	* @function
186 	* @name ch.Helper#position
187 	* @example
188 	* // Change component's position.
189 	* me.position({ 
190 	*	offset: "0 10",
191 	*	points: "lt lb"
192 	* });
193 	* @see ch.Object#position
194 	*/
195 
196 /**
197 *  Default event delegation
198 */
199 
200 	$("body").bind(ch.events.LAYOUT.CHANGE, function(){ 
201 		that.position("refresh");
202 	});
203 
204 	/**
205 	* Triggers when the component is ready to use.
206 	* @name ch.Helper#ready
207 	* @event
208 	* @public
209 	* @example
210 	* // Following the first example, using 'me' as Layer's instance controller:
211 	* me.on("ready",function(){
212 	*	this.show();
213 	* });
214 	*/
215 	that.trigger("ready");
216 
217 	return that;
218 };
