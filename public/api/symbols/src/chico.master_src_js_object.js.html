 54 	* Prevent propagation and default actions.
 55 	* @name ch.Object#prevent
 56 	* @function
 57 	* @protected
 58 	* @param {event} event Recieves a event object

 60 	that.prevent = function(event) {
 61 
 62 		if (event && typeof event == "object") {
 63 			event.preventDefault();
 64 			event.stopPropagation();
 65 		};
 66 
 67 		return that;
 68 	};

 70 	/**
 71 	* Set and get the content of a component. With no arguments will behave as a getter function. Send any kind of content and will be a setter function. Use a valid URL for AJAX content, use a CSS selector for a DOM content or just send a static content like HTML or Text.
 72 	* @name ch.Object#content
 73 	* @protected
 74 	* @function
 75 	* @param {string} [content] Could be a simple text, html or a url to get the content with ajax.
 76 	* @returns {string} content
 77 	* @requires ch.Cache
 78 	* @example
 79 	* // Simple static content
 80 	* $(element).layer().content("Some static content");
 81 	* @example
 82 	* // Get DOM content
 83 	* $(element).layer().content("#hiddenContent");
 84 	* @example
 85 	* // Get AJAX content
 86 	* $(element).layer().content("http://chico.com/content/layer.html");
 87 	*/
 88 	that.content = function(content) {
 89 
 90 		var _get = (content) ? false : true,
 91 
 92 			// Local argument
 93 			content = content,
 94 			// Local isURL
 95 			sourceIsUrl = ch.utils.isUrl(that.source),
 96 			// Local isSelector
 97 			sourceIsSelector = ch.utils.isSelector(that.source),
 98 			// Local inDom
 99 			sourceInDom = (!sourceIsUrl) ? ch.utils.inDom(that.source) : false,
100 			// Get context, could be a single component or a controller
101 			context = ( ch.utils.hasOwn(that, "controller") ) ? that.controller : that["public"],
102 			// undefined, for comparison.
103 			undefined,
104 			// Save cache configuration
105 			cache = ( ch.utils.hasOwn(conf, "cache") ) ? conf.cache : true;
106 
107 		/**
108 		* Get content
109 		*/
110 
111 		// return defined content
112 		if (_get) {
113 
114 			// no source, no content
115 			if (that.source === undefined) {
116 				that.staticContent = "<p>No content defined for this component</p>";
117 				that.trigger("contentLoad");
118 
119 				return;
120 			}
121 
122 			// First time we need to get the content.
123 			// Is cache is off, go and get content again.
124 			// Yeap, recursive.
125 			if (!cache || that.staticContent === undefined) {
126 				that.content(that.source);
127 				return;
128 			}
129 
130 			// Get data from cache if the staticContent was defined
131 			if (cache && that.staticContent) {
132 				var fromCache = ch.cache.get(that.source);
133 
134 				// Load content from cache
135 				if (fromCache && that.staticContent != fromCache) {
136 					that.staticContent = fromCache;
137 
138 					that.trigger("contentLoad");
139 
140 					// Return and load content from cache
141 					return;
142 				}
143 
144 				// Return and show the latest content that was loaded
145 				return;
146 			}
147 		}
148 
149 		/**
150 		* Set content
151 		*/
152 
153 		// Reset cache to overwrite content
154 		conf.cache = false;
155 
156 		// Local isURL
157 		var isUrl = ch.utils.isUrl(content),
158 			// Local isSelector
159 			isSelector = ch.utils.isSelector(content),
160 			// Local inDom
161 			inDom = (!isUrl) ? ch.utils.inDom(content) : false;
162 
163 		/* Evaluate static content*/
164 
165 		// Set 'that.staticContent' and overwrite 'that.source'
166 		// just in case you want to update DOM or AJAX Content.
167 
168 		that.staticContent = that.source = content;
169 
170 		/* Evaluate AJAX content*/
171 
172 		if (isUrl) {
173 
174 			// First check Cache
175 			// Check if this source is in our cache
176 			if (cache) {
177 				var fromCache = ch.cache.get(that.source);
178 				if (fromCache) {
179 					conf.cache = cache;
180 					that.staticContent = fromCache;
181 					that.trigger("contentLoad", context);
182 					return;
183 				}
184 			}
185 
186 			var _method, _serialized, _params = "x=x";
187 
188 			// If the trigger is a form button, serialize its parent to send data to the server.
189 			if (that.$element.attr('type') == 'submit') {
190 				_method = that.$element.parents('form').attr('method') || 'GET';
191 				_serialized = that.$element.parents('form').serialize();
192 				_params = _params + ((_serialized != '') ? '&' + _serialized : '');
193 			};
194 
195 			// Set ajax config
196 			// On IE (6-7) "that" reference losts for second time
197 			// Why?? I don't know... but with a setTimeOut() works fine!
198 			setTimeout(function(){
199 
200 				$.ajax({
201 					url: that.source,
202 					type: _method || 'GET',
203 					data: _params,
204 					// each component could have a different cache configuration
205 					cache: cache,
206 					async: true,
207 					beforeSend: function(jqXHR){
208 						// Ajax default HTTP headers
209 						jqXHR.setRequestHeader("X-Requested-With", "XMLHttpRequest");
210 					},
211 					success: function(data, textStatus, jqXHR){
212 						// Save data as staticContent
213 						that.staticContent = data;
214 
215 						// Use the contentLoad callback.
216 						that.trigger("contentLoad", context);
217 
218 						// Save new staticContent to the cache
219 						if (cache) {
220 							ch.cache.set(that.source, that.staticContent);
221 						}
222 
223 					},
224 					error: function(jqXHR, textStatus, errorThrown){
225 						that.staticContent = "<p>Error on ajax call.</p>";
226 
227 						var data = {
228 							"context": context,
229 							"jqXHR": jqXHR,
230 							"textStatus": textStatus,
231 							"errorThrown": errorThrown
232 						};
233 
234 						// Use the contentError callback.
235 						that.trigger("contentError", data);
236 					}
237 				});
238 
239 			}, 0);
240 
241 			// Return Spinner and wait for async callback
242 			that.$content.html("<div class=\"loading\"></div>");
243 			that.staticContent = undefined;
244 
245 		} else {
246 
247 			/* Evaluate DOM content*/
248 
249 			if (isSelector && inDom) {
250 
251 				// Save original DOMFragment.
252 				that.originalContent = $(content);
253 
254 				// Save DOMParent, so we know where to re-insert the content.
255 				that.DOMParent = that.originalContent.parent();
256 
257 				// Save a clone to original DOM content
258 				that.staticContent = that.originalContent.clone().removeClass("ch-hide");
259 
260 			}
261 
262 			// Save new data to the cache
263 			if (cache) {
264 				ch.cache.set(that.source, that.staticContent);
265 			}
266 
267 			// First time we need to set the callbacks that append and remove the original content.
268 			if (that.originalContent && that.originalContent.selector == that.source) {
269 
270 				// Remove DOM content from DOM to avoid ID duplications
271 				that["public"].on("show", function() {
272 					that.originalContent.detach();
273 				});
274 
275 				// Returns DOMelement to DOM
276 				that["public"].on("hide", function(){
277 					that.originalContent.appendTo(that.DOMParent||"body");
278 				});
279 			}
280 		}
281 
282 		/* Set previous cache configuration*/
283 		conf.cache = cache;
284 
285 		// trigger contentLoad callback for DOM and Static content.
286 		if (that.staticContent !== undefined) {
287 			that.trigger("contentLoad", context);
288 		}
289 
290 	};

306 	/**
307 	* Change component's position configuration. If a "refresh" {string} is recived, will refresh component's positioning with the same configuration. You can send an {object} with a new configuration.
308 	* @name ch.Object#position
309 	* @function
310 	* @protected
311 	* @param {string} ["refresh"] Refresh
312 	* @returns {object} Configuration object if no arguments are sended.
313 	* @see ch.Positioner
314 	*/
315 	// TODO: Add examples!!!
316 	that.position = function(o) {
317 
318 		switch(typeof o) {
319 
320 			case "object":
321 				conf.position.context = o.context || conf.position.context;
322 				conf.position.points = o.points || conf.position.points;
323 				conf.position.offset = o.offset || conf.position.offset;
324 				conf.position.fixed = o.fixed || conf.position.fixed;
325 
326 				ch.positioner(conf.position);
327 				return that["public"];
328 				break;
329 
330 			case "string":
331 				if(o != "refresh") {
332 					alert("Chico UI error: position() expected to find \"refresh\" parameter.");
333 				}
334 
335 				ch.positioner(conf.position);
336 
337 				return that["public"];
338 				break;
339 
340 			case "undefined":
341 				return conf.position;
342 				break;
343 		};
344 
345 	};

405 	/**
406 	* Sets and gets component content. To get the defined content just use the method without arguments, like 'me.content()'. To define a new content pass an argument to it, like 'me.content("new content")'. Use a valid URL to get content using AJAX. Use a CSS selector to get content from a DOM Element. Or just use a String with HTML code.
407 	* @public
408 	* @name ch.Object#content
409 	* @function
410 	* @param {string} content Static content, DOM selector or URL. If argument is empty then will return the content.
411 	* @example
412 	* // Get the defined content
413 	* me.content();
414 	* @example
415 	* // Set static content
416 	* me.content("Some static content");
417 	* @example
418 	* // Set DOM content
419 	* me.content("#hiddenContent");
420 	* @example
421 	* // Set AJAX content
422 	* me.content("http://chico.com/some/content.html");
423 	*/
424 	that["public"].content = function(content){
425 		if (content) { // sets
426 			// Reset content data
427 			that.source = content;
428 			that.staticContent = undefined;
429 
430 			if (that.active) {
431 				that.content(content);
432 			}
433 
434 			return that["public"];
435 
436 		} else { // gets
437 			return that.staticContent;
438 		}
439 	};

441 	/**
442 	* Add a callback function from specific event.
443 	* @public
444 	* @function
445 	* @name ch.Object#on
446 	* @param {string} event Event name.
447 	* @param {function} handler Handler function.
448 	* @returns itself
449 	* @since version 0.7.1
450 	* @example
451 	* // Will add a event handler to the "ready" event
452 	* me.on("ready", startDoingStuff);
453 	*/
454 	that["public"].on = function(event, handler) {
455 
456 		if (event && handler) {
457 			$(that["public"]).bind("ch-"+event, handler);
458 		}
459 
460 		return that["public"];
461 	};
