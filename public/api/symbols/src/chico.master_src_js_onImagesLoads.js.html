1 /**
  2 * Execute callback when images of a query selection loads
  3 * @abstract
  4 * @name onImagesLoads
  5 * @class onImagesLoads
  6 * @memberOf ch
  7 * @param array
  8 * @returns jQuery
  9 * @example
 10 * $("img").onImagesLoads(function(){ ... });
 11 */
 12 
 13 ch.onImagesLoads = function(conf){
 14 
 15 	/**
 16 	* Reference to a internal component instance, saves all the information and configuration properties.
 17 	* @private
 18 	* @name ch.onImagesLoads#that
 19 	* @type object
 20 	*/
 21 	var that = this;
 22 	conf = ch.clon(conf);
 23 	that.conf = conf;
 24 	
 25 	that.$element
 26 		// On load event
 27 		.bind("load", function(){
 28 			setTimeout(function(){
 29 				if (--that.$element.length <= 0) {
 30 					that.conf.lambda.call(that.$element, this);
 31 				};
 32 			}, 200);
 33 		})
 34 		// For each image
 35 		.each(function(){
 36 			// Cached images don't fire load sometimes, so we reset src.
 37 			if (this.complete || this.complete === undefined) {
 38 				var src = this.src;
 39 				
 40 				// Data uri fix bug in web-kit browsers
 41 				this.src = "data:image/gif;base64,R0lGODlhAQABAIAAAAAAAP///ywAAAAAAQABAAACAUwAOw==";
 42 				this.src = src;
 43 			};
 44 		});
 45 	
 46 	return that;
 47 };
 48 
 49 ch.factory("onImagesLoads");
