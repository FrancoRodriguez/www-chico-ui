  2 /**
  3 * Validate strings.
  4 * @name String 
  5 * @class String
  6 * @interface
  7 * @augments ch.Watcher
  8 * @memberOf ch
  9 * @param {string} msg Validation message
 10 * @returns itself
 11 * @see ch.Watcher
 12 * @see ch.Required
 13 * @see ch.Custom
 14 * @see ch.Number
 15 * @example
 16 * // Create a string validation
 17 * $("input").string("This field must be a string.");
 18 */
 19 
 20 ch.extend("watcher").as("string", function (conf) {
 21 
 22 	// $.string("message"); support
 23 	if ( !conf.string && !conf.email && !conf.url && !conf.maxLength && !conf.minLength ) {
 24 		conf.string = true;
 25 	};
 26 	
 27 	/**
 28 	* @public
 29 	* @name ch.String#conditions
 30 	* @type {Map}
 31 	*/
 32 	// Define the conditions of this interface
 33 	conf.conditions = [{
 34 			name: "string",
 35 			// the following regular expression has the utf code for the lating characters
 36 			// the ranges are A,EI,O,U,a,ei,o,u,รง,ร please for reference see http://www.fileformat.info/info/charset/UTF-8/list.htm
 37 			patt: /^([a-zA-Z\u00C0-\u00C4\u00C8-\u00CF\u00D2-\u00D6\u00D9-\u00DC\u00E0-\u00E4\u00E8-\u00EF\u00F2-\u00F6\u00E9-\u00FC\u00C7\u00E7\s]*)$/
 38 		},{
 39 			name:"email",
 40 			patt: /^(("[\w-\s]+")|([\w-]+(?:\.[\w-]+)*)|("[\w-\s]+")([\w-]+(?:\.[\w-]+)*))(@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$)|(@\[?((25[0-5]\.|2[0-4][0-9]\.|1[0-9]{2}\.|[0-9]{1,2}\.))((25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\.){2}(25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\]?$)/ 
 41 		},{
 42 			name: "url",
 43 			patt: /^((https?|ftp|file):\/\/|((www|ftp)\.)|(\/|.*\/)*)[a-z0-9-]+((\.|\/)[a-z0-9-]+)+([/?].*)?$/ 
 44 		},{
 45 			name: "minLength",
 46 			expr: function(a,b) { return a.length >= b } 
 47 		},{
 48 			name: "maxLength",
 49 			expr: function(a,b) { return a.length <= b } 
 50 		}];
 51 
 52 	return conf;
 53 
 54 });
