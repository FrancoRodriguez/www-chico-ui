  2 /** 
  3 * UI feedback utility, creates a visual highlight
  4 * changing background color from yellow to white.
  5 * @function
  6 * @name blink
  7 * @param {Selector} selector CSS Selector to blink a collection
  8 * @param {Number} [time] Amount of time to blink
  9 * @returns {jQuery Object}
 10 * @memberOf ch
 11 */
 12 ch.blink = function (conf) {
 13 
 14 	var that = this,
 15 		// Hex start level toString(16).
 16 		level = 1, 
 17 		// Time, 200 miliseconds by default.
 18 		t = conf.value || 200,
 19 		// Inner highlighter.
 20 		highlight = function (e) {
 21 			// Let know everyone we are active.
 22 			that.$element.addClass("ch-active");
 23 			// Color iteration.
 24 			function step () {
 25 				// New hex level.
 26 				var h = level.toString(16);
 27 				// Change background-color, redraw().
 28 				e.style.backgroundColor = '#FFFF' + h + h;
 29 				// Itearate for all hex levels.
 30 				if (level < 15) {
 31 					// Increment hex level.
 32 					level += 1;
 33 					// Inner recursion.
 34 					setTimeout(step, t);
 35 				} else {
 36 					// Stop right there...
 37 					that.$element.removeClass("ch-active");
 38 				}
 39 		};
 40 		// Begin steps.
 41 		setTimeout(step, t);
 42 	}
 43 	// Start a blink if the element isn't active.
 44 	if (!that.$element.hasClass("ch-active")) {
 45 		highlight(that.element);
 46 	}
 47 	// Return the element so keep chaining things.
 48 	return that.$element;
 49 }
