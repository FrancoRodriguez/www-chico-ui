  3 /**
  4  * Abstract class of all floats UI-Objects.
  5  * @abstract
  6  * @name Floats
  7  * @class Floats
  8  * @augments ch.Object
  9  * @memberOf ch
 10  * @requires ch.Positioner
 11  * @param {Configuration Object} conf Object with configuration properties
 12  * @returns {Chico-UI Object}
 13  * @see ch.Tooltip
 14  * @see ch.Layer
 15  * @see ch.Modal
 16  */ 
 17 
 18 ch.floats = function() {
 19 
 20     /**
 21      * Reference to a internal component instance, saves all the information and configuration properties.
 22      * @private
 23      * @name that
 24      * @type {Object}
 25      * @memberOf ch.Floats
 26      */ 
 27  	var that = this;
 28 	var conf = that.conf;
 29 
 30 /**
 31  *  Inheritance
 32  */
 33 
 34     that = ch.object.call(that);
 35     that.parent = ch.clon(that);
 36 
 37 /**
 38  *  Private Members
 39  */
 40 
 41     /**
 42      * Creates a 'cone', is a visual asset for floats.
 43      * @private
 44      * @name createCone
 45      * @function
 46      * @memberOf ch.Floats
 47      */ 
 48 	var createCone = function() {
 49 		$("<div class=\"ch-cone\">")
 50 			.prependTo( that.$container );
 51 	};
 52 
 53     /**
 54      * Creates close button.
 55      * @private
 56      * @name createClose
 57      * @function
 58      * @memberOf ch.Floats
 59      */ 
 60 	var createClose = function() {
 61 		// Close Button
 62 		$("<div class=\"btn close\" style=\"z-index:"+(ch.utils.zIndex+=1)+"\">")
 63 			.bind("click", function(event){ that.hide(event); })
 64 			.prependTo( that.$container );
 65 		
 66 		// ESC key
 67 		ch.utils.document.bind(ch.events.KEY.ESC, function(event){ that.hide(event); });
 68 		
 69 		return;
 70 	};
 71 
 72 /**
 73  *  Protected Members
 74  */ 
 75 			
 76 /**
 77  *  Public Members
 78  */
 79      /**
 80      * Flag that indicates if the float is active on the DOM tree.
 81      * @public
 82      * @name active
 83      * @type {Boolean}
 84      * @memberOf ch.Floats
 85      */ 
 86 	that.active = false;
 87 
 88     /**
 89      * Content configuration propertie.
 90      * @public
 91      * @name source
 92      * @type {String}
 93      * @memberOf ch.Floats
 94      */
 95 	that.source = conf.content || conf.msg || conf.ajax || that.$element.attr('href') || that.$element.parents('form').attr('action');
 96 
 97 
 98     /**
 99      * Container for UI Component.
100      * @public
101      * @name $container
102      * @type {jQuery Object}
103      * @memberOf ch.Floats
104      */ 
105     that.$container = (function() { // Create Layout
106 
107 		// Create the component container
108 		that.$container = $("<div class=\"ch-"+ that.type +"\" style=\"z-index:"+(ch.utils.zIndex+=1)+"\">");
109 				
110 		// Visual configuration
111 		if( ch.utils.hasOwn(conf, "classes") ) { that.$container.addClass(conf.classes); }
112 		if( ch.utils.hasOwn(conf, "width") ) { that.$container.css("width", conf.width); }
113 		if( ch.utils.hasOwn(conf, "height") ) { that.$container.css("height", conf.height); }
114 		if( ch.utils.hasOwn(conf, "closeButton") && conf.closeButton ) { createClose(); }
115 		if( ch.utils.hasOwn(conf, "cone") ) { createCone(); }
116 		if( ch.utils.hasOwn(conf, "fx") ) { conf.fx = conf.fx; } else { conf.fx = true; }
117 		
118 		// Cache - Default: true
119 		//conf.cache = ( ch.utils.hasOwn(conf, "cache") ) ? conf.cache : true;
120 
121 		// Position component
122 		conf.position = conf.position || {};
123 		//conf.position.element = that.$container;
124 		conf.position.hold = conf.hold || false;
125 		//ch.positioner.call(that); // Is this necesary?
126 
127 		// Return the entire Layout
128 		return that.$container;
129 
130     })(); 
131 
132     /**
133      * Container for UI Component's content.
134      * @public
135      * @name $content
136      * @type {jQuery Object}
137      * @memberOf ch.Floats
138      */ 
139 	that.$content = $("<div class=\"ch-"+ that.type +"-content\">").appendTo(that.$container);
140 
141     /**
142      * This callback is triggered when async data is loaded into component's content, when ajax content comes back.
143      * @public
144      * @name contentCallback
145      * @returns {Chico-UI Object}
146      * @memberOf ch.Floats
147      */ 
148     that.contentCallback = function(data) {
149         that.staticContent = data;
150         that.$content.html(that.staticContent);
151     	if ( ch.utils.hasOwn(conf, "position") ) {
152     	   ch.positioner(conf.position);
153     	}
154     }
155 
156     /**
157      * Renders the component in the display by adding it to the DOM tree.
158      * @public
159      * @name show
160      * @returns {Chico-UI Object}
161      * @memberOf ch.Floats
162      */ 
163 	that.show = function(event) {
164 
165 		if ( event ) {
166 			that.prevent(event);
167 		}
168 		
169 		// Avoid showing things that are already shown
170 		if ( that.active ) return;
171 		
172 		// Get content
173 		that.staticContent = that.content();
174 		// Saves content
175         that.$content.html(that.staticContent);
176 
177         // Add layout to DOM tree
178         // Increment zIndex
179 		that.$container
180 		    .appendTo("body")
181 			.css("z-index", ch.utils.zIndex++);
182 
183         /**
184          * Callback function
185          * @name onShow
186          * @type {Function}
187          * @memberOf ch.Floats
188          */
189 		// Show component with effects
190 		if( conf.fx ) {
191 			that.$container.fadeIn("fast", function(){ that.callbacks("onShow"); });
192 		} else { 
193         // Show component without effects
194 			that.$container.removeClass("ch-hide");
195 			that.callbacks("onShow");
196 		};
197 	
198 		// TODO: Positioner should recalculate the element's size (width and height) 
199 		conf.position.element = that.$container;
200 
201 		that.position("refresh");
202 
203 		that.active = true;
204 
205         return that;
206 	};
207 
208     /**
209      * Hides the component and detach it from DOM tree.
210      * @public
211      * @name hide
212      * @returns {Chico-UI Object}
213      * @memberOf ch.Floats
214      */ 
215 	that.hide = function(event) {
216 
217 		if (event) {
218 			that.prevent(event);
219 		}
220 		
221 		if (!that.active) {
222 			return;
223 		}
224 
225 		var afterHide = function(){ 
226 			 
227 			that.active = false;
228 			
229            /**
230             * Callback function
231             * @name onHide
232             * @type {Function}
233             * @memberOf ch.Floats
234             */
235 			that.callbacks('onHide');
236 			
237 			that.$container.detach();
238 			
239 			// TODO: This should be wrapped on Object.content() method
240 			// We need to be able to use interal callbacks...
241 			if (!ch.utils.isSelector(that.source)) {
242 				var original = $(that.staticContent).clone();
243 					original.appendTo(that.DOMParent||"body");
244 
245 			   if (!that.DOMContentIsVisible) {
246 					original.addClass("ch-hide");	     
247 		       }
248 		        
249 			};
250 		};
251 		
252 		// Show component with effects
253 		if( conf.fx ) {
254 			that.$container.fadeOut("fast", afterHide);
255 		
256 		// Show component without effects
257 		} else {
258 			that.$container.addClass("ch-hide");
259 			afterHide();
260 		};
261 		
262 		return that;
263 
264 	};
265 	
266 	/**
267      * Getter and setter for size attributes on any float component.
268      * @private
269      * @function
270      * @name size
271      * @param {String} prop Property that will be setted or getted, like "width" or "height".
272      * @param {String} [data] Only for setter. It's the new value of defined property.
273 	 * @returns {Internal component instance}
274      * @memberOf ch.Floats
275      */
276 	that.size = function(prop, data) {
277 		// Getter
278 		if (!data) {
279 			return that.conf[prop];
280 		};
281 
282 		// Setter
283 		that.conf[prop] = data;
284 		
285 		// Container
286 		that.$container[prop](data);
287 		
288 		that.position("refresh");
289 		
290 		return that["public"];
291 	};
292 
293     /**
294      * Shows float element that contains the zoomed image.
295      * @public
296      * @name show
297      * @function
298      * @returns {Chico-UI Object}
299      * @memberOf ch.Float
300      */
301 	that["public"].show = function(){
302 		that.show();
303 		
304 		return that["public"];
305 	};
306 	
307     /**
308      * Hides float element that contains the zoomed image.
309      * @public
310      * @name hide
311      * @function
312      * @returns {Chico-UI Object}
313      * @memberOf ch.Float
314      */
315 	that["public"].hide = function(){
316 		that.hide();
317 		
318 		return that["public"];
319 	};
320     /**
321      * Sets or gets the width of the Float element.
322      * @public
323      * @name width
324      * @function
325      * @returns {Chico-UI Object}
326      * @memberOf ch.Float
327      */
328 	that["public"].width = function(data) {
329 		
330 		return that.size("width", data) || that["public"];
331 	};
332     /**
333      * Sets or gets the height of the Float element.
334      * @public
335      * @name height
336      * @function
337      * @returns {Chico-UI Object}
338      * @memberOf ch.Float
339      */
340 	that["public"].height = function(data) {
341 			
342 		return that.size("height", data) || that["public"];
343 	};
344     
345     /**
346      * Returns true if the component is active.
347      * @public
348      * @name isActive
349      * @function
350      * @returns {Boolean}
351      * @memberOf ch.Floats
352      */
353 	that["public"].isActive = function() {
354 	   return that.active;
355 	};
356 	
357 	return that;
358 	
359 };
