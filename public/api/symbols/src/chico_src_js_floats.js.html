  2 /**
  3 * Abstract class of all floats UI-Objects.
  4 * @abstract
  5 * @name ch.Floats
  6 * @class Floats
  7 * @augments ch.Object
  8 * @requires ch.Positioner
  9 * @returns {ch Object}
 10 * @see ch.Tooltip
 11 * @see ch.Layer
 12 * @see ch.Modal
 13 */ 
 14 
 15 ch.floats = function() {
 16 
 17 	/**
 18 	* Reference to a internal component instance, saves all the information and configuration properties.
 19 	* @protected
 20 	* @name ch.Floats#that
 21 	* @type {Object}
 22 	*/ 
 23 	var that = this;
 24 	var conf = that.conf;
 25 
 26 /**
 27 * Inheritance
 28 */
 29 
 30 	that = ch.object.call(that);
 31 	that.parent = ch.clon(that);
 32 
 33 /**
 34 * Private Members
 35 */
 36 
 37 	/**
 38  	* Creates a 'cone', is a visual asset for floats.
 39  	* @private
 40  	* @name ch.Floats#createCone
 41  	* @function
 42  	*/ 
 43 	var createCone = function() {
 44 		$("<div class=\"ch-cone\">")
 45 			.prependTo( that.$container );
 46 	};
 47 
 48 	/**
 49  	* Creates close button.
 50  	* @private
 51  	* @name ch.Floats#createClose
 52  	* @function
 53  	*/ 
 54 	var createClose = function() {
 55 		// Close Button
 56 		$("<div class=\"btn close\" style=\"z-index:"+(ch.utils.zIndex+=1)+"\">")
 57 			.bind("click", function(event){ that.innerHide(event); })
 58 			.prependTo( that.$container );
 59 		
 60 		// ESC key
 61 		ch.utils.document.bind(ch.events.KEY.ESC, function(event){ that.innerHide(event); });
 62 		
 63 		return;
 64 	};
 65 
 66 /**
 67 * Protected Members
 68 */ 
 69 	/**
 70 	* Flag that indicates if the float is active and rendered on the DOM tree.
 71 	* @protected
 72 	* @name ch.Floats#active
 73 	* @type {Boolean}
 74 	*/ 
 75 	that.active = false;
 76 
 77 	/**
 78 	* Content configuration property.
 79 	* @protected
 80 	* @name ch.Floats#source
 81 	* @type {String}
 82 	*/
 83 	that.source = conf.content || conf.msg || conf.ajax || that.$element.attr('href') || that.$element.parents('form').attr('action');
 84 
 85 	/**
 86 	* Inner function that resolves the component's layout and returns a static reference.
 87 	* @protected
 88 	* @name ch.Floats#$container
 89 	* @type {jQuery Object}
 90 	*/ 
 91 	that.$container = (function() { // Create Layout
 92 
 93 		// Create the component container
 94 		that.$container = $("<div class=\"ch-"+ that.type +"\" style=\"z-index:"+(ch.utils.zIndex+=1)+"\">");
 95 
 96 		// Visual configuration
 97 		if( ch.utils.hasOwn(conf, "classes") ) { that.$container.addClass(conf.classes); }
 98 		if( ch.utils.hasOwn(conf, "width") ) { that.$container.css("width", conf.width); }
 99 		if( ch.utils.hasOwn(conf, "height") ) { that.$container.css("height", conf.height); }
100 		if( ch.utils.hasOwn(conf, "closeButton") && conf.closeButton ) { createClose(); }
101 		if( ch.utils.hasOwn(conf, "cone") ) { createCone(); }
102 		if( ch.utils.hasOwn(conf, "fx") ) { conf.fx = conf.fx; } else { conf.fx = true; }
103 		
104 		// Cache - Default: true
105 		//conf.cache = ( ch.utils.hasOwn(conf, "cache") ) ? conf.cache : true;
106 
107 		// Position component
108 		conf.position = conf.position || {};
109 		//conf.position.element = that.$container;
110 		conf.position.hold = conf.hold || false;
111 		//ch.positioner.call(that); // Is this necesary?
112 
113 		// Return the entire Layout
114 		return that.$container;
115 
116 	})(); 
117 
118 	/**
119 	* Inner reference to content container. Here is where the content will be added.
120 	* @protected
121 	* @name ch.Floats#$content
122 	* @type {jQuery Object}
123 	* @see ch.Object#content
124 	*/ 
125 	that.$content = $("<div class=\"ch-"+ that.type +"-content\">").appendTo(that.$container);
126 
127 	/**
128 	* This callback is triggered when async data is loaded into component's content, when ajax content comes back.
129 	* @protected
130 	* @name ch.Floats#contentCallback
131 	* @function
132 	* @returns {this}
133 	*/ 
134 	that.contentCallback = function(data) {
135 		that.staticContent = data;
136 		that.$content.html(that.staticContent);
137 		if ( ch.utils.hasOwn(conf, "position") ) {
138 		   ch.positioner(conf.position);
139 		}
140 	}
141 
142 	/**
143 	* Inner show method. Attach the component layout to the DOM tree.
144 	* @protected
145 	* @name ch.Floats#innerShow
146 	* @function
147 	* @returns {this}
148 	*/ 
149 	that.innerShow = function(event) {
150 
151 		if (event) {
152 			that.prevent(event);
153 		}
154 				
155 		// Avoid showing things that are already shown
156 		if ( that.active ) return;
157 
158 		// Get content
159 		that.staticContent = that.content();
160 		// Saves content
161 		that.$content.html(that.staticContent);
162 
163 		// Add layout to DOM tree
164 		// Increment zIndex
165 		that.$container
166 			.appendTo("body")
167 			.css("z-index", ch.utils.zIndex++);
168 
169 		/**
170 		* Triggers when component is visible.
171 		* @name ch.Floats#show
172 		* @event
173 		* @public
174 		*/
175 		// Show component with effects
176 		if( conf.fx ) {
177 			that.$container.fadeIn("fast", function(){ 
178 				// new callbacks
179 				that.trigger("show");
180 				// Old callback system
181 				that.callbacks('onShow');
182 			});
183 		} else { 
184 		// Show component without effects
185 			that.$container.removeClass("ch-hide");
186 			// new callbacks
187 			that.trigger("show");
188 			// Old callback system
189 			that.callbacks('onShow');
190 		};
191 	
192 		// TODO: Positioner should recalculate the element's size (width and height) 
193 		conf.position.element = that.$container;
194 
195 		that.position("refresh");
196 
197 		that.active = true;
198 
199 		return that;
200 	};
201 
202 	/**
203 	* Inner hide method. Hides the component and detach it from DOM tree.
204 	* @protected
205 	* @name ch.Floats#innerHide
206 	* @function
207 	* @returns {this}
208 	*/ 
209 	that.innerHide = function(event) {
210 
211 		if (event) {
212 			that.prevent(event);
213 		}
214 		
215 		if (!that.active) {
216 			return;
217 		}
218 
219 		var afterHide = function(){ 
220 			 
221 			that.active = false;
222 			
223 		/**
224 		* Triggers when component is not longer visible.
225 		* @name ch.Floats#hide
226 		* @event
227 		* @public
228 		*/
229 			// new callbacks
230 			that.trigger("hide");
231 			// Old callback system
232 			that.callbacks('onHide');
233 
234 			that.$container.detach();
235 
236 			// TODO: This should be wrapped on Object.content() method
237 			// We need to be able to use interal callbacks...
238 			if (ch.utils.isSelector(that.source) && !ch.utils.inDom(that.source) && !ch.utils.isUrl(that.source)) {
239 				var original = $(that.staticContent).clone();
240 					original.appendTo(that.DOMParent||"body");
241 
242 				if (!that.DOMContentIsVisible) {
243 					original.addClass("ch-hide");
244 				}
245 
246 			};
247 		};
248 		
249 		// Show component with effects
250 		if( conf.fx ) {
251 			that.$container.fadeOut("fast", afterHide);
252 		
253 		// Show component without effects
254 		} else {
255 			that.$container.addClass("ch-hide");
256 			afterHide();
257 		};
258 
259 		return that;
260 
261 	};
262 	
263 	/**
264 	* Getter and setter for size attributes on any float component.
265 	* @protected
266 	* @function
267 	* @name ch.Floats#size
268 	* @param {String} prop Property that will be setted or getted, like "width" or "height".
269 	* @param {String} [data] Only for setter. It's the new value of defined property.
270 	* @returns {this}
271 	*/
272 	that.size = function(prop, data) {
273 		// Getter
274 		if (!data) {
275 			return that.conf[prop];
276 		};
277 		// Setter
278 		that.conf[prop] = data;
279 		// Container
280 		that.$container[prop](data);
281 		that.position("refresh");
282 		return that["public"];
283 	};
284 
285 
286 /**
287 * Public Members
288 */
289  
290 	/**
291 	* Triggers the innerShow method and returns the public scope to keep method chaining.
292 	* @public
293 	* @name ch.Floats#show
294 	* @function
295 	* @returns {this}
296 	*/
297 	that["public"].show = function(){
298 		that.innerShow();
299 		return that["public"];
300 	};
301 	
302 	/**
303 	* Triggers the innerHide method and returns the public scope to keep method chaining.
304 	* @public
305 	* @name ch.Floats#hide
306 	* @function
307 	* @returns {this}
308 	*/
309 	that["public"].hide = function(){
310 		that.innerHide();
311 		return that["public"];
312 	};
313 	/**
314 	* Sets or gets the width property of the component's layout. Use it without arguments to get the value. To set a new value pass an argument, could be a Number or CSS value like '300' or '300px'.
315 	* @public
316 	* @name ch.Floats#width
317 	* @function
318 	* @returns {this}
319 	* @see ch.Floats#size
320 	* @example
321 	* // to set the width
322 	* me.width(700);
323 	* @example
324 	* // to get the width
325 	* me.width // 700
326 	*/
327 	that["public"].width = function(data) {
328 		return that.size("width", data) || that["public"];
329 	};
330 	/**
331 	* Sets or gets the height of the Float element.
332 	* @public
333 	* @name ch.Floats#height
334 	* @function
335 	* @returns {this}
336 	* @see ch.Floats#size
337 	* @example
338 	* // to set the heigth
339 	* me.height(300);
340 	* @example
341 	* // to get the heigth
342 	* me.height // 300
343 	*/
344 	that["public"].height = function(data) {
345 		return that.size("height", data) || that["public"];
346 	};
347 	
348 	/**
349 	* Returns a Boolean if the component's core behavior is active. That means it will return 'true' if the component is on and it will return false otherwise.
350 	* @public
351 	* @name ch.Floats#isActive
352 	* @function
353 	* @returns {Boolean}
354 	*/
355 	that["public"].isActive = function() {
356 		return that.active;
357 	};
358 
359 	return that;
360  
361 };
