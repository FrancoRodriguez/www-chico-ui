  2 /**
  3 * Is a contextual floated UI-Object.
  4 * @name Layer
  5 * @class Layer
  6 * @augments ch.Floats
  7 * @memberOf ch
  8 * @param {Object} conf Object with configuration properties
  9 * @returns {itself}
 10 * @see ch.Tooltip
 11 * @see ch.Modal
 12 * @example
 13 * // Create a simple contextual layer
 14 * var me = $(".some-element").layer("<p>Some content.</p>");
 15 * @example
 16 * // Now 'me' is a reference to the layer instance controller.
 17 * // You can set a new content by using 'me' like this: 
 18 * me.content("http://content.com/new/content");
 19 */
 20 
 21 ch.layer = function(conf) {
 22 
 23 	/**
 24 	* Reference to a internal component instance, saves all the information and configuration properties.
 25 	* @private
 26 	* @name ch.Layer#that
 27 	* @type {Object}
 28 	*/ 
 29 	var that = this;
 30 	
 31 	conf = ch.clon(conf);
 32 	conf.cone = true;
 33 	conf.classes = "box";
 34 	conf.closeButton =	(conf.event === 'click') ? true : false;
 35 	conf.position = {};
 36 	conf.position.context = that.$element;
 37 	conf.position.offset = conf.offset || "0 10";
 38 	conf.position.points = conf.points || "lt lb";
 39 
 40 	that.conf = conf;
 41 
 42 /**
 43 *	Inheritance
 44 */
 45 
 46 	that = ch.floats.call(that);
 47 	that.parent = ch.clon(that);
 48 	
 49 /**
 50 *	Private Members
 51 */
 52 
 53 	/**
 54 	* Delay time to show component's contents.
 55 	* @private
 56 	* @name ch.Layer#showTime
 57 	* @type {Number}
 58 	* @default 400
 59 	*/ 
 60 	var showTime = conf.showTime || 400,
 61 	/**
 62 	* Delay time to hide component's contents.
 63 	* @private
 64 	* @name ch.Layer#hideTime
 65 	* @type {Number}
 66 	* @default 400
 67 	*/ 
 68 	hideTime = conf.hideTime || 400,
 69 
 70 	/**
 71 	* Show timer instance.
 72 	* @private
 73 	* @name ch.Layer#st
 74 	* @type {Timer}
 75 	*/ 
 76 	st,
 77 	
 78 	/**
 79 	* Hide timer instance.
 80 	* @private
 81 	* @name ch.Layer#ht
 82 	* @type {Timer}
 83 	*/ 
 84 	ht,
 85 	
 86 	/**
 87 	* Starts show timer.
 88 	* @private
 89 	* @name ch.Layer#showTimer
 90 	* @function
 91 	*/ 
 92 	showTimer = function(){ st = setTimeout(that.innerShow, showTime) },
 93 	
 94 	/**
 95 	* Starts hide timer.
 96 	* @private
 97 	* @name ch.Layer#hideTimer
 98 	* @function
 99 	*/ 
100 	hideTimer = function(){ ht = setTimeout(that.innerHide, hideTime) },
101 	
102 	/**
103 	* Clear all timers.
104 	* @private
105 	* @name ch.Layer#clearTimers
106 	* @function
107 	*/ 
108 	clearTimers = function(){ clearTimeout(st); clearTimeout(ht); },
109 
110 	/**
111 	* Stop event bubble propagation to avoid hiding the layer by click on his own layout.
112 	* @private
113 	* @name ch.Layer#stopBubble
114 	* @function
115 	*/
116 	stopBubble = function(event){ event.stopPropagation(); };
117 /**
118 *	Protected Members
119 */
120 
121 	that.$trigger = that.$element;
122 
123 	/**
124 	* Inner show method. Attach the component layout to the DOM tree.
125 	* @protected
126 	* @name ch.Layer#innerShow
127 	* @function
128 	* @returns {itself}
129 	*/ 
130 	that.innerShow = function(event) {
131 	
132 		// Reset all layers
133 		$.each(ch.instances.layer, function(i, e){ e.hide(); });
134 		//conf.position.context = that.$element;
135 		that.parent.innerShow(event);
136 
137 		// Click
138 		if (conf.event == "click") {
139 			conf.close = true;
140 			// Document events
141 			$(document).one('click', that.innerHide);
142 			that.$container.bind('click', stopBubble);
143 		// Hover
144 		} else { 		
145 			clearTimers();	
146 			that.$container
147 				.one("mouseenter", clearTimers)
148 				.one("mouseleave", function(event){
149 					var target = event.srcElement || event.target;
150 					var relatedTarget = event.relatedTarget || event.toElement;
151 					var relatedParent = relatedTarget.parentNode;
152 					if ( target === relatedTarget || relatedParent === null || target.nodeName === "SELECT" ) return;
153 					hideTimer();
154 				});
155 		};
156 
157 		return that;
158 	};
159 	
160 	/**
161 	* Inner hide method. Hides the component and detach it from DOM tree.
162 	* @protected
163 	* @name ch.Layer#innerHide
164 	* @function
165 	* @returns {itself}
166 	*/ 
167 	that.innerHide = function(event) {
168 		that.$container.unbind('click', stopBubble);
169 		that.parent.innerHide(event);
170 	}

194 	* @type {String}

198 	* Sets and gets component content. To get the defined content just use the method without arguments, like 'me.content()'. To define a new content pass an argument to it, like 'me.content("new content")'. Use a valid URL to get content using AJAX. Use a CSS selector to get content from a DOM Element. Or just use a String with HTML code.
199 	* @public
200 	* @name ch.Layer#content
201 	* @function
202 	* @param {String} content Static content, DOM selector or URL. If argument is empty then will return the content.
203 	* @example
204 	* // Get the defined content
205 	* me.content();
206 	* @example
207 	* // Set static content

209 	* @example
210 	* // Set DOM content
211 	* me.content("#hiddenContent");
212 	* @example
213 	* // Set AJAX content
214 	* me.content("http://chico.com/some/content.html");
215 	* @see ch.Object#content
216 	*/
217

228 	* @public
229 	* @name ch.Layer#show
230 	* @function
231 	* @returns {itself}
232 	* @see ch.Floats#show
233 	* @example
234 	* // Following the first example, using 'me' as modal's instance controller:

242 	* @function

244 	* @see ch.Floats#hide
245 	* @example
246 	* // Following the first example, using 'me' as modal's instance controller:
247 	* me.hide();
248 	*/

251 	* Sets or gets the width property of the component's layout. Use it without arguments to get the value. To set a new value pass an argument, could be a Number or CSS value like '300' or '300px'.
252 	* @public

269 	* @function
270 	* @returns {itself}
271 	* @see ch.Floats#size

273 	* // to set the heigth

276 	* // to get the heigth

278 	*/
279

281 	* Sets or gets positioning configuration. Use it without arguments to get actual configuration. Pass an argument to define a new positioning configuration.
282 	* @public
283 	* @name ch.Layer#position
284 	* @example
285 	* // Change component's position.
286 	* me.position({ 
287 	*	offset: "0 10",
288 	*	points: "lt lb"
289 	* });
290 	* @see ch.Object#position
291 	*/
292 	
293 /**
294 *	Default event delegation
295 */
296 
297 	// Click
298 	if(conf.event === 'click') {
299 		that.$trigger
300 			.css('cursor', 'pointer')
301 			.bind('click', that.innerShow);
302 
303 	// Hover
304 	} else {
305 		that.$trigger
306 			.css('cursor', 'default')
307 			.bind('mouseenter', that.innerShow)
308 			.bind('mouseleave', hideTimer);
309 	};
310 
311 	// Fix: change layout problem
312 	$("body").bind(ch.events.LAYOUT.CHANGE, function(){ that.position("refresh") });
313 
314 	/**
315 	* Triggers when component is visible.
316 	* @name ch.Layer#show
317 	* @event
318 	* @public
319 	* @example
320 	* me.on("show",function(){
321 	*	this.content("Some new content");
322 	* });
323 	* @see ch.Floats#event:show
324 	*/
325 
326 	/**
327 	* Triggers when component is not longer visible.
328 	* @name ch.Layer#hide
329 	* @event
330 	* @public
331 	* @example
332 	* me.on("hide",function(){
333 	*	otherComponent.show();
334 	* });
335 	* @see ch.Floats#event:hide
336 	*/
337 		
338 	/**
339 	* Triggers when the component is ready to use.
340 	* @name ch.Layer#ready
341 	* @event
342 	* @public
343 	* @example
344 	* // Following the first example, using 'me' as Layer's instance controller:
345 	* me.on("ready",function(){
346 	*	this.show();
347 	* });
348 	*/
349 	that.trigger("ready");

351 	return that;
