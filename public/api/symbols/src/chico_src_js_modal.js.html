1 /**
  2 * Is a centered floated window with a dark gray dimmer background. This component let you handle its size, positioning and content.
  3 * @name Modal
  4 * @class Modal
  5 * @augments ch.Floats
  6 * @memberOf ch
  7 * @param {Object} conf Object with configuration properties
  8 * @returns {itself}
  9 * @see ch.Tooltip
 10 * @see ch.Layer
 11 * @example
 12 * // Create a new modal window triggered by an anchor with a class name 'example'.
 13 * var me = $("a.example").modal();
 14 * @example
 15 * // Now 'me' is a reference to the modal instance controller.
 16 * // You can set a new content by using 'me' like this: 
 17 * me.content("http://content.com/new/content");
 18 */
 19 
 20 ch.modal = function(conf){
 21 
 22 	/**
 23 	* Reference to a internal component instance, saves all the information and configuration properties.
 24 	* @private
 25 	* @name ch.Modal#that
 26 	* @type {Object}
 27 	*/
 28 	var that = this;
 29 	conf = ch.clon(conf);
 30 	conf.closeButton = (that.type == "modal") ? true : false;
 31 	conf.classes = "box";
 32 
 33 	that.conf = conf;
 34 
 35 /**
 36 *	Inheritance
 37 */
 38 
 39 	that = ch.floats.call(that);
 40 	that.parent = ch.clon(that);
 41 
 42 /**
 43 *	Private Members
 44 */
 45 
 46 	/**
 47 	* Reference to the dimmer object, the gray background element.
 48 	* @private
 49 	* @name ch.Modal#$dimmer
 50 	* @type {jQuery Object}
 51 	*/
 52 	var $dimmer = $("<div class=\"ch-dimmer\">");
 53 	
 54 	// Set dimmer height for IE6
 55 	if (ch.utils.html.hasClass("ie6")) { $dimmer.height( parseInt(document.documentElement.clientHeight, 10)* 3) };
 56 	
 57 	/**
 58 	* Reference to dimmer control, turn on/off the dimmer object.
 59 	* @private
 60 	* @name ch.Modal#dimmer
 61 	* @type {Object}
 62 	*/
 63 	var dimmer = {
 64 		on: function() { //TODO: posicionar el dimmer con el positioner
 65 			$dimmer
 66 				.css("z-index", ch.utils.zIndex += 1)
 67 				.appendTo("body")
 68 				.fadeIn();
 69 		
 70 			if (that.type == "modal") {
 71 				$dimmer.one("click", function(event){ that.innerHide(event) });
 72 			};
 73 			
 74 			if (!ch.features.fixed) {
 75 			 	ch.positioner({ element: $dimmer });
 76 			};
 77 
 78 			if ($("html").hasClass("ie6")) {
 79 				$("select, object").css("visibility", "hidden");
 80 			};
 81 		},
 82 		off: function() {
 83 			$dimmer.fadeOut("normal", function(){
 84 				$dimmer.detach();
 85 				if ($("html").hasClass("ie6")) {
 86 					$("select, object").css("visibility", "visible");
 87 				};
 88 			});
 89 		}
 90 	};
 91 
 92 /**
 93 *	Protected Members
 94 */ 
 95 
 96 	that.$trigger = that.$element;
 97 
 98 	/**
 99 	* Inner show method. Attach the component's layout to the DOM tree and load defined content.
100 	* @protected
101 	* @name ch.Modal#innerShow
102 	* @function
103 	* @returns {itself}
104 	*/ 
105 	that.innerShow = function(event) {	
106 		dimmer.on();
107 		that.parent.innerShow(event);		
108 		that.$trigger.blur();
109 		return that;
110 	};
111 	
112 	/**
113 	* Inner hide method. Hides the component's layout and detach it from DOM tree.
114 	* @protected
115 	* @name ch.Modal#innerHide
116 	* @function
117 	* @returns {itself}
118 	*/	
119 	that.innerHide = function(event) {
120 		dimmer.off();		
121 		that.parent.innerHide(event);
122 		return that;
123 	};

Or just use a String with HTML code.

250 		.css("cursor", "pointer")

272 	*	otherComponent.show();

275 	*/

281 	* @public

284 	* me.on("ready",function(){
285 	*	this.show();
286 	* });
287 	*/
288 	that.trigger("ready");
289 
290 	return that;
291 };

294 /**
295 * Transition
296 * @name Transition
297 * @class Transition
298 * @augments ch.Modal
299 * @memberOf ch
300 * @returns {itself}
301 */
302 ch.extend("modal").as("transition", function(conf) {
303 	conf.closeButton = false;
304 	conf.msg = conf.msg || conf.content || "Please wait...";
305 	conf.content = $("<div>")
306 		.addClass("loading")
307 		.after( $("<p>").html(conf.msg) );
308 	return conf;
309 });
