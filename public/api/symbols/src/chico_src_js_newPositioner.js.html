385 		/**
386 		 * Calculates left and top position from specific points.
387 		 * @function
388 		 * @name getCoordinates
389 		 * @memberOf ch.newPositioner
390 		 * @param {Points} points String with points to be calculated.
391 		 * @returns {Offset measures}
392 		 * @example
393 		 * var foo = getCoordinates("lt rt");
394 		 * 
395 		 * foo = {
396 		 *     left: Number,
397 		 *     top: Number
398 		 * };
399 		 */
400 			getCoordinates = function (points) {
401 				
402 				/**
403 				 * Calculates left or top position from points related to specific axis (X or Y).
404 				 * @function
405 				 * @memberOf getCoordinates
406 				 * @param {Axis Points} reference String with points of an axis.
407 				 * @returns {Position value}
408 				 * @example
409 				 * // Calculates X axis measure. It returns a left position value.
410 				 * _calculate("lr");
411 				 * @example
412 				 * // Calculates Y axis measure. It returns a top position value.
413 				 * _calculate("tt");
414 				 */
415 				var calculate = function (reference) {
416 					
417 					var r;
418 					
419 					switch (reference) {
420 					// X references
421 					// TODO: lc, rl, rc, cl, cr
422 					case "ll": r = context.left; break;
423 					case "lr": r = context.left + context.width; break;
424 					case "rr": r = context.left + context.width - $element.outerWidth(); break;
425 					case "cc": r = context.left + context.width / 2 - $element.outerWidth() / 2; break;
426 					// Y references
427 					// TODO: tm, bb, bm, mt, mb
428 					case "tt": r = context.top; break;
429 					case "tb": r = context.top + context.height; break;
430 					case "bt": r = context.top - $element.outerHeight(); break;
431 					case "mm": r = context.top + context.height / 2 - $element.outerHeight() / 2; break;
432 					}
433 					
434 					return r;
435 				},
436 				
437 				// Splitted points
438 					splittedPoints = points.split(" ");
439 				
440 				// Calculates left and top with references to X and Y axis points
441 				return {
442 					left: calculate(splittedPoints[0].charAt(0) + splittedPoints[1].charAt(0)),
443 					top: calculate(splittedPoints[0].charAt(1) + splittedPoints[1].charAt(1))
444 				};
445 			},

457 		 * Gets new coordinates and checks it's space into viewport.
458 		 * @function
459 		 * @name getPosition
460 		 * @memberOf ch.newPositioner
461 		 * @returns {Offset measures}
462 		 */
463 			getPosition = function () {
464 				
465 				// Gets coordinates from main points
466 				var coordinates = getCoordinates(points);
467 				
468 				// Default behavior: returns left and top offset related to main points
469 				if (!conf.reposition) {
470 					return {
471 						left: coordinates.left,
472 						top: coordinates.top
473 					};
474 				}
475 				
476 				// Intelligence
477 				// TODO: Unificar los 3 casos de inteligencia en uno (solo cambian las 2 condiciones)
478 				var newPoints, newCoordinates;
479 				
480 				// Elements positioned at bottom without space in viewport (Element bottom > Viewport bottom)
481 				// TODO: If points are in CORRELATION_MAP and last point is "b", then do it
482 				if (
483 					(points === "lt lb" || points === "rt rb") &&
484 						(coordinates.top + offset[1] + $element.outerHeight() > ch.viewport.height || repositioned)
485 				) {
486 					newPoints = repositioned || CORRELATION_MAP[points];
487 					newCoordinates = getCoordinates(newPoints);
488 					
489 					// Elements positioned at top without space in viewport (Element top < Viewport top)
490 					if (newCoordinates.top + offset[1] > ch.viewport.top) {
491 						coordinates = newCoordinates;
492 						repositioned = (repositioned) ? null : points;
493 						points = newPoints;
494 					}
495 				}
496 				
497 				if (
498 					(points === "lb lt" || points === "rb rt") &&
499 						(coordinates.top + offset[1] < ch.viewport.top || repositioned)
500 				) {
501 					newPoints = repositioned || CORRELATION_MAP[points];
502 					newCoordinates = getCoordinates(newPoints);
503 					
504 					// Elements positioned at top without space in viewport (Element top < Viewport top)
505 					if (newCoordinates.top + offset[1] + $element.outerHeight() < ch.viewport.height) {
506 						coordinates = newCoordinates;
507 						repositioned = (repositioned) ? null : points;
508 						points = newPoints;
509 					}
510 				}
511 				
512 				// Elements positioned at right without space in viewport (Element right > Viewport right)
513 				if (
514 					(points === "lt rt") &&
515 						(coordinates.left + offset[0] + $element.outerWidth() > ch.viewport.width || repositioned)
516 				) {
517 					newPoints = repositioned || CORRELATION_MAP[points];
518 					newCoordinates = getCoordinates(newPoints);
519 					
520 					// Elements positioned at top without space in viewport (Element top < Viewport top)
521 					if (newCoordinates.left + offset[0] > ch.viewport.left) {
522 						coordinates = newCoordinates;
523 						repositioned = (repositioned) ? null : points;
524 						points = newPoints;
525 					}
526 				}
527 				
528 				// Returns left and top offset related to modified points
529 				return {
530 					left: coordinates.left,
531 					top: coordinates.top
532 				};
533 			},

632 		/**
633 		 * Control panel that allows change configuration properties, refresh current position or get current configuration.
634 		 * @name position
635 		 * @function
636 		 * @param {Object} [o] Configuration object
637 		 * @param {String} ["refresh"] Refresh current position
638 		 * @memberOf ch.newPositioner
639 		 * @returns {Control Panel Object}
640 		 * @example
641 		 * // Sets a new configuration
642 		 * var foo = ch.positioner({ ... });
643 		 *     foo.position({ ... });
644 		 * @example
645 		 * // Refresh current position
646 		 *     foo.position("refresh");
647 		 * @example
648 		 * // Gets current configuration
649 		 *     foo.position();
650 		 */
651 		that.position = function (o) {
652 			
653 			var r;
654 			
655 			switch (typeof o) {
656 			
657 			// Changes configuration properties and re-positions element
658 			case "object":
659 				// New points
660 				if (ch.utils.hasOwn(o, "points")) { points = o.points; }
661 				
662 				// New reposition
663 				if (ch.utils.hasOwn(o, "reposition")) { conf.reposition = o.reposition; }
664 				
665 				// New offset (splitted)
666 				if (ch.utils.hasOwn(o, "offset")) { offset = o.offset.split(" "); }
667 				
668 				// New context
669 				if (ch.utils.hasOwn(o, "context")) {
670 					// Sets conf value
671 					conf.context = o.context;
672 					
673 					// Re-generate context object
674 					context = getContext();
675 				}
676 				
677 				// Reset
678 				init();
679 				
680 				r = that;
681 				
682 				break;
683 			
684 			// Refresh current position
685 			case "string":
686 				if (o !== "refresh") {
687 					alert("Chico-UI error: expected to find \"refresh\" parameter on position() method.");
688 				}
689 				
690 				// Reset
691 				init();
692 				
693 				r = that;
694 				
695 				break;
696 			
697 			// Gets current configuration
698 			case "undefined":
699 				r = {
700 					context: context.element,
