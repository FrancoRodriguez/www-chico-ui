  2 /**
  3  * Object represent the abstract class of all UI Objects.
  4  * @abstract
  5  * @name Object
  6  * @class Object 
  7  * @memberOf ch
  8  * @see ch.Controllers
  9  * @see ch.Floats
 10  * @see ch.Navs
 11  * @see ch.Watcher
 12  */
 13  
 14 ch.object = function(){
 15 	
 16     /**
 17      * Reference to a internal component instance, saves all the information and configuration properties.
 18      * @private
 19      * @name that
 20      * @type {Object}
 21      * @memberOf ch.Object
 22      */ 
 23 	var that = this;	
 24 	var conf = that.conf;
 25 		
 26 /**
 27  *  Public Members
 28  */
 29  
 30 
 31    /**
 32     * Component's static content.
 33     * @public
 34     * @name staticContent
 35 	* @type {String}
 36     * @memberOf ch.Object
 37     */ 
 38     that.staticContent;
 39     
 40    /**
 41     * DOM Parent of content, this is useful to attach DOM Content when float is hidding.
 42     * @public
 43     * @name DOMParent
 44     * @type {HTMLElement}
 45     * @memberOf ch.Object
 46     */ 
 47     that.DOMParent;
 48 
 49    /**
 50     * Flag to know if the DOM Content is visible or not.
 51     * @public
 52     * @name DOMContentIsVisible
 53     * @type {Boolean}
 54     * @memberOf ch.Object
 55     */ 
 56     that.DOMContentIsVisible;
 57 
 58    /**
 59     * Prevent propagation and default actions.
 60     * @name prevent
 61     * @function
 62     * @param {EventObject} event Recieves a event object
 63     * @memberOf ch.Object
 64     */
 65 	that.prevent = function(event) {
 66 	   
 67 		if (event && typeof event == "object") {
 68 		    event.preventDefault();
 69 			event.stopPropagation();
 70 		};
 71 		
 72 		return that;
 73 	};
 74 
 75 
 76    /**
 77     * Set and get the content of a component
 78     * @name content
 79     * @function
 80     * @param {String} [content] Could be a simple text, html or a url to get the content with ajax.
 81     * @returns {String} content
 82     * @memberOf ch.Object
 83     * @requires ch.Cache
 84     * @example
 85     * // Simple static content
 86     * $(element).layer("Some static content");
 87     * @example
 88     * // Get DOM content
 89     * $(element).layer("#hiddenContent");
 90     * @example
 91     * // Get AJAX content
 92 	* $(element).layer("http://chico.com/content/layer.html");
 93     */
 94     that.content = function(content) {
 95 
 96         var _get = (content) ? false : true,
 97 
 98             // Local argument
 99             content = content,
100             // Local isURL
101             sourceIsUrl = ch.utils.isUrl(that.source),
102             // Local isSelector
103             sourceIsSelector = ch.utils.isSelector(that.source),
104             // Get context, could be a single component or a controller
105             context = ( ch.utils.hasOwn(that, "controller") ) ? that.controller : that["public"],
106             // undefined, for comparison.
107             undefined,
108             // Save cache configuration
109             cache = ( ch.utils.hasOwn(conf, "cache") ) ? conf.cache : true;
110 
111     /**
112      * Get content
113      */
114 		// return defined content
115 		if ( _get ) {
116 
117 			// no source, no content
118             if ( that.source === undefined ) {
119                 return "<p>No content defined for this component</p>";    
120             }
121 
122             // Get data from cache for DOMContent or AJAXContent
123             if ( cache && ( sourceIsSelector || sourceIsUrl )) {
124                 var fromCache = ch.cache.get(that.source);
125                 if (fromCache) {
126                     return fromCache;
127                 }
128             }
129             
130             // First time we need to get the contemt.
131             // Is cache is off, go and get content again.
132             // Yeap, recursive.
133             if ( !cache || that.staticContent === undefined ) {
134             	return that.content(that.source);
135             }
136 
137             // Flag to remove DOM content and avoid ID duplication
138             if ( sourceIsSelector ) {
139             	$(that.source).detach();
140             }
141             
142 			// get at last
143             return that.staticContent;
144 
145 		}
146 
147     /**
148      * Set content
149      */	
150 
151     // Reset cache to overwrite content
152 	conf.cache = false;
153 
154 		// Local isURL
155 	var isUrl = ch.utils.isUrl(content),
156 		// Local isSelector
157 		isSelector = ch.utils.isSelector(content);
158 
159     /* Evaluate static content */  
160 
161         // Set 'that.staticContent' and overwrite 'that.source'
162 		// just in case you want to update DOM or AJAX Content.
163 		that.staticContent = that.source = content;
164 
165     /* Evaluate DOM content */
166 
167         if (isSelector) {
168 			
169 			// Save DOMParent, so we know where to re-insert the content.
170             that.DOMParent = $(content).parent();
171             // Check if content was visible or not.
172             that.DOMContentIsVisible = $(content).is(":visible")
173 
174             // Return DOM content, remove it from DOM to avoid ID duplications
175 			that.staticContent = $(content).removeClass("ch-hide").remove();
176 			
177 			// Save new data to the cache
178             if (cache) {
179             	ch.cache.set(that.source, that.staticContent);
180             }       
181         }
182 
183         // trigger onContentLoad callback for DOM and Static,
184         // Avoid trigger this callback on AJAX requests.
185 		if ( ch.utils.hasOwn(conf, "onContentLoad") && !isUrl) { conf.onContentLoad.call( context ); }
186 
187     /* Evaluate AJAX content */  
188 
189         if (isUrl) {
190             
191             // First check Cache
192             // Check if this source is in our cache
193             if ( cache ) {
194                 var fromCache = ch.cache.get(that.source);
195                 if (fromCache) {
196                     return fromCache;
197                 }
198             }
199 
200             var _method, _serialized, _params = "x=x";
201             
202 			// If the trigger is a form button, serialize its parent to send data to the server.
203 			if (that.$element.attr('type') == 'submit') {
204 				_method = that.$element.parents('form').attr('method') || 'GET';
205 				_serialized = that.$element.parents('form').serialize();
206 				_params = _params + ((_serialized != '') ? '&' + _serialized : '');
207 			};
208 			
209 			$.ajax({
210 				url: that.source,
211 				type: _method || 'GET',
212 				data: _params,
213 				// each component could have a different cache configuration
214 				cache: cache,
215 				async: true,
216 				beforeSend: function(jqXHR){
217 					// Ajax default HTTP headers
218 					jqXHR.setRequestHeader("X-Requested-With", "XMLHttpRequest");
219 				},
220 				success: function(data, textStatus, jqXHR){
221 					// TODO: It would be nice to re-use the onContentLoad callback.
222                     that.contentCallback.call(that,data);
223                     // Callback your way out
224 					if (ch.utils.hasOwn(conf, "onContentLoad")) {
225 					   conf.onContentLoad.call(context, data, textStatus, jqXHR);
226 					}
227 					// Save new data to the cache
228                     if (cache) {
229                         ch.cache.set(that.source,data);
230                     }
231 				},
232 				error: function(jqXHR, textStatus, errorThrown){
233 					// TODO: It would be nice to re-use the onContentError callback.                    
234                     that.contentCallback.call(that,"<p>Error on ajax call </p>");
235                     // Callback your way out                    
236 					if (ch.utils.hasOwn(conf, "onContentError")) {
237 					   conf.onContentError.call(context, jqXHR, textStatus, errorThrown)
238 					}
239 				}
240 			});
241 
242             // Return Spinner and wait for callbacks
243 			that.staticContent = '<div class="loading"></div>';
244 
245         }
246 
247      /* Set previous cache configuration */
248 
249 		conf.cache = cache;
250 
251      /* Finally return 'staticContent' */
252 		
253 		// Update Content
254 		that.contentCallback.call(that,that.staticContent);
255 
256 		return that.staticContent;
257     };
258 
259    /**
260     * Executes a specific callback
261     * @name callbacks
262     * @function
263     * @memberOf ch.Object
264     */
265     // TODO: Add examples!!!
266 	that.callbacks = function(when) {
267 		if( ch.utils.hasOwn(conf, when) ) {
268 			var context = ( that.controller ) ? that.controller["public"] : that["public"];
269 			return conf[when].call( context );
270 		};
271 	};
272 
273    /**
274     * Change components position
275     * @name position
276     * @function
277     * @param {Object} [o] Configuration object
278     * @param {String} ["refresh"] Refresh
279     * @memberOf ch.Object
280     * @returns {Object} o Configuration object is arguments are empty
281     */	
282     // TODO: Add examples!!!
283 	that.position = function(o) {
284 	
285 		switch(typeof o) {
286 		 
287 			case "object":
288 				conf.position.context = o.context || conf.position.context;
289 				conf.position.points = o.points || conf.position.points;
290 				conf.position.offset = o.offset || conf.position.offset;				
291 				conf.position.fixed = o.fixed || conf.position.fixed;
292 			
293 				ch.positioner(conf.position);
294 				return that["public"];
295 				break;
296 		
297 			case "string":
298 				if(o != "refresh") {
299 					alert("Chico UI error: position() expected to find \"refresh\" parameter.");
300 				};
301 
302 				ch.positioner(conf.position);
303 
304 				return that["public"];   			
305 				break;
306 		
307 			case "undefined":
308 				return conf.position;
309 			    break;
310 		};
311 		
312 	};
313 	
314 
315  	that["public"] = {};
316     /**
317      * The component's instance unique identifier.
318      * @public
319      * @name uid
320      * @type {Number}
321      * @memberOf ch.Object
322      */
323    	that["public"].uid = that.uid;
324     /**
325      * The element reference.
326      * @public
327      * @name element
328      * @type {HTMLElement}
329      * @memberOf ch.Object
330      */
331 	that["public"].element = that.element;
332     /**
333      * The component's type.
334      * @public
335      * @name type
336      * @type {String}
337      * @memberOf ch.Object
338      */
339 	that["public"].type = that.type;
340     
341     /**
342      * Positioning configuration.
343      * @public
344      * @name position
345      * @memberOf ch.Object
346      */
347 	that["public"].position = that.position;
348     /**
349      * Positioning configuration.
350      * @public
351      * @function
352      * @name content
353      * @memberOf ch.Object
354      */
355 	that["public"].content = that.content;
356 	
357 	return that;
358 };
