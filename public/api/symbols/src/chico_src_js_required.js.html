  2 /**
  3 * Creates required validations.
  4 * @name Required
  5 * @class Required
  6 * @interface
  7 * @augments ch.Watcher
  8 * @memberOf ch
  9 * @param {Object} conf Object with configuration properties
 10 * @returns {itself}
 11 * @see ch.Watcher

 13 * @see ch.Number
 14 * @see ch.String
 15 * @example
 16 * // Simple validation
 17 * $("input").required("This field is required");
 18 */
 19 
 20 ch.extend("watcher").as("required", function(conf) {
 21 
 22 	// Define the validation interface    
 23 	conf.required = true;
 24 	// Define the conditions of this interface
 25 	conf.conditions = [{
 26 		name: "required"
 27 	}];
 28 
 29 	return conf;
 30 
 31 });
