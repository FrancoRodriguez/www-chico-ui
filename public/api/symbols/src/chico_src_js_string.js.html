  2 /**
  3 * Validate strings.
  4 * @name String 
  5 * @class String
  6 * @interface
  7 * @augments ch.Watcher
  8 * @memberOf ch
  9 * @param {String} msg Validation message
 10 * @returns {Chico-UI Object}
 11 * @see ch.Watcher
 12 * @see ch.Required
 13 * @see ch.Custom
 14 * @see ch.Number
 15 * @example
 16 * // Create a string validation
 17 * $("input").string("This field must be a string.");
 18 */
 19 
 20 ch.extend("watcher").as("string", function (conf) {
 21 
 22 	// $.string("message"); support
 23 	if ( !conf.text && !conf.email && !conf.url && !conf.maxLength && !conf.minLength ) {
 24 		conf.text = true;
 25 	};
 26 	
 27 	/**
 28 	* @public
 29 	* @name ch.String#conditions
 30 	* @type {Map}
 31 	*/
 32 	// Define the conditions of this interface
 33 	conf.conditions = [{
 34 			name: "text", 
 35 			patt: /^([a-zA-Z\s]+)$/
 36 		},{
 37 			name:"email",
 38 			patt: /^(("[\w-\s]+")|([\w-]+(?:\.[\w-]+)*)|("[\w-\s]+")([\w-]+(?:\.[\w-]+)*))(@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$)|(@\[?((25[0-5]\.|2[0-4][0-9]\.|1[0-9]{2}\.|[0-9]{1,2}\.))((25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\.){2}(25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\]?$)/ 
 39 		},{
 40 			name: "url",
 41 			patt: /^((https?|ftp|file):\/\/|((www|ftp)\.)|(\/|.*\/)*)[a-z0-9-]+((\.|\/)[a-z0-9-]+)+([/?].*)?$/ 
 42 		},{
 43 			name: "minLength",
 44 			expr: function(a,b) { return a.length >= b } 
 45 		},{
 46 			name: "maxLength",
 47 			expr: function(a,b) { return a.length <= b } 
 48 		}];
 49 
 50 	return conf;
 51 
 52 });
