  2 /**
  3  * TabNavigator UI-Component for static and dinamic content.
  4  * @name TabNavigator
  5  * @class TabNavigator
  6  * @augments ch.Controllers
  7  * @memberOf ch
  8  * @param {Configuration Object} conf Object with configuration properties
  9  * @returns {Chico-UI Object}
 10  */
 11 
 12 ch.tabNavigator = function(conf){
 13 
 14     /**
 15      * Reference to a internal component instance, saves all the information and configuration properties.
 16      * @private
 17      * @name that
 18      * @type {Object}
 19      * @memberOf ch.TabNavigator
 20      */
 21 	var that = this;
 22 
 23 	conf = ch.clon(conf);
 24 
 25 	that.conf = conf;
 26 	
 27 /**
 28  *	Inheritance
 29  */
 30 
 31     that = ch.controllers.call(that);
 32     that.parent = ch.clon(that);
 33 
 34 /**
 35  *  Private Members
 36  */
 37 
 38     /**
 39      * The actual location hash, is used to know if there's a specific tab selected.
 40      * @private
 41      * @name hash
 42      * @type {String}
 43      * @memberOf ch.TabNavigator
 44      */
 45 	var hash = window.location.hash.replace("#!", "");
 46     /**
 47      * A boolean property to know if the some tag should be selected.
 48      * @private
 49      * @name hashed
 50      * @type {Boolean}
 51      * @default false
 52      * @memberOf ch.TabNavigator
 53      */
 54     var hashed = false;
 55     /**
 56      * Get wich tab is selected.
 57      * @private
 58      * @name selected
 59      * @type {Number}
 60      * @memberOf ch.TabNavigator
 61      */
 62     var selected = conf.selected - 1 || conf.value - 1 || 0;
 63     /**
 64      * Create controller's children.
 65      * @private
 66      * @name createTabs
 67      * @function
 68      * @memberOf ch.TabNavigator
 69      */
 70 	var createTabs = function(){
 71 
 72 		// Children
 73 		that.$element.children().eq(0).addClass("ch-tabNavigator-triggers").find("a").each(function(i, e){
 74 
 75 			// Tab context
 76 			var tab = {};
 77 				tab.uid = that.uid + "#" + i;
 78 				tab.type = "tab";
 79 				tab.element = e;
 80 				tab.$element = $(e);
 81 				tab.controller = that["public"];
 82 
 83 			// Tab configuration
 84 			var conf = {};
 85 				conf.open = (selected == i);
 86 				conf.onShow = function(){
 87 					selected = i;
 88 				};
 89 
 90            /**
 91             * Callback function
 92             * @name onContentLoad
 93             * @type {Function}
 94             * @memberOf ch.TabNavigator
 95             */
 96 			if ( ch.utils.hasOwn(that.conf, "onContentLoad") ) conf.onContentLoad = that.conf.onContentLoad;
 97            /**
 98             * Callback function
 99             * @name onContentError
100             * @type {Function}
101             * @memberOf ch.TabNavigator
102             */
103 			if ( ch.utils.hasOwn(that.conf, "onContentError") ) conf.onContentError = that.conf.onContentError;
104 
105 			// Create Tabs
106 			that.children.push(
107 				ch.tab.call(tab, conf)
108 			);
109 
110 			// Bind new click to have control
111 			$(e).unbind("click").bind("click", function(event){
112 				that.prevent(event);
113 				select(i + 1);
114 			});
115 
116 		});
117 
118 		return;
119 
120 	};
121     /**
122      * Select a child to show its content.
123      * @private
124      * @function
125      * @memberOf ch.TabNavigator
126      */
127 	var select = function(tab){
128 
129 		tab = that.children[tab - 1];
130 		
131 		if(tab === that.children[selected]) return; // Don't click me if I'm open
132 
133 		// Hide my open bro
134 		$.each(that.children, function(i, e){
135 			if(tab !== e) e.hide();
136 		});
137 
138 		tab.show();
139 
140         //Change location hash
141 		window.location.hash = "#!" + tab.$content.attr("id");	
142 		
143        /**
144         * Callback function
145         * @name onSelect
146         * @type {Function}
147         * @memberOf ch.TabNavigator
148         */
149 		that.callbacks("onSelect");
150 		
151         return that;
152 	};
153 
154 /**
155  *  Protected Members
156  */ 
157     /**
158      * The component's content.
159      * @private
160      * @name $content
161      * @type {jQuery Object}
162      * @memberOf ch.TabNavigator
163      */
164 	that.$content = that.$element.children().eq(1).addClass("ch-tabNavigator-content box");
165 
166     
167 /**
168  *  Public Members
169  */
170 
171     /**
172      * The component's instance unique identifier.
173      * @public
174      * @name uid
175      * @type {Number}
176      * @memberOf ch.TabNavigator
177      */
178 	that["public"].uid = that.uid;
179     /**
180      * The element reference.
181      * @public
182      * @name element
183      * @type {HTMLElement}
184      * @memberOf ch.TabNavigator
185      */
186 	that["public"].element = that.element;
187     /**
188      * The component's type.
189      * @public
190      * @name type
191      * @type {String}
192      * @memberOf ch.TabNavigator
193      */
194 	that["public"].type = that.type;
195     /**
196      * Children instances associated to this controller.
197      * @public
198      * @name children
199      * @type {Collection}
200      * @memberOf ch.TabNavigator
201      */
202 	that["public"].children = that.children;
203     /**
204      * Select a specific child.
205      * @public
206      * @function
207      * @name select
208      * @param {Number} tab Tab's index.
209      * @memberOf ch.TabNavigator
210      */
211 	that["public"].select = function(tab){
212 		select(tab);
213 		
214 		return that["public"];
215 	};
216     /**
217      * Returns the selected child's index.
218      * @public
219      * @function
220      * @name getSelected
221      * @returns {Number} selected Tab's index.
222      * @memberOf ch.TabNavigator
223      */	
224 	that["public"].getSelected = function(){ return (selected + 1); };
225 
226 /**
227  *  Default event delegation
228  */	
229 
230     	that.$element.addClass('ch-tabNavigator');
231 
232 	createTabs();
233 
234 	//Default: Load hash tab or Open first tab	
235 	for(var i = that.children.length; i--; ){
236 		if ( that.children[i].$content.attr("id") === hash ) {
237 			select(i + 1);
238 			
239 			hashed = true;
240 			
241 			break;
242 		};
243 	};
244 
245 	return that;
246 	
247 };

251 /**
252  * Simple unit of content for TabNavigators.
253  * @abstract
254  * @name Tab
255  * @class Tab
256  * @augments ch.Navs
257  * @memberOf ch
258  * @param {Configuration Object} conf Object with configuration properties
259  * @returns {Chico-UI Object}
260  */
261 
262 ch.tab = function(conf){
263     /**
264      * Reference to a internal component instance, saves all the information and configuration properties.
265      * @private
266      * @name that
267      * @type {Object}
268      * @memberOf ch.Tab
269      */
270     var that = this;
271 
272 	conf = ch.clon(conf);
273 	conf.icon = false;
274 	
275 	that.conf = conf;
276 
277 	
278 /**
279  *	Inheritance
280  */
281 
282 	that = ch.navs.call(that);
283 	that.parent = ch.clon(that);
284 
285 /**
286  *  Private Members
287  */
288     /**
289      * Creates the basic structure for the tab's content.
290      * @private
291      * @name createContent
292      * @function
293      * @memberOf ch.TabNavigator
294      */
295 	var createContent = function(){
296 		var href = that.element.href.split("#");
297 		var controller = that.$element.parents(".ch-tabNavigator");
298 		var content = controller.find("#" + href[1]);
299 		
300 		// If there are a tabContent...
301 		if ( content.length > 0 ) {
302 			
303 			return content;
304 		
305 		// If tabContent doesn't exists        
306 		} else {
307 			// Set ajax configuration
308 			conf.ajax = true;
309 
310 			// Create tabContent
311 			return $("<div>")
312 				.attr("id", (href.length == 2) ? href[1] : "ch-tab" + that.uid.replace("#","-") )
313 				.addClass("ch-hide")
314 				.appendTo( controller.children().eq(1) );
315 		};
316 
317 	};
318 
319 /**
320  *  Protected Members
321  */ 
322     /**
323      * Reference to the trigger element.
324      * @private
325      * @name $trigger
326      * @type {jQuery Object}
327      * @memberOf ch.Tab
328      */
329 	that.$trigger = that.$element;
330 
331     /**
332      * The component's content.
333      * @private
334      * @name $content
335      * @type {jQuery Object}
336      * @memberOf ch.Tab
337      */	
338 	that.$content = createContent();
339 
340     /**
341      * Process the show event.
342      * @private
343      * @function
344      * @name show
345      * @returns {jQuery Object}
346      * @memberOf ch.Tab
347      */ 
348 	that.show = function(event){
349 		that.prevent(event);
350 
351 		// Load my content if I'need an ajax request 
352 		if( that.$content.html() == "" ) that.$content.html( that.loadContent() );
353 
354 		// Show me
355 		that.parent.show(event);
356 		
357 		return that;
358 	};
359 
360 /**
361  *  Public Members
362  */
363 	
364 	
365 /**
366  *  Default event delegation
367  */
368 
369 	that.configBehavior();
370 
371 	
372 	return that;
373 }
