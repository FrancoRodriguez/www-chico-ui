1 /**
  2  * Viewport is a reference to positions and sizes of visible area into browser.
  3  * @abstract
  4  * @name Viewport
  5  * @class Viewport
  6  * @memberOf ch
  7  * @returns {Viewport Object}
  8  */
  9 ch.viewport = (function () {
 10 	
 11 	// Object to be returned
 12 	var self = {};
 13 	
 14 	/**
 15 	* Width of viewport.
 16 	* @public
 17 	* @name width
 18 	* @type {Number}
 19 	* @memberOf ch.Viewport
 20 	*/
 21 	self.width =
 22 	
 23 	/**
 24 	* Height of viewport.
 25 	* @public
 26 	* @name height
 27 	* @type {Number}
 28 	* @memberOf ch.Viewport
 29 	*/
 30 		self.height =
 31 	
 32 	/**
 33 	* Left offset of viewport.
 34 	* @public
 35 	* @name left
 36 	* @type {Number}
 37 	* @memberOf ch.Viewport
 38 	*/
 39 		self.left =
 40 	
 41 	/**
 42 	* Top offset of viewport.
 43 	* @public
 44 	* @name top
 45 	* @type {Number}
 46 	* @memberOf ch.Viewport
 47 	*/
 48 		self.top = 0;
 49 	
 50 	// Main browsers
 51 	if (typeof window.innerWidth !== "undefined") {
 52 		
 53 		/**
 54 	     * Viewport HTML Element.
 55 	     * @public
 56 	     * @name element
 57 	     * @type {HTMLElement}
 58 	     * @memberOf ch.Viewport
 59 	     */
 60 		self.element = window;
 61 		
 62 		/**
 63 	     * Re-calculates width and height of viewport and updates ch.viewport.width and ch.viewport.height.
 64 	     * @public
 65 	     * @function
 66 	     * @name getSize
 67 	     * @returns {Size Object}
 68 	     * @memberOf ch.Viewport
 69 	     */
 70 		self.getSize = function () {
 71 			return {
 72 				width: ch.viewport.width = self.element.innerWidth,
 73 				height: ch.viewport.height = self.element.innerHeight
 74 			};
 75 		};
 76 		
 77 		/**
 78 	     * Re-calculates left and top of viewport and updates ch.viewport.left and ch.viewport.top.
 79 	     * @public
 80 	     * @function
 81 	     * @name getOffset
 82 	     * @returns {Offset Object}
 83 	     * @memberOf ch.Viewport
 84 	     */
 85 		self.getOffset = function () {
 86 			return {
 87 				left: ch.viewport.left = self.element.pageXOffset,
 88 				top: ch.viewport.top = self.element.pageYOffset
 89 			};
 90 		};
 91 	
 92 	// Fallback
 93 	} else {
 94 		
 95 		/**
 96 	     * Viewport HTML Element.
 97 	     * @public
 98 	     * @type {HTMLElement}
 99 	     * @memberOf ch.Viewport
100 	     */
101 		self.element = document.documentElement;
102 		
103 		/**
104 	     * Re-calculates width and height of viewport and updates ch.viewport.width and ch.viewport.height.
105 	     * @public
106 	     * @function
107 	     * @returns {Size Object}
108 	     * @memberOf ch.Viewport
109 	     */
110 		self.getSize = function () {
111 			return {
112 				width: ch.viewport.width = self.element.clientWidth,
113 				height: ch.viewport.height = self.element.clientHeight
114 			};
115 		};
116 		
117 		/**
118 	     * Re-calculates left and top of viewport and updates ch.viewport.left and ch.viewport.top.
119 	     * @public
120 	     * @function
121 	     * @returns {Offset Object}
122 	     * @memberOf ch.Viewport
123 	     */
124 		self.getOffset = function () {
125 			return {
126 				left: self.element.scrollLeft,
127 				top: self.element.scrollTop
128 			};
129 		};
130 	}
131 	
132 	return self;
133 }());
134 
135 // Calculates size and position
136 ch.viewport.getSize();
137 ch.viewport.getOffset();
