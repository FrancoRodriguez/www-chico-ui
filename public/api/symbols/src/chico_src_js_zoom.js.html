  2 /**
  3 * Zoom is a standalone UI component that shows a contextual reference to an augmented version of main declared image.
  4 * @name Zoom
  5 * @class Zoom
  6 * @augments ch.Floats
  7 * @requires ch.Positioner
  8 * @requires ch.onImagesLoads
  9 * @memberOf ch
 10 * @param {Object} conf Object with configuration properties
 11 * @returns {self}
 12 */
 13 
 14 ch.zoom = function(conf) {
 15 	/**
 16 	* Reference to an internal component instance, saves all the information and configuration properties.
 17 	* @private
 18 	* @name ch.Zoom#that
 19 	* @type {Object}
 20 	*/
 21 	var that = this;
 22 
 23 	conf = ch.clon(conf);
 24 	conf.fx = false;
 25 	
 26 	conf.position = {};
 27 	conf.position.context = conf.context || that.$element;
 28 	conf.position.offset = conf.offset || "20 0";
 29 	conf.position.points = conf.points || "lt rt";
 30 	conf.position.hold = true;
 31 
 32 	that.conf = conf;
 33 
 34 /**
 35 *	Inheritance
 36 */
 37 
 38 	that = ch.floats.call(that);
 39 	that.parent = ch.clon(that);
 40 	
 41 /**
 42 *	Private Members

 44 
 45 	/**
 46 	* Reference to main image declared on HTML code snippet.
 47 	* @private
 48 	* @name ch.Zoom#original
 49 	* @type {Object}
 50 	*/
 51 	var original = {};
 52 		original.img = that.$element.children();
 53 		original["width"] = original.img.prop("width");
 54 		original["height"] = original.img.prop("height");
 55 
 56 	/**
 57 	* Reference to the augmented version of image, that will be displayed in context.
 58 	* @private
 59 	* @name ch.Zoom#zoomed
 60 	* @type {Object}
 61 	*/
 62 	var zoomed = {};
 63 		// Define the content source 
 64 		zoomed.img = that.source = $("<img>").prop("src", that.element.href);
 65 	
 66 	/**
 67 	* Seeker is the visual element that follows mouse movement for referencing to zoomable area into original image.
 68 	* @private
 69 	* @name ch.Zoom#seeker
 70 	* @type {Object}
 71 	*/
 72 	var seeker = {};
 73 		seeker.shape = $("<div>").addClass("ch-seeker ch-hide")
 74 	
 75 	/**
 76 	* Gets the mouse position relative to original image position, and accordingly moves the zoomed image.
 77 	* @private
 78 	* @function
 79 	* @name ch.Zoom#move
 80 	* @param {Event Object} event
 81 	*/
 82 	var move = function(event){
 83 		
 84 		// Cursor coordinates relatives to original image
 85 		var x = event.pageX - original.offset.left;
 86 		var y = event.pageY - original.offset.top;
 87 		
 88 		// Seeker axis
 89 		var limit = {};
 90 			limit.left = parseInt(x - seeker["width"]);
 91 			limit.right = parseInt(x + seeker["width"]);
 92 			limit.top = parseInt(y - seeker["height"]);
 93 			limit.bottom = parseInt(y + seeker["height"]);
 94 
 95 		// Horizontal: keep seeker into limits
 96 		if(limit.left >= 0 && limit.right < original["width"] - 1) {
 97 			zoomed.img.css("left", -( (parseInt(zoomed["width"]* x) / original["width"]) - (conf.width / 2) ));
 98 			seeker.shape.css("left", limit.left);
 99 		};
100 		
101 		// Vertical: keep seeker into limits
102 		if(limit.top >= 0 && limit.bottom < original["height"] - 1) {
103 			zoomed.img.css("top", -( (parseInt(zoomed["height"]* y) / original["height"]) - (conf.height / 2) ));
104 			seeker.shape.css("top", limit.top);
105 		};
106 		
107 	};
108 
109 	/**
110 	* Calculates zoomed image sizes and adds event listeners to trigger of float element
111 	* @private
112 	* @function
113 	* @name ch.Zoom#init
114 	*/
115 	var init = function(){
116 		// Zoomed image size
117 		zoomed["width"] = zoomed.img.prop("width");
118 		zoomed["height"] = zoomed.img.prop("height");
119 		
120 		// Anchor
121 		that.$element
122 			// Apend Seeker
123 			.append( seeker.shape )
124 			
125 			// Show
126 			.bind("mouseenter", that.show)
127 			
128 			// Hide
129 			.bind("mouseleave", that.hide)
130 	};

132 /**

135 
136 	/**
137 	* Anchor that wraps the main image and links to zoomed image file.
138 	* @public
139 	* @name ch.Zoom#$trigger
140 	* @type {Object}
141 	*/
142 	that.$trigger = that.$element;
143 
144 	that.innerShow = function(){
145 		// Recalc offset of original image
146 		original.offset = original.img.offset();
147 
148 		// Move
149 		that.$element.bind("mousemove", function(event){ 
150 			move(event); 
151 		});
152 
153 		// Seeker
154 		seeker.shape.removeClass("ch-hide");
155 		
156 		// Floats show
157 		that.parent.innerShow();
158 
159 		return that;
160 	};
161 
162 	that.innerHide = function(){
163 		// Move
164 		that.$element.unbind("mousemove");
165 		
166 		// Seeker
167 		seeker.shape.addClass("ch-hide");
168 		
169 		// Floats hide
170 		that.parent.innerHide();
171 		
172 		return that;
173 	};

175 	/**
176 	* Triggered on anchor click, it prevents redirection to zoomed image file.
177 	* @protected
178 	* @function
179 	* @name ch.Zoom#enlarge
180 	* @param {Mouse Event Object} event
181 	* @returns {itself}
182 	*/
183 	that.enlarge = function(event){
184 		that.prevent(event);
185 		// Do what you want...
186 		return that;
187 	};
188 	
189 	/**
190 	* Getter and setter for size attributes of float that contains the zoomed image.
191 	* @protected
192 	* @function
193 	* @name size
194 	* @param {String} prop Property that will be setted or getted, like "width" or "height".
195 	* @param {String} [data] Only for setter. It's the new value of defined property.
196 	* @returns {Internal component instance}
197 	* @memberOf ch.Zoom
198 	*/
199 	that.size = function(prop, data) {
200 
201 		if (data) {
202 
203 			// Seeker: shape size relative to zoomed image respect zoomed area
204 			var size = (original[prop]* data) / zoomed[prop];
205 
206 			// Seeker: sets shape size
207 			seeker.shape[prop](size);
208 
209 			// Seeker: save shape half size for position it respect cursor
210 			seeker[prop] = size / 2;
211 
212 		};
213 
214 		return that.parent.size(prop, data);
215 	};
