  2 /**
  3  * Zoom is a standalone UI component that shows a contextual reference to an augmented version of main declared image.
  4  * @name Zoom
  5  * @class Zoom
  6  * @augments ch.Floats
  7  * @requires ch.Positioner
  8  * @requires ch.onImagesLoads
  9  * @memberOf ch
 10  * @param {Configuration Object} conf Object with configuration properties
 11  * @returns {Chico-UI Object}
 12  */
 13 
 14 ch.zoom = function(conf) {
 15 
 16     /**
 17      * Reference to an internal component instance, saves all the information and configuration properties.
 18      * @private
 19      * @name that
 20      * @type {Object}
 21      * @memberOf ch.Zoom
 22      */
 23 	var that = this;
 24 
 25 	conf = ch.clon(conf);
 26 	conf.fx = false;
 27 	
 28 	conf.position = {};
 29 	conf.position.context = conf.context || that.$element;
 30 	conf.position.offset = conf.offset || "20 0";
 31 	conf.position.points = conf.points || "lt rt";
 32 	conf.position.hold = true;
 33 
 34 	that.conf = conf;
 35 
 36 /**
 37  *	Inheritance
 38  */
 39 
 40     that = ch.floats.call(that);
 41     that.parent = ch.clon(that);
 42     
 43 /**
 44  *  Private Members
 45  */
 46 
 47     /**
 48      * Reference to main image declared on HTML code snippet.
 49      * @private
 50      * @name original
 51      * @type {Object}
 52      * @memberOf ch.Zoom
 53      */
 54 	var original = {};
 55 		original.img = that.$element.children();
 56 		original["width"] = original.img.prop("width");
 57 		original["height"] = original.img.prop("height");
 58 
 59     /**
 60      * Reference to the augmented version of image, that will be displayed in context.
 61      * @private
 62      * @name zoomed
 63      * @typeÂ {Object}
 64      * @memberOf ch.Zoom
 65      */
 66 	var zoomed = {};
 67 		// Define the content source 
 68 		zoomed.img = that.source = $("<img>").prop("src", that.element.href);
 69 	
 70     /**
 71      * Seeker is the visual element that follows mouse movement for referencing to zoomable area into original image.
 72      * @private
 73      * @name seeker
 74      * @typeÂ {Object}
 75      * @memberOf ch.Zoom
 76      */
 77 	var seeker = {};
 78 		seeker.shape = $("<div>").addClass("ch-seeker ch-hide")
 79     
 80     /**
 81      * Gets the mouse position relative to original image position, and accordingly moves the zoomed image.
 82      * @private
 83      * @function
 84      * @name move
 85      * @param {Mouse Event Object} event
 86 	 * @returns {void}
 87      * @memberOf ch.Zoom
 88      */
 89 	var move = function(event){
 90 		
 91 		// Cursor coordinates relatives to original image
 92 		var x = event.pageX - original.offset.left;
 93 		var y = event.pageY - original.offset.top;
 94 		
 95 		// Seeker axis
 96 		var limit = {};
 97 			limit.left = parseInt(x - seeker["width"]);
 98 			limit.right = parseInt(x + seeker["width"]);
 99 			limit.top = parseInt(y - seeker["height"]);
100 			limit.bottom = parseInt(y + seeker["height"]);
101 
102 		// Horizontal: keep seeker into limits
103 		if(limit.left >= 0 && limit.right < original["width"] - 1) {
104 			zoomed.img.css("left", -( (parseInt(zoomed["width"] * x) / original["width"]) - (conf.width / 2) ));
105 			seeker.shape.css("left", limit.left);
106 		};
107 		
108 		// Vertical: keep seeker into limits
109 		if(limit.top >= 0 && limit.bottom < original["height"] - 1) {
110 			zoomed.img.css("top", -( (parseInt(zoomed["height"] * y) / original["height"]) - (conf.height / 2) ));
111 			seeker.shape.css("top", limit.top);
112 		};
113 		
114 		return;
115 	};
116 		
117 	/**
118      * Calculates zoomed image sizes and adds event listeners to trigger of float element
119      * @private
120      * @function
121      * @name init
122 	 * @returns {void}
123      * @memberOf ch.Zoom
124      */
125 	var init = function(){
126 		// Zoomed image size
127 		zoomed["width"] = zoomed.img.prop("width");
128 		zoomed["height"] = zoomed.img.prop("height");
129 		
130 		// Anchor
131 		that.$element
132 			// Apend Seeker
133 			.append( seeker.shape )
134 			
135 			// Show
136 			.bind("mouseenter", that.show)
137 			
138 			// Hide
139 			.bind("mouseleave", that.hide)
140 		
141 		return;
142 	};
143 	
144 /**
145  *  Protected Members
146  */
147 	
148 	/**
149      * Anchor that wraps the main image and links to zoomed image file.
150      * @public
151      * @name $trigger
152      * @typeÂ {Object}
153      * @memberOf ch.Zoom
154      */
155 	that.$trigger = that.$element;
156 	
157 	that.show = function(){
158 		// Recalc offset of original image
159 		original.offset = original.img.offset();
160 
161 		// Move
162 		that.$element.bind("mousemove", function(event){ 
163 			move(event); 
164 		});
165 
166 		// Seeker
167 		seeker.shape.removeClass("ch-hide");
168 		
169 		// Floats show
170 		that.parent.show();
171 
172 		return that;
173 	};
174 	
175 	that.hide = function(){
176 		// Move
177 		that.$element.unbind("mousemove");
178 		
179 		// Seeker
180 		seeker.shape.addClass("ch-hide");
181 		
182 		// Floats hide
183 		that.parent.hide();
184 		
185 		return that;
186 	};
187 
188     /**
189      * Triggered on anchor click, it prevents redirection to zoomed image file.
190      * @private
191      * @function
192      * @name enlarge
193      * @param {Mouse Event Object} event
194 	 * @returns {Internal component instance}
195      * @memberOf ch.Zoom
196      */
197 	that.enlarge = function(event){
198 		that.prevent(event);
199 		
200 		// Do what you want...
201 		
202 		return that;
203 	};
204 	
205     /**
206      * Getter and setter for size attributes of float that contains the zoomed image.
207      * @private
208      * @function
209      * @name size
210      * @param {String} prop Property that will be setted or getted, like "width" or "height".
211      * @param {String} [data] Only for setter. It's the new value of defined property.
212 	 * @returns {Internal component instance}
213      * @memberOf ch.Zoom
214      */
215 	that.size = function(prop, data) {
216 
217 		if (data) {
218 
219 			// Seeker: shape size relative to zoomed image respect zoomed area
220 			var size = (original[prop] * data) / zoomed[prop];
221 		
222 			// Seeker: sets shape size
223 			seeker.shape[prop](size);
224 		
225 			// Seeker: save shape half size for position it respect cursor
226 			seeker[prop] = size / 2;
227 
228 		};
229 
230 		return that.parent.size(prop, data);
231 	};

331      * Gets and sets the height size of float element.
332      * @private
333      * @name height
334      * @function
335      * @returns {Chico-UI Object}
336      * @param {Number} data Height value.
337      * @memberOf ch.Zoom
338      * @example
339      * // Gets height of Zoom float element.
340      * foo.height();
341      * @example
342      * // Sets height of Zoom float element and update the seeker size to keep these relation.
343      * foo.height(500);
344      */

349  */
350 	
351 	// Anchor
352 	that.$element
353 		.addClass("ch-zoom-trigger")
354 		
355 		// Size (same as image)
356 		.css({"width": original["width"], "height": original["height"]})
357 		
358 		// Enlarge
359 		.bind("click", function(event){ that.enlarge(event); });
360 	
361 	// Initialize when zoomed image loads...
362 	zoomed.img.onImagesLoads( init );
363 	
364 	return that;
